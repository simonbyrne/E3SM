38 void ElementsDerivedState::randomize(const int seed, const Real dp3d_min) {
39  // Check derived state was inited
40  assert (m_num_elems>0);
41 
42  // Sanity check
43  assert (dp3d_min>0);
44 
45  // arbitrary minimum value to generate
46  constexpr const Real min_value = 0.015625;
47  std::mt19937_64 engine(seed);
48  std::uniform_real_distribution<Real> random_dist(min_value, 1.0 / min_value);
49 
50  genRandArray(m_omega_p, engine, random_dist);
51  genRandArray(m_vn0, engine, random_dist);
52 
53  // Generate eta_dot_dpdn so that it is << dp3d
54  genRandArray(m_eta_dot_dpdn, engine, std::uniform_real_distribution<Real>(0.01*dp3d_min,0.1*dp3d_min));
55 
56  // During remap, in the case of rsplit=0, the target thickness is computed as
57  // dp_star(k) = dp(k) + dt*(eta_dot_dpdn(k+1)-eta_dot_dpdn(k))
58  // Such dp_star should verify sum(dp_star) = sum(dp). Real runs should always
59  // verify this (up to roundoff errors), so we need our "random" eta_dot_dpdn
60  // to also verify this. In the unit tests, we check this with a somewhat
61  // loose tolerance, so it is enough to ensure this up to roundoff errors.
62  // One way to enforce this is to compute the sum of delta(eta_dot_dpdn),
63  // and add the sum to the first entry of eta_dot_dpdn.
64  auto eta_dot_dpdn = m_eta_dot_dpdn;
65  auto policy = Homme::get_default_team_policy<ExecSpace>(m_num_elems);
66  TeamUtils<ExecSpace> tu(policy);
67  const int nslots = tu.get_num_ws_slots();
68  ExecViewManaged<Scalar *[NP][NP][NUM_LEV]> delta_eta("",nslots);
69  Kokkos::parallel_for(policy, KOKKOS_LAMBDA(const TeamMember& team) {
70  KernelVariables kv(team, tu);
71  Kokkos::parallel_for(Kokkos::TeamThreadRange(kv.team,NP*NP),
72  [&](const int idx) {
73  const int igp = idx / NP;
74  const int jgp = idx % NP;
75 
76  auto eta_dot_dpdn_ij = Homme::subview(eta_dot_dpdn,kv.ie,igp,jgp);
77  auto delta_eta_ij = Homme::subview(delta_eta,kv.team_idx,igp,jgp);
78 
79  Kokkos::single(Kokkos::PerThread(kv.team),[&](){
80  constexpr int LAST_PACK = ColInfo<NUM_INTERFACE_LEV>::LastPack;
81  constexpr int LAST_PACK_LEN = ColInfo<NUM_INTERFACE_LEV>::LastPackLen;
82  constexpr int LAST_PACK_END = ColInfo<NUM_INTERFACE_LEV>::LastPackEnd;
83 
84  // Last entry of eta_dot_dpdn should be 0
85  eta_dot_dpdn_ij(LAST_PACK)[LAST_PACK_END] = 0.0;
86 
87  // If VECTOR_SIZE does not divide NUM_INTERFACE_LEV, zero out the
88  // garbage at the end
89  if (LAST_PACK_LEN!=VECTOR_SIZE) {
90  for (int i=LAST_PACK_LEN; i<VECTOR_SIZE; ++i) {
91  eta_dot_dpdn_ij(LAST_PACK)[i] = 0.0;
92  }
93  }
94  });
95 
96  Real sum;
97  // Compute delta_eta over the column
98  ColumnOps::compute_midpoint_delta(kv,eta_dot_dpdn_ij,delta_eta_ij);
99 
100  // Integrate delta_eta over the column, and subtract that value
101  // to the first entry of eta_dot_dpdn
102  ColumnOps::column_reduction<NUM_PHYSICAL_LEV>(kv.team,delta_eta_ij,sum);
103 
104  Kokkos::single(Kokkos::PerThread(kv.team),[&](){
105  eta_dot_dpdn_ij(0)[0] += sum;
106  });
107  });
108  });
109 }
