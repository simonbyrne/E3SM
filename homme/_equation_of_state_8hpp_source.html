83  // Compute:
84  // 1) p_over_exner = -Rgas*vtheta_dp/delta(phi_i)
85  // 2) pnh = p0 (p_over_exner/p0)^(1/(1-kappa))
86  // 3) exner = pnh/p_over_exner
87  template<typename Scalar>
88  KOKKOS_INLINE_FUNCTION
89  static void compute_pnh_and_exner (const Scalar& vtheta_dp, const Scalar& dphi,
90  Scalar& pnh, Scalar& exner) {
91  exner = (-PhysicalConstants::Rgas)*vtheta_dp / dphi;
92  pnh = exner/PhysicalConstants::p0;
93 #ifndef HOMMEXX_BFB_TESTING
94  pnh = pow(pnh,1.0/(1.0-PhysicalConstants::kappa));
95 #else
96  pnh = bfb_pow(pnh,1.0/(1.0-PhysicalConstants::kappa));
97 #endif
98  pnh *= PhysicalConstants::p0;
99  exner = pnh/exner;
100  }

103  void compute_dpnh_dp_i (const KernelVariables& kv,
104  const ExecViewUnmanaged<const Scalar[NUM_LEV ]>& pnh,
105  const ExecViewUnmanaged<const Scalar[NUM_LEV_P]>& dp_i,
106  const ExecViewUnmanaged< Scalar[NUM_LEV_P]>& dpnh_dp_i) const
107  {
108  if (m_theta_hydrostatic_mode) {
109  // Set dpnh_dp_i to 1.0
110  Kokkos::parallel_for(Kokkos::ThreadVectorRange(kv.team,NUM_LEV_P),
111  [&](const int ilev) {
112  dpnh_dp_i(ilev) = 1.0;
113  });
114  } else {
115  // dp_i CANNOT alias dpnh_dp_i
116  assert(dpnh_dp_i.data()!=dp_i.data());
117 
118  // Start with dpnh_dp_i = delta(pnh)/dp_i.
119  ColumnOps::compute_interface_delta<CombineMode::Replace>(kv.team,pnh,dpnh_dp_i);
120 
121  // Note: top and bottom need special treatment, so we may as well stop at NUM_LEV here (rather than NUM_LEV_P)
122  Kokkos::parallel_for(Kokkos::ThreadVectorRange(kv.team,NUM_LEV),
123  [&](const int ilev) {
124  dpnh_dp_i(ilev) /= dp_i(ilev);
125  });
126 
127  // Boundaries: delta(x) = 2*(x_m(last)-x_i(last)).
128  // Top: pnh_i = pi_i = hyai(0)*ps0.
129  // Bottom: approximate with hydrostatic, so that dpnh_dp_i=1
130  dpnh_dp_i(0)[0] = 2*(pnh(0)[0] - m_hvcoord.hybrid_ai(0)*m_hvcoord.ps0)/dp_i(0)[0];
131  const Real pnh_last = pnh(LAST_MID_PACK)[LAST_MID_PACK_END];
132  const Real dp_last = dp_i(LAST_INT_PACK)[LAST_INT_PACK_END];
133  const Real pnh_i_last = pnh_last + dp_last/2;
134 
135  dpnh_dp_i(LAST_INT_PACK)[LAST_INT_PACK_END] = 2*(pnh_i_last - pnh_last)/dp_last;
136  }
