31 // A simple struct, storing a connection info. In addition to LidGidPos (on both local and
32 // remote element), it stores also whether the ordering is the same on both the element
33 // (relevant only for edge-type connections), and the process id of the remote element,
34 // which is only used if the remote element is on a different process.
35 // Note: we store kind, sharing and direction already converted to ints
36 struct ConnectionInfo
37 {
38  LidGidPos local;
39  LidGidPos remote;
40 
41  int kind; // etoi(ConnectionKind::EDGE)=0, etoi(ConnectionKind::CORNER)=1, etoi(ConnectionSharing::MISSING)=2
42  int sharing; // etoi(ConnectionSharing::LOCAL)=0, etoi(ConnectionSharing::SHARED)=1, etoi(ConnectionSharing::MISSING)=2
43 
44 
45  // The following is needed only for W/E/S/N edges, in case the ordering of the NP points is different in the two elements
46  int direction; //0=forward, 1=backward
47 
48  // This is only needed if the neighboring element is owned by a different process
49  int remote_pid; // Process id owning the other side of the connection
50 };
