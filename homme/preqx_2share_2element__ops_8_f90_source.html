379  subroutine set_forcing_rayleigh_friction(elem, zm, zi, ztop, zc, tau, u0,v0, n)
380  !
381  ! test cases which use rayleigh friciton will call this with the relaxation coefficient
382  ! f_d, and the reference state u0,v0. Currently assume w0 = 0
383  !
384  implicit none
385 
386  type(element_t), intent(inout):: elem
387  real(real_kind), intent(in) :: zm(nlev) ! height at layer midpoints
388  real(real_kind), intent(in) :: zi(nlevp) ! height at interfaces
389  real(real_kind), intent(in) :: ztop ! top of atm height
390  real(real_kind), intent(in) :: zc ! cutoff height
391  real(real_kind), intent(in) :: tau ! damping timescale
392  real(real_kind), intent(in) :: u0(np,np,nlev) ! reference u
393  real(real_kind), intent(in) :: v0(np,np,nlev) ! reference v
394  integer, intent(in) :: n ! timestep
395 
396  real(real_kind):: f_d(nlev)
397  integer :: k
398 
399  ! Compute damping as a function of layer-midpoint height
400  f_d=0.0d0
401  where(zm .ge. zc); f_d = sin(dd_pi/2 *(zm - zc)/(ztop - zc))**2; end where
402  where(zm .ge. ztop); f_d = 1.0d0; end where
403  f_d = -f_d/tau
404 
405  do k=1,nlev
406  elem%derived%FM(:,:,1,k) = f_d(k) * ( elem%state%v(:,:,1,k,n) - u0(:,:,k) )
407  elem%derived%FM(:,:,2,k) = f_d(k) * ( elem%state%v(:,:,2,k,n) - v0(:,:,k) )
408  enddo
409  end subroutine
