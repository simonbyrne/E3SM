121 ! Generate the supercell initial conditions
122 !=======================================================================
123  SUBROUTINE supercell_init() &
124  BIND(c, name = "supercell_init")
125 
126  IMPLICIT NONE
127 
128  ! d/dphi and int(dphi) operators
129  REAL(8), DIMENSION(nphi,nphi) :: ddphi, intphi
130 
131  ! d/dz and int(dz) operators
132  REAL(8), DIMENSION(nz, nz) :: ddz, intz
133 
134  ! Buffer matrices for computing SVD of d/dphi operator
135  REAL(8), DIMENSION(nphi,nphi) :: ddphibak
136  REAL(8), DIMENSION(nphi,nphi) :: svdpu, svdpvt
137  REAL(8), DIMENSION(nphi) :: svdps
138  REAL(8), DIMENSION(5*nphi) :: pwork
139 
140  ! Buffer matrices for computing SVD of d/dz operator
141  REAL(8), DIMENSION(nz, nz) :: ddzbak
142  REAL(8), DIMENSION(nz, nz) :: svdzu, svdzvt
143  REAL(8), DIMENSION(nz) :: svdzs
144  REAL(8), DIMENSION(5*nz) :: zwork
145 
146  ! Buffer data for calculation of SVD
147  INTEGER(4) :: lwork, info
148 
149  ! Sampled values of ueq**2 and d/dz(ueq**2)
150  REAL(8), DIMENSION(nphi, nz) :: ueq2, dueq2
151 
152  ! Buffer matrices for iteration
153  REAL(8), DIMENSION(nphi, nz) :: phicoordmat, dztheta, rhs, irhs
154 
155  ! Buffer for sampled potential temperature at equator
156  REAL(8), DIMENSION(nz) :: thetaeq
157 
158  ! Buffer for computed equatorial Exner pressure and relative humidity
159  REAL(8), DIMENSION(nz) :: exnereq, H
160 
161  ! Variables for calculation of equatorial profile
162  REAL(8) :: exnereqs, p, T, qvs, qv
163 
164  ! Error metric
165  REAL(8) :: err
166 
167  ! Loop indices
168  INTEGER(4) :: i, k, iter
169 
170  ! Chebyshev nodes in the phi direction
171  do i = 1, nphi
172  phicoord(i) = - cos(dble(i-1) * pi / dble(nphi-1))
173  phicoord(i) = 0.25d0 * pi * (phicoord(i) + 1.0d0)
174  end do
175 
176  ! Matrix of phis
177  do k = 1, nz
178  phicoordmat(:,k) = phicoord
179  end do
180 
181  ! Chebyshev nodes in the z direction
182  do k = 1, nz
183  zcoord(k) = - cos(dble(k-1) * pi / dble(nz-1))
184  zcoord(k) = z1 + 0.5d0*(z2-z1)*(zcoord(k)+1.0d0)
185  end do
186 
187  ! Compute the d/dphi operator
188  do i = 1, nphi
189  call diff_lagrangian_polynomial_coeffs( &
190  nphi, phicoord, ddphi(:,i), phicoord(i))
191  end do
192 
193  ! Zero derivative at pole
194  ddphi(:,nphi) = 0.0d0
195 
196  ! Compute the d/dz operator
197  do k = 1, nz
198  call diff_lagrangian_polynomial_coeffs( &
199  nz, zcoord, ddz(:,k), zcoord(k))
200  end do
201 
202  ! Compute the int(dphi) operator via pseudoinverse
203  lwork = 5*nphi
204 
205  ddphibak = ddphi
206  call dgesvd('A', 'A', &
207  nphi, nphi, ddphibak, nphi, &
208  svdps, svdpu, nphi, svdpvt, nphi, &
209  pwork, lwork, info)
210 
211  if (info .ne. 0) then
212  write(*,*) 'Unable to compute SVD of d/dphi matrix'
213  stop
214  end if
215 
216  do i = 1, nphi
217  if (abs(svdps(i)) .le. 1.0d-12) then
218  call dscal(nphi, 0.0d0, svdpu(1,i), 1)
219  else
220  call dscal(nphi, 1.0d0 / svdps(i), svdpu(1,i), 1)
221  end if
222  end do
223  call dgemm('T', 'T', &
224  nphi, nphi, nphi, 1.0d0, svdpvt, nphi, svdpu, nphi, 0.0d0, &
225  intphi, nphi)
226 
227  ! Compute the int(dz) operator via pseudoinverse
228  lwork = 5*nz
229 
230  ddzbak = ddz
231  call dgesvd('A', 'A', &
232  nz, nz, ddzbak, nz, &
233  svdzs, svdzu, nz, svdzvt, nz, &
234  zwork, lwork, info)
235 
236  if (info .ne. 0) then
237  write(*,*) 'Unable to compute SVD of d/dz matrix'
238  stop
239  end if
240 
241  do i = 1, nz
242  if (abs(svdzs(i)) .le. 1.0d-12) then
243  call dscal(nz, 0.0d0, svdzu(1,i), 1)
244  else
245  call dscal(nz, 1.0d0 / svdzs(i), svdzu(1,i), 1)
246  end if
247  end do
248  call dgemm('T', 'T', &
249  nz, nz, nz, 1.0d0, svdzvt, nz, svdzu, nz, 0.0d0, &
250  intz, nz)
251 
252  ! Sample the equatorial velocity field and its derivative
253  do k = 1, nz
254  ueq2(1,k) = zonal_velocity(zcoord(k), 0.0d0)
255  ueq2(1,k) = ueq2(1,k)**2
256  end do
257  do k = 1, nz
258  dueq2(1,k) = dot_product(ddz(:,k), ueq2(1,:))
259  end do
260  do i = 2, nphi
261  ueq2(i,:) = ueq2(1,:)
262  dueq2(i,:) = dueq2(1,:)
263  end do
264 
265  ! Initialize potential temperature at equator
266  do k = 1, nz
267  thetaeq(k) = equator_theta(zcoord(k))
268  h(k) = equator_relative_humidity(zcoord(k))
269  end do
270  thetavyz(1,:) = thetaeq
271 
272  ! Exner pressure at the equatorial surface
273  exnereqs = (pseq / p0)**(rd/cp)
274 
275  ! Iterate on equatorial profile
276  do iter = 1, 12
277 
278  ! Calculate Exner pressure in equatorial column (p0 at surface)
279  rhs(1,:) = - g / cp / thetavyz(1,:)
280  do k = 1, nz
281  exnereq(k) = dot_product(intz(:,k), rhs(1,:))
282  end do
283  do k = 2, nz
284  exnereq(k) = exnereq(k) + (exnereqs - exnereq(1))
285  end do
286  exnereq(1) = exnereqs
287 
288  ! Calculate new pressure and temperature
289  do k = 1, nz
290  p = p0 * exnereq(k)**(cp/rd)
291  t = thetaeq(k) * exnereq(k)
292 
293  qvs = saturation_mixing_ratio(p, t)
294  qveq(k) = qvs * h(k)
295 
296  !thetavyz(1,k) = thetaeq(k) * (1.d0 + 0.61d0 * qveq(k))
297  thetavyz(1,k) = thetaeq(k) * (1.d0 + mvap * qveq(k))
298 
299  end do
300  end do
301 
302  !do k = 1, nz
303  ! write(*,*) exnereq(k) * thetaeq(k)
304  !end do
305 
306  ! Iterate on remainder of domain
307  do iter = 1, 12
308 
309  ! Compute d/dz(theta)
310  do i = 1, nphi
311  do k = 1, nz
312  dztheta(i,k) = dot_product(ddz(:,k), thetavyz(i,:))
313  end do
314  end do
315 
316  ! Compute rhs
317  rhs = sin(2.0d0*phicoordmat)/(2.0d0*g) &
318  * (ueq2 * dztheta - thetavyz * dueq2)
319 
320  ! Integrate
321  do k = 1, nz
322  do i = 1, nphi
323  irhs(i,k) = dot_product(intphi(:,i), rhs(:,k))
324  end do
325  end do
326 
327  ! Apply boundary conditions (fixed Dirichlet condition at equator)
328  do i = 2, nphi
329  irhs(i,:) = irhs(i,:) + (thetavyz(1,:) - irhs(1,:))
330  end do
331  irhs(1,:) = thetavyz(1,:)
332 
333  ! Compute difference after iteration
334  !err = sum(irhs - thetavyz)
335  !write(*,*) iter, err
336 
337  ! Update iteration
338  thetavyz = irhs
339  end do
340 
341  ! Calculate pressure through remainder of domain
342  rhs = - ueq2 * sin(phicoordmat) * cos(phicoordmat) / cp / thetavyz
343 
344  do k = 1, nz
345  do i = 1, nphi
346  exneryz(i,k) = dot_product(intphi(:,i), rhs(:,k))
347  end do
348  do i = 2, nphi
349  exneryz(i,k) = exneryz(i,k) + (exnereq(k) - exneryz(1,k))
350  end do
351 
352  exneryz(1,k) = exnereq(k)
353  end do
354 
355  ! Initialization successful
356  initialized = 1
357 
358  END SUBROUTINE supercell_init

440 !-----------------------------------------------------------------------
441 ! Calculate pointwise pressure and temperature
442 !-----------------------------------------------------------------------
443  SUBROUTINE supercell_z(lon, lat, z, p, thetav, rho, q, pert)
444 
445  REAL(8), INTENT(IN) :: &
446  lon, & ! Longitude (radians)
447  lat, & ! Latitude (radians)
448  z ! Altitude (m)
449 
450  INTEGER, INTENT(IN) :: pert ! 1 if perturbation should be included
451  ! 0 if no perturbation should be included
452 
453  ! Evaluated variables
454  REAL(8), INTENT(OUT) :: p, thetav, rho, q
455 
456  ! Northern hemisphere latitude
457  REAL(8) :: nh_lat
458 
459  ! Pointwise Exner pressure
460  REAL(8) :: exner
461 
462  ! Assembled variable values in a column
463  REAL(8), DIMENSION(nz) :: varcol
464 
465  ! Coefficients for computing a polynomial fit in each coordinate
466  REAL(8), DIMENSION(nphi) :: fitphi
467  REAL(8), DIMENSION(nz) :: fitz
468 
469  ! Loop indices
470  INTEGER(4) :: k
471 
472  ! Northern hemisphere latitude
473  if (lat .le. 0.0d0) then
474  nh_lat = -lat
475  else
476  nh_lat = lat
477  end if
478 
479  ! Perform fit
480  CALL lagrangian_polynomial_coeffs(nz, zcoord, fitz, z)
481  CALL lagrangian_polynomial_coeffs(nphi, phicoord, fitphi, nh_lat)
482 
483  ! Obtain exner pressure of background state
484  do k = 1, nz
485  varcol(k) = dot_product(fitphi, exneryz(:,k))
486  end do
487  exner = dot_product(fitz, varcol)
488  p = p0 * exner**(cp/rd)
489 
490  ! Sample the initialized fit at this point for theta_v
491  do k = 1, nz
492  varcol(k) = dot_product(fitphi, thetavyz(:,k))
493  end do
494  thetav = dot_product(fitz, varcol)
495 
496  ! Sample water vapor mixing ratio
497  q = dot_product(fitz, qveq)
498 
499  ! Fixed density
500  rho = p / (rd * exner * thetav)
501 
502  ! Modified virtual potential temperature
503  if (pert .ne. 0) then
504  thetav = thetav + thermal_perturbation(lon, lat, z) * (1.d0 + mvap * q)
505  end if
506 
507  ! Updated pressure
508  p = p0 * (rho * rd * thetav / p0)**(cp/(cp-rd))
509 
510  END SUBROUTINE supercell_z

512 !-----------------------------------------------------------------------
513 ! Calculate pointwise z and temperature given pressure
514 !-----------------------------------------------------------------------
515  SUBROUTINE supercell_p(lon, lat, p, z, thetav, rho, q, pert)
516 
517  REAL(8), INTENT(IN) :: &
518  lon, & ! Longitude (radians)
519  lat, & ! Latitude (radians)
520  p ! Pressure (Pa)
521 
522  INTEGER, INTENT(IN) :: pert ! 1 if perturbation should be included
523  ! 0 if no perturbation should be included
524 
525  ! Evaluated variables
526  REAL(8), INTENT(OUT) :: z, thetav, rho, q
527 
528  ! Bounding interval and sampled values
529  REAL(8) :: za, zb, zc, pa, pb, pc
530 
531  ! Iterate
532  INTEGER(4) :: iter
533 
534  za = z1
535  zb = z2
536 
537  CALL supercell_z(lon, lat, za, pa, thetav, rho, q, pert)
538  CALL supercell_z(lon, lat, zb, pb, thetav, rho, q, pert)
539 
540  if (pa .lt. p) then
541  write(*,*) 'Requested pressure out of range on bottom, adjust sample interval'
542  write(*,*) pa, p
543  stop
544  end if
545  if (pb .gt. p) then
546  write(*,*) 'Requested pressure out of range on top, adjust sample interval'
547  write(*,*) pb, p
548  stop
549  end if
550 
551  ! Iterate using fixed point method
552  do iter = 1, 400
553 
554  zc = (za * (pb - p) - zb * (pa - p)) / (pb - pa)
555 
556  CALL supercell_z(lon, lat, zc, pc, thetav, rho, q, pert)
557 
558  !print *,"pc=",pc," zc=",zc
559 
560  !if (abs((pc - p) / p) .lt. 1.d-14) then
561  if (abs((pc - p) / p) .lt. 1.d-12) then
562  exit
563  end if
564 
565  if (pc .gt. p) then
566  za = zc
567  pa = pc
568  else
569  zb = zc
570  pb = pc
571  end if
572  end do
573 
574  if (iter .eq. 201) then
575  write(*,*) 'Iteration failed to converge'
576  print *,"pc=",pc," zc=",zc
577  stop
578  end if
579 
580  z = zc
581 
582  END SUBROUTINE supercell_p

584 !-----------------------------------------------------------------------
585 ! Calculate pointwise z and temperature given pressure
586 !-----------------------------------------------------------------------
587  REAL(8) function thermal_perturbation(lon, lat, z)
588 
589  REAL(8), INTENT(IN) :: &
590  lon, & ! Longitude (radians)
591  lat, & ! Latitude (radians)
592  z ! Altitude (m)
593 
594  ! Great circle radius from the perturbation centerpoint
595  REAL(8) :: gr
596 
597  ! Approximately spherical radius from the perturbation centerpoint
598  REAL(8) :: rtheta
599 
600  gr = rearth*acos(sin(pert_latc*deg2rad)*sin(lat) + &
601  (cos(pert_latc*deg2rad)*cos(lat)*cos(lon-pert_lonc*deg2rad)))
602 
603  rtheta = sqrt((gr/pert_rh)**2 + ((z - pert_zc) / pert_rz)**2)
604 
605  if (rtheta .le. 1.d0) then
606  thermal_perturbation = pert_dtheta * (cos(0.5d0 * pi * rtheta))**2
607  else
608  thermal_perturbation = 0.0d0
609  end if
610 
611  END FUNCTION thermal_perturbation

613 !-----------------------------------------------------------------------
614 ! Calculate the reference zonal velocity
615 !-----------------------------------------------------------------------
616  REAL(8) function zonal_velocity(z, lat)
617 
618  IMPLICIT NONE
619 
620  REAL(8), INTENT(IN) :: z, lat
621 
622  if (z .le. zs - zt) then
623  zonal_velocity = us * (z / zs) - uc
624  elseif (abs(z - zs) .le. zt) then
625  zonal_velocity = &
626  (-4.0d0/5.0d0 + 3.0d0*z/zs - 5.0d0/4.0d0*(z**2)/(zs**2)) * us - uc
627  else
628  zonal_velocity = us - uc
629  end if
630 
631  zonal_velocity = zonal_velocity * cos(lat)
632 
633  END FUNCTION zonal_velocity

637 !-----------------------------------------------------------------------

639 
640  IMPLICIT NONE
641 
642  REAL(8), INTENT(IN) :: z
643

794  cycle
795  end if
