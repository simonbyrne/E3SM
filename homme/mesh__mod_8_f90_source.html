753  enddo
754  enddo

759  ! Mesh in Mesh2_map. elements in Mesh2 which are not mapped get assigned

765  y = centroids(i,2)

767  ! [ -pi/4, pi/4 ] -> [ 0, ne2 ]

771  if (face == 1 .or. face == 2 .or. face == 6) j2 = ne2-j2+1
772  if (i2<1 ) i2=1

774  if (j2<1 ) j2=1

776  mesh2_map(i2,j2)=i
777  end if

780  ! which map to Mesh.

783  j2 = sfcij(j,2)
784  i = mesh2_map(i2,j2)

786  ! (i2,j2) element maps to element
787  sfcidx(i) = sfc_index

789  endif
790  enddo
791  enddo

851  subroutine gen_mesh_data(nx, ny, centroids, h)
852  integer, intent(in) :: nx, ny
853  real(kind=real_kind), intent(out) :: centroids(ny,nx,2), h
854  integer :: iy, ix, d, i, j, iy1, ix1
855  real(kind=real_kind) :: dx, dy, xo, yo, x, y, r(2), pmin(2), pmax(2), x0, y0, x1, y1
856  real(kind=real_kind), parameter :: pi = 3.141592653589793116d0, delta = 0.01d0
857 
858  dx = 1.0d0 / (nx-1)
859  dy = 1.0d0 / (ny-1)
860 
861  pmin = (/ 1, 1 /)
862  pmax = (/ 0, 0 /)
863 
864  ! Generate an interesting mesh.
865  do ix = 1, nx
866  do iy = 1, ny
867  call random_number(r)
868  r = 2*(r - 0.5d0)
869  xo = (ix-1)*dx
870  yo = (iy-1)*dy
871  x = xo + 0.2*dx*r(1) + 0.1d0*cos(pi*yo)
872  y = yo + 0.2*dy*r(2) + 0.1d0*sin(pi*xo)
873  centroids(iy,ix,:) = (/ x, y /)
874  do d = 1, 2
875  pmin(d) = min(pmin(d), centroids(iy,ix,d))
876  pmax(d) = max(pmax(d), centroids(iy,ix,d))
877  end do
878  end do
879  end do
880  ! Normalize it to fit in (0,1)x(0,1)
881  do ix = 1, nx
882  do iy = 1, ny
883  do d = 1, 2
884  centroids(iy,ix,d) = delta + (centroids(iy,ix,d) - pmin(d)) / &
885  (2*delta + (pmax(d) - pmin(d)))
886  end do
887  end do
888  end do
889  ! Get the smallest element dimension.
890  h = 1
891  do ix = 1, nx
892  do iy = 1, ny
893  x0 = centroids(iy,ix,1)
894  y0 = centroids(iy,ix,2)
895  do i = -1, 1
896  do j = -1, 1
897  if (i == 0 .and. j == 0) cycle
898  ix1 = modulo(ix + i - 1, nx) + 1
899  iy1 = modulo(iy + j - 1, ny) + 1
900  x1 = centroids(iy1,ix1,1)
901  y1 = centroids(iy1,ix1,2)
902  h = min(h, sqrt((x1 - x0)**2 + (y1 - y0)**2))
903  end do
904  end do
905  end do
906  end do
907  ! Scale to [-pi/4, pi/4]
908  do ix = 1, nx
909  do iy = 1, ny
910  do d = 1, 2
911  centroids(iy,ix,d) = 0.5d0*pi*(centroids(iy,ix,d) - 0.5d0)
912  end do
913  end do
914  end do
915  end subroutine gen_mesh_data

954  subroutine find_corner_neighbors (GridVertex, normal_to_homme_ordering, element_nodes, corner_wgt, index_table)
955  use parallel_mod, only : abortmp
956  use gridgraph_mod, only : gridvertex_t
957  use dimensions_mod, only : max_elements_attached_to_node, max_corner_elem
958  use control_mod, only: north, south, east, west, neast,seast, nwest,swest
959  implicit none
960 
961  type (gridvertex_t), intent(inout) :: gridvertex(:)
962  integer , intent(in) :: normal_to_homme_ordering(8)
963  integer , intent(in) :: element_nodes(p_number_elements, 4)
964  integer , intent(in) :: corner_wgt
965  integer , intent(in) :: index_table(:,:)
966 
967  integer :: node_elements (2*max_elements_attached_to_node)
968  integer :: elem_neighbor (4*max_elements_attached_to_node)
969  integer :: nbr_cnt(4)
970  integer :: elem_nbr_start, start
971  integer :: i, j, k, ll, jj, kk
972  integer :: node, loc, cnt, cnt_index
973  integer :: corner_array(max_corner_elem), orig_pos(max_corner_elem)
974  integer :: face_array(max_corner_elem), a_corner_elems(max_corner_elem)
975  integer :: corner_sides(2)
976  integer :: side_elem, corner_elem, tmp_s
977 
978  !the last column in the index table is a count of the number of elements
979  cnt_index = max_elements_attached_to_node + 1
980 
981  do i=1, p_number_elements !loop through all elements
982  node_elements(:) = 0
983  elem_neighbor(:) = 0
984  elem_nbr_start = 0
985  nbr_cnt(:) = 0
986 
987  do j=1,4 !check each of the 4 nodes at the element corners
988  node = element_nodes(i,j)
989  cnt = index_table(node, cnt_index)
990  if (cnt < 3 .or. max_elements_attached_to_node < cnt) then
991  call abortmp('find_corner_neighbors: Number of elements attached to node less than 3 or greater than maximum.')
992  endif
993 
994  node_elements(1:cnt) = index_table(node, 1:cnt)
995 
996  !now node_elements contains the element neighbors to that node - so grab the
997  ! corner neighbors - these are the ones that are not already side neighbors (or myself)
998  k = 0
999  do ll=1,cnt
1000  if ( i /= node_elements(ll) .and. & !not me
1001  gridvertex(i)%nbrs(1) /= node_elements(ll) .and. & !not side 1
1002  gridvertex(i)%nbrs(2) /= node_elements(ll) .and. & ! etc ...
1003  gridvertex(i)%nbrs(3) /= node_elements(ll) .and. &
1004  gridvertex(i)%nbrs(4) /= node_elements(ll)) then
1005  k = k + 1
1006  elem_neighbor(elem_nbr_start + k) = node_elements(ll)
1007  end if
1008  end do ! end of ll loop for multiplicity
1009 
1010  !keep track of where we are starting in elem_neighbor for each corner j
1011  elem_nbr_start = elem_nbr_start + k
1012  nbr_cnt(j) = k !how many neighbors in this corner
1013  end do ! end of j loop through 4 nodes
1014 
1015 
1016  ! now that we have done the 4 corners we can populate nbrs and nbrs_ptr
1017  ! with the corners in the proper order (clockwise) in neighbors
1018  ! also we can add the corner weight
1019 
1020  do j=5,8 !loop through 4 corners
1021  elem_nbr_start = 1
1022  !easiest to do the corner in ascending order - find loc
1023  do jj = 5,8
1024  ll = normal_to_homme_ordering(jj)
1025  if (j == ll) then
1026  loc = jj
1027  exit
1028  end if
1029  elem_nbr_start = elem_nbr_start + nbr_cnt(jj-4)
1030  end do
1031 
1032  start = gridvertex(i)%nbrs_ptr(j)
1033  cnt = nbr_cnt(loc - 4)
1034  gridvertex(i)%nbrs_ptr(j+1) = start + cnt
1035 
1036  if (cnt > 0) then
1037  gridvertex(i)%nbrs(start : start + cnt-1) = &
1038  elem_neighbor(elem_nbr_start : elem_nbr_start + cnt -1)
1039  gridvertex(i)%nbrs_face(start : start + cnt - 1) = &
1040  gridvertex(elem_neighbor(elem_nbr_start : elem_nbr_start + cnt -1))%face_number
1041  gridvertex(i)%nbrs_wgt(start : start + cnt-1) = corner_wgt
1042 
1043  end if
1044 
1045  ! within each corner neighbor, lets list the corners in clockwise order
1046  if (cnt > 1) then !cnt is the number of neighbors in this corner j
1047  !there can be at most max_corner element of these
1048 
1049  a_corner_elems = 0
1050  a_corner_elems(1:cnt) = elem_neighbor(elem_nbr_start : elem_nbr_start + cnt -1)
1051  !corner-sides(2) is clockwise of corner_side(1)
1052  corner_array= 0
1053  orig_pos = 0
1054  select case (j)
1055  case(neast)
1056  corner_sides(1) = north
1057  corner_sides(2) = east
1058  case(seast)
1059  corner_sides(1) = east
1060  corner_sides(2) = south
1061  case(swest)
1062  corner_sides(1) = south
1063  corner_sides(2) = west
1064  case(nwest)
1065  corner_sides(1) = west
1066  corner_sides(2) = north
1067  end select
1068 
1069  !so the first element to list touches corner_sides(1) element
1070  side_elem = gridvertex(i)%nbrs(corner_sides(1))
1071 
1072  !loop though the corner elements and see if any have a side neighbor
1073  !that = side_elem
1074  do k = 1,cnt !number of corner elements
1075  corner_elem = a_corner_elems(k)
1076  do kk = 1,4 !number of sides to check
1077  loc = gridvertex(corner_elem)%nbrs_ptr(kk)
1078  tmp_s = gridvertex(corner_elem)%nbrs(loc)
1079  if (tmp_s == side_elem) then
1080  corner_array(1) = corner_elem
1081  orig_pos(1) = k
1082  exit
1083  endif
1084  enddo
1085  if (corner_array(1)> 0) exit
1086  enddo
1087  if (corner_array(1)==0) then
1088  print *, i, cnt
1089  call abortmp('find_corner_neighbors (1) : mistake finding corner neighbor order')
1090  endif
1091 
1092  !if cnt == 2, we are done (we know the order of neighbors)
1093  if (cnt ==2) then
1094  if (corner_array(1) == a_corner_elems(1)) then
1095  corner_array(2) = a_corner_elems(2)
1096  orig_pos(2) = 2
1097  else
1098  corner_array(2) = a_corner_elems(1)
1099  orig_pos(2) = 1
1100  end if
1101  else !cnt = 3 or 4
1102  !find which corner element borders corner_sides(2)
1103  side_elem = gridvertex(i)%nbrs(corner_sides(2))
1104  do k = 1,cnt
1105  corner_elem = a_corner_elems(k)
1106  do kk = 1,4
1107  loc = gridvertex(corner_elem)%nbrs_ptr(kk)
1108  tmp_s = gridvertex(corner_elem)%nbrs(loc)
1109  if (tmp_s == side_elem) then
1110  corner_array(4) = corner_elem
1111  orig_pos(4) = k
1112  exit
1113  endif
1114  enddo
1115  if (corner_array(4)> 0) exit
1116  enddo
1117  if (corner_array(4)==0 .or. corner_array(4) == corner_array(1)) then
1118  print *, i, cnt
1119  call abortmp('find_corner_neighbors (2) : mistake finding corner neighbor order')
1120  endif
1121 
1122  !now if cnt = 3 then we are done
1123  if (cnt ==3) then
1124  corner_array(3) = corner_array(4)
1125  orig_pos(3) = orig_pos(4)
1126 
1127  do k = 1,cnt !find the "middle" element
1128  if (k /= orig_pos(1) .and. k /= orig_pos(3)) then
1129  orig_pos(2) = k
1130  corner_array(2) = a_corner_elems(k)
1131  exit
