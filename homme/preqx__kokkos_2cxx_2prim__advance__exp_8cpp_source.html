25 void prim_advance_exp (TimeLevel& tl, const Real dt, const bool compute_diagnostics)
26 {
27  GPTLstart("tl-ae prim_advance_exp");
28  // Get simulation params
29  SimulationParams& params = Context::singleton().get<SimulationParams>();
30 
31  // Note: In the following, all the checks are superfluous, since we already check that
32  // the options are supported when we init the simulation params. However, this way
33  // we remind ourselves that in these cases there is some missing code to convert from Fortran
34 
35  // Determine the tracers time level
36  tl.n0_qdp= -1;
37  if (params.moisture == MoistDry::MOIST) {
38  tl.update_tracers_levels(params.qsplit);
39  }
40 
41  // Set eta_ave_w
42  Real eta_ave_w = 1.0/params.qsplit;
43 
44 #ifndef CAM
45  // if "prescribed wind" set dynamics explicitly and skip time-integration
46  if (params.prescribed_wind) {
47  Errors::runtime_abort("'prescribed wind' functionality not yet available in C++ build.\n",
48  Errors::err_not_implemented);
49  }
50 #endif
51 
52  // Perform time-advance
53  switch (params.time_step_type) {
54  case TimeStepType::ULLRICH_RK35:
55  // Perform RK stages
56  u3_5stage_timestep(tl, dt, eta_ave_w);
57  break;
58  default:
59  {
60  std::string msg = "[prim_advance_exp_iter]:";
61  msg += "missing some code for this time step method. ";
62  msg += "The program should have errored out earlier though. Plese, investigate.";
63  Errors::runtime_abort(msg,Errors::err_not_implemented);
64  }
65  }
66 
67 #ifdef ENERGY_DIAGNOSTICS
68  if (compute_diagnostics) {
69  Errors::runtime_abort("'compute diagnostic' functionality not yet available in C++ build.\n",
70  Errors::err_not_implemented);
71  }
72 #endif
73 
74  if (!is_implicit(params.time_step_type)) {
75  // Get and run the HVF
76  HyperviscosityFunctor& functor = Context::singleton().get<HyperviscosityFunctor>();
77  GPTLstart("tl-ae advance_hypervis_dp");
78  functor.run(tl.np1,dt,eta_ave_w);
79  GPTLstop("tl-ae advance_hypervis_dp");
80  }
81 
82 #ifdef ENERGY_DIAGNOSTICS

93 void u3_5stage_timestep(const TimeLevel& tl, const Real dt, const Real eta_ave_w)
94 {
95  GPTLstart("tl-ae U3-5stage_timestep");
96  // Get elements structure
97  Elements& elements = Context::singleton().get<Elements>();
98 
99  // Create the functor
100  CaarFunctor& functor = Context::singleton().get<CaarFunctor>();
101 
102  // ===================== RK STAGES ===================== //
103 
104  // Stage 1: u1 = u0 + dt/5 RHS(u0), t_rhs = t
105  functor.run(RKStageData(tl.n0,tl.n0,tl.nm1,tl.n0_qdp,dt/5.0,eta_ave_w/4.0));
106 
107  // Stage 2: u2 = u0 + dt/5 RHS(u1), t_rhs = t + dt/5
108  functor.run(RKStageData(tl.n0,tl.nm1,tl.np1,tl.n0_qdp,dt/5.0,0.0));
109 
110  // Stage 3: u3 = u0 + dt/3 RHS(u2), t_rhs = t + dt/5 + dt/5
111  functor.run(RKStageData(tl.n0,tl.np1,tl.np1,tl.n0_qdp,dt/3.0,0.0));
112 
113  // Stage 4: u4 = u0 + 2dt/3 RHS(u3), t_rhs = t + dt/5 + dt/5 + dt/3
114  functor.run(RKStageData(tl.n0,tl.np1,tl.np1,tl.n0_qdp,2.0*dt/3.0,0.0));
115 
116  // Compute (5u1-u0)/4 and store it in timelevel nm1
117  {
118  const auto t = elements.m_state.m_t;
119  const auto v = elements.m_state.m_v;
120  const auto dp3d = elements.m_state.m_dp3d;
121  const auto nm1 = tl.nm1;
122  const auto n0 = tl.n0;
123  Kokkos::parallel_for(
124  Kokkos::RangePolicy<ExecSpace>(0, elements.num_elems()*NP*NP*NUM_LEV),
125  KOKKOS_LAMBDA(const int it) {
126  const int ie = it / (NP*NP*NUM_LEV);
127  const int igp = (it / (NP*NUM_LEV)) % NP;
128  const int jgp = (it / NUM_LEV) % NP;
129  const int ilev = it % NUM_LEV;
130  t(ie,nm1,igp,jgp,ilev) = (5.0*t(ie,nm1,igp,jgp,ilev)-t(ie,n0,igp,jgp,ilev))/4.0;
131  v(ie,nm1,0,igp,jgp,ilev) = (5.0*v(ie,nm1,0,igp,jgp,ilev)-v(ie,n0,0,igp,jgp,ilev))/4.0;
132  v(ie,nm1,1,igp,jgp,ilev) = (5.0*v(ie,nm1,1,igp,jgp,ilev)-v(ie,n0,1,igp,jgp,ilev))/4.0;
133  dp3d(ie,nm1,igp,jgp,ilev) = (5.0*dp3d(ie,nm1,igp,jgp,ilev)-dp3d(ie,n0,igp,jgp,ilev))/4.0;
134  });
135  }
136  ExecSpace::impl_static_fence();
137 
138  // Stage 5: u5 = (5u1-u0)/4 + 3dt/4 RHS(u4), t_rhs = t + dt/5 + dt/5 + dt/3 + 2dt/3
139  functor.run(RKStageData(tl.nm1,tl.np1,tl.np1,tl.n0_qdp,3.0*dt/4.0,3.0*eta_ave_w/4.0));
140  GPTLstop("tl-ae U3-5stage_timestep");
141 }
