80  subroutine divergence_sphere_wk_openacc(v,deriv,elem,div,len,nets,nete,ntl,tl)
81  use element_mod, only: element_t
82  use physical_constants, only: rrearth
83  implicit none
84 ! input: v = velocity in lat-lon coordinates
85 ! ouput: div(v) spherical divergence of v, integrated by parts
86 ! Computes -< grad(psi) dot v >
87 ! (the integrated by parts version of < psi div(v) > )
88 ! note: after DSS, divergence_sphere () and divergence_sphere_wk()
89 ! are identical to roundoff, as theory predicts.
90  real(kind=real_kind), intent(in) :: v(np,np,2,len,ntl,nelemd) ! in lat-lon coordinates
91  type (derivative_t) , intent(in) :: deriv
92  type (element_t) , intent(in) :: elem(:)
93  real(kind=real_kind), intent(out):: div(np,np,len,ntl,nelemd)
94  integer , intent(in) :: len
95  integer , intent(in) :: nets , nete , ntl , tl
96  ! Local
97  integer, parameter :: kchunk = 8
98  integer :: i,j,l,k,ie,kc,kk
99  real(kind=real_kind) :: vtemp(np,np,2,kchunk), tmp, deriv_tmp(np,np)
100  ! latlon- > contra
101  !$acc parallel loop gang collapse(2) present(v,elem(:),div,deriv) private(vtemp,deriv_tmp)
102  do ie = nets , nete
103  do kc = 1 , len/kchunk+1
104  !$acc cache(vtemp,deriv_tmp)
105  !$acc loop vector collapse(3)
106  do kk = 1 , kchunk
107  do j = 1 , np
108  do i = 1 , np
109  k = (kc-1)*kchunk+kk
110  if (k <= len) then
111  vtemp(i,j,1,kk)=elem(ie)%spheremp(i,j)*(elem(ie)%Dinv(i,j,1,1)*v(i,j,1,k,tl,ie) + elem(ie)%Dinv(i,j,1,2)*v(i,j,2,k,tl,ie))
112  vtemp(i,j,2,kk)=elem(ie)%spheremp(i,j)*(elem(ie)%Dinv(i,j,2,1)*v(i,j,1,k,tl,ie) + elem(ie)%Dinv(i,j,2,2)*v(i,j,2,k,tl,ie))
113  endif
114  if (kk == 1) deriv_tmp(i,j) = deriv%Dvv(i,j)
115  enddo
116  enddo
117  enddo
118  !$acc loop vector collapse(3) private(tmp)
119  do kk = 1 , kchunk
120  do j = 1 , np
121  do i = 1 , np
122  k = (kc-1)*kchunk+kk
123  if (k <= len) then
124  tmp = 0.
125  do l = 1 , np
126  tmp = tmp - ( vtemp(l,j,1,kk)*deriv_tmp(i,l) + vtemp(i,l,2,kk)*deriv_tmp(j,l) )
127  enddo
128  div(i,j,k,tl,ie) = tmp * rrearth
129  endif
130  enddo
131  enddo
132  enddo
133  enddo
134  enddo

192 ! ouput: div(v) spherical divergence of v
193  use element_mod , only: element_t

217  gv(i,j,kk,2)=elem(ie)%metdet(i,j)*(elem(ie)%Dinv(i,j,2,1)*v(i,j,1,k,tl,ie) + elem(ie)%Dinv(i,j,2,2)*v(i,j,2,k,tl,ie))
218  endif

220  enddo
221  enddo
222  enddo
223  ! compute d/dx and d/dy
224  !$acc loop vector collapse(3) private(dudx00,dvdy00,k)
225  do kk = 1 , kchunk
226  do j = 1 , np
227  do i = 1 , np
228  k = (kc-1)*kchunk+kk

233  dudx00 = dudx00 + deriv_tmp(l,i)*gv(l,j,kk,1)
234  dvdy00 = dvdy00 + deriv_tmp(l,j)*gv(i,l,kk,2)
235  enddo
