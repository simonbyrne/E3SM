36 void init_simulation_params_c (const int& remap_alg, const int& limiter_option, const int& rsplit, const int& qsplit,
37  const int& time_step_type, const int& qsize, const int& state_frequency,
38  const Real& nu, const Real& nu_p, const Real& nu_q, const Real& nu_s, const Real& nu_div, const Real& nu_top,
39  const int& hypervis_order, const int& hypervis_subcycle, const double& hypervis_scaling,
40  const int& ftype, const bool& prescribed_wind, const bool& moisture, const bool& disable_diagnostics,
41  const bool& use_cpstar, const bool& use_semi_lagrangian_transport)
42 {
43  // Check that the simulation options are supported. This helps us in the future, since we
44  // are currently 'assuming' some option have/not have certain values. As we support for more
45  // options in the C++ build, we will remove some checks
46  Errors::check_option("init_simulation_params_c","vert_remap_q_alg",remap_alg,{1,3});
47  Errors::check_option("init_simulation_params_c","prescribed_wind",prescribed_wind,{false});
48  Errors::check_option("init_simulation_params_c","hypervis_order",hypervis_order,{2});
49  Errors::check_option("init_simulation_params_c","use_semi_lagrangian_transport",use_semi_lagrangian_transport,{false});
50  Errors::check_option("init_simulation_params_c","time_step_type",time_step_type,{5});
51  Errors::check_option("init_simulation_params_c","qsize",qsize,0,Errors::ComparisonOp::GE);
52  Errors::check_option("init_simulation_params_c","qsize",qsize,QSIZE_D,Errors::ComparisonOp::LE);
53  Errors::check_option("init_simulation_params_c","limiter_option",limiter_option,{8,9});
54  Errors::check_option("init_simulation_params_c","ftype",ftype, {-1, 0, 2});
55  Errors::check_option("init_simulation_params_c","nu_p",nu_p,0.0,Errors::ComparisonOp::GT);
56  Errors::check_option("init_simulation_params_c","nu",nu,0.0,Errors::ComparisonOp::GT);
57  Errors::check_option("init_simulation_params_c","nu_div",nu_div,0.0,Errors::ComparisonOp::GT);
58 
59  // Get the simulation params struct
60  SimulationParams& params = Context::singleton().create<SimulationParams>();
61 
62  if (remap_alg==1) {
63  params.remap_alg = RemapAlg::PPM_MIRRORED;
64  } else if (remap_alg == 2) {
65  params.remap_alg = RemapAlg::PPM_FIXED_PARABOLA;
66  } else if (remap_alg == 3) {
67  params.remap_alg = RemapAlg::PPM_FIXED_MEANS;
68  }
69  if (time_step_type==5) {
70  params.time_step_type = TimeStepType::ULLRICH_RK35;;
71  }
72 
73  params.limiter_option = limiter_option;
74  params.rsplit = rsplit;
75  params.qsplit = qsplit;
76  params.prescribed_wind = prescribed_wind;
77  params.state_frequency = state_frequency;
78  params.qsize = qsize;
79  params.nu = nu;
80  params.nu_p = nu_p;
81  params.nu_q = nu_q;
82  params.nu_s = nu_s;
83  params.nu_div = nu_div;
84  params.nu_top = nu_top;
85  params.hypervis_order = hypervis_order;
86  params.hypervis_subcycle = hypervis_subcycle;
87  params.hypervis_scaling = hypervis_scaling;
88  params.disable_diagnostics = disable_diagnostics;
89  params.moisture = (moisture ? MoistDry::MOIST : MoistDry::DRY);
90  params.use_cpstar = use_cpstar;
91  params.use_semi_lagrangian_transport = use_semi_lagrangian_transport;
92 
93  //set nu_ratios values
94  if (params.nu != params.nu_div) {
95  Real ratio = params.nu_div / params.nu;
96  if (params.hypervis_scaling != 0.0) {
97  params.nu_ratio1 = ratio * ratio;
98  params.nu_ratio2 = 1.0;
99  }else{
100  params.nu_ratio1 = ratio;
101  params.nu_ratio2 = ratio;
102  }
103  }else{
104  params.nu_ratio1 = 1.0;
105  params.nu_ratio2 = 1.0;
106  }
107 
108  if (ftype == -1) {
109  params.ftype = ForcingAlg::FORCING_OFF;
110  } else if (ftype == 0) {
111  params.ftype = ForcingAlg::FORCING_DEBUG;
112  } else if (ftype == 2) {
113  params.ftype = ForcingAlg::FORCING_2;
114  }
115 
116  // TODO Parse a fortran string and set this properly. For now, our code does
117  // not depend on this except to throw an error in apply_test_forcing.
118  params.test_case = TestCase::JW_BAROCLINIC;
119 
120  // Now this structure can be used safely
121  params.params_set = true;
122 
123 }

132 void cxx_push_results_to_f90(F90Ptr &elem_state_v_ptr, F90Ptr &elem_state_temp_ptr,
133  F90Ptr &elem_state_dp3d_ptr, F90Ptr &elem_state_Qdp_ptr,

135  F90Ptr &elem_derived_omega_p_ptr) {

138 
139  Tracers &tracers = Context::singleton().get<Tracers>();
140  tracers.push_qdp(elem_state_Qdp_ptr);
141 
142  // F90 ptrs to arrays (np,np,num_time_levels,nelemd) can be stuffed directly
143  // in an unmanaged view
144  // with scalar Real*[NUM_TIME_LEVELS][NP][NP] (with runtime dimension nelemd)
145  HostViewUnmanaged<Real * [NUM_TIME_LEVELS][NP][NP]> ps_v_f90(
146  elem_state_ps_v_ptr, elements.num_elems());
147 
148  auto ps_v_host = Kokkos::create_mirror_view(elements.m_state.m_ps_v);
149 
150  Kokkos::deep_copy(ps_v_host, elements.m_state.m_ps_v);
151  Kokkos::deep_copy(ps_v_f90, ps_v_host);
152 
153  sync_to_host(elements.m_derived.m_omega_p,
154  HostViewUnmanaged<Real * [NUM_PHYSICAL_LEV][NP][NP]>(
155  elem_derived_omega_p_ptr, elements.num_elems()));
156  sync_to_host(tracers.Q,
157  HostViewUnmanaged<Real * [QSIZE_D][NUM_PHYSICAL_LEV][NP][NP]>(
158  elem_Q_ptr, elements.num_elems()));
159 }

229 void init_elements_c (const int& num_elems)
230 {
231  auto& c = Context::singleton();
232  Elements& e = c.create<Elements> ();
233  const SimulationParams& params = c.get<SimulationParams>();
234 
235  const bool consthv = (params.hypervis_scaling==0.0);
236  e.init (num_elems, consthv, /* alloc_gradphis = */ false);
237 
238  // Init also the tracers structure
239  Tracers& t = c.create<Tracers> ();
240  t.init(num_elems,params.qsize);
241 
242  // In the context, we register also Elements[Geometry|State|DerivedState|Forcing],
243  // making sure they store the same views as in the subobjects of Elements.
244  // This allows objects that need only a piece of Elements, to grab it from the Context,
245  // while still knowing that what they grab contains the same views as the object stored in the
246  // Elements inside the Context
247  // WARNING: after this point, you should NOT do things like
248  // e.m_geometry.m_phis = ...
249  // since they would NOT be reflected into the ElementsGeometry stored in the Context.
250  // In other words, you cannot reset the views. If you *really* need to do it,
251  // you must reset the view in both c.get<Elements>().m_geometry AND
252  // c.get<ElementsGeometry>()
253  c.create_ref<ElementsGeometry>(e.m_geometry);
254  c.create_ref<ElementsState>(e.m_state);
255  c.create_ref<ElementsDerivedState>(e.m_derived);
256  c.create_ref<ElementsForcing>(e.m_forcing);
257 }

259 void init_functors_c ()
260 {
261  // We init all the functors in the Context, so that every call to
262  // Context::singleton().get<[FunctorName]>() is allowed (otherwise
263  // Context would throw because the requested object is not found).
264 
265  auto& elems = Context::singleton().get<Elements>();
266  auto& tracers = Context::singleton().get<Tracers>();
267  auto& ref_FE = Context::singleton().get<ReferenceElement>();
268  auto& hvcoord = Context::singleton().get<HybridVCoord>();
269  auto& params = Context::singleton().get<SimulationParams>();
270  auto& fbm = Context::singleton().create<FunctorsBuffersManager>();
271 
272  // Check that the above structures have been inited
273  Errors::runtime_check(elems.inited(), "Error! You must initialize the Elements structure before initializing the functors.\n", -1);
274  Errors::runtime_check(tracers.inited(), "Error! You must initialize the Tracers structure before initializing the functors.\n", -1);
275  Errors::runtime_check(ref_FE.inited(), "Error! You must initialize the ReferenceElement structure before initializing the functors.\n", -1);
276  Errors::runtime_check(hvcoord.m_inited, "Error! You must initialize the HybridVCoord structure before initializing the functors.\n", -1);
277  Errors::runtime_check(params.params_set, "Error! You must initialize the SimulationParams structure before initializing the functors.\n", -1);
278 
279  // First, sphere operators, then all the functors
280  auto& sph_op = Context::singleton().create<SphereOperators>(elems.m_geometry,ref_FE);
281  auto& caar = Context::singleton().create<CaarFunctor>(elems,tracers,ref_FE,hvcoord,sph_op,params);
282  auto& esf = Context::singleton().create<EulerStepFunctor>();
283  auto& hvf = Context::singleton().create<HyperviscosityFunctor>();
284  Context::singleton().create<VerticalRemapManager>();
285 
286  // Ask the functors to request buffers to the buffers manager
287  fbm.request_size(caar.requested_buffer_size());
288  fbm.request_size(esf.requested_buffer_size());
289  fbm.request_size(hvf.requested_buffer_size());
290 
291  // Allocate the buffers
292  fbm.allocate();
293 
294  // Tell the functors to grab their buffers
295  caar.init_buffers(fbm);
296  esf.init_buffers(fbm);
297  hvf.init_buffers(fbm);
298 }
