161  KOKKOS_INLINE_FUNCTION
162  void operator() (const TagUpdateStates&, const int idx) const {
163  const int ie = idx / (NP*NP*NUM_LEV);
164  const int igp = (idx / (NP*NUM_LEV)) % NP;
165  const int jgp = (idx / NUM_LEV) % NP;
166  const int ilev = idx % NUM_LEV;
167 
168  // Apply inverse mass matrix
169  m_buffers.vtens(ie,0,igp,jgp,ilev) = (m_data.dt * m_buffers.vtens(ie,0,igp,jgp,ilev) *
170  m_geometry.m_rspheremp(ie,igp,jgp));
171  m_buffers.vtens(ie,1,igp,jgp,ilev) = (m_data.dt * m_buffers.vtens(ie,1,igp,jgp,ilev) *
172  m_geometry.m_rspheremp(ie,igp,jgp));
173  m_state.m_v(ie,m_data.np1,0,igp,jgp,ilev) += m_buffers.vtens(ie,0,igp,jgp,ilev);
174  m_state.m_v(ie,m_data.np1,1,igp,jgp,ilev) += m_buffers.vtens(ie,1,igp,jgp,ilev);
175 
176  m_buffers.ttens(ie,igp,jgp,ilev) = (m_data.dt*m_buffers.ttens(ie,igp,jgp,ilev) *
177  m_geometry.m_rspheremp(ie,igp,jgp));
178  const Scalar heating = m_buffers.vtens(ie,0,igp,jgp,ilev)*m_state.m_v(ie,m_data.np1,0,igp,jgp,ilev)
179  + m_buffers.vtens(ie,1,igp,jgp,ilev)*m_state.m_v(ie,m_data.np1,1,igp,jgp,ilev);
180  m_state.m_t(ie,m_data.np1,igp,jgp,ilev) =
181  m_state.m_t(ie,m_data.np1,igp,jgp,ilev) + m_buffers.ttens(ie,igp,jgp,ilev) -
182  heating/PhysicalConstants::cp;
183 
184  m_state.m_dp3d(ie,m_data.np1,igp,jgp,ilev) = (m_buffers.dptens(ie,igp,jgp,ilev) *
185  m_geometry.m_rspheremp(ie,igp,jgp));
186  }

188  KOKKOS_INLINE_FUNCTION
189  void operator()(const TagHyperPreExchange, const TeamMember &team) const {
190  KernelVariables kv(team);
191  Kokkos::parallel_for(Kokkos::TeamThreadRange(kv.team, NP * NP),
192  [&](const int &point_idx) {
193  const int igp = point_idx / NP;
194  const int jgp = point_idx % NP;
195  Kokkos::parallel_for(Kokkos::ThreadVectorRange(kv.team, NUM_LEV),
196  [&](const int &lev) {
197 
198  m_derived.m_dpdiss_ave(kv.ie, igp, jgp, lev) +=
199  m_data.eta_ave_w * m_state.m_dp3d(kv.ie, m_data.np1, igp, jgp, lev) /
200  m_data.hypervis_subcycle;
201  m_derived.m_dpdiss_biharmonic(kv.ie, igp, jgp, lev) +=
202  m_data.eta_ave_w * m_buffers.dptens(kv.ie, igp, jgp, lev) /
203  m_data.hypervis_subcycle;
204  });
205  });
206  kv.team_barrier();
207 
208  // laplace subfunctors cannot be called from a TeamThreadRange or
209  // ThreadVectorRange
210  if (m_data.nu_top > 0) {
211 
212  //for top 3 levels and laplace, there is trivial nu_ratio only
213  m_sphere_ops.vlaplace_sphere_wk_contra<NUM_BIHARMONIC_LEV,NUM_LEV>(
214  kv, 1.0,
215  // input
216  Homme::subview(m_state.m_v, kv.ie, m_data.np1),
217  // output
218  Homme::subview(m_buffers.laplace_v, kv.team_idx));
219 
220  m_sphere_ops.laplace_simple<NUM_BIHARMONIC_LEV,NUM_LEV>(
221  kv,
222  // input
223  Homme::subview(m_state.m_t, kv.ie, m_data.np1),
224  // output
225  Homme::subview(m_buffers.laplace_t, kv.team_idx));
226 
227  m_sphere_ops.laplace_simple<NUM_BIHARMONIC_LEV,NUM_LEV>(
228  kv,
229  // input
230  Homme::subview(m_state.m_dp3d, kv.ie, m_data.np1),
231  // output
232  Homme::subview(m_buffers.laplace_dp, kv.team_idx));
233  }//if nu_top>0
234  kv.team_barrier();
235 
236  Kokkos::parallel_for(Kokkos::TeamThreadRange(kv.team, NP * NP),
237  [&](const int &point_idx) {
238  const int igp = point_idx / NP;
239  const int jgp = point_idx % NP;
240  Kokkos::parallel_for(Kokkos::ThreadVectorRange(kv.team, NUM_LEV),
241  [&](const int &lev) {
242  m_buffers.vtens(kv.ie, 0, igp, jgp, lev) *= -m_data.nu;
243  m_buffers.vtens(kv.ie, 1, igp, jgp, lev) *= -m_data.nu;
244  m_buffers.ttens(kv.ie, igp, jgp, lev) *= -m_data.nu_s;
245  m_buffers.dptens(kv.ie, igp, jgp, lev) *= -m_data.nu_p;
246  });
247 
248  if (m_data.nu_top > 0) {
249  Kokkos::parallel_for(Kokkos::ThreadVectorRange(kv.team, int(NUM_BIHARMONIC_LEV)),
250  [&](const int ilev) {
251  m_buffers.vtens(kv.ie, 0, igp, jgp, ilev) +=
252  m_nu_scale_top[ilev] *
253  m_buffers.laplace_v(kv.team_idx, 0, igp, jgp, ilev);
254  m_buffers.vtens(kv.ie, 1, igp, jgp, ilev) +=
255  m_nu_scale_top[ilev] *
256  m_buffers.laplace_v(kv.team_idx, 1, igp, jgp, ilev);
257 
258  m_buffers.ttens(kv.ie, igp, jgp, ilev) +=
259  m_nu_scale_top[ilev] *
260  m_buffers.laplace_t(kv.team_idx, igp, jgp, ilev);
261 
262  m_buffers.dptens(kv.ie, igp, jgp, ilev) +=
263  m_nu_scale_top[ilev] *
264  m_buffers.laplace_dp(kv.team_idx, igp, jgp, ilev);
265  });
266  }
267  // While for T and v we exchange the tendencies, for dp3d we exchange the updated state.
268  // However, since the BE structure already has registerd the *tens quantities, we store
269  // the updated state in dptens.
270  Kokkos::parallel_for(Kokkos::ThreadVectorRange(kv.team, NUM_LEV),
271  [&](const int &lev) {
272  m_buffers.dptens(kv.ie, igp, jgp, lev) *= m_data.dt;
273  m_buffers.dptens(kv.ie, igp, jgp, lev) += m_state.m_dp3d(kv.ie,m_data.np1,igp,jgp,lev)
274  * m_geometry.m_spheremp(kv.ie,igp,jgp);
275  });
276  });
277  }
