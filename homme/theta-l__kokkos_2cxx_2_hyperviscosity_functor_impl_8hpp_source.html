100  // first iter of laplace, const hv
101  KOKKOS_INLINE_FUNCTION
102  void operator() (const TagFirstLaplaceHV&, const TeamMember& team) const {
103  using IntColumn = decltype(Homme::subview(m_state.m_w_i,0,0,0,0));
104 
105  KernelVariables kv(team, m_tu);
106  // Subtract the reference states from the states
107  Kokkos::parallel_for(Kokkos::TeamThreadRange(kv.team,NP*NP),
108  [&](const int idx) {
109  const int igp = idx / NP;
110  const int jgp = idx % NP;
111 
112  auto vtheta = Homme::subview(m_state.m_vtheta_dp,kv.ie,m_data.np1,igp,jgp);
113  auto dp = Homme::subview(m_state.m_dp3d,kv.ie,m_data.np1,igp,jgp);
114 
115  auto theta_ref = Homme::subview(m_state.m_ref_states.theta_ref,kv.ie,igp,jgp);
116  auto dp_ref = Homme::subview(m_state.m_ref_states.dp_ref,kv.ie,igp,jgp);
117 
118  IntColumn phi_i, phi_i_ref;
119 
120  if (m_process_nh_vars) {
121  phi_i = Homme::subview(m_state.m_phinh_i,kv.ie,m_data.np1,igp,jgp);
122  phi_i_ref = Homme::subview(m_state.m_ref_states.phi_i_ref,kv.ie,igp,jgp);
123  }
124  Kokkos::parallel_for(Kokkos::ThreadVectorRange(kv.team,NUM_LEV),
125  [&](const int ilev) {
126  vtheta(ilev) -= theta_ref(ilev);
127  dp(ilev) -= dp_ref(ilev);
128  if (m_process_nh_vars) {
129  phi_i(ilev) -= phi_i_ref(ilev);
130  }
131  });
132 
133 #ifndef XX_NONBFB_COMING
134  // It would be fine to not even bother with the surface level, since
135  // phitens is only NUM_LEV long, so all the hv stuff does not even happen
136  // at NUM_LEV_P (unless NUM_LEV_P==NUM_LEV). However, removing the subtraction
137  // and addition of phi_i_ref at NUM_LEV_P introduces NON BFB diffs.
138  if (m_process_nh_vars && NUM_LEV!=NUM_LEV_P) {
139  Kokkos::single(Kokkos::PerThread(kv.team),[&](){
140  phi_i(NUM_LEV_P-1) -= phi_i_ref(NUM_LEV_P-1);
141  });
142  }
143 #endif
144  });
145 
146  // Laplacian of layer thickness
147  m_sphere_ops.laplace_simple(kv,
148  Homme::subview(m_state.m_dp3d,kv.ie,m_data.np1),
149  Homme::subview(m_buffers.dptens,kv.ie));
150  // Laplacian of theta
151  m_sphere_ops.laplace_simple(kv,
152  Homme::subview(m_state.m_vtheta_dp,kv.ie,m_data.np1),
153  Homme::subview(m_buffers.ttens,kv.ie));
154 
155  if (m_process_nh_vars) {
156  // Laplacian of vertical velocity (do not compute last interface)
157  m_sphere_ops.laplace_simple<NUM_LEV,NUM_LEV_P>(kv,
158  Homme::subview(m_state.m_w_i,kv.ie,m_data.np1),
159  Homme::subview(m_buffers.wtens,kv.ie));
160  // Laplacian of geopotential (do not compute last interface)
161  m_sphere_ops.laplace_simple<NUM_LEV,NUM_LEV_P>(kv,
162  Homme::subview(m_state.m_phinh_i,kv.ie,m_data.np1),
163  Homme::subview(m_buffers.phitens,kv.ie));
164  }
165 
166  // Laplacian of velocity
167  m_sphere_ops.vlaplace_sphere_wk_contra(kv, m_data.nu_ratio1,
168  Homme::subview(m_state.m_v,kv.ie,m_data.np1),
169  Homme::subview(m_buffers.vtens,kv.ie));
170  }

238  KOKKOS_INLINE_FUNCTION
239  void operator() (const TagUpdateStates&, const TeamMember& team) const {
240  KernelVariables kv(team, m_tu);
241 
242  using MidColumn = decltype(Homme::subview(m_buffers.wtens,0,0,0));
243  using IntColumn = decltype(Homme::subview(m_state.m_w_i,0,0,0,0));
244  Kokkos::parallel_for(Kokkos::TeamThreadRange(kv.team,NP*NP),
245  [&](const int idx) {
246  const int igp = idx / NP;
247  const int jgp = idx % NP;
248 
249  // Add Xtens quantities back to the states, except for vtheta
250  auto u = Homme::subview(m_state.m_v,kv.ie,m_data.np1,0,igp,jgp);
251  auto v = Homme::subview(m_state.m_v,kv.ie,m_data.np1,1,igp,jgp);
252  auto vtheta = Homme::subview(m_state.m_vtheta_dp,kv.ie,m_data.np1,igp,jgp);
253  auto dp = Homme::subview(m_state.m_dp3d,kv.ie,m_data.np1,igp,jgp);
254 
255  auto utens = Homme::subview(m_buffers.vtens,kv.ie,0,igp,jgp);
256  auto vtens = Homme::subview(m_buffers.vtens,kv.ie,1,igp,jgp);
257  auto ttens = Homme::subview(m_buffers.ttens,kv.ie,igp,jgp);
258  auto dptens = Homme::subview(m_buffers.dptens,kv.ie,igp,jgp);
259  const auto& rspheremp = m_geometry.m_rspheremp(kv.ie,igp,jgp);
260 
261  MidColumn wtens, phitens;
262  IntColumn w, phi_i;
263 
264  if (m_process_nh_vars) {
265  wtens = Homme::subview(m_buffers.wtens,kv.ie,igp,jgp);
266  phitens = Homme::subview(m_buffers.phitens,kv.ie,igp,jgp);
267  w = Homme::subview(m_state.m_w_i,kv.ie,m_data.np1,igp,jgp);
268  phi_i = Homme::subview(m_state.m_phinh_i,kv.ie,m_data.np1,igp,jgp);
269  }
270 
271  Kokkos::parallel_for(Kokkos::ThreadVectorRange(kv.team,NUM_LEV),
272  [&](const int ilev) {
273  utens(ilev) *= m_data.dt*rspheremp;
274  vtens(ilev) *= m_data.dt*rspheremp;
275  ttens(ilev) *= m_data.dt*rspheremp;
276  dptens(ilev) *= m_data.dt*rspheremp;
277 
278  u(ilev) += utens(ilev);
279  v(ilev) += vtens(ilev);
280  vtheta(ilev) += ttens(ilev);
281  dp(ilev) += dptens(ilev);
282  if (m_process_nh_vars) {
283  wtens(ilev) *= m_data.dt*rspheremp;
284  phitens(ilev) *= m_data.dt*rspheremp;
285 
286  w(ilev) += wtens(ilev);
287  phi_i(ilev) += phitens(ilev);
288  }
289  });
290  });
291  }

293  KOKKOS_INLINE_FUNCTION
294  void operator()(const TagHyperPreExchange, const TeamMember &team) const {
295  using IntColumn = decltype(Homme::subview(m_state.m_w_i,0,0,0,0));
296 
297  KernelVariables kv(team, m_tu);
298  Kokkos::parallel_for(Kokkos::TeamThreadRange(kv.team, NP * NP),
299  [&](const int &point_idx) {
300  const int igp = point_idx / NP;
301  const int jgp = point_idx % NP;
302 
303  const auto dpdiss_ave = Homme::subview(m_derived.m_dpdiss_ave,kv.ie, igp, jgp);
304  const auto dpdiss_bih = Homme::subview(m_derived.m_dpdiss_biharmonic,kv.ie, igp, jgp);
305  const auto dp3d = Homme::subview(m_state.m_dp3d,kv.ie, m_data.np1, igp, jgp);
306  const auto dp_ref = Homme::subview(m_state.m_ref_states.dp_ref,kv.ie, igp, jgp);
307  const auto theta = Homme::subview(m_state.m_vtheta_dp,kv.ie, m_data.np1, igp, jgp);
308  const auto theta_ref = Homme::subview(m_state.m_ref_states.theta_ref,kv.ie, igp, jgp);
309  const auto dptens = Homme::subview(m_buffers.dptens,kv.ie, igp, jgp);
310 
311  IntColumn phi, phi_ref;
312 
313  if (m_process_nh_vars) {
314  phi = Homme::subview(m_state.m_phinh_i,kv.ie, m_data.np1, igp, jgp);
315  phi_ref = Homme::subview(m_state.m_ref_states.phi_i_ref,kv.ie, igp, jgp);
316  }
317 
318  Kokkos::parallel_for(Kokkos::ThreadVectorRange(kv.team, NUM_LEV),
319  [&](const int &ilev) {
320 
321  dp3d(ilev) += dp_ref(ilev);
322  theta(ilev) += theta_ref(ilev);
323  if (m_process_nh_vars) {
324  phi(ilev) += phi_ref(ilev);
325  }
326  if (m_data.nu_p>0) {
327  dpdiss_ave(ilev) += m_data.eta_ave_w*dp3d(ilev) / m_data.hypervis_subcycle;
328  dpdiss_bih(ilev) += m_data.eta_ave_w*dptens(ilev) / m_data.hypervis_subcycle;
329  }
330  });
331 #ifndef XX_NONBFB_COMING
332  // It would be fine to not even bother with the surface level, since
333  // phitens is only NUM_LEV long, so all the hv stuff does not even happen
334  // at NUM_LEV_P (unless NUM_LEV_P==NUM_LEV). However, removing the subtraction
335  // and addition of phi_i_ref at NUM_LEV_P introduces NON BFB diffs.
336  if (m_process_nh_vars && NUM_LEV!=NUM_LEV_P) {
337  Kokkos::single(Kokkos::PerThread(kv.team),[&](){
338  phi(NUM_LEV_P-1) += phi_ref(NUM_LEV_P-1);
339  });
340  }
341 #endif
342  });
343  kv.team_barrier();
344 
345  // laplace subfunctors cannot be called from a TeamThreadRange or
346  // ThreadVectorRange
347  if (m_data.nu_top > 0) {
348 
349  //for top 3 levels and laplace, there is trivial nu_ratio only
350 
351  m_sphere_ops.laplace_simple<NUM_BIHARMONIC_LEV,NUM_LEV>(
352  kv, Homme::subview(m_state.m_dp3d, kv.ie, m_data.np1),
353  Homme::subview(m_buffers.lapl_dp, kv.team_idx));
354 
355  m_sphere_ops.laplace_simple<NUM_BIHARMONIC_LEV,NUM_LEV>(
356  kv, Homme::subview(m_state.m_vtheta_dp, kv.ie, m_data.np1),
357  Homme::subview(m_buffers.lapl_theta, kv.team_idx));
358 
359  if (m_process_nh_vars) {
360  m_sphere_ops.laplace_simple<NUM_BIHARMONIC_LEV,NUM_LEV_P>(
361  kv, Homme::subview(m_state.m_w_i, kv.ie, m_data.np1),
362  Homme::subview(m_buffers.lapl_w, kv.team_idx));
363 
364  m_sphere_ops.laplace_simple<NUM_BIHARMONIC_LEV,NUM_LEV_P>(
365  kv, Homme::subview(m_state.m_phinh_i, kv.ie, m_data.np1),
366  Homme::subview(m_buffers.lapl_phi, kv.team_idx));
367  }
368 
369  m_sphere_ops.vlaplace_sphere_wk_contra<NUM_BIHARMONIC_LEV,NUM_LEV>(
370  kv, 1.0, Homme::subview(m_state.m_v, kv.ie, m_data.np1),
371  Homme::subview(m_buffers.lapl_v, kv.team_idx));
372  }//if nu_top>0
373  kv.team_barrier();
374 
375  Kokkos::parallel_for(Kokkos::TeamThreadRange(kv.team, NP * NP),
376  [&](const int &point_idx) {
377  const int igp = point_idx / NP;
378  const int jgp = point_idx % NP;
379  Kokkos::parallel_for(Kokkos::ThreadVectorRange(kv.team, NUM_LEV),
380  [&](const int &lev) {
381  m_buffers.vtens(kv.ie, 0, igp, jgp, lev) *= -m_data.nu;
382  m_buffers.vtens(kv.ie, 1, igp, jgp, lev) *= -m_data.nu;
383  m_buffers.ttens(kv.ie, igp, jgp, lev) *= -m_data.nu;
384  m_buffers.dptens(kv.ie, igp, jgp, lev) *= -m_data.nu_p;
385  if (m_process_nh_vars) {
386  m_buffers.wtens(kv.ie, igp, jgp, lev) *= -m_data.nu;
387  m_buffers.phitens(kv.ie, igp, jgp, lev) *= -m_data.nu_s;
388  }
389  });
390 
391  if (m_data.nu_top > 0) {
392  // Need to loop on the actual number of levels, not packs, to avoid adding more sponge layers
393  Kokkos::parallel_for(Kokkos::ThreadVectorRange(kv.team, int(NUM_BIHARMONIC_PHYSICAL_LEVELS)),
394  [&](const int k) {
395  const int ilev = k / VECTOR_SIZE;
396  const int ivec = k % VECTOR_SIZE;
397 
398  m_buffers.vtens(kv.ie, 0, igp, jgp, ilev)[ivec] +=
399  m_nu_scale_top[ilev][ivec] *
400  m_buffers.lapl_v(kv.team_idx, 0, igp, jgp, ilev)[ivec];
401 
402  m_buffers.vtens(kv.ie, 1, igp, jgp, ilev)[ivec] +=
403  m_nu_scale_top[ilev][ivec] *
404  m_buffers.lapl_v(kv.team_idx, 1, igp, jgp, ilev)[ivec];
405 
406  m_buffers.ttens(kv.ie, igp, jgp, ilev)[ivec] +=
407  m_nu_scale_top[ilev][ivec] *
408  m_buffers.lapl_theta(kv.team_idx, igp, jgp, ilev)[ivec];
409 
410  m_buffers.dptens(kv.ie, igp, jgp, ilev)[ivec] +=
411  m_nu_scale_top[ilev][ivec] *
412  m_buffers.lapl_dp(kv.team_idx, igp, jgp, ilev)[ivec];
413 
414  if (m_process_nh_vars) {
415  m_buffers.wtens(kv.ie, igp, jgp, ilev)[ivec] +=
416  m_nu_scale_top[ilev][ivec] *
417  m_buffers.lapl_w(kv.team_idx, igp, jgp, ilev)[ivec];
418 
419  m_buffers.phitens(kv.ie, igp, jgp, ilev)[ivec] +=
420  m_nu_scale_top[ilev][ivec] *
421  m_buffers.lapl_phi(kv.team_idx, igp, jgp, ilev)[ivec];
422  }
423  });
424  }
425  });
426  }
