9 // Set the least significant n bits of the fraction to 0. Use this for
10 // BFB-testing between GPU and host. Zero the n least significant bits of y in
11 // y = pow(x, e),
12 // for example.
13 // Table 8 of Appendix E of the Cuda programming guide gives the accuracy of
14 // various functions. However, we also have to account for carries in the
15 // calculations. Thus, even if an answer is accurate to 2 ULP, carries can make
16 // more bits differ. Thus, zeroing (or any sort of rounding) is not assured to
17 // work. However, you can reduce the probability of a diff to a small number by
18 // choosing n to be large. Full double precision range is 53 bits; choose
19 // something like n = 20 to get a carry-based diff every 1 in a million calls,
20 // still leaving 33 bits.
21 // To remain sensitive to the full bits in the argument to the non-BFB
22 // function, such as x in pow above, pass replace = x. The n least significant
23 // bits of the fraction part of x is used to fill the 0s introduced in this
24 // function. Thus, the final result z of
25 // y = pow(x, e);
26 // z = zeroulpn(y, n, x)
27 // is sensitive to all bits of x.
28 KOKKOS_INLINE_FUNCTION
29 double zeroulpn (double a, const int n, double replace) {
30  const auto f = static_cast<double>(1LL << 54);
31  int e;
32  a = std::frexp(a, &e);
33  const int sign = a >= 0 ? 1 : -1;
34  a = std::abs(a);
35  long long int x = f*a;
36  x >>= n;
37  x <<= n;
38  { // Replace the n least significant bits with those from the fraction part of
39  // 'replace'.
40  const long long int ones = 0xffffffffffffffff;
41  auto mask = ones;
42  mask >>= n;
43  mask <<= n;
44  mask = ones & (mask ^ ones);
45  int er;
46  replace = std::abs(std::frexp(replace, &er));
47  long long int r = f*replace;
48  x = x | (mask & r);
49  }
50  a = x/f;
51  return std::ldexp(sign*a, e);
52 }
