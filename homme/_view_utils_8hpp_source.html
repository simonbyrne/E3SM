31 // ================ Reinterpret a view of Scalar as a view of Real ======================= //
32 // Note: we template on ScalarType to allow both const and non-const, but the underlying
33 // type (the one you get with std::remove_const) *must* be Scalar (as defined in Types.hpp).
34 template <typename ScalarType, int DIM1, typename... Properties>
35 KOKKOS_INLINE_FUNCTION
36 typename
37 std::enable_if<std::is_same<typename std::remove_const<ScalarType>::type,Scalar>::value,
38  Unmanaged<ViewType<RealType<ScalarType>[DIM1*VECTOR_SIZE],Properties...>>
39  >::type
40 viewAsReal(ViewType<ScalarType [DIM1], Properties...> v_in) {
41  using ReturnST = RealType<ScalarType>;
42  using ReturnView = Unmanaged<ViewType<RealType<ScalarType>[DIM1*VECTOR_SIZE],Properties...>>;
43  return ReturnView(reinterpret_cast<ReturnST*>(v_in.data()));
44 }
