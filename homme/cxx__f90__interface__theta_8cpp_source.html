249 void init_elements_c (const int& num_elems)
250 {
251  auto& c = Context::singleton();
252 
253  Elements& e = c.create<Elements> ();
254  const SimulationParams& params = c.get<SimulationParams>();
255 
256  const bool consthv = (params.hypervis_scaling==0.0);
257  e.init (num_elems, consthv, /* alloc_gradphis = */ true);
258 
259  // Init also the tracers structure
260  Tracers& t = c.create<Tracers> ();
261  t.init(num_elems,params.qsize);
262 
263  // In the context, we register also Elements[Geometry|State|DerivedState|Forcing],
264  // making sure they store the same views as in the subobjects of Elements.
265  // This allows objects that need only a piece of Elements, to grab it from the Context,
266  // while still knowing that what they grab contains the same views as the object stored in the
267  // Elements inside the Context
268  // WARNING: after this point, you should NOT do things like
269  // e.m_geometry.m_phis = ...
270  // since they would NOT be reflected into the ElementsGeometry stored in the Context.
271  // In other words, you cannot reset the views. If you *really* need to do it,
272  // you must reset the view in both c.get<Elements>().m_geometry AND
273  // c.get<ElementsGeometry>()
274  c.create_ref<ElementsGeometry>(e.m_geometry);
275  c.create_ref<ElementsState>(e.m_state);
276  c.create_ref<ElementsDerivedState>(e.m_derived);
277  c.create_ref<ElementsForcing>(e.m_forcing);
278 }

280 void init_functors_c ()
281 {
282  // We init all the functors in the Context, so that every call to
283  // Context::singleton().get<[FunctorName]>()
284  // will return a functor already initialized.
285  // This avoids the risk of having a class doing
286  // FunctorName f = Context::singleton().get<FunctorName>();
287  // f.init(some_args);
288  // and then, somewhere else, we find
289  // FunctorName f = Context::singleton().get<FunctorName>(some_args);
290  // The problem is that the first call created an uninitialized functor,
291  // *copied it*, and initialized the copy. The second call to get,
292  // sees that there is *already* an object of type FunctorName in the
293  // Context, and therefore does *not* create a FunctorName object.
294  // A solution would be to use a reference in the first call, or to always
295  // use the get method with the initialization arguments. However, if the functor
296  // is already init-ed, what do we do? Ignore: there's the risk that the user thinks
297  // that the functor was created with the inputs he/she provided Throw: we create
298  // the headache of first finding out if a functor already exists, and, if not,
299  // create it and init it.
300  // It is way easier to create all functors here once and for all,
301  // so that the user does not have to initialize them when calling them.
302 
303  auto& elems = Context::singleton().get<Elements>();
304  auto& tracers = Context::singleton().get<Tracers>();
305  auto& ref_FE = Context::singleton().get<ReferenceElement>();
306  auto& hvcoord = Context::singleton().get<HybridVCoord>();
307  auto& params = Context::singleton().get<SimulationParams>();
308 
309  // Check that the above structures have been inited
310  Errors::runtime_check(elems.inited(), "Error! You must initialize the Elements structure before initializing the functors.\n", -1);
311  Errors::runtime_check(tracers.inited(), "Error! You must initialize the Tracers structure before initializing the functors.\n", -1);
312  Errors::runtime_check(ref_FE.inited(), "Error! You must initialize the ReferenceElement structure before initializing the functors.\n", -1);
313  Errors::runtime_check(hvcoord.m_inited, "Error! You must initialize the HybridVCoord structure before initializing the functors.\n", -1);
314  Errors::runtime_check(params.params_set, "Error! You must initialize the SimulationParams structure before initializing the functors.\n", -1);
315 
316  // First, sphere operators, then the others
317  auto& sph_op = Context::singleton().create<SphereOperators>(elems.m_geometry,ref_FE);
318  auto& caar = Context::singleton().create<CaarFunctor>(elems,tracers,ref_FE,hvcoord,sph_op,params);
319  auto& esf = Context::singleton().create<EulerStepFunctor>();
320  auto& hvf = Context::singleton().create<HyperviscosityFunctor>();
321  auto& fbm = Context::singleton().create<FunctorsBuffersManager>();
322  auto& ff = Context::singleton().create<ForcingFunctor>();
323  auto& diag = Context::singleton().create<Diagnostics> (elems.num_elems());
324  auto& vrm = Context::singleton().create<VerticalRemapManager>();
325 
326  const bool need_dirk = (params.time_step_type==TimeStepType::IMEX_KG243 ||
327  params.time_step_type==TimeStepType::IMEX_KG254 ||
328  params.time_step_type==TimeStepType::IMEX_KG255 ||
329  params.time_step_type==TimeStepType::IMEX_KG355);
330 
331  if (need_dirk) {
332  // Create dirk functor only if needed
333  Context::singleton().create<DirkFunctor>(elems.num_elems());
334  }
335 
336  // Make the functor request their buffer to the buffers manager
337  // Note: diagnostics also needs buffers
338  fbm.request_size(caar.requested_buffer_size());
339  fbm.request_size(esf.requested_buffer_size());
340  fbm.request_size(hvf.requested_buffer_size());
341  fbm.request_size(diag.requested_buffer_size());
342  fbm.request_size(ff.requested_buffer_size());
343  fbm.request_size(vrm.requested_buffer_size());
344  if (need_dirk) {
345  const auto& dirk = Context::singleton().get<DirkFunctor>();
346  fbm.request_size(dirk.requested_buffer_size());
347  }
348 
349  // Allocate the buffers in the FunctorsBuffersManager, then tell the functors to grab their buffers
350  fbm.allocate();
351 
352  caar.init_buffers(fbm);
353  esf.init_buffers(fbm);
354  hvf.init_buffers(fbm);
355  diag.init_buffers(fbm);
356  ff.init_buffers(fbm);
357  vrm.init_buffers(fbm);
358  if (need_dirk) {
359  auto& dirk = Context::singleton().get<DirkFunctor>();
360  dirk.init_buffers(fbm);
361  }
362 }
