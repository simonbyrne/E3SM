200  type(n_vector), pointer :: sunvec_ycur
201  real(c_double) :: tcur(1)
202  real(c_double) :: gamma(1)
203  type(nvec_t), pointer :: ycur
204  type(nvec_t), pointer :: b
205  type(nvec_t), pointer :: x
206  integer :: fcolumnsolsolve
207 
208  ! initialize return flag to success
209  ier = 0
210 
211  ! access current time
212  ier = farkstepgetcurrenttime(arkmem, tcur)
213 
214  ! access current state
215  ier = farkstepgetcurrentstate(arkmem, ycur_ptr)
216  sunvec_ycur => fn_vgetvecatindexvectorarray(ycur_ptr, 0)
217 
218  ! access current gamma value
219  ier = farkstepgetcurrentgamma(arkmem, gamma)
220 
221  ! call Fortran routine to do operation
222  ycur => fn_vgetcontent(sunvec_ycur)
223  b => fn_vgetcontent(sunvec_b)
224  x => fn_vgetcontent(sunvec_x)
225  ier = fcolumnsolsolve(b, tcur, ycur, gamma, x)
226 
227  return
