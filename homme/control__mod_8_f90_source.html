314  function timestep_make_subcycle_parameters_consistent(par, rsplit, qsplit, &
315  dt_remap_factor, dt_tracer_factor, abort, silent) result(status)
316 
317  use parallel_mod, only: abortmp, parallel_t
318  use kinds, only: iulog
319 
320  type (parallel_t), intent(in) :: par
321  integer, intent(inout) :: rsplit, qsplit, dt_remap_factor, dt_tracer_factor
322  logical, intent(in), optional :: abort, silent
323  integer :: status
324 
325  integer :: qsplit_prev, rsplit_prev
326  logical :: split_specified, factor_specified, split_is_master, abort_in, silent_in
327 
328  abort_in = .true.
329  if (present(abort)) abort_in = abort
330  silent_in = .false.
331  if (present(silent)) silent_in = silent
332 
333  status = -1 ! error value for early returns on error
334 
335  split_specified = rsplit >= 0 .and. qsplit >= 1
336  factor_specified = dt_remap_factor >= 0 .and. dt_tracer_factor >= 1
337 
338  if (.not. split_specified .and. .not. factor_specified) then
339  if (par%masterproc .and. .not. silent_in) then
340  write(iulog,*) 'Neither rsplit,qsplit nor dt_remap_factor,dt_tracer_factor &
341  &are specified; one set must be.'
342  end if
343  if (abort_in) call abortmp('timestep_make_parameters_consistent: input error')
344  return
345  end if
346 
347  !! Process rsplit, qsplit, dt_remap_factor, dt_tracer_factor.
348 
349  ! To support namelists with defaulted qsplit, rsplit values, we
350  ! permit (split_specified .and. factor_specified). In this case,
351  ! factor_specified means factor values are used.
352 
353  split_is_master = .not. factor_specified
354 
355  if (split_is_master) then
356  dt_remap_factor = rsplit*qsplit
357  dt_tracer_factor = qsplit
358  else
359  if (dt_remap_factor > 0) then
360  if (.not. (modulo(dt_remap_factor, dt_tracer_factor) == 0 .or. &
361  modulo(dt_tracer_factor, dt_remap_factor) == 0)) then
362  if (par%masterproc .and. .not. silent_in) then
363  write(iulog,*) 'dt_remap_factor and dt_tracer_factor were specified, &
364  &but neither divides the other.'
365  end if
366  if (abort_in) call abortmp('timestep_make_parameters_consistent: divisibility error')
367  return
368  end if
369  end if
370  qsplit_prev = qsplit
371  rsplit_prev = rsplit
372  qsplit = dt_tracer_factor
373  ! This is the only inconsistent setting. But I want to keep
374  ! this here b/c almost all uses of rsplit is simply for whether
375  ! it's == 0, and I don't want to touch all those lines in this
376  ! PR.
377  if (dt_tracer_factor <= dt_remap_factor) then
378  rsplit = dt_remap_factor/dt_tracer_factor
379  else
380  ! If rsplit cannot be set consistently (because
381  ! dt_tracer_factor < dt_remap_factor), then just preserve
382  ! the sign to distinguish between vertically Eulerian and
383  ! Lagrangian methods.
384  if (dt_remap_factor > 0) then
385  rsplit = 1
386  else
387  rsplit = 0
388  end if
389  end if
390  if (split_specified .and. (qsplit /= qsplit_prev .or. rsplit /= rsplit_prev) .and. &
391  par%masterproc .and. .not. silent_in) then
392  write(iulog,'(a,i2,a,i2,a,i2,a,i2,a)') &
393  'dt_remap_factor and dt_tracer_factor were specified; changing qsplit from ', &
394  qsplit_prev, ' to ', qsplit, ' and rsplit from ', rsplit_prev, ' to ', rsplit, '.'
395  end if
396  end if
397 
398  status = 0 ! success value
399  end function timestep_make_subcycle_parameters_consistent

401  function timestep_make_eam_parameters_consistent(par, dt_remap_factor, dt_tracer_factor, &
402  nsplit, nstep_factor, tstep, dtime, abort, silent) result(status)
403 
404  use parallel_mod, only: abortmp, parallel_t
405  use kinds, only: iulog
406 
407  type (parallel_t), intent(in) :: par
408  integer, intent(in) :: dt_remap_factor, dt_tracer_factor
409  integer, intent(inout) :: nsplit
410  integer, intent(out) :: nstep_factor
411  real(kind=real_kind), intent(inout) :: tstep
412  integer, intent(inout) :: dtime
413  logical, intent(in), optional :: abort, silent
414  integer :: status
415 
416  real(kind=real_kind), parameter :: &
417  zero = 0.0_real_kind, &
418  eps = epsilon(1.0_real_kind), &
419  divisible_tol = 1e3_real_kind*eps
420 
421  real(kind=real_kind) :: nsplit_real, tmp
422  integer :: dt_max_factor
423  logical :: abort_in, silent_in
424 
425  abort_in = .true.
426  if (present(abort)) abort_in = abort
427  silent_in = .false.
428  if (present(silent)) silent_in = silent
429 
430  status = -1 ! error value for early returns on error
431 
432  !! Process dtime, tstep, nsplit.
433  dt_max_factor = max(dt_remap_factor, dt_tracer_factor)
434 
435  ! Every 'if' has an 'else', so every case is covered.
436  if (nsplit > 0) nstep_factor = dt_max_factor*nsplit
437  if (dtime > 0) then
438  if (nsplit > zero) then
439  tmp = real(dtime, real_kind)/real(nstep_factor, real_kind)
440  if (tstep > zero) then
441  if (abs(tstep - tmp) > divisible_tol*tmp) then
442  if (par%masterproc .and. .not. silent_in) then
443  write(iulog,'(a,a,i6,a,i2,a,es11.4,a,i2)') &
444  'dtime, nsplit, tstep were all >0 on input, but they disagree: ', &
445  'dtime ', dtime, ' nsplit ', nsplit, ' tstep ', tstep, ' nstep_factor ', nstep_factor
446  end if
447  if (abort_in) call abortmp('timestep_make_parameters_consistent: divisibility error')
448  return
449  end if
450  end if
451  tstep = tmp
452  elseif (tstep > zero) then
453  nsplit_real = real(dtime, real_kind)/(dt_max_factor*tstep)
454  nsplit = idnint(nsplit_real)
455  nstep_factor = dt_max_factor*nsplit
456  if (abs(nsplit_real - nsplit) > divisible_tol*nsplit_real) then
457  if (par%masterproc .and. .not. silent_in) then
458  write(iulog,'(a,es11.4,a,i7,a,es11.4,a,i2,a,i2,a,i2,a)') &
459  'nsplit was computed as ', nsplit_real, ' based on dtime ', dtime, &
460  ', tstep ', tstep, &
461  ', and dt_max_factor = max(dt_remap_factor, dt_tracer_factor) = max(', &
462  dt_remap_factor, ',', dt_tracer_factor, ') =', dt_max_factor, &
463  ', which is outside the divisibility tolerance. &
464  &Set tstep, dt_remap_factor, and dt_tracer_factor so that &
465  &tstep and dt_max_factor*tstep divide dtime.'
466  end if
467  if (abort_in) call abortmp('timestep_make_parameters_consistent: divisibility error')
468  return
469  end if
470  else
471  if (par%masterproc .and. .not. silent_in) then
472  write(iulog,*) 'If dtime is set to >0, then either nsplit or tstep must be >0.'
473  end if
474  if (abort_in) call abortmp('timestep_make_parameters_consistent: input error')
475  return
476  end if
477  else
478  if (tstep > zero) then
479  if (nsplit > 0) then
480  dtime = tstep*nstep_factor
481  else
482 #ifdef CAM
483  if (par%masterproc .and. .not. silent_in) then
484  write(iulog,*) 'If dtime is set to <=0 and tstep >0, then nsplit must be >0.'
485  end if
486  if (abort_in) call abortmp('timestep_make_parameters_consistent: input error')
487  return
488 #endif
489  end if
490  else
491  if (par%masterproc .and. .not. silent_in) then
492  write(iulog,*) 'If dtime is set to <=0, then tstep must be >0.'
493  end if
494  if (abort_in) call abortmp('timestep_make_parameters_consistent: input error')
495  return
496  end if
497  end if
498 
499  status = 0 ! success value
500  end function timestep_make_eam_parameters_consistent
