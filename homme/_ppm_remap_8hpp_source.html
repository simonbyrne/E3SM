211  KOKKOS_INLINE_FUNCTION
212  void compute_remap_phase(KernelVariables &kv,
213  ExecViewUnmanaged<Scalar[NP][NP][NUM_LEV]> remap_var)
214  const {
215  // From here, we loop over tracers for only those portions which depend on
216  // tracer data, which includes PPM limiting and mass accumulation
217  // More parallelism than we need here, maybe break it up?
218  Kokkos::parallel_for(Kokkos::TeamThreadRange(kv.team, NP * NP),
219  [&](const int &loop_idx) {
220  const int igp = loop_idx / NP;
221  const int jgp = loop_idx % NP;
222 
223  Kokkos::parallel_for(Kokkos::ThreadVectorRange(kv.team, NUM_PHYSICAL_LEV),
224  [&](const int k) {
225  const int ilevel = k / VECTOR_SIZE;
226  const int ivector = k % VECTOR_SIZE;
227  m_ao(kv.team_idx, igp, jgp, k + _ppm_consts::INITIAL_PADDING) =
228  remap_var(igp, jgp, ilevel)[ivector] /
229  m_dpo(kv.ie, igp, jgp, k + _ppm_consts::INITIAL_PADDING);
230  });
231 
232  boundaries::fill_cell_means_gs(kv,Homme::subview(m_ao, kv.team_idx, igp, jgp));
233 
234  Dispatch<ExecSpace>::parallel_scan(
235  kv.team, NUM_PHYSICAL_LEV,
236  [=](const int &k, Real &accumulator, const bool last) {
237  // Accumulate the old mass up to old grid cell interface locations
238  // to simplify integration during remapping. Also, divide out the
239  // grid spacing so we're working with actual tracer values and can
240  // conserve mass.
241  const int ilevel = k / VECTOR_SIZE;
242  const int ivector = k % VECTOR_SIZE;
243  accumulator += remap_var(igp, jgp, ilevel)[ivector];
244  if (last) {
245  m_mass_o(kv.team_idx, igp, jgp, k + 1) = accumulator;
246  }
247  });
248 
249  // Computes a monotonic and conservative PPM reconstruction
250  compute_ppm(kv,
251  Homme::subview(m_ao, kv.team_idx, igp, jgp),
252  Homme::subview(m_ppmdx, kv.ie, igp, jgp),
253  Homme::subview(m_dma, kv.team_idx, igp, jgp),
254  Homme::subview(m_ai, kv.team_idx, igp, jgp),
255  Homme::subview(m_parabola_coeffs, kv.team_idx, igp, jgp));
256 
257  compute_remap(kv,
258  Homme::subview(m_kid, kv.ie, igp, jgp),
259  Homme::subview(m_z2, kv.ie, igp, jgp),
260  Homme::subview(m_parabola_coeffs, kv.team_idx, igp, jgp),
261  Homme::subview(m_mass_o, kv.team_idx, igp, jgp),
262  Homme::subview(m_dpo, kv.ie, igp, jgp),
263  Homme::subview(remap_var, igp, jgp));
264  }); // End team thread range
265  kv.team_barrier();
266  }

268  KOKKOS_FORCEINLINE_FUNCTION
269  Real compute_mass(const Real sq_coeff, const Real lin_coeff,
270  const Real const_coeff, const Real prev_mass,
271  const Real prev_dp, const Real x2) const {
272  // This remapping assumes we're starting from the left interface of an
273  // old grid cell
274  // In fact, we're usually integrating very little or almost all of the
275  // cell in question
276  const Real x1 = -0.5;
277  const Real integral =
278  integrate_parabola(sq_coeff, lin_coeff, const_coeff, x1, x2);
279  const Real mass = prev_mass + integral * prev_dp;
280  return mass;
281  }

283  template <typename ExecSpaceType = ExecSpace>
284  KOKKOS_INLINE_FUNCTION
285  typename std::enable_if<!Homme::OnGpu<ExecSpaceType>::value, void>::type
286  compute_remap(KernelVariables &/* kv */,
287  ExecViewUnmanaged<const int[NUM_PHYSICAL_LEV]> k_id,
288  ExecViewUnmanaged<const Real[NUM_PHYSICAL_LEV]> integral_bounds,
289  ExecViewUnmanaged<const Real[3][NUM_PHYSICAL_LEV]> parabola_coeffs,
290  ExecViewUnmanaged<Real[_ppm_consts::MASS_O_PHYSICAL_LEV]> mass,
291  ExecViewUnmanaged<const Real[_ppm_consts::DPO_PHYSICAL_LEV]> prev_dp,
292  ExecViewUnmanaged<Scalar[NUM_LEV]> remap_var) const {
293  // Compute tracer values on the new grid by integrating from the old cell
294  // bottom to the new cell interface to form a new grid mass accumulation.
295  // Store the mass in the integral bounds for that level
296  // Then take the difference between accumulation at successive interfaces
297  // gives the mass inside each cell. Since Qdp is supposed to hold the full
298  // mass this needs no normalization.
299  Real mass1 = 0;
300  Real mass2;
301  ExecViewUnmanaged<Real[NUM_PHYSICAL_LEV]> rvar(reinterpret_cast<Real*>(remap_var.data()));
302  for (int k=0; k<NUM_PHYSICAL_LEV; ++k) {
303  const int kk_cur_lev = k_id(k);
304  assert(kk_cur_lev < parabola_coeffs.extent_int(1));
305 
306  const Real x2_cur_lev = integral_bounds(k);
307  // Repurpose the mass buffer to store the new mass.
308  // WARNING: This may not be thread safe in future architectures which
309  // use this level of parallelism!!!
310  mass2 = compute_mass(
311  parabola_coeffs(2, kk_cur_lev), parabola_coeffs(1, kk_cur_lev),
312  parabola_coeffs(0, kk_cur_lev), mass(kk_cur_lev),
313  prev_dp(kk_cur_lev + _ppm_consts::INITIAL_PADDING), x2_cur_lev);
314  rvar(k) = mass2 - mass1;
315  mass1 = mass2;
316  }
317  }

319  template <typename ExecSpaceType = ExecSpace>
320  KOKKOS_INLINE_FUNCTION
321  typename std::enable_if<Homme::OnGpu<ExecSpaceType>::value, void>::type
322  compute_remap(KernelVariables &kv,
323  ExecViewUnmanaged<const int[NUM_PHYSICAL_LEV]> k_id,
324  ExecViewUnmanaged<const Real[NUM_PHYSICAL_LEV]> integral_bounds,
325  ExecViewUnmanaged<const Real[3][NUM_PHYSICAL_LEV]> parabola_coeffs,
326  ExecViewUnmanaged<Real[_ppm_consts::MASS_O_PHYSICAL_LEV]> prev_mass,
327  ExecViewUnmanaged<const Real[_ppm_consts::DPO_PHYSICAL_LEV]> prev_dp,
328  ExecViewUnmanaged<Scalar[NUM_LEV]> remap_var) const {
329  // This duplicates work, but the parallel gain on CUDA is >> 2
330  assert(VECTOR_SIZE==1);
331  Kokkos::parallel_for(Kokkos::ThreadVectorRange(kv.team, NUM_PHYSICAL_LEV),
332  [&](const int k) {
333  const Real mass_1 =
334  (k > 0)
335  ? compute_mass(
336  parabola_coeffs(2, k_id(k - 1)),
337  parabola_coeffs(1, k_id(k - 1)),
338  parabola_coeffs(0, k_id(k - 1)), prev_mass(k_id(k - 1)),
339  prev_dp(k_id(k - 1) + _ppm_consts::INITIAL_PADDING),
340  integral_bounds(k - 1))
341  : 0.0;
342 
343  const Real x2_cur_lev = integral_bounds(k);
344 
345  const int kk_cur_lev = k_id(k);
346  assert(kk_cur_lev < parabola_coeffs.extent_int(1));
347 
348  const Real mass_2 = compute_mass(
349  parabola_coeffs(2, kk_cur_lev), parabola_coeffs(1, kk_cur_lev),
350  parabola_coeffs(0, kk_cur_lev), prev_mass(kk_cur_lev),
351  prev_dp(kk_cur_lev + _ppm_consts::INITIAL_PADDING), x2_cur_lev);
352 
353  remap_var(k)[0] = mass_2 - mass_1;
354  }); // k loop
355  }

357  KOKKOS_INLINE_FUNCTION
358  void compute_grids(KernelVariables &kv,
359  const ExecViewUnmanaged<const Real[_ppm_consts::DPO_PHYSICAL_LEV]> dx,
360  const ExecViewUnmanaged<Real[10][_ppm_consts::PPMDX_PHYSICAL_LEV]> grids) const
361  {
362  constexpr int dpo_offset = _ppm_consts::INITIAL_PADDING - _ppm_consts::gs;
363  Kokkos::parallel_for(Kokkos::ThreadVectorRange(kv.team,
364  NUM_PHYSICAL_LEV + 2),
365  [&](const int j) {
366  grids(0, j) = dx(j + 1 + dpo_offset) /
367  (dx(j + dpo_offset) + dx(j + 1 + dpo_offset) +
368  dx(j + 2 + dpo_offset));
369 
370  grids(1, j) = (2.0 * dx(j + dpo_offset) + dx(j + 1 + dpo_offset)) /
371  (dx(j + 1 + dpo_offset) + dx(j + 2 + dpo_offset));
372 
373  grids(2, j) = (dx(j + 1 + dpo_offset) + 2.0 * dx(j + 2 + dpo_offset)) /
374  (dx(j + dpo_offset) + dx(j + 1 + dpo_offset));
375  });
376 
377  Kokkos::parallel_for(Kokkos::ThreadVectorRange(kv.team,
378  NUM_PHYSICAL_LEV + 1),
379  [&](const int j) {
380  grids(3, j) = dx(j + 1 + dpo_offset) /
381  (dx(j + 1 + dpo_offset) + dx(j + 2 + dpo_offset));
382 
383  grids(4, j) = 1.0 / (dx(j + dpo_offset) + dx(j + 1 + dpo_offset) +
384  dx(j + 2 + dpo_offset) + dx(j + 3 + dpo_offset));
385 
386  grids(5, j) = (2.0 * dx(j + 1 + dpo_offset) * dx(j + 2 + dpo_offset)) /
387  (dx(j + 1 + dpo_offset) + dx(j + 2 + dpo_offset));
388 
389  grids(6, j) = (dx(j + dpo_offset) + dx(j + 1 + dpo_offset)) /
390  (2.0 * dx(j + 1 + dpo_offset) + dx(j + 2 + dpo_offset));
391 
392  grids(7, j) = (dx(j + 3 + dpo_offset) + dx(j + 2 + dpo_offset)) /
393  (2.0 * dx(j + 2 + dpo_offset) + dx(j + 1 + dpo_offset));
394 
395  grids(8, j) = dx(j + 1 + dpo_offset) *
396  (dx(j + dpo_offset) + dx(j + 1 + dpo_offset)) /
397  (2.0 * dx(j + 1 + dpo_offset) + dx(j + 2 + dpo_offset));
398 
399  grids(9, j) = dx(j + 2 + dpo_offset) *
400  (dx(j + 2 + dpo_offset) + dx(j + 3 + dpo_offset)) /
401  (dx(j + 1 + dpo_offset) + 2.0 * dx(j + 2 + dpo_offset));
402  });
403  }

405  KOKKOS_INLINE_FUNCTION
406  void compute_ppm(KernelVariables &kv,
407  // input views
408  ExecViewUnmanaged<const Real[_ppm_consts::AO_PHYSICAL_LEV]> cell_means,
409  ExecViewUnmanaged<const Real[10][_ppm_consts::PPMDX_PHYSICAL_LEV]> dx,
410  // buffer views
411  ExecViewUnmanaged<Real[_ppm_consts::DMA_PHYSICAL_LEV]> dma,
412  ExecViewUnmanaged<Real[_ppm_consts::AI_PHYSICAL_LEV]> ai,
413  // result view
414  ExecViewUnmanaged<Real[3][NUM_PHYSICAL_LEV]> parabola_coeffs) const
415  {
416  const auto INITIAL_PADDING = _ppm_consts::INITIAL_PADDING;
417 
418  Kokkos::parallel_for(Kokkos::ThreadVectorRange(kv.team,
419  NUM_PHYSICAL_LEV + 2),
420  [&](const int j) {
421  if ((cell_means(j + INITIAL_PADDING) -
422  cell_means(j + INITIAL_PADDING - 1)) *
423  (cell_means(j + INITIAL_PADDING - 1) -
424  cell_means(j + INITIAL_PADDING - gs)) >
425  0.0) {
426  Real da =
427  dx(0, j) * (dx(1, j) * (cell_means(j + INITIAL_PADDING) -
428  cell_means(j + INITIAL_PADDING - 1)) +
429  dx(2, j) * (cell_means(j + INITIAL_PADDING - 1) -
430  cell_means(j + INITIAL_PADDING - gs)));
431 
432  dma(j) = min(fabs(da), 2.0 * fabs(cell_means(j + INITIAL_PADDING - 1) -
433  cell_means(j + INITIAL_PADDING - gs)),
434  2.0 * fabs(cell_means(j + INITIAL_PADDING) -
435  cell_means(j + INITIAL_PADDING - 1))) *
436  copysign(1.0, da);
437  } else {
438  dma(j) = 0.0;
439  }
440  });
441 
442  Kokkos::parallel_for(
443  Kokkos::ThreadVectorRange(kv.team, NUM_PHYSICAL_LEV + 1),
444  [&](const int j) {
445  ai(j) = cell_means(j + INITIAL_PADDING - 1) +
446  dx(3, j) * (cell_means(j + INITIAL_PADDING) -
447  cell_means(j + INITIAL_PADDING - 1)) +
448  dx(4, j) * (dx(5, j) * (dx(6, j) - dx(7, j)) *
449  (cell_means(j + INITIAL_PADDING) -
450  cell_means(j + INITIAL_PADDING - 1)) -
451  dx(8, j) * dma(j + 1) + dx(9, j) * dma(j));
452  });
453 
454  Kokkos::parallel_for(Kokkos::ThreadVectorRange(kv.team, NUM_PHYSICAL_LEV),
455  [&](const int j_prev) {
456  const int j = j_prev + 1;
457  Real al = ai(j - 1);
458  Real ar = ai(j);
459  if ((ar - cell_means(j + INITIAL_PADDING - 1)) *
460  (cell_means(j + INITIAL_PADDING - 1) - al) <=
461  0.) {
462  al = cell_means(j + INITIAL_PADDING - 1);
463  ar = cell_means(j + INITIAL_PADDING - 1);
464  }
465  if ((ar - al) * (cell_means(j + INITIAL_PADDING - 1) - (al + ar) / 2.0) >
466  (ar - al) * (ar - al) / 6.0) {
467  al = 3.0 * cell_means(j + INITIAL_PADDING - 1) - 2.0 * ar;
468  }
469  if ((ar - al) * (cell_means(j + INITIAL_PADDING - 1) - (al + ar) / 2.0) <
470  -(ar - al) * (ar - al) / 6.0) {
471  ar = 3.0 * cell_means(j + INITIAL_PADDING - 1) - 2.0 * al;
472  }
473 
474  // Computed these coefficients from the edge values
475  // and cell mean in Maple. Assumes normalized
476  // coordinates: xi=(x-x0)/dx
477 
478  assert(parabola_coeffs.data() != nullptr);
479  assert(j - 1 < parabola_coeffs.extent_int(1));
480  assert(2 < parabola_coeffs.extent_int(0));
481

484  parabola_coeffs(1, j - 1) = ar - al;
485  parabola_coeffs(2, j - 1) =

494  KOKKOS_INLINE_FUNCTION
495  void compute_partitions(
496  KernelVariables &kv,
497  ExecViewUnmanaged<const Scalar[NP][NP][NUM_LEV]> src_layer_thickness,
498  ExecViewUnmanaged<const Scalar[NP][NP][NUM_LEV]> tgt_layer_thickness)
499  const {
500  Kokkos::parallel_for(Kokkos::TeamThreadRange(kv.team, NP * NP),
501  [&](const int &loop_idx) {
502  const int igp = loop_idx / NP;
503  const int jgp = loop_idx % NP;
504  ExecViewUnmanaged<Real[_ppm_consts::PIO_PHYSICAL_LEV]> pt_pio =
505  Homme::subview(m_pio, kv.ie, igp, jgp);
506  ExecViewUnmanaged<Real[_ppm_consts::PIN_PHYSICAL_LEV]> pt_pin =
507  Homme::subview(m_pin, kv.ie, igp, jgp);
508  ExecViewUnmanaged<const Scalar[NUM_LEV]> pt_src_thickness =
509  Homme::subview(src_layer_thickness, igp, jgp);
510  ExecViewUnmanaged<const Scalar[NUM_LEV]> pt_tgt_thickness =
511  Homme::subview(tgt_layer_thickness, igp, jgp);
512 
513  Dispatch<ExecSpace>::parallel_scan(
514  kv.team, NUM_PHYSICAL_LEV,
515  [&](const int &level, Real &accumulator, const bool last) {
516  if (last) {
517  pt_pio(level) = accumulator;
518  }
519  const int ilev = level / VECTOR_SIZE;
520  const int vlev = level % VECTOR_SIZE;
521  accumulator += pt_src_thickness(ilev)[vlev];
522  });
523  Dispatch<ExecSpace>::parallel_scan(
524  kv.team, NUM_PHYSICAL_LEV,
525  [&](const int &level, Real &accumulator, const bool last) {
526  if (last) {
527  pt_pin(level) = accumulator;
528  }
529  const int ilev = level / VECTOR_SIZE;
530  const int vlev = level % VECTOR_SIZE;
531  accumulator += pt_tgt_thickness(ilev)[vlev];
532  });
533 
534  Kokkos::single(Kokkos::PerThread(kv.team), [&]() {
535  const int ilast = (NUM_PHYSICAL_LEV - 1) / VECTOR_SIZE;
536  const int vlast = (NUM_PHYSICAL_LEV - 1) % VECTOR_SIZE;
537  pt_pio(NUM_PHYSICAL_LEV) =
538  pt_pio(NUM_PHYSICAL_LEV - 1) + pt_src_thickness(ilast)[vlast];
539  pt_pin(NUM_PHYSICAL_LEV) =
540  pt_pin(NUM_PHYSICAL_LEV - 1) + pt_tgt_thickness(ilast)[vlast];
541  // This is here to allow an entire block of k
542  // threads to run in the remapping phase. It makes
543  // sure there's an old interface value below the
544  // domain that is larger.
545  assert(fabs(m_pio(kv.ie, igp, jgp, NUM_PHYSICAL_LEV) -
546  m_pin(kv.ie, igp, jgp, NUM_PHYSICAL_LEV)) < 1.0);
547 
548  pt_pio(_ppm_consts::PIO_PHYSICAL_LEV - 1) =
549  pt_pio(_ppm_consts::PIO_PHYSICAL_LEV - 2) + 1.0;
550 
551  // The total mass in a column does not change.
552  // Therefore, the pressure of that mass cannot
553  // either.
554  pt_pin(NUM_PHYSICAL_LEV) = pt_pio(NUM_PHYSICAL_LEV);
555  });
556 
557  Kokkos::parallel_for(Kokkos::ThreadVectorRange(kv.team, NUM_PHYSICAL_LEV),
558  [&](const int &k) {
559  int ilevel = k / VECTOR_SIZE;
560  int ivector = k % VECTOR_SIZE;
561  m_dpo(kv.ie, igp, jgp, k + _ppm_consts::INITIAL_PADDING) =
562  src_layer_thickness(igp, jgp, ilevel)[ivector];
563  });
564  });
565  kv.team_barrier();
566  // Fill in the ghost regions with mirrored values.
567  // if vert_remap_q_alg is defined, this is of no
568  // consequence.
569  // Note that the range of k makes this completely parallel,
570  // without any data dependencies
571  Kokkos::parallel_for(Kokkos::TeamThreadRange(kv.team, NP * NP),
572  [&](const int &loop_idx) {
573  const int igp = loop_idx / NP;
574  const int jgp = loop_idx % NP;
575  Kokkos::parallel_for(Kokkos::ThreadVectorRange(kv.team, gs),
576  [&](const int &k) {
577  m_dpo(kv.ie, igp, jgp, _ppm_consts::INITIAL_PADDING - 1 - k) =
578  m_dpo(kv.ie, igp, jgp, k + _ppm_consts::INITIAL_PADDING);
579  m_dpo(kv.ie, igp, jgp,
580  NUM_PHYSICAL_LEV + _ppm_consts::INITIAL_PADDING + k) =
581  m_dpo(kv.ie, igp, jgp,
582  NUM_PHYSICAL_LEV + _ppm_consts::INITIAL_PADDING - 1 - k);
583  });
584  });
585  kv.team_barrier();
586  }

588  KOKKOS_INLINE_FUNCTION
589  void compute_integral_bounds(KernelVariables &kv) const {
590  Kokkos::parallel_for(Kokkos::TeamThreadRange(kv.team, NP * NP),
591  [&](const int &loop_idx) {
592  const int igp = loop_idx / NP;
593  const int jgp = loop_idx % NP;
594  Kokkos::parallel_for(Kokkos::ThreadVectorRange(kv.team, NUM_PHYSICAL_LEV),
595  [&](const int k) {
596  // Compute remapping intervals once for all
597  // tracers. Find the old grid cell index in which
598  // the k-th new cell interface resides. Then
599  // integrate from the bottom of that old cell to
600  // the new interface location. In practice, the
601  // grid never deforms past one cell, so the search
602  // can be simplified by this. Also, the interval
603  // of integration is usually of magnitude close to
604  // zero or close to dpo because of minimial
605  // deformation. Numerous tests confirmed that the
606  // bottom and top of the grids match to machine
607  // precision, so set them equal to each other.
608  int kk = k;// + 1;
609  // This reduces the work required to find the index where this
610  // fails at, and is typically less than NUM_PHYSICAL_LEV^2 Since
611  // the top bounds match anyway, the value of the coefficients
612  // don't matter, so enforcing kk <= NUM_PHYSICAL_LEV doesn't
613  // affect anything important
614  //
615  // Note that because we set
616  // pio(:, :, :, NUM_PHYSICAL_LEV + 1) = pio(:, :, :,
617  // NUM_PHYSICAL_LEV) + 1.0 and pin(:, :, :, NUM_PHYSICAL_LEV) =
618  // pio(:, :, :, NUM_PHYSICAL_LEV) this loop ensures kk <=
619  // NUM_PHYSICAL_LEV + 2 Furthermore, since we set pio(:, :, :,
620  // 0) = 0.0 and pin(:, :, :, 0) = 0.0 kk must be incremented at
621  // least once
622  assert(m_pio(kv.ie, igp, jgp, _ppm_consts::PIO_PHYSICAL_LEV - 1) >
623  m_pin(kv.ie, igp, jgp, k + 1));
624  if (m_pio(kv.ie,igp,jgp,kk) <= m_pin(kv.ie,igp,jgp,k+1)) {
625  while (m_pio(kv.ie, igp, jgp, kk) <= m_pin(kv.ie, igp, jgp, k + 1)) {
626  kk++;
627  assert(kk < m_pio.extent_int(3));
628  }
629  --kk;
630  } else {
631  binary_search(Homme::subview(m_pio,kv.ie,igp,jgp),m_pin(kv.ie,igp,jgp,k+1),kk);
632  }
633 
634  // This is to keep the indices in bounds.
635  if (kk == NUM_PHYSICAL_LEV) {
636  --kk;
637  }
638  // kk is now the cell index we're integrating over.
639 
640  // Save kk for reuse
641  m_kid(kv.ie, igp, jgp, k) = kk;
642  // PPM interpolants are normalized to an independent coordinate
643  // domain
644  // [-0.5, 0.5].
645  m_z2(kv.ie, igp, jgp, k) =
646  (m_pin(kv.ie, igp, jgp, k + 1) -
647  (m_pio(kv.ie, igp, jgp, kk) + m_pio(kv.ie, igp, jgp, kk+1)) * 0.5) /
648  m_dpo(kv.ie, igp, jgp, kk + _ppm_consts::INITIAL_PADDING);
649  });
650 
651  auto point_dpo = Homme::subview(m_dpo, kv.ie, igp, jgp);
652  auto point_ppmdx = Homme::subview(m_ppmdx, kv.ie, igp, jgp);
653  compute_grids(kv, point_dpo, point_ppmdx);
654  });
655  }
