5  ! Function for setting up idealized tropical cyclone initial conditions
6  !
7  ! Given a point specified by:
8  ! longitude (radians)
9  ! latitude (radians)

23  ! Initial data are currently identical to:
24  !
25  ! Reed, K. A., and C. Jablonowski, 2011: An analytic
26  ! vortex initialization technique for idealized tropical
27  ! cyclone studies in AGCMs. Mon. Wea. Rev., 139, 689-710.

77  ttrop = t0 - gamma*ztrop, & ! Tropopause temp

87 SUBROUTINE test5_tropical_cyclone (lon,lat,p,z,zcoords,u,v,w,t,phis,ps,rho,q)

91 !-----------------------------------------------------------------------
92 ! input/output params parameters at given location
93 !-----------------------------------------------------------------------

95  real(8), intent(in) :: lon, & ! Longitude (radians)
96  lat, & ! Latitude (radians)
97  z ! Height (m)
98 
99  real(8), intent(inout) :: p ! Pressure (Pa)
100 
101  integer, intent(in) :: zcoords ! 0 or 1 see below
102  real(8), intent(out) :: u, & ! Zonal wind (m s^-1)
103  v, & ! Meridional wind (m s^-1)
104  w, & ! Vertical velocity (m s^-1)
105  t, & ! Temperature (K)
106  phis, & ! Surface Geopotential (m^2 s^-2)
107  ps, & ! Surface Pressure (Pa)
108  rho, & ! density (kg m^-3)
109  q ! Specific Humidity (kg/kg)
110 
111  ! if zcoords = 1, then we use z and output p
112  ! if zcoords = 0, then we use p
113 
114 !-----------------------------------------------------------------------
115 ! Additional parameters
116 !-----------------------------------------------------------------------
117  real(8) :: d1, d2, d, vfac, ufac, height, zhere, gr, f, zn
118 
119  integer n
120 
121 !-----------------------------------------------------------------------
122 ! Define Great circle distance (gr) and Coriolis parameter (f)
123 !-----------------------------------------------------------------------
124 
125  f = 2.d0*omega*sin(cen_lat/convert) ! Coriolis parameter
126  gr = a*acos(sin(cen_lat/convert)*sin(lat) + & ! Great circle distance
127  (cos(cen_lat/convert)*cos(lat)*cos(lon-cen_lon/convert)))
128 
129 !-----------------------------------------------------------------------
130 ! initialize PS (surface pressure of moist air)
131 !-----------------------------------------------------------------------
132  ps = p00-dp*exp(-(gr/rp)**exppr)
133 
134 !-----------------------------------------------------------------------
135 ! initialize height field if provided pressure or pressure if provided z
136 !-----------------------------------------------------------------------
137 
138  if (zcoords .eq. 1) then
139 
140  height = z
141 
142  if (height > ztrop) then
143  p = ptrop*exp(-(g*(height-ztrop))/(rd*ttrop))
144  else
145  p = (p00-dp*exp(-(gr/rp)**exppr)*exp(-(height/zp)**exppz)) &
146  * ((t0-gamma*height)/t0)**(1/exponent)
147  end if
148 
149  else
150 
151  height = (t0/gamma)*(1.d0-(p/ps)**exponent)
152 
153  ! If inside a certain distance of the center of the storm
154  ! perform a Fixed-point iteration to calculate the height
155  ! more accurately
156 
157  if (gr < rfpi ) then
158  zhere = height
159  n = 1
160  20 continue
161  n = n+1
162  zn = zhere - fpif(p,gr,zhere)/fpidfdz(gr,zhere)
163  if (n.gt.20) then
164  print *,'FPI did not converge after 20 interations in q & T!!!'
165  else if ( abs(zn-zhere)/abs(zn) > deltaz) then
166  zhere = zn
167  goto 20
168  end if
169  height = zn
170  end if
171 
172  end if
173 
174 !-----------------------------------------------------------------------
175 ! initialize U and V (wind components)
176 !-----------------------------------------------------------------------
177 
178  d1 = sin(cen_lat/convert)*cos(lat) - &
179  cos(cen_lat/convert)*sin(lat)*cos(lon-cen_lon/convert)
180  d2 = cos(cen_lat/convert)*sin(lon-cen_lon/convert)
181  d = max(epsilon, sqrt(d1**2.d0 + d2**2.d0))
182  ufac = d1/d
183  vfac = d2/d
184  if (height > ztrop) then
185  u = 0.d0
186  v = 0.d0
187  else
188  v = vfac*(-f*gr/2.d0+sqrt((f*gr/2.d0)**(2.d0) &
189  - exppr*(gr/rp)**exppr*rd*(t0-gamma*height) &
190  /(exppz*height*rd*(t0-gamma*height)/(g*zp**exppz) &
191  +(1.d0-p00/dp*exp((gr/rp)**exppr)*exp((height/zp)**exppz)))))
192  u = ufac*(-f*gr/2.d0+sqrt((f*gr/2.d0)**(2.d0) &
193  - exppr*(gr/rp)**exppr*rd*(t0-gamma*height) &
194  /(exppz*height*rd*(t0-gamma*height)/(g*zp**exppz) &
195  +(1.d0-p00/dp*exp((gr/rp)**exppr)*exp((height/zp)**exppz)))))
196  end if
197 
198 !-----------------------------------------------------------------------
199 ! set the vertical velocity to zero (only required for non-hydrostatic models)
200 !-----------------------------------------------------------------------
201 
202  w = 0.d0
203 
204 !-----------------------------------------------------------------------
205 ! tracer q (specific humidity)
206 !-----------------------------------------------------------------------
207 
208  if (height > ztrop) then
209  q = qtrop
210  else
211  q = q0*exp(-height/zq1)*exp(-(height/zq2)**exppz)
212  end if
213 
214 !-----------------------------------------------------------------------
215 ! initialize T (temperature)
216 !-----------------------------------------------------------------------
217 
218  if (height > ztrop) then
219  t = ttrop
220  else
221  t = (t0-gamma*height)/(1.d0+consttv*q)/(1.d0+exppz*rd*(t0-gamma*height)*height &
222  /(g*zp**exppz*(1.d0-p00/dp*exp((gr/rp)**exppr)*exp((height/zp)**exppz))))
223  end if
224 
225 !-----------------------------------------------------------------------
226 ! initialize PHIS (surface geopotential)
227 !-----------------------------------------------------------------------
228 
229  phis = 0.d0 ! constant
230 
231 !-----------------------------------------------------------------------
232 ! initialize RHO (density of moist air)
233 !-----------------------------------------------------------------------
234 
235  rho = p/(rd*t*(1.d0+consttv*q))
236 
237  END SUBROUTINE test5_tropical_cyclone

250  END FUNCTION fpif
