32 void RefStates::compute(const bool hydrostatic,
33  const HybridVCoord& hvcoord,
34  const ExecViewUnmanaged<Real *[NP][NP]>& phis) {
35  EquationOfState eos;
36  eos.init(hydrostatic,hvcoord);
37 
38  ElementOps elem_ops;
39  elem_ops.init(hvcoord);
40 
41  const int num_elems = dp_ref.extent_int(0);
42  assert(num_elems==m_num_elems);
43 
44  // Local copies, to avoid cuda issues with *this
45  auto l_dp_ref = dp_ref;
46  auto l_phi_i_ref = phi_i_ref;
47  auto l_theta_ref = theta_ref;
48 
49  constexpr Real Rgas = PhysicalConstants::Rgas;
50  constexpr Real Tref = PhysicalConstants::Tref;
51 
52  const int num_slots = m_tu.get_num_ws_slots();
53  const auto tu = m_tu;
54 
55  ExecViewManaged<Scalar*[NP][NP][NUM_LEV]> buf_p("",num_slots);
56  ExecViewManaged<Scalar*[NP][NP][NUM_LEV_P]> buf_p_i("",num_slots);
57  Kokkos::parallel_for(m_policy,KOKKOS_LAMBDA(const TeamMember& team){
58  KernelVariables kv(team, tu);
59  Kokkos::parallel_for(Kokkos::TeamThreadRange(team,NP*NP),
60  [&](const int idx){
61  const int igp = idx / NP;
62  const int jgp = idx % NP;
63 
64  auto dpRef = Homme::subview(l_dp_ref,kv.ie,igp,jgp);
65  auto phiRef = Homme::subview(l_phi_i_ref,kv.ie,igp,jgp);
66  auto thetaRef = Homme::subview(l_theta_ref,kv.ie,igp,jgp);
67  auto p = Homme::subview(buf_p,kv.team_idx,igp,jgp);
68  auto p_i = Homme::subview(buf_p_i,kv.team_idx,igp,jgp);
69 
70  // Step 0: compute ps_ref
71  const Real ps_ref = hvcoord.ps0 * exp(-phis(kv.ie,igp,jgp)/(Rgas*Tref));
72 
73  // Step 1: compute dp_ref = delta_hyai(k)*ps0 + delta_hybi(k)*ps_ref
74  hvcoord.compute_dp_ref(kv,ps_ref,dpRef);
75 
76  // Step 2: compute p_ref = p(p_i(dp))
77  p_i(0)[0] = hvcoord.hybrid_ai0*hvcoord.ps0;
78  ColumnOps::column_scan_mid_to_int<true>(kv,dpRef,p_i);
79  ColumnOps::compute_midpoint_values(kv,p_i,p);
80 
81  // Step 3: compute theta_ref = theta(exner(p_ref))
82  elem_ops.compute_theta_ref(kv,p,thetaRef);
83 
84  // Step 3: compute phi_i_ref = phi(theta_ref,dp_ref,p_ref)
85  // TODO: you could exploit exner computed in compute_theta_ref, to replace (p/p0)^(k-1) with exner*p0/p.
86  auto theta_dp = [&thetaRef,&dpRef](const int ilev)->Scalar {
87  return thetaRef(ilev)*dpRef(ilev);
88  };
89  eos.compute_phi_i(kv,phis(kv.ie,igp,jgp),
90  theta_dp,p,phiRef);
91  });
92  });
93  Kokkos::fence();
94  Kokkos::deep_copy(theta_ref,0.0);
95  Kokkos::deep_copy(dp_ref,0.0);
96 }

123 void ElementsState::randomize(const int seed,
124  const Real max_pressure,
125  const Real ps0, const Real hyai0,
126  const ExecViewUnmanaged<const Real*[NP][NP]>& phis) {
127  randomize(seed,max_pressure,ps0,hyai0);
128 
129  // Re-do phinh so it satisfies phinh_i(bottom)=phis
130 
131  // Sanity check
132  assert(phis.extent_int(0)==m_num_elems);
133 
134  std::mt19937_64 engine(seed);
135 
136  // Note: to avoid errors in the equation of state, we need phi to be increasing.
137  // Rather than using a constraint (which may call the function many times,
138  // we simply ask that there are no duplicates, then we sort it later.
139  auto sort_and_chek = [](const ExecViewManaged<Real[NUM_PHYSICAL_LEV]>::HostMirror v)->bool {
140  Real* start = reinterpret_cast<Real*>(v.data());
141  Real* end = reinterpret_cast<Real*>(v.data()) + NUM_PHYSICAL_LEV;
142  std::sort(start,end);
143  std::reverse(start,end);
144  auto it = std::unique(start,end);
145  return it==end;
146  };
147 
148  auto h_phis = Kokkos::create_mirror_view(phis);
149  Kokkos::deep_copy(h_phis,phis);
150  for (int ie=0; ie<m_num_elems; ++ie) {
151  for (int igp=0; igp<NP; ++igp) {
152  for (int jgp=0; jgp<NP; ++ jgp) {
153  const Real phis_ij = h_phis(ie,igp,jgp);
154  // Ensure generated values are larger than phis
155  std::uniform_real_distribution<Real> random_dist(1.001*phis_ij,100.0*phis_ij);
156  for (int itl=0; itl<NUM_TIME_LEVELS; ++itl) {
157  // Get column
158  auto phi_col = Homme::viewAsReal(Homme::subview(m_phinh_i,ie,itl,igp,jgp));
159 
160  // Stuff phis at the bottom
161  Kokkos::deep_copy(Kokkos::subview(phi_col,NUM_PHYSICAL_LEV),phis_ij);
162 
163  // Generate values except at bottom
164  ExecViewUnmanaged<Real[NUM_PHYSICAL_LEV]> phi_no_bottom(phi_col.data());
165  genRandArray(phi_no_bottom,engine,random_dist,sort_and_chek);
166  }
167  }
168  }
169  }
170 }

172 void ElementsState::randomize(const int seed,
173  const Real max_pressure,
174  const Real ps0,
175  const Real hyai0) {
176  // Check elements were inited
177  assert (m_num_elems>0);
178 
179  // Check data makes sense
180  assert (max_pressure>ps0);
181  assert (ps0>0);
182  assert (hyai0>=0);
183 
184  // Arbitrary minimum value to generate
185  constexpr const Real min_value = 0.015625;
186 
187  std::mt19937_64 engine(seed);
188  std::uniform_real_distribution<Real> random_dist(min_value, 1.0 / min_value);
189 
190  genRandArray(m_v, engine, random_dist);
191  genRandArray(m_w_i, engine, random_dist);
192  genRandArray(m_vtheta_dp, engine, random_dist);
193  // Note: to avoid errors in the equation of state, we need phi to be increasing.
194  // Rather than using a constraint (which may call the function many times,
195  // we simply ask that there are no duplicates, then we sort it later.
196  auto sort_and_chek = [](const ExecViewManaged<Scalar[NUM_LEV_P]>::HostMirror v)->bool {
197  Real* start = reinterpret_cast<Real*>(v.data());
198  Real* end = reinterpret_cast<Real*>(v.data()) + NUM_LEV_P*VECTOR_SIZE;
199  std::sort(start,end);
200  std::reverse(start,end);
201  auto it = std::unique(start,end);
202  return it==end;
203  };
204  for (int ie=0; ie<m_num_elems; ++ie) {
205  for (int itl=0; itl<NUM_TIME_LEVELS; ++itl) {
206  for (int igp=0; igp<NP; ++igp) {
207  for (int jgp=0; jgp<NP; ++ jgp) {
208  genRandArray(Homme::subview(m_phinh_i,ie,itl,igp,jgp),engine,random_dist,sort_and_chek);
209  }
210  }
211  }
212  }
213 
214  // This ensures the pressure in a single column is monotonically increasing
215  // and has fixed upper and lower values
216  const auto make_pressure_partition = [=](
217  ExecViewUnmanaged<Scalar[NUM_LEV]> pt_dp) {
218 
219  auto h_pt_dp = Kokkos::create_mirror_view(pt_dp);
220  Kokkos::deep_copy(h_pt_dp,pt_dp);
221  Real* data = reinterpret_cast<Real*>(h_pt_dp.data());
222  Real* data_end = data + NUM_PHYSICAL_LEV;
223 
224  Real p[NUM_INTERFACE_LEV];
225  Real* p_start = &p[0];
226  Real* p_end = p_start+NUM_INTERFACE_LEV;
227 
228  for (int i=0; i<NUM_PHYSICAL_LEV; ++i) {
229  p[i+1] = data[i];
230  }
231  p[0] = ps0*hyai0;
232  p[NUM_INTERFACE_LEV-1] = max_pressure;
233 
234  // Put in monotonic order
235  std::sort(p_start, p_end);
236 
237  // Check for no repetitions
238  if (std::unique(p_start,p_end)!=p_end) {
239  return false;
240  }
241 
242  // Compute dp from p (we assume p(last interface)=max_pressure)
243  for (int i=0; i<NUM_PHYSICAL_LEV; ++i) {
244  data[i] = p[i+1]-p[i];
245  }
246 
247  // Check that dp>=dp_min
248  const Real min_dp = std::numeric_limits<Real>::epsilon()*1000;
249  for (auto it=data; it!=data_end; ++it) {
250  if (*it < min_dp) {
251  return false;
252  }
253  }
254 
255  // Fill remainder of last vector pack with quiet nan's
256  Real* alloc_end = data+NUM_LEV*VECTOR_SIZE;
257  for (auto it=data_end; it!=alloc_end; ++it) {
258  *it = std::numeric_limits<Real>::quiet_NaN();
259  }
260 
261  Kokkos::deep_copy(pt_dp,h_pt_dp);
262 
263  return true;
264  };
265 
266  std::uniform_real_distribution<Real> pressure_pdf(min_value, max_pressure);
267 
268  for (int ie = 0; ie < m_num_elems; ++ie) {
269  // Because this constraint is difficult to satisfy for all of the tensors,
270  // incrementally generate the view
271  for (int igp = 0; igp < NP; ++igp) {
272  for (int jgp = 0; jgp < NP; ++jgp) {
273  for (int tl = 0; tl < NUM_TIME_LEVELS; ++tl) {
274  ExecViewUnmanaged<Scalar[NUM_LEV]> pt_dp3d =
275  Homme::subview(m_dp3d, ie, tl, igp, jgp);
276  do {
277  genRandArray(pt_dp3d, engine, pressure_pdf);
278  } while (make_pressure_partition(pt_dp3d)==false);
279  }
280  }
281  }
282  }
283 
284  // Generate ps_v so that it is equal to sum(dp3d).
285  HybridVCoord hvcoord;
286  hvcoord.ps0 = ps0;
287  hvcoord.hybrid_ai0 = hyai0;
288  hvcoord.m_inited = true;
289  auto dp = m_dp3d;
290  auto ps = m_ps_v;
291  const auto tu = m_tu;
292  Kokkos::parallel_for(m_policy, KOKKOS_LAMBDA(const TeamMember& team) {
293  KernelVariables kv(team, tu);
294  const int ie = kv.ie / NUM_TIME_LEVELS;
295  const int tl = kv.ie % NUM_TIME_LEVELS;
296  hvcoord.compute_ps_ref_from_dp(kv,Homme::subview(dp,ie,tl),
297  Homme::subview(ps,ie,tl));
298  });
299  ExecSpace::impl_static_fence();
300 }
