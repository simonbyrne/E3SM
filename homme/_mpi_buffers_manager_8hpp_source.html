25 /*
26  * MpiBuffersManager: a class to handle the buffers needed by BoundaryExchange
27  *
28  * This class (BM) is responsible mainly for allocating buffers to be
29  * used by the BoundaryExchange (BE) class. A single BM object can
30  * be the buffers provider for several different BE objects, which
31  * are referred to as 'customers'; in this case, the stored buffers
32  * will have a size big enough to accommodate the most demanding needs
33  * among those of all the customers. For more details about why one BM
34  * may have more than one customer, see the explanation in the header
35  * BoundaryExchange.hpp.
36  *
37  * The BM class stores 4 types of buffers, in the form of Kokkos View's:
38  *
39  * - a local buffer: this can be non empty only if there is more than
40  * one element per processor. This buffer stores values that the BE
41  * customer needs to exchange with other elements on the same rank
42  * - a send and a recv buffer: these buffers are non empty only if
43  * there is more than one rank. These buffers will store the outgoing
44  * and incoming values that the BE customer is exchanging with
45  * other processes.
46  * - a blackhole send and recv buffer: these buffers are always used
47  * (regardless of the number of ranks), and they will be used to
48  * dump discarded values or read dummy zero values. Their need
49  * arises from the fact that there are missing connections between
50  * elements (see ConnectivityHelpers.hpp). In particular, there
51  * are some elements corners that are not shared with any other
52  * element per se. They are shared with other elements AS PART
53  * OF AN EDGE, but there is no other element that thouches the
54  * current one ONLY through that point. In order to avoid if's
55  * inside BE, we treat these corners as 'normal' corners, but
56  * the outgoing values will be stored in this blackhole send
57  * buffer (which is never read from), and the incoming values
58  * will be read from the blackhole recv buffer (which is filled
59  * with zeros).
60  * - an mpi send and recv buffer: these buffers are strictly linked
61  * to the send and recv ones. They indeed store the same pointers
62  * if the MPI can be performed using pointers in the Kokkos'
63  * Execution Space. This is always true for CPU/KNL builds, but
64  * it may or may not be true for GPU builds.
65  * The send/recv buffers are used to pack/unpack the data, while
66  * the mpi_send/mpi_recv buffers are used by MPI.
67  *
68  * The BM class also takes care of syncing the send/recv buffers
69  * with the mpi_send/mpi_recv buffers, via a call to Kokkos::deep_copy,
70  * which is a no-op if the MPIMemSpace=ExecMemSpace, that is, if
71  * the MPI is performed using pointers on the Execution Space.
72  *
73  */

186 inline void MpiBuffersManager::sync_send_buffer (BoundaryExchange* customer)
187 {
188  // Only customers can call this
189  assert (m_customers.find(customer)!=m_customers.end());
190 
191  const size_t customer_mpi_buffer_size = m_customers.find(customer)->second.mpi_buffer_size;
192  if (customer_mpi_buffer_size<m_mpi_buffer_size) {
193  // Avoid copying more than we need
194  MPIViewUnmanaged<Real*> mpi_send_view(m_mpi_send_buffer.data(),customer_mpi_buffer_size);
195  ExecViewUnmanaged<const Real*> send_view(m_send_buffer.data(),customer_mpi_buffer_size);
196  Kokkos::deep_copy(mpi_send_view, send_view);
197  } else {
198  Kokkos::deep_copy(m_mpi_send_buffer, m_send_buffer);
199  }
200 }
