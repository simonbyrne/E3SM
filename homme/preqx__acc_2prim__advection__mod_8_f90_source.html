153  ! (complete with DSS), and then applies a vertical remap
154  !
155  ! This routine may use dynamics fields at timelevel np1

157  ! explicitly saved by the dynamics: (in elem(ie)%derived struct)

159  ! Fields required from dynamics: (in
160  ! omega_p it will be DSS'd here, for later use by CAM physics
161  ! we DSS omega here because it can be done for "free"

167  ! Euler step from t -> t+.5

305  subroutine advance_hypervis_scalar( elem , hvcoord , hybrid , deriv , nt , nt_qdp , nets , nete , dt2 )

307  ! take one timestep of:
308  ! Q(:,:,:,np) = Q(:,:,:,np) + dt2*nu*laplacian**order ( Q )
309  ! For correct scaling, dt2 should be the same 'dt2' used in the leapfrog advace

311  use dimensions_mod , only: np, nlev
312  use hybrid_mod , only: hybrid_t

314  use element_state , only: derived_divdp_proj, state_qdp

317  use hybvcoord_mod , only: hvcoord_t
318  use control_mod , only: nu_q, hypervis_order, hypervis_subcycle_q, nu_p
319  use viscosity_mod, only: biharmonic_wk_scalar_openacc
320  use edge_mod , only: edgevpack_openacc, edgevunpack_openacc
321  use bndry_mod , only: bndry_exchangev => bndry_exchangev_simple_overlap
322  implicit none
323  type (element_t) , intent(inout), target :: elem(:)
324  type (hvcoord_t) , intent(in ) :: hvcoord
325  type (hybrid_t) , intent(in ) :: hybrid
326  type (derivative_t) , intent(in ) :: deriv
327  integer , intent(in ) :: nt
328  integer , intent(in ) :: nt_qdp
329  integer , intent(in ) :: nets
330  integer , intent(in ) :: nete
331  real (kind=real_kind), intent(in ) :: dt2
332  ! local
333  real (kind=real_kind), dimension( nlev,qsize,nets:nete) :: min_neigh
334  real (kind=real_kind), dimension( nlev,qsize,nets:nete) :: max_neigh
335  integer :: k , kptr , i , j , ie , ic , q
336  real (kind=real_kind), dimension(np,np) :: lap_p
337  real (kind=real_kind) :: v1,v2,dt
338  real (kind=real_kind) :: dp
339  integer :: density_scaling = 0
340  if ( nu_q == 0 ) return
341  if ( hypervis_order /= 2 ) return
342  call t_startf('advance_hypervis_scalar')

344  ! hyper viscosity
345  !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!

347

361  ! contribution from dynamics.
362  do j = 1 , np

368  enddo
369  else
370  do q = 1 , qsize
371  qtens(i,j,k,q,ie) = hvcoord%dp0(k)*state_qdp(i,j,k,q,nt_qdp,ie) / dp
372  enddo
373  endif
374  enddo
375  enddo
376  enddo
377  enddo
378  !$omp end master

386  do q = 1 , qsize
387  do k = 1 , nlev
388  do j = 1 , np
389  do i = 1 , np

391  ! DIFF( Qdp) ~ dp0 DIFF (Q) = dp0 DIFF ( Qdp/dp )
392  state_qdp(i,j,k,q,nt_qdp,ie) = state_qdp(i,j,k,q,nt_qdp,ie) * elem(ie)%spheremp(i,j) - dt * nu_q * qtens(i,j,k,q,ie)

394  enddo

403

417  do j = 1 , np

496  ! local

498  real(kind=real_kind) :: tmp, mintmp, maxtmp, dp
499  integer :: ie,q,i,j,k

505  data_pack2(:,:,:,ie) = elem(ie)%derived%dp
506  enddo

510  !$acc update device(derived_divdp_proj,derived_vn0) async(2)

520  enddo
521  enddo

527  ! compute Q min/max values for lim8
528  ! compute biharmonic mixing term f
529  !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
530  rhs_viss = 0
531  if ( limiter_option == 8 ) then
532  ! when running lim8, we also need to limit the biharmonic, so that term needs
533  ! to be included in each euler step. three possible algorithms here:
534  ! 1) most expensive:
535  ! compute biharmonic (which also computes qmin/qmax) during all 3 stages
536  ! be sure to set rhs_viss=1
537  ! cost: 3 biharmonic steps with 3 DSS
538  !
539  ! 2) cheapest:
540  ! compute biharmonic (which also computes qmin/qmax) only on first stage
541  ! be sure to set rhs_viss=3
542  ! reuse qmin/qmax for all following stages (but update based on local qmin/qmax)
543  ! cost: 1 biharmonic steps with 1 DSS
544  ! main concern: viscosity
545  !
546  ! 3) compromise:
547  ! compute biharmonic (which also computes qmin/qmax) only on last stage
548  ! be sure to set rhs_viss=3
549  ! compute qmin/qmax directly on first stage
550  ! reuse qmin/qmax for 2nd stage stage (but update based on local qmin/qmax)
551  ! cost: 1 biharmonic steps, 2 DSS
552  !
553  ! NOTE when nu_p=0 (no dissipation applied in dynamics to dp equation), we should
554  ! apply dissipation to Q (not Qdp) to preserve Q=1
555  ! i.e. laplace(Qdp) ~ dp0 laplace(Q)
556  ! for nu_p=nu_q>0, we need to apply dissipation to Q * diffusion_dp
557  !
558  ! initialize dp, and compute Q from Qdp (and store Q in Qtens_biharmonic)
559  !$omp barrier
560  !$omp master
561  !$acc parallel loop gang vector collapse(4) private(tmp) present(elem(:),derived_divdp_proj,state_qdp,qtens_biharmonic)
562  do ie = 1 , nelemd
563  ! add hyperviscosity to RHS. apply to Q at timelevel n0, Qdp(n0)/dp
564  do k = 1 , nlev ! Loop index added with implicit inversion (AAM)
565  do j = 1 , np
566  do i = 1 , np
567  tmp = elem(ie)%derived%dp(i,j,k) - rhs_multiplier*dt*derived_divdp_proj(i,j,k,ie)
568  do q = 1 , qsize
569  qtens_biharmonic(i,j,k,q,ie) = state_qdp(i,j,k,q,n0_qdp,ie) / tmp
570  enddo
571  enddo
572  enddo
573  enddo
574  enddo
575  !$acc parallel loop gang vector collapse(3) private(mintmp,maxtmp) present(qmin,qmax,qtens_biharmonic)
576  do ie = 1 , nelemd
577  do q = 1 , qsize
578  do k = 1 , nlev
579  if (rhs_multiplier == 0 .or. rhs_multiplier == 2) then
580  mintmp = 1.d20
581  maxtmp = -1.d20
582  else

584  maxtmp = qmax(k,q,ie)
585  endif
586  do j = 1 , np
587  do i = 1 , np
588  mintmp = min(mintmp,qtens_biharmonic(i,j,k,q,ie))
589  maxtmp = max(maxtmp,qtens_biharmonic(i,j,k,q,ie))
590  enddo
591  enddo
592  qmin(k,q,ie) = max(mintmp,0d0)
593  qmax(k,q,ie) = maxtmp
594  enddo
595  enddo
596  enddo
597  !$omp end master
598  !$omp barrier
599  if ( rhs_multiplier == 0 ) call neighbor_minmax_openacc(elem,hybrid,edgeminmax,1,nelemd,qmin,qmax)
600  ! compute biharmonic mixing term
601  if ( rhs_multiplier == 2 ) then
602  rhs_viss = 3
603  ! two scalings depending on nu_p:
604  ! nu_p=0: qtens_biharmonic *= dp0 (apply viscsoity only to q)
605  ! nu_p>0): qtens_biharmonc *= elem()%psdiss_ave (for consistency, if nu_p=nu_q)
606  if ( nu_p > 0 ) then
607  !$omp barrier
608  !$omp master
609  !$acc parallel loop gang vector collapse(4) present(elem(:),qtens_biharmonic,hvcoord%dp0)
610  do ie = 1 , nelemd
611  do k = 1 , nlev
612  do j = 1 , np
613  do i = 1 , np
614  tmp = elem(ie)%derived%dpdiss_ave(i,j,k) / hvcoord%dp0(k)
615  do q = 1 , qsize
616  ! NOTE: divide by dp0 since we multiply by dp0 below
617  qtens_biharmonic(i,j,k,q,ie) = qtens_biharmonic(i,j,k,q,ie) * tmp
618  enddo
619  enddo
620  enddo
621  enddo
622  enddo
623  !$omp end master
624  !$omp barrier
625  endif
626  call biharmonic_wk_scalar_openacc( elem , qtens_biharmonic , grads_tracer , deriv , edge_g , hybrid , 1 , nelemd )
627  call neighbor_minmax_openacc( elem , hybrid , edgeminmax , 1 , nelemd , qmin , qmax )
628  !$omp barrier
629  !$omp master
630  !$acc parallel loop gang vector collapse(4) present(qtens_biharmonic,hvcoord%dp0,elem(:))
631  do ie = 1 , nelemd
632  do k = 1 , nlev ! Loop inversion (AAM)
633  do j = 1 , np
634  do i = 1 , np
635  tmp = -rhs_viss*dt*nu_q*hvcoord%dp0(k) / elem(ie)%spheremp(i,j)
636  do q = 1 , qsize
637  ! note: biharmonic_wk() output has mass matrix already applied. Un-apply since we apply again below:
638  qtens_biharmonic(i,j,k,q,ie) = tmp*qtens_biharmonic(i,j,k,q,ie)
639  enddo
640  enddo
641  enddo
642  enddo
643  enddo
644  !$omp end master
645  !$omp barrier
646  endif
647  endif ! compute biharmonic mixing term and qmin/qmax

651  ! 2D Advection step
652  !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
653  ! Compute velocity used to advance Qdp
654 
655  if ( limiter_option == 8 .and. nu_p > 0 .and. rhs_viss /= 0 ) then
656  do ie = nets , nete
657  data_pack(:,:,:,ie) = elem(ie)%derived%dpdiss_biharmonic
658  enddo
659  !$omp barrier
660  !$omp master
661  !$acc update device(data_pack) async(1)
662  !$acc parallel loop gang vector collapse(4) present(elem,data_pack) async(1)
663  do ie = 1 , nelemd
664  do k = 1 , nlev
665  do j = 1 , np
666  do i = 1 , np
667  elem(ie)%derived%dpdiss_biharmonic(i,j,k) = data_pack(i,j,k,ie)
668  enddo
669  enddo
670  enddo
671  enddo
672  !$acc wait(1)
673  !$omp end master
674  !$omp barrier
675  endif

678  !$omp master
679  if (limiter_option == 8) then
680  !$acc update device(derived_divdp)
681  endif
682  !$acc parallel loop gang vector collapse(4) present(elem(:),derived_divdp_proj,derived_vn0,dp_star,derived_divdp,grads_tracer,state_qdp) &
683  !$acc& private(dp,vstar)
684  do ie = 1 , nelemd
685  do k = 1 , nlev ! Loop index added (AAM)
686  do j = 1 , np
687  do i = 1 , np
688  ! derived variable divdp_proj() (DSS'd version of divdp) will only be correct on 2nd and 3rd stage
689  ! but that's ok because rhs_multiplier=0 on the first stage:
690  dp = elem(ie)%derived%dp(i,j,k) - rhs_multiplier * dt * derived_divdp_proj(i,j,k,ie)
691  vstar(1) = derived_vn0(i,j,1,k,ie) / dp
692  vstar(2) = derived_vn0(i,j,2,k,ie) / dp
693  if ( limiter_option == 8 ) then
694  ! UN-DSS'ed dp at timelevel n0+1:
695  dp_star(i,j,k,ie) = dp - dt * derived_divdp(i,j,k,ie)
696  if ( nu_p > 0 .and. rhs_viss /= 0 ) then
697  ! add contribution from UN-DSS'ed PS dissipation
698  dp_star(i,j,k,ie) = dp_star(i,j,k,ie) - rhs_viss * dt * nu_q * elem(ie)%derived%dpdiss_biharmonic(i,j,k) / elem(ie)%spheremp(i,j)
699  endif
700  endif
701  do q = 1 , qsize

703  grads_tracer(i,j,2,k,q,ie) = vstar(2) * state_qdp(i,j,k,q,n0_qdp,ie)
704  enddo
705  enddo
706  enddo
707  enddo
708  enddo

710  !$acc parallel loop gang vector collapse(5) present(qtens,state_qdp,qtens_biharmonic)
711  do ie = 1 , nelemd
712  ! advance Qdp
713  do q = 1 , qsize
714  do k = 1 , nlev ! dp_star used as temporary instead of divdp (AAM)
715  do j = 1 , np
716  do i = 1 , np
717  tmp = state_qdp(i,j,k,q,n0_qdp,ie) - dt * qtens(i,j,k,q,ie)
718  ! optionally add in hyperviscosity computed above:
719  if ( rhs_viss /= 0 ) tmp = tmp + qtens_biharmonic(i,j,k,q,ie)
720  qtens(i,j,k,q,ie) = tmp
721  enddo
722  enddo
723  enddo
724  enddo
725  enddo
726  if ( limiter_option == 8 ) then
727  call limiter_optim_iter_full( qtens , elem(:) , qmin , qmax , dp_star ) ! apply limiter to Q = Qtens / dp_star
728  endif

730  do ie = 1 , nelemd
731  ! apply mass matrix, overwrite np1 with solution:
732  ! dont do this earlier, since we allow np1_qdp == n0_qdp
733  ! and we dont want to overwrite n0_qdp until we are done using it
734  do k = 1 , nlev
735  do j = 1 , np
736  do i = 1 , np
737  tmp = elem(ie)%spheremp(i,j)
738  do q = 1 , qsize
739  state_qdp(i,j,k,q,np1_qdp,ie) = tmp * qtens(i,j,k,q,ie)
740  enddo
741  enddo
742  enddo
743  enddo
744  enddo
745  if ( limiter_option == 4 ) then
746  !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!1
747  ! sign-preserving limiter, applied after mass matrix
748  !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!1
749  call limiter2d_zero( state_qdp , 2 , np1_qdp )
750  endif
751  ! note: eta_dot_dpdn is actually dimension nlev+1, but nlev+1 data is
752  ! all zero so we only have to DSS 1:nlev
753  call t_startf('eus_PEU')
754  call edgevpack_openacc(edge_g , state_qdp , nlev*qsize , 0 , nlev*qsize , 1 , nelemd , 2 , np1_qdp )
755  !$omp end master
756  !$omp barrier
757 
758  call t_startf('eus_exch')
759  call bndry_exchangev( hybrid , edge_g )
760  call t_stopf('eus_exch')
761 
762  !$omp barrier
763  !$omp master
764  call edgevunpack_openacc( edge_g , state_qdp , nlev*qsize , 0 , nlev*qsize , 1 , nelemd , 2 , np1_qdp )
765  call t_stopf('eus_PEU')
766  !$acc parallel loop gang vector collapse(4) present(state_Qdp,elem(:))
767  do ie = 1 , nelemd
768  do k = 1 , nlev ! Potential loop inversion (AAM)
769  do j = 1 , np
770  do i = 1 , np
771  tmp = elem(ie)%rspheremp(i,j)
772  do q = 1 , qsize
773  state_qdp(i,j,k,q,np1_qdp,ie) = tmp * state_qdp(i,j,k,q,np1_qdp,ie)
774  enddo
775  enddo
776  enddo
777  enddo
778  enddo
779  !$omp end master
780  !$omp barrier
781  end subroutine euler_step

783  subroutine limiter2d_zero(Qdp,tdim,tl)
784  ! mass conserving zero limiter (2D only). to be called just before DSS
785  !
786  ! this routine is called inside a DSS loop, and so Q had already
787  ! been multiplied by the mass matrix. Thus dont include the mass
788  ! matrix when computing the mass = integral of Q over the element
789  !
790  ! ps is only used when advecting Q instead of Qdp
791  ! so ps should be at one timelevel behind Q
792  implicit none
793  integer , intent(in ) :: tdim
794  integer , intent(in ) :: tl
795  real (kind=real_kind), intent(inout) :: qdp(np,np,nlev,qsize,tdim,nelemd)
796  ! local
797  real (kind=real_kind) :: mass,mass_new
798  real (kind=real_kind) :: qtmp(np,np)
799  integer i,j,k,q,ie
800  !$acc parallel loop gang vector collapse(3) present(qdp) private(qtmp)
801  do ie = 1 , nelemd
802  do q = 1 , qsize
803  do k = nlev , 1 , -1
804  !$acc cache(qtmp)
805  qtmp = qdp(:,:,k,q,tl,ie)
806  mass = 0
807  do j = 1 , np
808  do i = 1 , np
809  mass = mass + qtmp(i,j)
810  enddo
811  enddo
812  ! negative mass. so reduce all postive values to zero
813  ! then increase negative values as much as possible
814  if ( mass < 0 ) qtmp = -qtmp
815  mass_new = 0
816  do j = 1 , np
817  do i = 1 , np
818  if ( qtmp(i,j) < 0 ) then
819  qtmp(i,j) = 0
820  else
821  mass_new = mass_new + qtmp(i,j)
822  endif
823  enddo
824  enddo
825  ! now scale the all positive values to restore mass
826  if ( mass_new > 0 ) qtmp = qtmp * abs(mass) / mass_new
827  if ( mass < 0 ) qtmp = -qtmp
828  qdp(:,:,k,q,tl,ie) = qtmp
829  enddo
830  enddo
831  enddo
832  end subroutine limiter2d_zero

834  subroutine limiter_optim_iter_full(ptens,elem,minp,maxp,dpmass)
835  use element_mod, only: element_t
836  use kinds , only : real_kind
837  use dimensions_mod, only : np, np, nlev
838  implicit none
839  !THIS IS A NEW VERSION OF LIM8, POTENTIALLY FASTER BECAUSE INCORPORATES KNOWLEDGE FROM
840  !PREVIOUS ITERATIONS
841  !The idea here is the following: We need to find a grid field which is closest
842  !to the initial field (in terms of weighted sum), but satisfies the min/max constraints.
843  !So, first we find values which do not satisfy constraints and bring these values
844  !to a closest constraint. This way we introduce some mass change (addmass),
845  !so, we redistribute addmass in the way that l2 error is smallest.
846  !This redistribution might violate constraints thus, we do a few iterations.
847  real (kind=real_kind), intent(inout) :: ptens(np*np,nlev,qsize,nelemd)
848  type(element_t) , intent(in ) :: elem (:)
849  real (kind=real_kind), intent(inout) :: minp( nlev,qsize,nelemd)
850  real (kind=real_kind), intent(inout) :: maxp( nlev,qsize,nelemd)
851  real (kind=real_kind), intent(in ) :: dpmass(np*np,nlev ,nelemd)
852  integer :: k1, k, i, j, iter, i1, i2, q, ie
853  real (kind=real_kind) :: addmass, weightssum, mass, mintmp, maxtmp, sumc
854  real (kind=real_kind) :: x(np*np),c(np*np)
855  integer :: maxiter = np*np-1
856  real (kind=real_kind) :: tol_limiter = 5e-14
857 
858  !$acc parallel loop gang vector collapse(3) present(ptens,elem(:),minp,maxp,dpmass) private(c,x,mintmp,maxtmp,addmass,weightssum,mass,sumc) vector_length(512)
859  do ie = 1 , nelemd
860  do q = 1 , qsize
861  do k = 1 , nlev
862  !$acc cache(c,x)
863  do k1 = 1 , np*np
864  i = modulo(k1-1,np)+1
865  j = (k1-1)/np+1
866  c(k1) = elem(ie)%spheremp(i,j) * dpmass(k1,k,ie)
867  x(k1) = ptens(k1,k,q,ie) / dpmass(k1,k,ie)
868  enddo
869  sumc = sum(c)
870  mintmp = minp(k,q,ie)
871  maxtmp = maxp(k,q,ie)
872  if (sumc <= 0 ) cycle ! this should never happen, but if it does, dont limit
873  mass=sum(c*x)
874  ! relax constraints to ensure limiter has a solution:
875  ! This is only needed if runnign with the SSP CFL>1 or
876  ! due to roundoff errors
877  if( mass < mintmp*sumc ) mintmp = mass / sumc
878  if( mass > maxtmp*sumc ) maxtmp = mass / sumc
879  do iter=1,maxiter
880  addmass=0.0d0
881  do k1=1,np*np
882  if((x(k1)>maxtmp)) then
883  addmass=addmass+(x(k1)-maxtmp)*c(k1)
884  x(k1)=maxtmp
885  endif
886  if((x(k1)<mintmp)) then
887  addmass=addmass-(mintmp-x(k1))*c(k1)
888  x(k1)=mintmp
889  endif
890  enddo !k1
891  if(abs(addmass)<=tol_limiter*abs(mass)) exit
892  weightssum=0.0d0
893  if(addmass>0)then
894  do k1=1,np*np
895  if(x(k1)<maxtmp)then
896  weightssum=weightssum+c(k1)
897  endif
898  enddo !k1
899  do k1=1,np*np
900  if(x(k1)<maxtmp)then
901  x(k1)=x(k1)+addmass/weightssum
902  endif
903  enddo
904  else
905  do k1=1,np*np
906  if(x(k1)>mintmp)then
907  weightssum=weightssum+c(k1)
908  endif
909  enddo
910  do k1=1,np*np
911  if(x(k1)>mintmp)then
912  x(k1)=x(k1)+addmass/weightssum
913  endif
914  enddo
915  endif
916  enddo!end of iteration
917  minp(k,q,ie) = mintmp
918  maxp(k,q,ie) = maxtmp
919  ptens(:,k,q,ie) = x * dpmass(:,k,ie)
920  enddo
921  enddo
922  enddo
923  end subroutine limiter_optim_iter_full

925  subroutine precompute_divdp( elem , hybrid , deriv , dt , nets , nete , n0_qdp )
926  use element_mod , only: element_t
927  use element_state , only: derived_vn0, derived_divdp, derived_divdp_proj
928  use hybrid_mod , only: hybrid_t
929  use derivative_mod , only: derivative_t
930  use edge_mod , only: edgevpack_nlyr, edgevunpack_nlyr
931  use bndry_mod , only: bndry_exchangev
932  use control_mod , only: limiter_option
933  use derivative_mod, only: divergence_sphere_openacc
934  use openacc_utils_mod , only: copy_ondev
935  use perf_mod , only: t_startf, t_stopf
936  implicit none
937  type(element_t) , intent(inout) :: elem(:)
938  type (hybrid_t) , intent(in ) :: hybrid
939  type (derivative_t) , intent(in ) :: deriv
940  real(kind=real_kind) , intent(in ) :: dt
941  integer , intent(in ) :: nets , nete , n0_qdp
942  integer :: ie , k
943  real(kind=real_kind), pointer, dimension(:,:,:) :: dssvar
944  call copy_qdp1_h2d(elem,n0_qdp,nets,nete)
945  !$omp barrier
946  !$omp master
947  !$acc update device(derived_vn0)
948  call divergence_sphere_openacc(derived_vn0,deriv,elem,derived_divdp,nlev,1,nelemd,1,1)
949  call copy_ondev(derived_divdp_proj,derived_divdp,product(shape(derived_divdp)))
950  !$acc update host(derived_divdp,derived_divdp_proj)
951  !$omp end master
952  !$omp barrier
953  call t_startf('derived PEU')
954  do ie = nets , nete
955  ! note: eta_dot_dpdn is actually dimension nlev+1, but nlev+1 data is
956  ! all zero so we only have to DSS 1:nlev
957  do k = 1 , nlev
958  elem(ie)%derived%eta_dot_dpdn(:,:,k) = elem(ie)%spheremp(:,:) * elem(ie)%derived%eta_dot_dpdn(:,:,k)

960  elem(ie)%derived%divdp_proj(:,:,k) = elem(ie)%spheremp(:,:) * elem(ie)%derived%divdp_proj(:,:,k)

963  call edgevpack_nlyr( edge_g , elem(ie)%desc, elem(ie)%derived%omega_p(:,:,1:nlev) , nlev , nlev , 3*nlev )
964  call edgevpack_nlyr( edge_g , elem(ie)%desc, elem(ie)%derived%divdp_proj(:,:,1:nlev) , nlev , nlev*2 , 3*nlev )
965  enddo
966 
967  call bndry_exchangev( hybrid , edge_g )
968 
969  do ie = nets , nete
970  call edgevunpack_nlyr( edge_g , elem(ie)%desc, elem(ie)%derived%eta_dot_dpdn(:,:,1:nlev) , nlev , 0 , 3*nlev )
971  call edgevunpack_nlyr( edge_g , elem(ie)%desc, elem(ie)%derived%omega_p(:,:,1:nlev) , nlev , nlev , 3*nlev )
972  call edgevunpack_nlyr( edge_g , elem(ie)%desc, elem(ie)%derived%divdp_proj(:,:,1:nlev) , nlev , nlev*2 , 3*nlev )
973  do k = 1 , nlev
974  elem(ie)%derived%eta_dot_dpdn(:,:,k) = elem(ie)%rspheremp(:,:) * elem(ie)%derived%eta_dot_dpdn(:,:,k)
975  elem(ie)%derived%omega_p(:,:,k) = elem(ie)%rspheremp(:,:) * elem(ie)%derived%omega_p(:,:,k)
976  elem(ie)%derived%divdp_proj(:,:,k) = elem(ie)%rspheremp(:,:) * elem(ie)%derived%divdp_proj(:,:,k)
977  enddo
978  enddo
979  call t_stopf('derived PEU')
980  end subroutine precompute_divdp
