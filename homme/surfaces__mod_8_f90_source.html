827  function find_intersect( x1in,x2in,y1in,y2in ) result(sect)
828  implicit none
829  type (cartesian2d_t) :: x1in,x2in,y1in,y2in,sect,x,y,b
830  type (cartesian2d_t) :: x1,x2,y1,y2
831 ! integer :: nvert
832  real(kind=real_kind) :: s1,s2,deta
833 
834 ! x1 + (x2-x1)*s1 = y1 + (y2-y1)*s2
835 ! b = y1-x1
836 ! x=x2-x1
837 ! y=y2-y1
838 ! x s1 - y s2 = b
839 ! x(1) s1 - y(1) s2 = b(1)
840 ! x(2) s1 - y(2) s2 = b(2)
841 !
842 ! x(1) -y(1) s1 = b(1) A s = b
843 ! x(2) -y(2) s2 = b(2)
844 !
845 ! A2= -y(2) y(1)
846 ! -x(2) x(1) s = A2 * b /detA
847 
848  ! convert to gnomonic
849  x1%x = tan(x1in%x)
850  x2%x = tan(x2in%x)
851  y1%x = tan(y1in%x)
852  y2%x = tan(y2in%x)
853  x1%y = tan(x1in%y)
854  x2%y = tan(x2in%y)
855  y1%y = tan(y1in%y)
856  y2%y = tan(y2in%y)
857 
858  x%x=x2%x-x1%x
859  x%y=x2%y-x1%y
860  y%x=y2%x-y1%x
861  y%y=y2%y-y1%y
862  b%x=y1%x-x1%x
863  b%y=y1%y-x1%y
864 
865  deta = -x%x*y%y + x%y*y%x
866 
867  s1 = (-y%y*b%x + y%x*b%y )/deta
868  s2 = (-x%y*b%x + x%x*b%y )/deta
869 
870  sect%x = x1%x + (x2%x-x1%x)*s1
871  sect%y = x1%y + (x2%y-x1%y)*s1
872 
873  sect%x = (sect%x + y1%x + (y2%x-y1%x)*s2)/2
874  sect%y = (sect%y + y1%y + (y2%y-y1%y)*s2)/2
875 
876  if (s1<0 .or. s1>1) then
877  print *,'failed: intersection: ',s1,s2
878  stop
879  endif
880 
881  ! convert back to equal angle:
882  sect%x = atan(sect%x)
883  sect%y = atan(sect%y)
884  end function find_intersect

890  subroutine pentagon_iteration(sq1,sq2,pent,asq1,asq2,apent,faceno,anorm)
891 ! sq2
892 ! 4 3
893 ! 1 2
894 !
895 ! sq1 4 3
896 ! 2 1 5 pent
897 ! 3 4 1 2
898 !
899 !
900 ! d/dt sq1(1) = (area(sq1)-asq1) * [ com(sq1)-sq1(1) ]
901 ! +(area(sq2)-asq2) * [ com(sq2)-sq1(1) ]
902 ! +(area(pent)-apent) * [ com(pent)-sq1(1) ]
903 !
904 !
905 !
906  type (cartesian2d_t) :: sq1(4),sq2(4),pent(5)
907  real (kind=real_kind) :: asq1,asq2,apent,anorm
908  integer :: faceno
909

911  real (kind=real_kind) :: isq1,isq2,ipent,dt,diff1,diff2,diffp,err
912  real (kind=real_kind) :: tol_pentagon_iteration=1e-10
913  type (cartesian2d_t) :: sq1com,sq2com,pentcom,ds1,ds2
914  integer :: i,iter,iter_max
915 
916  ! compute center of mass:
917  sq1com%x = sum(sq1(:)%x)/4
918  sq1com%y = sum(sq1(:)%y)/4
919  sq2com%x = sum(sq2(:)%x)/4
920  sq2com%y = sum(sq2(:)%y)/4
921  pentcom%x = sum(pent(:)%x)/5
922  pentcom%y = sum(pent(:)%y)/5
923 
924  do i=1,4
925  sq1_3d(i)=cubedsphere2cart(sq1(i),faceno )
926  sq2_3d(i)=cubedsphere2cart(sq2(i),faceno )
927  pent_3d(i)=cubedsphere2cart(pent(i),faceno )
928  enddo
929  pent_3d(5)=cubedsphere2cart(pent(5),faceno )
930 
931  dt=.5
932  iter_max = 10000
933  do iter=1,iter_max
934  isq1 = surfarea(sq1_3d,4)
935  isq2 = surfarea(sq2_3d,4)
936  ipent = surfarea(pent_3d,5)
937 
938 ! d/dt sq1(1) = (area(sq1)-asq1) * [ com(sq1)-sq1(1) ]
939 ! +(area(sq2)-asq2) * [ com(sq2)-sq1(1) ]
940 ! +(area(pent)-apent) * [ com(pent)-sq1(1) ]
941 !
942  diff1 = (isq1-asq1)/anorm
943  diff2 = (isq2-asq2)/anorm
944  diffp = (ipent-apent)/anorm
945 
946  err = abs(diff1)+abs(diff2)+abs(diffp)
947  if (err< tol_pentagon_iteration) exit
948  !write(*,'(i5,3e18.5)') iter,diff1,diff2,diffp
949  !if (mod(iter,1000).eq.0) write(*,'(i5,3e18.5)') iter,err
950

952  ds1%y = diff1* ( sq1com%y - sq1(1)%y )

954  ds1%y = ds1%y + diffp* ( pentcom%y - sq1(1)%y )

957  ds2%y = diff2* ( sq2com%y - sq2(1)%y )

959  ds2%y = ds2%y + diffp* ( pentcom%y - sq2(1)%y )

962  sq1(1)%y = sq1(1)%y + dt*ds1%y
963  sq2(1)%x = sq2(1)%x + dt*ds2%x
964  sq2(1)%y = sq2(1)%y + dt*ds2%y

966  pent(5)=sq1(1)
967  sq1_3d(1)=cubedsphere2cart(sq1(1),faceno )
968  sq2_3d(1)=cubedsphere2cart(sq2(1),faceno )
969  pent_3d(4)=sq2_3d(1)
970  pent_3d(5)=sq1_3d(1)
