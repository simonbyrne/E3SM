81 // What follows provides utilities to parameterize the parallel machine (CPU/KNL
82 // cores within a rank, GPU attached to a rank) optimally. The parameterization
83 // is a nontrivial function of available resources, number of parallel
84 // iterations to be performed, and kernel-specific preferences regarding team

153 // Return a TeamPolicy using defaults that, so far, have been good for all use
154 // cases. Use of this function means you don't have to use
155 // DefaultThreadsDistribution.
156 template <typename ExecSpace, typename... Tags>
157 Kokkos::TeamPolicy<ExecSpace, Tags...>
158 get_default_team_policy(const int num_parallel_iterations,
159  const ThreadPreferences tp = ThreadPreferences()) {
160  const auto threads_vectors =
161  DefaultThreadsDistribution<ExecSpace>::team_num_threads_vectors(
162  num_parallel_iterations, tp);
163  auto policy = Kokkos::TeamPolicy<ExecSpace, Tags...>(num_parallel_iterations,
164  threads_vectors.first,
165  threads_vectors.second);
166  policy.set_chunk_size(1);
167  return policy;
168 }

314  template<typename LoopBdyType, class Lambda, typename ValueType>
315  KOKKOS_FORCEINLINE_FUNCTION
316  static void parallel_reduce (
317  const Kokkos::TeamPolicy<ExeSpace>::member_type& team,
318  const LoopBdyType& loop_boundaries,
319  const Lambda& lambda, ValueType& result)
320  {
321 #ifdef HOMMEXX_BFB_TESTING
322  // We want to get C++ on GPU to match F90 on CPU. Thus, need to
323  // serialize parallel reductions.
324 
325  // All threads init result.
326  // NOTE: we *need* an automatic temporary, since we do not know
327  // where 'result' comes from. If result itself is an automatic
328  // variable, using it would be fine (only one vector lane would
329  // actually have a nonzero value after the single). But if
330  // result is taken from a view, then all vector lanes would
331  // see the updated value before the vector_reduce call,
332  // which will cause the final answer to be multiplied by the
333  // size of the warp.
334  auto local_tmp = Kokkos::reduction_identity<ValueType>::sum();
335  // One thread sums.
336  Kokkos::single(Kokkos::PerThread(team), [&] () {
337  for (auto i = loop_boundaries.start; i < loop_boundaries.end; ++i)
338  lambda(i, local_tmp);
339  });
340  // Broadcast result to all threads by doing sum of one thread's
341  // non-0 value and the rest of the 0s.
342  Kokkos::Impl::CudaTeamMember::vector_reduce(
343  Kokkos::Sum<ValueType>(local_tmp));
344  result = local_tmp;
345 #else
346  Kokkos::parallel_reduce(loop_boundaries, lambda, result);
347 #endif
348  }

370  template<class Lambda>
371  static KOKKOS_FORCEINLINE_FUNCTION
372  void parallel_scan (
373  const typename Kokkos::TeamPolicy<ExeSpace>::member_type& team,
374  const int num_iters,
375  const Lambda& lambda)
376  {
377 #ifdef HOMMEXX_BFB_TESTING
378  // We want to get C++ on GPU to match F90 on CPU. Thus, need to
379  // serialize parallel scans.
380 
381  // Detect the value type
382  using value_type =
383  typename Kokkos::Impl::FunctorAnalysis
384  < Kokkos::Impl::FunctorPatternInterface::SCAN
385  , void
386  , Lambda >::value_type ;
387 
388  // All threads init result.
389  value_type accumulator = Kokkos::reduction_identity<value_type>::sum();
390  // Only one thread does the work, i.e., only one sweeps (so last arg to lambda is true)
391  Kokkos::single(Kokkos::PerThread(team), [&] () {
392  for (int i = 0; i < num_iters; ++i) {
393  lambda(i, accumulator, true);
394  }
395  });
396 #else
397  Kokkos::parallel_scan(Kokkos::ThreadVectorRange(team,num_iters), lambda);
398 #endif
399  }
