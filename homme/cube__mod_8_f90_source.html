188  ! Initialize cube-sphere metric terms:
189  ! equal angular elements (atomic)
190  ! initialize:
191  ! metdet, rmetdet (analytic) = detD, 1/detD
192  ! met (analytic) D^t D (symmetric)
193  ! metdet (analytic) = detD
194  ! metinv (analytic) Dinv Dinv^t (symmetic)
195  ! D (from subroutine vmap)
196  ! Dinv (computed directly from D)
197  !
198  ! ucontra = Dinv * u = metinv * ucov
199  ! ucov = D^t * u = met * ucontra
200  !
201  ! we also compute DE = D*E, where
202  ! E = eigenvectors of metinv as a basis metinv = E LAMBDA E^t
203  !
204  ! ueig = E^t ucov = E^t D^t u = (DE)^t u
205  !
206  !
207  ! so if we want to tweak the mapping by a factor alpha (so he weights add up to 4pi, for example)
208  ! we take:
209  ! NEW OLD
210  ! D = sqrt(alpha) D and then rederive all quantities.
211  ! detD = alpha detD
212  !
213  ! where alpha = 4pi/SEMarea, SEMarea = global sum elem(ie)%mv(i,j)*elem(ie)%metdet(i,j)

242  ! ==============================================
243  ! Initialize differential mapping operator
244  ! to and from vector fields on the sphere to

246  ! i.e. dM/dx^i in Sadourney (1972) and it's

251  max_normdinv = 0.0d0
252  min_svd = 1d99

254  do i=1,np

256  x2=gll_points(j)
257  call dmap(elem%D(i,j,:,:),x1,x2,elem%corners3D,cubed_sphere_map,elem%cartp,elem%facenum)

260  ! Numerical metric tensor based on analytic D: met = D^T times D
261  ! (D maps between sphere and reference element)
262  elem%met(i,j,1,1) = elem%D(i,j,1,1)*elem%D(i,j,1,1) + &
263  elem%D(i,j,2,1)*elem%D(i,j,2,1)
264  elem%met(i,j,1,2) = elem%D(i,j,1,1)*elem%D(i,j,1,2) + &
265  elem%D(i,j,2,1)*elem%D(i,j,2,2)
266  elem%met(i,j,2,1) = elem%D(i,j,1,1)*elem%D(i,j,1,2) + &
267  elem%D(i,j,2,1)*elem%D(i,j,2,2)
268  elem%met(i,j,2,2) = elem%D(i,j,1,2)*elem%D(i,j,1,2) + &
269  elem%D(i,j,2,2)*elem%D(i,j,2,2)
270 
271  ! compute D^-1...
272  ! compute determinant of D mapping matrix... if not zero compute inverse
273 
274  detd = elem%D(i,j,1,1)*elem%D(i,j,2,2) - elem%D(i,j,1,2)*elem%D(i,j,2,1)
275 
276  elem%Dinv(i,j,1,1) = elem%D(i,j,2,2)/detd
277  elem%Dinv(i,j,1,2) = -elem%D(i,j,1,2)/detd
278  elem%Dinv(i,j,2,1) = -elem%D(i,j,2,1)/detd
279  elem%Dinv(i,j,2,2) = elem%D(i,j,1,1)/detd
280 
281  ! L2 norm = sqrt max eigenvalue of metinv
282  ! = 1/sqrt(min eigenvalue of met)
283  ! l1 and l2 are eigenvalues of met
284  ! (should both be positive, l1 > l2)
285  l1 = (elem%met(i,j,1,1) + elem%met(i,j,2,2) + sqrt(4.0d0*elem%met(i,j,1,2)*elem%met(i,j,2,1) + &
286  (elem%met(i,j,1,1) - elem%met(i,j,2,2))**2))/2.0d0
287  l2 = (elem%met(i,j,1,1) + elem%met(i,j,2,2) - sqrt(4.0d0*elem%met(i,j,1,2)*elem%met(i,j,2,1) + &
288  (elem%met(i,j,1,1) - elem%met(i,j,2,2))**2))/2.0d0
289  ! Max L2 norm of Dinv is sqrt of max eigenvalue of metinv
290  ! max eigenvalue of metinv is 1/min eigenvalue of met
291  norm = 1.0d0/sqrt(min(abs(l1),abs(l2)))
292  max_svd = max(norm, max_svd)
293  ! Min L2 norm of Dinv is sqrt of min eigenvalue of metinv
294  ! min eigenvalue of metinv is 1/max eigenvalue of met
295  norm = 1.0d0/sqrt(max(abs(l1),abs(l2)))
296  min_svd = min(norm, min_svd)
297 
298  ! some kind of pseudo-norm of Dinv
299  ! C = 1/sqrt(2) sqrt( |g^x|^2 + |g^y|^2 + 2*|g^x dot g^y|)
300  ! = 1/sqrt(2) sqrt( |g_x|^2 + |g_y|^2 + 2*|g_x dot g_y|) / J
301  ! g^x = Dinv(:,1) g_x = D(1,:)
302  ! g^y = Dinv(:,2) g_y = D(2,:)
303  norm = (2*abs(sum(elem%Dinv(i,j,:,1)*elem%Dinv(i,j,:,2))) + sum(elem%Dinv(i,j,:,1)**2) + sum(elem%Dinv(i,j,:,2)**2))
304  norm = sqrt(norm)
305 ! norm = (2*abs(sum(elem%D(1,:,i,j)*elem%D(2,:,i,j))) + sum(elem%D(1,:,i,j)**2) + sum(elem%D(2,:,i,j)**2))
306 ! norm = sqrt(norm)/detD
307  max_normdinv = max(norm,max_normdinv)
308 
309 
310  ! Need inverse of met if not calculated analytically
311  elem%metdet(i,j) = abs(detd)
312  elem%rmetdet(i,j) = 1.0d0/abs(detd)
313 
314  elem%metinv(i,j,1,1) = elem%met(i,j,2,2)/(detd*detd)
315  elem%metinv(i,j,1,2) = -elem%met(i,j,1,2)/(detd*detd)
316  elem%metinv(i,j,2,1) = -elem%met(i,j,2,1)/(detd*detd)
317  elem%metinv(i,j,2,2) = elem%met(i,j,1,1)/(detd*detd)
318 
319  ! matricies for tensor hyper-viscosity
320  ! compute eigenvectors of metinv (probably same as computed above)
321  m = elem%metinv(i,j,:,:)
322 
323  eig(1) = (m(1,1) + m(2,2) + sqrt(4.0d0*m(1,2)*m(2,1) + &
324  (m(1,1) - m(2,2))**2))/2.0d0
325  eig(2) = (m(1,1) + m(2,2) - sqrt(4.0d0*m(1,2)*m(2,1) + &
326  (m(1,1) - m(2,2))**2))/2.0d0
327 
328  ! use DE to store M - Lambda, to compute eigenvectors
329  de=m
330  de(1,1)=de(1,1)-eig(1)
331  de(2,2)=de(2,2)-eig(1)
332 
333  imaxm = maxloc(abs(de))
334  if (maxval(abs(de))==0) then
335  e(1,1)=1; e(2,1)=0;
336  elseif ( imaxm(1)==1 .and. imaxm(2)==1 ) then
337  e(2,1)=1; e(1,1) = -de(2,1)/de(1,1)
338  else if ( imaxm(1)==1 .and. imaxm(2)==2 ) then
339  e(2,1)=1; e(1,1) = -de(2,2)/de(1,2)
340  else if ( imaxm(1)==2 .and. imaxm(2)==1 ) then
341  e(1,1)=1; e(2,1) = -de(1,1)/de(2,1)
342  else if ( imaxm(1)==2 .and. imaxm(2)==2 ) then
343  e(1,1)=1; e(2,1) = -de(1,2)/de(2,2)
344  else
345  call abortmp('Impossible error in cube_mod.F90::metric_atomic()')
346  endif
347 
348  ! the other eigenvector is orthgonal:
349  e(1,2)=-e(2,1)
350  e(2,2)= e(1,1)
351 
352 !normalize columns
353  e(:,1)=e(:,1)/sqrt(sum(e(:,1)*e(:,1)));
354  e(:,2)=e(:,2)/sqrt(sum(e(:,2)*e(:,2)));

357 ! OBTAINING TENSOR FOR HV:
358 
359 ! Instead of the traditional scalar Laplace operator \grad \cdot \grad
360 ! we introduce \grad \cdot V \grad
361 ! where V = D E LAM LAM^* E^T D^T.
362 ! Recall (metric_tensor)^{-1}=(D^T D)^{-1} = E LAM E^T.
363 ! Here, LAM = diag( 4/((np-1)dx)^2 , 4/((np-1)dy)^2 ) = diag( 4/(dx_elem)^2, 4/(dy_elem)^2 )
364 ! Note that metric tensors and LAM correspondingly are quantities on a unit sphere.
365 
366 ! This motivates us to use V = D E LAM LAM^* E^T D^T
367 ! where LAM^* = diag( nu1, nu2 ) where nu1, nu2 are HV coefficients scaled like (dx)^{hv_scaling/2}, (dy)^{hv_scaling/2}.
368 ! (Halves in powers come from the fact that HV consists of two Laplace iterations.)
369 
370 ! Originally, we took LAM^* = diag(
371 ! 1/(eig(1)**(hypervis_scaling/4.0d0))*(rearth**(hypervis_scaling/2.0d0))
372 ! 1/(eig(2)**(hypervis_scaling/4.0d0))*(rearth**(hypervis_scaling/2.0d0)) ) =
373 ! = diag( lamStar1, lamStar2)
374 ! \simeq ((np-1)*dx_sphere / 2 )^hv_scaling/2 = SQRT(OPERATOR_HV)
375 ! because 1/eig(...) \simeq (dx_on_unit_sphere)^2 .
376 ! Introducing the notation OPERATOR = lamStar^2 is useful for conversion formulas.
377 
378 ! This leads to the following conversion formula: nu_const is nu used for traditional HV on uniform grids
379 ! nu_tensor = nu_const * OPERATOR_HV^{-1}, so
380 ! nu_tensor = nu_const *((np-1)*dx_sphere / 2 )^{ - hv_scaling} or
381 ! nu_tensor = nu_const *(2/( (np-1) * dx_sphere) )^{hv_scaling} .
382 ! dx_sphere = 2\pi *rearth/(np-1)/4/NE
383 ! [nu_tensor] = [meter]^{4-hp_scaling}/[sec]
384 
385 ! (1) Later developments:
386 ! Apply tensor V only at the second Laplace iteration. Thus, LAM^* should be scaled as (dx)^{hv_scaling}, (dy)^{hv_scaling},
387 ! see this code below:
388 ! DEL(1:2,1) = (lamStar1**2) *eig(1)*DE(1:2,1)
389 ! DEL(1:2,2) = (lamStar2**2) *eig(2)*DE(1:2,2)
390 
391 ! (2) Later developments:
392 ! Bringing [nu_tensor] to 1/[sec]:
393 ! lamStar1=1/(eig(1)**(hypervis_scaling/4.0d0)) *(rearth**2.0d0)
394 ! lamStar2=1/(eig(2)**(hypervis_scaling/4.0d0)) *(rearth**2.0d0)
395 ! OPERATOR_HV = ( (np-1)*dx_unif_sphere / 2 )^{hv_scaling} * rearth^4
396 ! Conversion formula:
397 ! nu_tensor = nu_const * OPERATOR_HV^{-1}, so
398 ! nu_tensor = nu_const *( 2*rearth /((np-1)*dx))^{hv_scaling} * rearth^{-4.0}.
399 
400 ! For the baseline coefficient nu=1e15 for NE30,
401 ! nu_tensor=7e-8 (BUT RUN TWICE AS SMALL VALUE FOR NOW) for hv_scaling=3.2
402 ! and
403 ! nu_tensor=1.3e-6 for hv_scaling=4.0.
404

407  de(1,1)=sum(elem%D(i,j,1,:)*e(:,1))
408  de(1,2)=sum(elem%D(i,j,1,:)*e(:,2))

413  lamstar2=1/(eig(2)**(hypervis_scaling/4.0d0)) *(rearth**2.0d0)

417 ! DEL(1:2,2) = lamStar2*eig(2)*DE(1:2,2)

422  del(1:2,2) = (lamstar2**2) *eig(2)*de(1:2,2)
423

426  v(1,2)=sum(del(1,:)*de(2,:))
427  v(2,1)=sum(del(2,:)*de(1,:))
428  v(2,2)=sum(del(2,:)*de(2,:))
429 
430  elem%tensorVisc(i,j,:,:)=v(:,:)

432  end do
433  end do

449  ! agreement with geometric area.
450  ! Three different cases:

452  ! called with alpha parameter there will be no correction,

478  elem%metinv(i,j,:,:) = elem%metinv(i,j,:,:) / alpha
479  enddo
480  enddo
481  endif ! end of alpha/eps. bubble correction

596  subroutine vmap(D, x1, x2, face_no)
597  real (kind=real_kind), intent(inout) :: d(2,2)
598  real (kind=real_kind), intent(in) :: x1
599  real (kind=real_kind), intent(in) :: x2
600  integer , intent(in) :: face_no
601 
602  ! Local variables
603 
604  real (kind=real_kind) :: poledist ! SQRT(TAN(x1)**2 +TAN(x2)**2)
605  real (kind=real_kind) :: r ! distance from cube point to center of sphere
606

608  real (kind=real_kind) :: d12
609  real (kind=real_kind) :: d21
610  real (kind=real_kind) :: d22

613 
614  if (face_no >= 1 .and. face_no <= 4) then
615 
616  d11 = 1.0d0/(r*cos(x1))
617  d12 = 0.0d0
618  d21 = -tan(x1)*tan(x2)/(cos(x1)*r*r)
619  d22 = 1.0d0/(r*r*cos(x1)*cos(x2)*cos(x2))
620 
621  d(1,1) = d11
622  d(1,2) = d12
623  d(2,1) = d21
624  d(2,2) = d22
625 
626 
627  else if (face_no==6) then
628  poledist=sqrt( tan(x1)**2 + tan(x2)**2)
629  if ( poledist <= dist_threshold ) then
630 
631  ! we set the D transform to the identity matrix
632  ! which works ONLY for swtc1, phi starting at
633  ! 3*PI/2... assumes lon at pole == 0
634 
635  d(1,1) = 1.0d0
636  d(1,2) = 0.0d0
637  d(2,1) = 0.0d0
638  d(2,2) = 1.0d0
639 
640  else
641 
642  d11 = -tan(x2)/(poledist*cos(x1)*cos(x1)*r)
643  d12 = tan(x1)/(poledist*cos(x2)*cos(x2)*r)
644  d21 = -tan(x1)/(poledist*cos(x1)*cos(x1)*r*r)
645  d22 = -tan(x2)/(poledist*cos(x2)*cos(x2)*r*r)
646 
647  d(1,1) = d11
648  d(1,2) = d12
649  d(2,1) = d21
650  d(2,2) = d22
651 
652  end if
653  else if (face_no==5) then
654  poledist=sqrt( tan(x1)**2 + tan(x2)**2)
655  if ( poledist <= dist_threshold ) then
656 
657  ! we set the D transform to the identity matrix
658  ! which works ONLY for swtc1, phi starting at
659  ! 3*PI/2... assumes lon at pole == 0, i.e. very specific
660 
661  d(1,1) = 1.0d0
662  d(1,2) = 0.0d0
663  d(2,1) = 0.0d0
664  d(2,2) = 1.0d0
665 
666  else
667 
668  d11 = tan(x2)/(poledist*cos(x1)*cos(x1)*r)
669  d12 = -tan(x1)/(poledist*cos(x2)*cos(x2)*r)
670  d21 = tan(x1)/(poledist*cos(x1)*cos(x1)*r*r)
671  d22 = tan(x2)/(poledist*cos(x2)*cos(x2)*r*r)
672 
673  d(1,1) = d11
674  d(1,2) = d12
675  d(2,1) = d21
676  d(2,2) = d22
677 
678  end if
679  end if

696  subroutine dmap_elementlocal(D, a,b, corners3D)
697  real (kind=real_kind), intent(out) :: d(2,2)
698  real (kind=real_kind), intent(in) :: a,b
699  type (cartesian3d_t) :: corners3d(4)
700 
701  type (spherical_polar_t) :: sphere
702 
703  real(kind=real_kind) :: c(3,4), q(4), xx(3), r, lam, th, dd(4,2)
704  real(kind=real_kind) :: sinlam, sinth, coslam, costh
705  real(kind=real_kind) :: d1(2,3), d2(3,3), d3(3,2), d4(3,2)
706  integer :: i,j
707 
708  sphere = ref2sphere(a,b,corners3d,2) ! use element local map, ref_map=2
709 
710  c(1,1)=corners3d(1)%x; c(2,1)=corners3d(1)%y; c(3,1)=corners3d(1)%z;
711  c(1,2)=corners3d(2)%x; c(2,2)=corners3d(2)%y; c(3,2)=corners3d(2)%z;
712  c(1,3)=corners3d(3)%x; c(2,3)=corners3d(3)%y; c(3,3)=corners3d(3)%z;
713  c(1,4)=corners3d(4)%x; c(2,4)=corners3d(4)%y; c(3,4)=corners3d(4)%z;
714 
715  q(1)=(1-a)*(1-b); q(2)=(1+a)*(1-b); q(3)=(1+a)*(1+b); q(4)=(1-a)*(1+b);
716  q=q/4.0d0;
717 
718  do i=1,3
719  xx(i)=sum(c(i,:)*q(:))
720  enddo
721 
722  r=sqrt(xx(1)**2+xx(2)**2+xx(3)**2)
723 
724  lam=sphere%lon; th=sphere%lat;
725  sinlam=sin(lam); sinth=sin(th);
726  coslam=cos(lam); costh=cos(th);
727 
728  d1(1,1)=-sinlam; d1(1,2)=coslam; d1(1,3)=0.0d0;
729  d1(2,1)=0.0d0; d1(2,2)=0.0d0; d1(2,3)=1.0d0;
730 
731  d2(1,1)=(sinlam**2)*(costh**2)+sinth**2; d2(1,2)=-sinlam*coslam*(costh**2); d2(1,3)=-coslam*sinth*costh;
732  d2(2,1)=-sinlam*coslam*(costh**2); d2(2,2)=(coslam**2)*(costh**2)+sinth**2; d2(2,3)=-sinlam*sinth*costh;
733  d2(3,1)=-coslam*sinth; d2(3,2)=-sinlam*sinth; d2(3,3)=costh;
734 
735  dd(1,1)=-1+b; dd(1,2)=-1+a;
736  dd(2,1)=1-b; dd(2,2)=-1-a;
737  dd(3,1)=1+b; dd(3,2)=1+a;
738  dd(4,1)=-1-b; dd(4,2)=1-a;
739 
740  dd=dd/4.0d0
741 
742  do i=1,3
743  do j=1,2
744  d3(i,j)=sum(c(i,:)*dd(:,j))
745  enddo
746  enddo
747 
748  do i=1,3
749  do j=1,2
750  d4(i,j)=sum(d2(i,:)*d3(:,j))
751  enddo
752  enddo
753 
754  do i=1,2
755  do j=1,2
756  d(i,j)=sum(d1(i,:)*d4(:,j))
757  enddo
758  enddo
759 
760  d=d/r
761  end subroutine dmap_elementlocal
