178 void ElementsGeometry::randomize(const int seed) {
179  // Check geometry was inited
180  assert (m_num_elems>0);
181 
182  // Arbitrary minimum value to generate and minimum determinant allowed
183  constexpr const Real min_value = 0.015625;
184  std::mt19937_64 engine(seed);
185  std::uniform_real_distribution<Real> random_dist(min_value, 1.0 / min_value);
186 
187  genRandArray(m_fcor, engine, random_dist);
188 
189  genRandArray(m_spheremp, engine, random_dist);
190  genRandArray(m_tensorvisc, engine, random_dist);
191  genRandArray(m_vec_sph2cart, engine, random_dist);
192 
193  genRandArray(m_phis, engine, random_dist);
194  if (m_gradphis.size()!=0) {
195  genRandArray(m_gradphis, engine, random_dist);
196  }
197 
198  // Lambdas used to constrain the metric tensor and its inverse
199  const auto compute_det = [](HostViewUnmanaged<Real[2][2]> mtx) {
200  return mtx(0, 0) * mtx(1, 1) - mtx(0, 1) * mtx(1, 0);
201  };
202 
203  // 2d tensors
204  // Generating lots of matrices with reasonable determinants can be difficult
205  // So instead of generating them all at once and verifying they're correct,
206  // generate them one at a time, verifying them individually
207  HostViewManaged<Real[2][2]> h_matrix("single host metric matrix");
208 
209  auto h_d = Kokkos::create_mirror_view(m_d);
210  auto h_dinv = Kokkos::create_mirror_view(m_dinv);
211  auto h_metinv = Kokkos::create_mirror_view(m_metinv);
212  auto h_metdet = Kokkos::create_mirror_view(m_metdet);
213  auto h_rspheremp = Kokkos::create_mirror_view(m_rspheremp);
214  auto h_spheremp = Kokkos::create_mirror_view(m_spheremp);
215  Kokkos::deep_copy(h_spheremp, m_spheremp);
216 
217  for (int ie = 0; ie < m_num_elems; ++ie) {
218  // Because this constraint is difficult to satisfy for all of the tensors,
219  // incrementally generate the view
220  for (int igp = 0; igp < NP; ++igp) {
221  for (int jgp = 0; jgp < NP; ++jgp) {
222 
223  // To avoid physical inconsistencies (which may trigger errors in the EOS):
224  // 1) rspheremp = 1/spheremp
225  // 2.1) det(d)>0
226  // 2.1) dinv = (d)^-1
227  // 2.1) det(metinv)>0
228  // 2.1) metdet = 1/det(metinv)
229  h_rspheremp(ie,igp,jgp) = 1./h_spheremp(ie,igp,jgp);
230 
231  do {
232  genRandArray(h_matrix, engine, random_dist);
233  } while (compute_det(h_matrix)<=0.0);
234 
235  for (int i = 0; i < 2; ++i) {
236  for (int j = 0; j < 2; ++j) {
237  h_d(ie, i, j, igp, jgp) = h_matrix(i, j);
238  }
239  }
240  const Real determinant = compute_det(h_matrix);
241  h_dinv(ie, 0, 0, igp, jgp) = h_matrix(1, 1) / determinant;
242  h_dinv(ie, 1, 0, igp, jgp) = -h_matrix(1, 0) / determinant;
243  h_dinv(ie, 0, 1, igp, jgp) = -h_matrix(0, 1) / determinant;
244  h_dinv(ie, 1, 1, igp, jgp) = h_matrix(0, 0) / determinant;
245 
246  do {
247  genRandArray(h_matrix, engine, random_dist);
248  } while (compute_det(h_matrix)<=0.0);
249  h_metdet(ie,igp,jgp) = compute_det(h_matrix);
250  h_metinv(ie, 0, 0, igp, jgp) = h_matrix(1, 1);
251  h_metinv(ie, 1, 0, igp, jgp) = h_matrix(1, 0);
252  h_metinv(ie, 0, 1, igp, jgp) = h_matrix(0, 1);
253  h_metinv(ie, 1, 1, igp, jgp) = h_matrix(0, 0);
254  }
255  }
256  }
257 
258  Kokkos::deep_copy(m_d, h_d);
259  Kokkos::deep_copy(m_dinv, h_dinv);
260  Kokkos::deep_copy(m_metinv, h_metinv);
261  Kokkos::deep_copy(m_metdet, h_metdet);
262  Kokkos::deep_copy(m_rspheremp, h_rspheremp);
263 }
