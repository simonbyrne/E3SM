466  function interpolate_2d(cart, f, interp, npts, fillvalue) result(fxy)
467  integer, intent(in) :: npts
468  type (cartesian2d_t), intent(in) :: cart
469  real (kind=real_kind), intent(in) :: f(npts,npts)
470  type (interpolate_t) :: interp
471  real (kind=real_kind) :: fxy ! value of f interpolated to (x,y)
472  real (kind=real_kind), intent(in), optional :: fillvalue
473  ! local variables
474 
475  real (kind=real_kind) :: tmp_1,tmp_2
476  real (kind=real_kind) :: fk0,fk1
477  real (kind=real_kind) :: pk
478 
479  integer :: l,j,k
480 
481  if(present(fillvalue)) then
482  if (any(f==fillvalue)) then
483  fxy = fillvalue
484  return
485  endif
486  endif
487 
488 
489  do l=1,npts,2
490 
491  ! Compute Pk(cart%x) for Legendre order 0
492 
493  pk = 1.0d0
494 
495  fk0=0.0d0
496  fk1=0.0d0
497  do j=1,npts
498  fk0 = fk0 + interp%Imat(j,1)*f(j,l )
499  fk1 = fk1 + interp%Imat(j,1)*f(j,l+1)
500  end do
501  interp%vtemp(l ) = pk*fk0
502  interp%vtemp(l+1) = pk*fk1
503 
504  ! Compute Pk(cart%x) for Legendre order 1
505 
506  tmp_2 = pk
507  pk = cart%x
508 
509  fk0=0.0d0
510  fk1=0.0d0
511  do j=1,npts
512  fk0 = fk0 + interp%Imat(j,2)*f(j,l )
513  fk1 = fk1 + interp%Imat(j,2)*f(j,l+1)
514  end do
515  interp%vtemp(l ) = interp%vtemp(l ) + pk*fk0
516  interp%vtemp(l+1) = interp%vtemp(l+1) + pk*fk1
517 
518  ! Compute Pk(cart%x) for Legendre order 2 to npts-1
519 
520  do k = 2,npts-1
521 
522  tmp_1 = tmp_2
523  tmp_2 = pk
524  pk = ( (2*k-1)*cart%x*tmp_2 - (k-1)*tmp_1 )*interp%rk(k)
525 
526  fk0=0.0d0
527  fk1=0.0d0
528  do j=1,npts
529  fk0 = fk0 + interp%Imat(j,k+1)*f(j,l )
530  fk1 = fk1 + interp%Imat(j,k+1)*f(j,l+1)
531  end do
532  interp%vtemp(l ) = interp%vtemp(l ) + pk*fk0
533  interp%vtemp(l+1) = interp%vtemp(l+1) + pk*fk1
534 
535  end do
536 
537  end do
538 
539  ! Compute Pk(cart%y) for Legendre order 0
540 
541  pk = 1.0
542 
543  fk0 = 0.0d0
544  do j=1,npts
545  fk0 = fk0 + interp%Imat(j,1)*interp%vtemp(j)
546  end do
547  fxy = pk*fk0
548 
549  ! Compute Pk(cart%y) for Legendre order 1
550 
551  tmp_2 = pk
552  pk = cart%y
553 
554  fk0=0.0d0
555  do j=1,npts
556  fk0 = fk0 + interp%Imat(j,2)*interp%vtemp(j)
557  end do
558  fxy = fxy + pk*fk0
559 
560  ! Compute Pk(cart%y) for Legendre order 2, npts-1
561 
562  do k = 2,npts-1
563  tmp_1 = tmp_2
564  tmp_2 = pk
565  pk = ( (2*k-1)*cart%y*tmp_2 - (k-1)*tmp_1 )*interp%rk(k)
566 
567  fk0 = 0.0d0
568  do j=1,npts
569  fk0 = fk0 + interp%Imat(j,k+1)*interp%vtemp(j)
570  end do
571 
572  fxy = fxy + pk*fk0
573 
574  end do

582  function interpol_bilinear(cart, f, interp, npts, fillvalue) result(fxy)
583

585  type (cartesian2d_t), intent(in) :: cart
586  real (kind=real_kind), intent(in) :: f(npts,npts)
587  type (interpolate_t) :: interp
588  real (kind=real_kind) :: fxy ! value of f interpolated to (x,y)
589  real (kind=real_kind), intent(in), optional :: fillvalue
590  ! local variables
591 
592  real (kind=real_kind) :: xoy(npts)
593  real (kind=real_kind) :: p,q,xp,yp ,y4(4)
594 
595  integer :: l,j,k, ii, jj, na,nb,nm
596 
597  xp = cart%x
598  yp = cart%y
599 
600  xoy(:) = interp%glp(:)
601 
602  ! Search index along "x" (bisection method)
603 
604  na = 1
605  nb = npts
606  do
607  if ((nb-na) <= 1) exit
608  nm = (nb + na)/2
609  if (xp > xoy(nm)) then
610  na = nm
611  else
612  nb = nm
613  endif
614  enddo
615  ii = na
616 
617  ! Search index along "y"
618 
619  na = 1
620  nb = npts
621  do
622  if ((nb-na) <= 1) exit
623  nm = (nb + na)/2
624  if (yp > xoy(nm)) then
625  na = nm
626  else
627  nb = nm
628  endif
629  enddo
630  jj = na
631 
632  ! GLL cell containing (xp,yp)
633 
634  y4(1) = f(ii,jj)
635  y4(2) = f(ii+1,jj)
636  y4(3) = f(ii+1,jj+1)
637  y4(4) = f(ii,jj+1)
638 
639  if(present(fillvalue)) then
640  if (any(y4==fillvalue)) then
641  fxy = fillvalue
642  return
643  endif
644  endif
645  p = (xp - xoy(ii))/(xoy(ii+1) - xoy(ii))
646  q = (yp - xoy(jj))/(xoy(jj+1) - xoy(jj))

653  function parametric_coordinates(sphere, corners3D,ref_map_in, corners,cartp,facenum) result (ref)
654 
655  implicit none
656  type (spherical_polar_t), intent(in) :: sphere
657  type (cartesian2d_t) :: ref
658 
659  type (cartesian3d_t) :: corners3d(4) !x,y,z coords of element corners
660  integer,optional :: ref_map_in ! default is global variable 'cubed_sphere_map'
661  ! optional arguments, only needed for ref_map=1 (equi-angle gnomonic projection):
662  type (cartesian2d_t),optional :: corners(4) ! gnomonic coords of element corners
663  type (cartesian2d_t),optional :: cartp(np,np) ! gnomonic coords of element data
664  integer,optional :: facenum

668  integer :: i, max_nr_iter=10
669  real(kind=real_kind) :: d(2,2),dinv(2,2),detd,a,b,resa,resb,dela,delb,costh
670  real(kind=real_kind) :: tol_sq=1e-26
671  type (spherical_polar_t) :: sphere1, sphere_tmp
672  integer :: ref_map
673 
674  !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
675  ! newton iteration on: ref=ref - df^-1 (ref2sphere(ref) - sphere)
676  !
677  ! Generic version written in terms of HOMME's 'ref2sphere' and 'Dmap' operaters,
678  ! with no assumption as to the type of map (gnomonic, equi-angular, parametric)
679  !
680  ! Note that the coordinate increment from newton iterations is not a direction and thus
681  ! should not be converted into motion along a great circle arc - this routine
682  ! correclty applies the increment by just adding it to the coordintes
683  !
684  ! f = ref2sphere(xvec) - sphere
685  ! df = d(ref2sphere)
686  !
687  ! D = diag(cos(theta),1) * d(ref2sphere) d(ref2sphere) = diag(1/cos(theta),1)*D
688  !
689  ! df = diag(1/cos(theta),1)*D
690  ! df^-1 = D^-1 * diag(cos(theta),1)
691  !
692  !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
693  if (present(ref_map_in)) then
694  ref_map=ref_map_in
695  else
696  ref_map=cubed_sphere_map
697  endif
698  costh=cos(sphere%lat)
699  a=0
700  b=0
701  i=0
702  do
703  sphere1 = ref2sphere(a,b,corners3d,ref_map,corners,facenum)
704  resa = sphere1%lon - sphere%lon
705  if (resa>dd_pi) resa=resa-2*dd_pi
706  if (resa<-dd_pi) resa=resa+2*dd_pi
707 
708  resb = sphere1%lat - sphere%lat
709 
710  call dmap(d,a,b,corners3d,ref_map,cartp,facenum)
711  detd = d(1,1)*d(2,2) - d(1,2)*d(2,1)
712  dinv(1,1) = d(2,2)/detd
713  dinv(1,2) = -d(1,2)/detd
714  dinv(2,1) = -d(2,1)/detd
715  dinv(2,2) = d(1,1)/detd
716 
717  dela = dinv(1,1)*costh*resa + dinv(1,2)*resb
718  delb = dinv(2,1)*costh*resa + dinv(2,2)*resb
719  a = a - dela
720  b = b - delb
721  i=i+1
722  if ( (costh*resa)**2 + resb**2 < tol_sq .or. max_nr_iter < i) exit
723  end do
724  ref%x=a
725  ref%y=b
726
