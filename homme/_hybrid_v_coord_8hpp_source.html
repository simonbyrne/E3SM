67  const ExecViewUnmanaged<const Real[NP][NP]>& ps,
68  const ExecViewUnmanaged<Scalar [NP][NP][NUM_LEV]>& dp) const
69  {
70  Kokkos::parallel_for(Kokkos::TeamThreadRange(kv.team,NP*NP),
71  [&](const int& point_idx) {
72  const int igp = point_idx / NP;
73  const int jgp = point_idx % NP;
74  auto point_dp = Homme::subview(dp,igp,jgp);
75  const Real point_ps = ps(igp,jgp);
76  Kokkos::parallel_for(Kokkos::ThreadVectorRange(kv.team,NUM_LEV),
77  [&](const int& ilev) {
