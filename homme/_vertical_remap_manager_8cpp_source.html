40  remapper = std::make_shared<RemapFunctor<
41  false, PpmVertRemap<PpmFixedMeans>> >(
42  params.qsize, e, t, h);
43  }
44  } else if (params.remap_alg == RemapAlg::PPM_MIRRORED) {
45  if (params.rsplit != 0) {
46  remapper = std::make_shared<RemapFunctor<
47  true, PpmVertRemap<PpmMirrored>> >(
