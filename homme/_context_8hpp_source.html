20 /* A Context manages resources that are morally singletons. Context is
21  * meant to have two roles. First, a Context singleton is the only singleton in
22  * the program. Second, a context need not be a singleton, and each Context
23  * object can have different Elements, ReferenceElement, etc., objects. (That
24  * probably isn't needed, but Context immediately supports it.)
25  *
26  * Finally, Context has two singleton functions: singleton(), which returns
27  * Context&, and finalize_singleton(). The second is called in a unit test exe
28  * main before Kokkos::finalize().
29  */
30 class Context {
31 public:

33  // Getters for a managed object.
34  template<typename ConcreteType>
35  bool has () const;
36 
37  // Setters for a managed object.
38  template<typename ConcreteType, typename... Args>
39  ConcreteType& create (Args&&... args);
40 
41  template<typename ConcreteType>
42  void create_ref (ConcreteType& src);
43 
44  // More relaxed than create, it won't throw if the
45  // object already exists, and simply return it.
46  // NOTE: this is allows more flexibility in the cxx-f90
47  // interfaces, in that you have more freedom in
48  // the order of certain function calls.
49  // With 'create', you have to either check if an
50  // object was already created (and create it if it wasn't),
51  // or make sure that the function that calls 'create<T>'
52  // is *always* invoked *before* those that call 'get<T>'.
53  template<typename ConcreteType, typename... Args>
54  ConcreteType& create_if_not_there (Args&&... args);
55 
56  // Getters for a managed object.

58  ConcreteType& get () const;
59

85 template<typename ConcreteType, typename... Args>
86 ConcreteType& Context::create_if_not_there (Args&&... args) {
87  // This is needed for emplacing a type whose constructor takes no arguments.
88  // We could do emplace(name,ConcreteType()), but then we would be assuming
89  // that ConcreteType *has* a move constructor. This implementation here is
90  // probably the most cumbersome, but also the safest.
91  auto it_bool = m_members.emplace(typeid(ConcreteType).name(),Homme::any());
92  if (it_bool.second) {
93  // We created it, so init it.
94  it_bool.first->second.reset<ConcreteType>(args...);
95  m_is_ref_wrapper[typeid(ConcreteType).name()] = false;
96  }
97  return *any_ptr_cast<ConcreteType>(it_bool.first->second);
98 }

120  // This is needed for emplacing a type whose constructor takes no arguments.
121  // We could do emplace(name,ConcreteType()), but then we would be assuming
122  // that ConcreteType *has* a move constructor. This implementation here is
123  // probably the most cumbersome, but also the safest.

133 template<typename ConcreteType>
134 ConcreteType& Context::get () const {
135  const std::string& name = typeid(ConcreteType).name();
136  auto it = m_members.find(name);
137  Errors::runtime_check(it!=m_members.end(), "Error! Context member '" + name + "' not found.\n", -1);
138  if (m_is_ref_wrapper.at(name)) {
139  auto ref = any_ptr_cast<std::reference_wrapper<ConcreteType>>(it->second);
140  return ref->get();
141  } else {
142  return *get_ptr<ConcreteType>();
143  }
144 }
