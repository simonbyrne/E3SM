459  function curl_sphere_wk_testcov(s,deriv,elem) result(ds)
460 !
461 ! integrated-by-parts gradient, w.r.t. COVARIANT test functions
462 ! input s: scalar (assumed to be s*khat)
463 ! output ds: weak curl, lat/lon coordinates
464 !
465 ! starting with:
466 ! PHIcov1 = (PHI,0) covariant vector
467 ! PHIcov2 = (0,PHI) covariant vector
468 !
469 ! ds1 = integral[ PHIcov1 dot curl(s*khat) ]
470 ! ds2 = integral[ PHIcov2 dot curl(s*khat) ]
471 ! integrate by parts:
472 ! ds1 = integral[ vor(PHIcov1) * s ]
473 ! ds2 = integral[ vor(PHIcov1) * s ]
474 !
475 ! PHIcov1 = (PHI^mn,0)
476 ! PHIcov2 = (0,PHI^mn)
477 ! vorticity() acts on covariant vectors:
478 ! ds1 = sum wij g s_ij 1/g ( (PHIcov1_2)_x - (PHIcov1_1)_y )
479 ! = -sum wij s_ij d/dy (PHI^mn )
480 ! for d/dy component, only sum over i=m
481 ! = -sum w_mj s_mj d( PHI^n)(j)
482 ! j
483 !
484 ! ds2 = sum wij g s_ij 1/g ( (PHIcov2_2)_x - (PHIcov2_1)_y )
485 ! = +sum wij s_ij d/dx (PHI^mn )
486 ! for d/dx component, only sum over j=n
487 ! = +sum w_in s_in d( PHI^m)(i)
488 ! i
489 !
490  type (derivative_t), intent(in) :: deriv
491  type (element_t), intent(in) :: elem
492  real(kind=real_kind), intent(in) :: s(np,np)
493 
494  real(kind=real_kind) :: ds(np,np,2)
495 
496  integer i,j,l,m,n
497  real(kind=real_kind) :: dscontra(np,np,2)
498 
499  dscontra=0
500  do n=1,np
501  do m=1,np
502 !DIR$ UNROLL(NP)
503  do j=1,np
504  ! phi(n)_y sum over second index, 1st index fixed at m
505  dscontra(m,n,1)=dscontra(m,n,1)-(elem%mp(m,j)*s(m,j)*deriv%Dvv(n,j) )
506  ! phi(m)_x sum over first index, second index fixed at n
507  dscontra(m,n,2)=dscontra(m,n,2)+(elem%mp(j,n)*s(j,n)*deriv%Dvv(m,j) )

509  enddo

522  function gradient_sphere_wk_testcov(s,deriv,elem) result(ds)
523 !
524 ! integrated-by-parts gradient, w.r.t. COVARIANT test functions
525 ! input s: scalar
526 ! output ds: weak gradient, lat/lon coordinates
527 ! ds = - integral[ div(PHIcov) s ]
528 !
529 ! PHIcov1 = (PHI^mn,0)
530 ! PHIcov2 = (0,PHI^mn)
531 ! div() acts on contra components, so convert test function to contra:
532 ! PHIcontra1 = metinv PHIcov1 = (a^mn,b^mn)*PHI^mn
533 ! a = metinv(1,1) b=metinv(2,1)
534 !
535 ! ds1 = sum wij g s_ij 1/g ( g a PHI^mn)_x + ( g b PHI^mn)_y )
536 ! = sum wij s_ij ag(m,n) d/dx( PHI^mn ) + bg(m,n) d/dy( PHI^mn)
537 ! i,j
538 ! for d/dx component, only sum over j=n
539 ! = sum w_in s_in ag(m,n) d( PHI^m)(i)
540 ! i
541 ! for d/dy component, only sum over i=m
542 ! = sum w_mj s_mj bg(m,n) d( PHI^n)(j)
543 ! j
544 !
545 !
546 ! This formula is identical to gradient_sphere_wk_testcontra, except that
547 ! g(m,n) is replaced by a(m,n)*g(m,n)
548 ! and we have two terms for each componet of ds
549 !
550 !
551  type (derivative_t), intent(in) :: deriv
552  type (element_t), intent(in) :: elem
553  real(kind=real_kind), intent(in) :: s(np,np)
554 
555  real(kind=real_kind) :: ds(np,np,2)
556 
557  integer i,j,l,m,n
558  real(kind=real_kind) :: dscontra(np,np,2)
559 
560 
561  dscontra=0
562  do n=1,np
563  do m=1,np
564 !DIR$ UNROLL(NP)
565  do j=1,np
566  dscontra(m,n,1)=dscontra(m,n,1)-(&
567  (elem%mp(j,n)*elem%metinv(m,n,1,1)*elem%metdet(m,n)*s(j,n)*deriv%Dvv(m,j) ) +&
568  (elem%mp(m,j)*elem%metinv(m,n,2,1)*elem%metdet(m,n)*s(m,j)*deriv%Dvv(n,j) ) &
569  )
570 
571  dscontra(m,n,2)=dscontra(m,n,2)-(&
572  (elem%mp(j,n)*elem%metinv(m,n,1,2)*elem%metdet(m,n)*s(j,n)*deriv%Dvv(m,j) ) +&
573  (elem%mp(m,j)*elem%metinv(m,n,2,2)*elem%metdet(m,n)*s(m,j)*deriv%Dvv(n,j) ) &
574  )
575  enddo
576  enddo
577  enddo

579  do j=1,np
580  do i=1,np
581  ds(i,j,1)=(elem%D(i,j,1,1)*dscontra(i,j,1) + elem%D(i,j,1,2)*dscontra(i,j,2)) *scale_factor_inv
582  ds(i,j,2)=(elem%D(i,j,2,1)*dscontra(i,j,1) + elem%D(i,j,2,2)*dscontra(i,j,2)) *scale_factor_inv
583  enddo
584  enddo
585 
586  end function gradient_sphere_wk_testcov
