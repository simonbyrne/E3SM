11 ! The KESSLER subroutine implements the Kessler (1969) microphysics
12 ! parameterization as described by Soong and Ogura (1973) and Klemp
13 ! and Wilhelmson (1978, KW). KESSLER is called at the end of each
14 ! time step and makes the final adjustments to the potential
15 ! temperature and moisture variables due to microphysical processes
16 ! occurring during that time step. KESSLER is called once for each
17 ! vertical column of grid cells. Increments are computed and added
18 ! into the respective variables. The Kessler scheme contains three
19 ! moisture categories: water vapor, cloud water (liquid water that
20 ! moves with the flow), and rain water (liquid water that falls
21 ! relative to the surrounding air). There are no ice categories.
22 ! Variables in the column are ordered from the surface to the top.
23 !
24 ! SUBROUTINE KESSLER(theta, qv, qc, qr, rho, pk, dt, z, nz, rainnc)
25 !
26 ! Input variables:
27 ! theta - potential temperature (K)
28 ! qv - water vapor mixing ratio (gm/gm)
29 ! qc - cloud water mixing ratio (gm/gm)
30 ! qr - rain water mixing ratio (gm/gm)
31 ! rho - dry air density (not mean state as in KW) (kg/m^3)
32 ! pk - Exner function (not mean state as in KW) (p/p0)**(R/cp)
33 ! dt - time step (s)
34 ! z - heights of thermodynamic levels in the grid column (m)
35 ! nz - number of thermodynamic levels in the column
36 ! precl - Precipitation rate (m_water/s)
37 !
38 ! Output variables:
39 ! Increments are added into t, qv, qc, qr, and rainnc which are
40 ! returned to the routine from which KESSLER was called. To obtain
41 ! the total precip qt, after calling the KESSLER routine, compute:
42 !
43 ! qt = sum over surface grid cells of (rainnc * cell area) (kg)
44 ! [here, the conversion to kg uses (10^3 kg/m^3)*(10^-3 m/mm) = 1]
45 !
46 !
47 ! Authors: Paul Ullrich
48 ! University of California, Davis
49 ! Email: paullrich@ucdavis.edu
50 !
51 ! Based on a code by Joseph Klemp
52 ! (National Center for Atmospheric Research)
53 !
54 ! Reference:
55 !
56 ! Klemp, J. B., W. C. Skamarock, W. C., and S.-H. Park, 2015:
57 ! Idealized Global Nonhydrostatic Atmospheric Test Cases on a Reduced
58 ! Radius Sphere. Journal of Advances in Modeling Earth Systems.
59 ! doi:10.1002/2015MS000435
60 !

63 SUBROUTINE kessler(theta, qv, qc, qr, rho, pk, dt, z, nz, precl)
64 
65 use physical_constants, only: kappa
66 
67  IMPLICIT NONE
68 
69  !------------------------------------------------
70  ! Input / output parameters
71  !------------------------------------------------
72 
73  REAL(8), DIMENSION(nz), INTENT(INOUT) :: &
74  theta , & ! Potential temperature (K)
75  qv , & ! Water vapor mixing ratio (gm/gm)
76  qc , & ! Cloud water mixing ratio (gm/gm)
77  qr ! Rain water mixing ratio (gm/gm)
78 
79  REAL(8), DIMENSION(nz), INTENT(IN) :: &
80  rho ! Dry air density (not mean state as in KW) (kg/m^3)
81 
82  REAL(8), INTENT(OUT) :: &
83  precl ! Precipitation rate (m_water / s)
84 
85  REAL(8), DIMENSION(nz), INTENT(IN) :: &
86  z , & ! Heights of thermo. levels in the grid column (m)
87  pk ! Exner function (p/p0)**(R/cp)
88 
89  REAL(8), INTENT(IN) :: &
90  dt ! Time step (s)
91 
92  INTEGER, INTENT(IN) :: nz ! Number of thermodynamic levels in the column
93 
94  !------------------------------------------------
95  ! Local variables
96  !------------------------------------------------
97  REAL, DIMENSION(nz) :: r, rhalf, velqr, sed, pc
98 
99  REAL(8) :: f5, f2x, xk, ern, qrprod, prod, qvs, psl, rhoqr, dt_max, dt0
100 
101  INTEGER :: k, rainsplit, nt
102 
103  !------------------------------------------------
104  ! Begin calculation
105  !------------------------------------------------
106  f2x = 17.27d0
107  f5 = 237.3d0 * f2x * 2500000.d0 / 1003.d0
108  xk = .2875d0 ! kappa (r/cp)
109  !xk = kappa ! kappa (r/cp)
110 
111  psl = 1000.d0 ! pressure at sea level (mb)
112  rhoqr = 1000.d0 ! density of liquid water (kg/m^3)
113 
114  do k=1,nz
115  r(k) = 0.001d0*rho(k)
116  rhalf(k) = sqrt(rho(1)/rho(k))
117  pc(k) = 3.8d0/(pk(k)**(1./xk)*psl)
118 
119  ! Liquid water terminal velocity (m/s) following KW eq. 2.15
120  velqr(k) = 36.34d0*(qr(k)*r(k))**0.1364*rhalf(k)
121 
122  end do
123 
124  ! Maximum time step size in accordance with CFL condition
125  if (dt .le. 0.d0) then
126  write(*,*) 'kessler.f90 called with nonpositive dt'
127  stop
128  end if
129 
130  dt_max = dt
131  do k=1,nz-1
132  if (velqr(k) .ne. 0.d0) then
133  dt_max = min(dt_max, 0.8d0*(z(k+1)-z(k))/velqr(k))
134  end if
135  end do
136 
137  ! Number of subcycles
138  rainsplit = ceiling(dt / dt_max)
139  dt0 = dt / real(rainsplit,8)
140 
141  ! Subcycle through rain process
142  precl = 0.d0
143 
144  do nt=1,rainsplit
145 
146  ! Precipitation rate (m/s)
147  precl = precl + rho(1) * qr(1) * velqr(1) / rhoqr
148 
149  ! Sedimentation term using upstream differencing
150  do k=1,nz-1
151  sed(k) = dt0*(r(k+1)*qr(k+1)*velqr(k+1)-r(k)*qr(k)*velqr(k))/(r(k)*(z(k+1)-z(k)))
152  end do
153  sed(nz) = -dt0*qr(nz)*velqr(nz)/(.5*(z(nz)-z(nz-1)))
154 
155  ! Adjustment terms
156  do k=1,nz
157 
158  ! Autoconversion and accretion rates following KW eq. 2.13a,b
159  qrprod = qc(k) - (qc(k)-dt0*max(.001*(qc(k)-.001d0),0.d0))/(1.d0+dt0*2.2d0*qr(k)**.875)
160  qc(k) = max(qc(k)-qrprod,0.d0)
161  qr(k) = max(qr(k)+qrprod+sed(k),0.d0)
162 
163  ! Saturation vapor mixing ratio (gm/gm) following KW eq. 2.11
164  qvs = pc(k)*exp(f2x*(pk(k)*theta(k)-273.d0) &
165  /(pk(k)*theta(k)- 36.d0))
166  prod = (qv(k)-qvs)/(1.d0+qvs*f5/(pk(k)*theta(k)-36.d0)**2)
167 
168  ! Evaporation rate following KW eq. 2.14a,b
169  ern = min(dt0*(((1.6d0+124.9d0*(r(k)*qr(k))**.2046) &
170  *(r(k)*qr(k))**.525)/(2550000d0*pc(k) &
171  /(3.8d0 *qvs)+540000d0))*(dim(qvs,qv(k)) &
172  /(r(k)*qvs)),max(-prod-qc(k),0.d0),qr(k))
173 
174  ! Saturation adjustment following KW eq. 3.10
175  theta(k)= theta(k) + 2500000d0/(1003.d0*pk(k))*(max( prod,-qc(k))-ern)
176  qv(k) = max(qv(k)-max(prod,-qc(k))+ern,0.d0)
177  qc(k) = qc(k)+max(prod,-qc(k))
178  qr(k) = qr(k)-ern
179  end do
180 
181  ! Recalculate liquid water terminal velocity
182  if (nt .ne. rainsplit) then
183  do k=1,nz
184  velqr(k) = 36.34d0*(qr(k)*r(k))**0.1364*rhalf(k)
185  end do
186  end if
187  end do
188 
189  precl = precl / dble(rainsplit)
190 
191 END SUBROUTINE kessler
