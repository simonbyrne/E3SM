70 int HyperviscosityFunctorImpl::requested_buffer_size () const {
71  constexpr int size_mid_scalar = NP*NP*NUM_LEV*VECTOR_SIZE;
72  constexpr int size_mid_vector = 2*NP*NP*NUM_LEV*VECTOR_SIZE;
73  constexpr int size_int_scalar = NP*NP*NUM_LEV_P*VECTOR_SIZE;
74  constexpr int size_bhm_scalar = NP*NP*NUM_BIHARMONIC_LEV*VECTOR_SIZE;
75  constexpr int size_bhm_vector = 2*NP*NP*NUM_BIHARMONIC_LEV*VECTOR_SIZE;
76 
77  const int nelems = m_geometry.num_elems();
78  const int nteams = m_tu.get_num_ws_slots();
79 
80  // Number of scalar/vector int/mid/bhm buffers needed, with size nteams/nelems
81  const int mid_vectors_nelems = 1;
82  const int int_scalars_nelems = 0;
83  const int mid_scalars_nelems = 2 + (m_process_nh_vars ? 2 : 0);
84 
85  const int bhm_scalars_nteams = 2 + (m_process_nh_vars ? 2 : 0);
86  const int bhm_vectors_nteams = 1;
87 
88  const int size = nelems*(mid_scalars_nelems*size_mid_scalar +
89  mid_vectors_nelems*size_mid_vector +
90  int_scalars_nelems*size_int_scalar) +
91  nteams*(bhm_scalars_nteams*size_bhm_scalar +
92  bhm_vectors_nteams*size_bhm_vector);
93 
94  return size;
95 }

97 void HyperviscosityFunctorImpl::init_buffers (const FunctorsBuffersManager& fbm) {
98  Errors::runtime_check(fbm.allocated_size()>=requested_buffer_size(), "Error! Buffers size not sufficient.\n");
99 
100  constexpr int size_mid_scalar = NP*NP*NUM_LEV;
101  constexpr int size_mid_vector = 2*NP*NP*NUM_LEV;
102  constexpr int size_bhm_scalar = NP*NP*NUM_BIHARMONIC_LEV;
103  constexpr int size_bhm_vector = 2*NP*NP*NUM_BIHARMONIC_LEV;
104 
105  auto mem_in = fbm.get_memory();
106  Scalar* mem = reinterpret_cast<Scalar*>(fbm.get_memory());
107  const int nelems = m_geometry.num_elems();
108  const int nteams = m_tu.get_num_ws_slots();
109 
110  // Tens quantities (persistent views => nelems)
111  m_buffers.dptens = decltype(m_buffers.dptens)(mem,nelems);
112  mem += size_mid_scalar*nelems;
113 
114  m_buffers.ttens = decltype(m_buffers.ttens)(mem,nelems);
115  mem += size_mid_scalar*nelems;
116 
117  if (m_process_nh_vars) {
118  m_buffers.wtens = decltype(m_buffers.wtens)(mem,nelems);
119  mem += size_mid_scalar*nelems;
120 
121  m_buffers.phitens = decltype(m_buffers.phitens)(mem,nelems);
122  mem += size_mid_scalar*nelems;
123  }
124 
125  m_buffers.vtens = decltype(m_buffers.vtens)(mem,nelems);
126  mem += size_mid_vector*nelems;
127 
128  // Biharmonic views (non-persistent views => nteams)
129  m_buffers.lapl_dp = decltype(m_buffers.lapl_dp)(mem,nteams);
130  mem += size_bhm_scalar*nteams;
131 
132  if (m_process_nh_vars) {
133  m_buffers.lapl_w = decltype(m_buffers.lapl_w)(mem,nteams);
134  mem += size_bhm_scalar*nteams;
135 
136  m_buffers.lapl_phi = decltype(m_buffers.lapl_phi)(mem,nteams);
137  mem += size_bhm_scalar*nteams;
138  }
139 
140  m_buffers.lapl_theta = decltype(m_buffers.lapl_theta)(mem,nteams);
141  mem += size_bhm_scalar*nteams;
142 
143  m_buffers.lapl_v = decltype(m_buffers.lapl_v)(mem,nteams);
144  mem += size_bhm_vector*nteams;
145 
146  const int used_mem = reinterpret_cast<Real*>(mem)-mem_in;
147  if (used_mem < requested_buffer_size()) {
148  printf("[HyperviscosityFunctorImpl] Warning! We used less memory than we said we would: %d instead of %d\n",
149  used_mem, requested_buffer_size());
150  } else if (used_mem > requested_buffer_size()) {
151  std::string msg = "[HyperviscosityFunctorImpl] Error! We used more memory than we said we would: "
152  + std::to_string(used_mem) + " instead of "
153  + std::to_string(requested_buffer_size()) + "\n";
154  Errors::runtime_abort(msg);
155  }
156 }

178 void HyperviscosityFunctorImpl::run (const int np1, const Real dt, const Real eta_ave_w)
179 {
180  m_data.np1 = np1;
181  m_data.dt = dt/m_data.hypervis_subcycle;
182  m_data.eta_ave_w = eta_ave_w;
183 
184  // Convert vtheta_dp -> theta
185  auto state = m_state;
186  Kokkos::parallel_for(Homme::get_default_team_policy<ExecSpace>(state.num_elems()),
187  KOKKOS_LAMBDA(const TeamMember& team) {
188  const int ie = team.league_rank();
189  Kokkos::parallel_for(Kokkos::TeamThreadRange(team,NP*NP),
190  [&](const int idx) {
191  const int igp = idx / NP;
192  const int jgp = idx % NP;
193 
194  // theta->vtheta
195  auto vtheta = Homme::subview(state.m_vtheta_dp,ie,np1,igp,jgp);
196  auto dp = Homme::subview(state.m_dp3d,ie,np1,igp,jgp);
197  Kokkos::parallel_for(Kokkos::ThreadVectorRange(team,NUM_LEV),
198  [&](const int ilev) {
199  vtheta(ilev) /= dp(ilev);
200  });
201  });
202  });
203  Kokkos::fence();
204 
205  for (int icycle = 0; icycle < m_data.hypervis_subcycle; ++icycle) {
206  GPTLstart("hvf-bhwk");
207  biharmonic_wk_theta ();
208  GPTLstop("hvf-bhwk");
209 
210  // dispatch parallel_for for first kernel
211  Kokkos::parallel_for(m_policy_pre_exchange, *this);
212  Kokkos::fence();
213 
214  // Exchange
215  assert (m_be->is_registration_completed());
216  GPTLstart("hvf-bexch");
217  m_be->exchange();
218  GPTLstop("hvf-bexch");
219 
220  // Update states
221  Kokkos::parallel_for(m_policy_update_states, *this);
222  Kokkos::fence();
223  }
224 
225  // Finally, convert theta back to vtheta, and adjust w at surface
226  auto geo = m_geometry;
227  Kokkos::parallel_for(Homme::get_default_team_policy<ExecSpace>(state.num_elems()),
228  KOKKOS_LAMBDA(const TeamMember& team) {
229  const int ie = team.league_rank();
230  Kokkos::parallel_for(Kokkos::TeamThreadRange(team,NP*NP),
231  [&](const int idx) {
232  const int igp = idx / NP;
233  const int jgp = idx % NP;
234 
235  // theta->vtheta
236  auto vtheta = Homme::subview(state.m_vtheta_dp,ie,np1,igp,jgp);
237  auto dp = Homme::subview(state.m_dp3d,ie,np1,igp,jgp);
238  Kokkos::parallel_for(Kokkos::ThreadVectorRange(team,NUM_LEV),
239  [&](const int ilev) {
240  vtheta(ilev) *= dp(ilev);
241  });
242 
243  // Fix w at surface:
244  // Adjust w_i at the surface, since velocity has changed
245  if (m_process_nh_vars) {
246  Kokkos::single(Kokkos::PerThread(team),[&](){
247  using InfoI = ColInfo<NUM_INTERFACE_LEV>;
248  using InfoM = ColInfo<NUM_PHYSICAL_LEV>;
249  constexpr int LAST_MID_PACK = InfoM::LastPack;
250  constexpr int LAST_MID_PACK_END = InfoM::LastPackEnd;
251  constexpr int LAST_INT_PACK = InfoI::LastPack;
252  constexpr int LAST_INT_PACK_END = InfoI::LastPackEnd;
253  constexpr Real g = PhysicalConstants::g;
254 
255  const auto& grad_x = geo.m_gradphis(ie,0,igp,jgp);
256  const auto& grad_y = geo.m_gradphis(ie,1,igp,jgp);
257  const auto& u = state.m_v(ie,np1,0,igp,jgp,LAST_MID_PACK)[LAST_MID_PACK_END];
258  const auto& v = state.m_v(ie,np1,1,igp,jgp,LAST_MID_PACK)[LAST_MID_PACK_END];
259 
260  auto& w = state.m_w_i(ie,np1,igp,jgp,LAST_INT_PACK)[LAST_INT_PACK_END];
261 
262  w = (u*grad_x+v*grad_y) / g;
263  });
264  }
265  });
266  });
267 }

269 void HyperviscosityFunctorImpl::biharmonic_wk_theta() const
270 {
271  // For the first laplacian we use a differnt kernel, which uses directly the states
272  // at timelevel np1 as inputs, and subtracts the reference states.
273  // This way we avoid copying the states to *tens buffers.
274  Kokkos::parallel_for(m_policy_first_laplace, *this);
275  Kokkos::fence();
276 
277  // Exchange
278  assert (m_be->is_registration_completed());
279  GPTLstart("hvf-bexch");
280  m_be->exchange(m_geometry.m_rspheremp);
281  GPTLstop("hvf-bexch");
282 
283  // Compute second laplacian, tensor or const hv
284  const int ne = m_geometry.num_elems();
285  if ( m_data.consthv ) {
286  auto policy = Homme::get_default_team_policy<ExecSpace,TagSecondLaplaceConstHV>(ne);
287  Kokkos::parallel_for(policy, *this);
288  }else{
289  auto policy = Homme::get_default_team_policy<ExecSpace,TagSecondLaplaceTensorHV>(ne);
290  Kokkos::parallel_for(policy, *this);
291  }
292  Kokkos::fence();
293 }
