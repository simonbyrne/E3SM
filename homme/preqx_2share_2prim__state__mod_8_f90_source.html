670 subroutine prim_energy_halftimes(elem,hvcoord,tl,n,t_before_advance,nets,nete)
671 !
672 ! called at the end of a timestep, before timelevel update.
673 ! Solution is known at two timelevels. We originally tried to
674 ! compute the energy at the midpoint between these levels, but now
675 ! we compute the energy at:
676 !
677 ! t_before_advance==.true. tl%n0 begining of timestep
678 ! t_before_advance==.false. tl%np1 completed timestep, before tl update
679 !
680 ! This routine is called 4 times: for historical reasons they are out
681 ! of sequence.
682 !
683 ! n=1: after CAM forcing, before timestep
684 ! n=2: after timestep, including remap, before tl update
685 ! n=3: after CAM forcing, before timestep
686 ! n=4: (not used)
687 !
688 ! LF case we use staggered formulas:
689 ! compute the energies at time half way between timelevels t1 and t2,
690 ! using timelevels t1,t2. store in location n of our elem()%accum%* variables
691 !
692 ! in the Dry case, PE (and similar for IE) will be conserved exactly by Leapfrog,
693 ! if it is defined as:
694 !
695 ! PE(n+.5) = .5*( T(n+1) dp(n) + T(n) dp(n+1) )
696 !
697 ! KE cannot be exactly conserved with Leapfrog. but we still use:
698 !
699 ! KE(n+.5) = .5*( .5 u(n+1)^2 dp(n) + .5 u(n)^2 dp(n+1) )
700 !
701 !
702  use kinds, only : real_kind
703  use dimensions_mod, only : np, np, nlev
704  use control_mod, only : use_cpstar
705  use hybvcoord_mod, only : hvcoord_t
706  use element_mod, only : element_t
707  use physical_constants, only : cp, cpwater_vapor
708  use physics_mod, only : virtual_specific_heat
709  use prim_si_mod, only : preq_hydrostatic
710 
711  integer :: t1,n,nets,nete
712  type (element_t) , intent(inout), target :: elem(:)
713  type (hvcoord_t) :: hvcoord
714  type (timelevel_t), intent(in) :: tl
715  logical :: t_before_advance
716 
717  integer :: ie,k,i,j
718  real (kind=real_kind), dimension(np,np,nlev) :: dpt1 ! delta pressure
719  real (kind=real_kind), dimension(np,np) :: e
720  real (kind=real_kind), dimension(np,np) :: suml,suml2,v1,v2
721  real (kind=real_kind), dimension(np,np,nlev) :: sumlk, suml2k
722  real (kind=real_kind), dimension(np,np,nlev) :: p,phi
723  real (kind=real_kind) :: cp_star1,qval_t1,qval_t2
724 
725 
726  integer:: tmp, t1_qdp ! the time pointers for Qdp are not the same
727 
728  if (t_before_advance) then
729  t1=tl%n0
730  call timelevel_qdp( tl, qsplit, t1_qdp) !get n0 level into t2_qdp
731  else
732  t1=tl%np1
733  call timelevel_qdp(tl, qsplit, tmp, t1_qdp) !get np1 into t2_qdp
734  endif
735 
736  ! IE Cp*dpdn*T + (Cpv-Cp) Qdpdn*T
737  ! Cp*dpdn(n)*T(n+1) + (Cpv-Cp) Qdpdn(n)*T(n+1)
738  ! [Cp + (Cpv-Cp) Q(n)] *dpdn(n)*T(n+1)
739  do ie=nets,nete
740 
741  dpt1=elem(ie)%state%dp3d(:,:,:,t1)
742 #if (defined COLUMN_OPENMP)
743 !$omp parallel do private(k,i,j,cp_star1,qval_t1)
744 #endif
745  do k=1,nlev
746  sumlk(:,:,k)=0
747  suml2k(:,:,k)=0
748  do i=1,np
749  do j=1,np
750  if(use_cpstar == 1) then
751  ! Cp_star = cp + (Cpwater_vapor - cp)*qval
752  qval_t1 = elem(ie)%state%Qdp(i,j,k,1,t1_qdp)/dpt1(i,j,k)
753  cp_star1= virtual_specific_heat(qval_t1)
754  else
755  cp_star1=cp
756  endif
757  sumlk(i,j,k) = sumlk(i,j,k) + cp_star1*elem(ie)%state%T(i,j,k,t1) *dpt1(i,j,k)
758  suml2k(i,j,k) = suml2k(i,j,k) + (cp_star1-cp)*elem(ie)%state%T(i,j,k,t1) *dpt1(i,j,k)
759  enddo
760  enddo
761  enddo
762  suml=0
763  suml2=0
764  do k=1,nlev
765  suml(:,:) = suml(:,:) + sumlk(:,:,k)
766  suml2(:,:) = suml2(:,:) + suml2k(:,:,k)
767  enddo
768  elem(ie)%accum%IEner(:,:,n)=suml(:,:)
769  elem(ie)%accum%IEner_wet(:,:,n)=suml2(:,:)
770 
771 
772  ! KE .5 dp/dn U^2
773 #if (defined COLUMN_OPENMP)
774 !$omp parallel do private(k,E)
775 #endif
776  do k=1,nlev
777  e = (elem(ie)%state%v(:,:,1,k,t1)**2 + &
778  elem(ie)%state%v(:,:,2,k,t1)**2 ) / 2
779  sumlk(:,:,k) = e*dpt1(:,:,k)
780  enddo
781  suml=0
782  do k=1,nlev
783  suml(:,:) = suml(:,:) + sumlk(:,:,k)
784  enddo
785  elem(ie)%accum%KEner(:,:,n)=suml(:,:)
786 
787 
788 
789  ! PE dp/dn PHIs
790  suml=0
791  do k=1,nlev
792  suml = suml + elem(ie)%state%phis(:,:)*dpt1(:,:,k)
793  enddo
794  elem(ie)%accum%PEner(:,:,n)=suml(:,:)
795 
796 
797 
798  enddo
799 
800 end subroutine prim_energy_halftimes
