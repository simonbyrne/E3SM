6 !SUBROUTINES:
7 ! prim_advec_tracers_remap_rk2()
8 ! SEM 2D RK2 + monotone remap + hyper viscosity
9 ! SEM 2D RK2 can use sign-preserving or monotone reconstruction
10 !
11 !For RK2 advection of Q: (example of 2 stage RK for tracers): dtq = qsplit*dt
12 !For consistency, if Q=1
13 ! dp1 = dp(t)- dtq div[ U1 dp(t)]
14 ! dp2 = dp1 - dtq div[ U2 dp1 ] + 2*dtq D( dpdiss_ave )
15 ! dp* = (dp(t) + dp2 )/2
16 ! = dp(t) - dtq div[ U1 dp(t) + U2 dp1 ]/2 + dtq D( dpdiss_ave )
17 !
18 !so we require:
19 ! U1 = Udp_ave / dp(t)
20 ! U2 = Udp_ave / dp1
21 !
22 !For tracer advection:
23 ! Qdp1 = Qdp(t)- dtq div[ U1 Qdp(t)]
24 ! Qdp2 = Qdp1 - dtq div[ U2 Qdp1 ] + 2*dtq D( Q dpdiss_ave )
25 ! Qdp* = (Qdp(t) + Qdp2 )/2
26 ! = Qdp(t) - dtq div[ U1 Qdp(t) + U2 Qdp1 ] + dtq D( Q dpdiss_ave )
27 !
28 !Qdp1: limit Q, with Q = Qdp1-before-DSS/(dp1-before-DSS) with dp1 as computed above
29 !Qdp2: limit Q, with Q = Qdp2-before-DSS/(dp2-before-DSS) with dp2 as computed above
30 !
31 !For dissipation: Q = Qdp1-after-DSS / dp1-after-DSS
32 !
33 !
34 !last step:
35 ! remap Qdp* to Qdp(t+1) [ dp_star(t+1) -> dp(t+1) ]

137  subroutine prim_advec_tracers_remap_rk2( elem , deriv , hvcoord , hybrid , dt , tl , nets , nete )
138  use perf_mod , only : t_startf, t_stopf ! _EXTERNAL
139  use derivative_mod, only : divergence_sphere
140  use control_mod , only : qsplit, dcmip16_mu_s, dcmip16_mu_q
141  implicit none
142  type (element_t) , intent(inout) :: elem(:)
143  type (derivative_t) , intent(in ) :: deriv
144  type (hvcoord_t) , intent(in ) :: hvcoord
145  type (hybrid_t) , intent(in ) :: hybrid
146  real(kind=real_kind) , intent(in ) :: dt
147  type (timelevel_t) , intent(inout) :: tl
148  integer , intent(in ) :: nets
149  integer , intent(in ) :: nete
150 
151  integer :: i,j,k,l,ie,q,nmin
152  integer :: nfilt,rkstage,rhs_multiplier
153  integer :: n0_qdp, np1_qdp
154 
155  call t_barrierf('sync_prim_advec_tracers_remap_rk2', hybrid%par%comm)
156  call t_startf('prim_advec_tracers_remap_rk2')
157 ! call extrae_user_function(1)
158  call timelevel_qdp( tl, qsplit, n0_qdp, np1_qdp) !time levels for qdp are not the same
159  rkstage = 3 ! 3 stage RKSSP scheme, with optimal SSP CFL
160 
161  !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
162  ! RK2 2D advection step
163  ! note: stage 3 we take the oppertunity to DSS omega
164  !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
165  ! use these for consistent advection (preserve Q=1)
166  ! derived%vdp_ave = mean horiz. flux: U*dp
167  ! derived%eta_dot_dpdn = mean vertical velocity (used for remap)
168  ! derived%omega_p = advection code will DSS this for the physics, but otherwise
169  ! it is not needed
170  ! Also: save a copy of div(U dp) in derived%div(:,:,:,1), which will be DSS'd
171  ! and a DSS'ed version stored in derived%div(:,:,:,2)
172 
173  call t_startf('precomput_divdp')
174  call precompute_divdp( elem , hybrid , deriv , dt , nets , nete , n0_qdp )
175  call t_stopf('precomput_divdp')
176 
177  !rhs_multiplier is for obtaining dp_tracers at each stage:
178  !dp_tracers(stage) = dp - rhs_multiplier*dt*divdp_proj
179 
180  call t_startf('euler_step_0')
181  rhs_multiplier = 0
182  call euler_step( np1_qdp , n0_qdp , dt/2 , elem , hvcoord , hybrid , deriv , nets , nete , dssdiv_vdp_ave , rhs_multiplier )
183  call t_stopf('euler_step_0')
184 
185  call t_startf('euler_step_1')
186  rhs_multiplier = 1
187  call euler_step( np1_qdp , np1_qdp , dt/2 , elem , hvcoord , hybrid , deriv , nets , nete , dsseta , rhs_multiplier )
188  call t_stopf('euler_step_1')
189 
190  call t_startf('euler_step_2')
191  rhs_multiplier = 2
192  call euler_step( np1_qdp , np1_qdp , dt/2 , elem , hvcoord , hybrid , deriv , nets , nete , dssomega , rhs_multiplier )
193  call t_stopf('euler_step_2')
194 
195  !to finish the 2D advection step, we need to average the t and t+2 results to get a second order estimate for t+1.
196  call t_startf('qdp_tavg')
197  call qdp_time_avg( elem , rkstage , n0_qdp , np1_qdp , limiter_option , nu_p , nets , nete )
198  call t_stopf('qdp_tavg')
199 
200  !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
201  ! Dissipation
202  !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
203  if ( limiter_option == 8 .or. limiter_option == 9 ) then
204  ! dissipation was applied in RHS.
205  else
206  call t_startf('ah_scalar')
207  call advance_hypervis_scalar(elem,hvcoord,hybrid,deriv,tl%np1,np1_qdp,nets,nete,dt)
208  call t_stopf('ah_scalar')
209  endif
210 ! call extrae_user_function(0)
211 
212  ! physical viscosity for supercell test case
213  if (dcmip16_mu_q>0) then
214  call advance_physical_vis(elem,hvcoord,hybrid,deriv,tl%np1,np1_qdp,nets,nete,dt,dcmip16_mu_q)
215  endif
216 
217  call t_stopf('prim_advec_tracers_remap_rk2')
218 
219  end subroutine prim_advec_tracers_remap_rk2

262  subroutine euler_step( np1_qdp , n0_qdp , dt , elem , hvcoord , hybrid , deriv , nets , nete , DSSopt , rhs_multiplier )
263  ! ===================================
264  ! This routine is the basic foward
265  ! euler component used to construct RK SSP methods
266  !
267  ! u(np1) = u(n0) + dt2*DSS[ RHS(u(n0)) ]
268  !
269  ! n0 can be the same as np1.
270  !
271  ! DSSopt = DSSeta or DSSomega: also DSS eta_dot_dpdn or omega
272  !
273  ! ===================================
274  use kinds , only : real_kind
275  use dimensions_mod , only : np, nlev
276  use hybrid_mod , only : hybrid_t
277  use element_mod , only : element_t
278  use derivative_mod , only : derivative_t, divergence_sphere, gradient_sphere, vorticity_sphere, &
279  limiter_optim_iter_full, limiter_clip_and_sum
280  use bndry_mod , only : bndry_exchangev
281  use hybvcoord_mod , only : hvcoord_t
282  implicit none
283  integer , intent(in ) :: np1_qdp, n0_qdp
284  real (kind=real_kind), intent(in ) :: dt
285  type (element_t) , intent(inout), target :: elem(:)
286  type (hvcoord_t) , intent(in ) :: hvcoord
287  type (hybrid_t) , intent(in ) :: hybrid
288  type (derivative_t) , intent(in ) :: deriv
289  integer , intent(in ) :: nets
290  integer , intent(in ) :: nete
291  integer , intent(in ) :: DSSopt
292  integer , intent(in ) :: rhs_multiplier
293 
294  ! local
295  real(kind=real_kind), dimension(np,np ) :: divdp, dpdiss
296  real(kind=real_kind), dimension(np,np,nlev) :: dpdissk
297  real(kind=real_kind), dimension(np,np,2 ) :: gradq
298  real(kind=real_kind), dimension(np,np,2,nlev ) :: vstar
299  real(kind=real_kind), dimension(np,np ,nlev ) :: qtens
300  real(kind=real_kind), dimension(np,np ,nlev ) :: dp,dp_star
301  real(kind=real_kind), dimension(np,np ,nlev,qsize,nets:nete) :: qtens_biharmonic
302  real(kind=real_kind), pointer, dimension(:,:,:) :: dssvar
303  integer :: ie,q,i,j,k, kptr
304  integer :: rhs_viss
305 
306 ! call t_barrierf('sync_euler_step', hybrid%par%comm)
307 !pw call t_startf('euler_step')
308 
309  !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
310  ! compute Q min/max values for lim8
311  ! compute biharmonic mixing term f
312  !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
313  rhs_viss = 0
314  if ( limiter_option == 8 .or. limiter_option == 9 ) then
315  call t_startf('bihmix_qminmax')
316  ! when running lim8, we also need to limit the biharmonic, so that term needs
317  ! to be included in each euler step. three possible algorithms here:
318  ! 1) most expensive:
319  ! compute biharmonic (which also computes qmin/qmax) during all 3 stages
320  ! be sure to set rhs_viss=1
321  ! cost: 3 biharmonic steps with 3 DSS
322  !
323  ! 2) cheapest:
324  ! compute biharmonic (which also computes qmin/qmax) only on first stage
325  ! be sure to set rhs_viss=3
326  ! reuse qmin/qmax for all following stages (but update based on local qmin/qmax)
327  ! cost: 1 biharmonic steps with 1 DSS
328  ! main concern: viscosity
329  !
330  ! 3) compromise:
331  ! compute biharmonic (which also computes qmin/qmax) only on last stage
332  ! be sure to set rhs_viss=3
333  ! compute qmin/qmax directly on first stage
334  ! reuse qmin/qmax for 2nd stage stage (but update based on local qmin/qmax)
335  ! cost: 1 biharmonic steps, 2 DSS
336  !
337  ! NOTE when nu_p=0 (no dissipation applied in dynamics to dp equation), we should
338  ! apply dissipation to Q (not Qdp) to preserve Q=1
339  ! i.e. laplace(Qdp) ~ dp0 laplace(Q)
340  ! for nu_p=nu_q>0, we need to apply dissipation to Q * diffusion_dp
341  !
342  ! initialize dp, and compute Q from Qdp (and store Q in Qtens_biharmonic)
343  do ie = nets , nete
344  ! add hyperviscosity to RHS. apply to Q at timelevel n0, Qdp(n0)/dp
345 omp_simd
346  do k = 1 , nlev ! Loop index added with implicit inversion (AAM)
347  dp(:,:,k) = elem(ie)%derived%dp(:,:,k) - rhs_multiplier*dt*elem(ie)%derived%divdp_proj(:,:,k)
348  enddo
349 #if (defined COLUMN_OPENMP)
350 !$omp parallel do private(q,k) collapse(2)
351 #endif
352  do q = 1 , qsize
353  do k=1,nlev
354  qtens_biharmonic(:,:,k,q,ie) = elem(ie)%state%Qdp(:,:,k,q,n0_qdp)/dp(:,:,k)
355  if ( rhs_multiplier == 1 ) then
356  ! for this stage, we skip neighbor_minmax() call, but update
357  ! qmin/qmax with any new local extrema:
358  qmin(k,q,ie)=min(qmin(k,q,ie),minval(qtens_biharmonic(:,:,k,q,ie)))
359  qmax(k,q,ie)=max(qmax(k,q,ie),maxval(qtens_biharmonic(:,:,k,q,ie)))
360  else
361  ! for rhs_multiplier=0,2 we will call neighbor_minmax and compute
362  ! the correct min/max values
363  qmin(k,q,ie)=minval(qtens_biharmonic(:,:,k,q,ie))
364  qmax(k,q,ie)=maxval(qtens_biharmonic(:,:,k,q,ie))
365  endif
366  enddo
367  enddo
368  enddo
369 
370  ! compute element qmin/qmax
371  if ( rhs_multiplier == 0 ) then
372  ! update qmin/qmax based on neighbor data for lim8
373  call t_startf('eus_neighbor_minmax1')
374  call neighbor_minmax(hybrid,edgeadvqminmax,nets,nete,qmin(:,:,nets:nete),qmax(:,:,nets:nete))
375  call t_stopf('eus_neighbor_minmax1')
376  endif
377 
378  ! get niew min/max values, and also compute biharmonic mixing term
379  if ( rhs_multiplier == 2 ) then
380  rhs_viss = 3
381  ! two scalings depending on nu_p:
382  ! nu_p=0: qtens_biharmonic *= dp0 (apply viscsoity only to q)
383  ! nu_p>0): qtens_biharmonc *= elem()%psdiss_ave (for consistency, if nu_p=nu_q)
384  if ( nu_p > 0 ) then
385  do ie = nets , nete
386 #if (defined COLUMN_OPENMP)
387  !$omp parallel do private(k, q) collapse(2)
388 #endif
389  do k = 1 , nlev
390  do q = 1 , qsize
391  ! NOTE: divide by dp0 since we multiply by dp0 below
392  qtens_biharmonic(:,:,k,q,ie)=qtens_biharmonic(:,:,k,q,ie)&
393  *elem(ie)%derived%dpdiss_ave(:,:,k)/hvcoord%dp0(k)
394  enddo
395  enddo
396  enddo ! ie loop
397  endif ! nu_p > 0
398 
399 ! Previous version of biharmonic_wk_scalar_minmax included a min/max
400 ! calculation into the boundary exchange. This was causing cache issues.
401 ! Split the single operation into two separate calls
402 ! call neighbor_minmax()
403 ! call biharmonic_wk_scalar()
404 !
405 ! call biharmonic_wk_scalar_minmax( elem , qtens_biharmonic , deriv , edgeAdvQ3 , hybrid , &
406 ! nets , nete , qmin(:,:,nets:nete) , qmax(:,:,nets:nete) )
407  call neighbor_minmax_start(hybrid,edgeadvqminmax,nets,nete,qmin(:,:,nets:nete),qmax(:,:,nets:nete))
408  call biharmonic_wk_scalar(elem,qtens_biharmonic,deriv,edge_g,hybrid,nets,nete)
409  do ie = nets , nete
410 #if (defined COLUMN_OPENMP_notB4B)
411 !$omp parallel do private(k, q)
412 #endif
413  do q = 1 , qsize
414  do k = 1 , nlev ! Loop inversion (AAM)
415  ! note: biharmonic_wk() output has mass matrix already applied. Un-apply since we apply again below:
416  qtens_biharmonic(:,:,k,q,ie) = &
417  -rhs_viss*dt*nu_q*hvcoord%dp0(k)*qtens_biharmonic(:,:,k,q,ie) / elem(ie)%spheremp(:,:)
418  enddo
419  enddo
420  enddo
421  call neighbor_minmax_finish(hybrid,edgeadvqminmax,nets,nete,qmin(:,:,nets:nete),qmax(:,:,nets:nete))
422  endif
423  call t_stopf('bihmix_qminmax')
424  endif ! compute biharmonic mixing term and qmin/qmax
425  ! end of limiter_option == 8 .or. limiter_option == 9
426 
427  !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
428  ! 2D Advection step
429  !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
430  call t_startf('eus_2d_advec')
431  do ie = nets , nete
432  ! note: eta_dot_dpdn is actually dimension nlev+1, but nlev+1 data is
433  ! all zero so we only have to DSS 1:nlev
434  if ( dssopt == dsseta ) dssvar => elem(ie)%derived%eta_dot_dpdn(:,:,:)
435  if ( dssopt == dssomega ) dssvar => elem(ie)%derived%omega_p(:,:,:)
436  if ( dssopt == dssdiv_vdp_ave ) dssvar => elem(ie)%derived%divdp_proj(:,:,:)
437 
438  ! Compute velocity used to advance Qdp
439 #if (defined COLUMN_OPENMP)
440  !$omp parallel do private(k,q)
441 #endif
442  do k = 1 , nlev ! Loop index added (AAM)
443  ! derived variable divdp_proj() (DSS'd version of divdp) will only be correct on 2nd and 3rd stage
444  ! but that's ok because rhs_multiplier=0 on the first stage:
445  dp(:,:,k) = elem(ie)%derived%dp(:,:,k) - rhs_multiplier * dt * elem(ie)%derived%divdp_proj(:,:,k)
446  vstar(:,:,1,k) = elem(ie)%derived%vn0(:,:,1,k) / dp(:,:,k)
447  vstar(:,:,2,k) = elem(ie)%derived%vn0(:,:,2,k) / dp(:,:,k)
448 
449  if ( limiter_option == 8 .or. limiter_option == 9 ) then
450  ! Note that the term dpdissk is independent of Q
451  ! UN-DSS'ed dp at timelevel n0+1:
452  dpdissk(:,:,k) = dp(:,:,k) - dt * elem(ie)%derived%divdp(:,:,k)
453  if ( nu_p > 0 .and. rhs_viss /= 0 ) then
454  ! add contribution from UN-DSS'ed PS dissipation
455 ! dpdiss(:,:) = ( hvcoord%hybi(k+1) - hvcoord%hybi(k) ) *
456 ! elem(ie)%derived%psdiss_biharmonic(:,:)
457  dpdissk(:,:,k) = dpdissk(:,:,k) - rhs_viss * dt * nu_q &
458  * elem(ie)%derived%dpdiss_biharmonic(:,:,k) / elem(ie)%spheremp(:,:)
459  endif
460  ! IMPOSE ZERO THRESHOLD. do this here so it can be turned off for
461  ! testing
462  do q=1,qsize
463  qmin(k,q,ie)=max(qmin(k,q,ie),0d0)
464  enddo
465  endif ! limiter == 8 or 9
466 
467  ! also DSS extra field
468  dssvar(:,:,k) = elem(ie)%spheremp(:,:) * dssvar(:,:,k)
469  enddo
470  call edgevpack_nlyr( edge_g , elem(ie)%desc, dssvar(:,:,1:nlev) , nlev , nlev*qsize , nlev*(qsize+1))
471 
472  ! advance Qdp
473 #if (defined COLUMN_OPENMP)
474  !$omp parallel do private(q,k,gradQ,dp_star,qtens)
475 #endif
476  do q = 1 , qsize
477  do k = 1 , nlev ! dp_star used as temporary instead of divdp (AAM)
478  ! div( U dp Q),
479  gradq(:,:,1) = vstar(:,:,1,k) * elem(ie)%state%Qdp(:,:,k,q,n0_qdp)
480  gradq(:,:,2) = vstar(:,:,2,k) * elem(ie)%state%Qdp(:,:,k,q,n0_qdp)

528 omp_simd

531  enddo

538  do k = 1 , nlev ! Potential loop inversion (AAM)

555  subroutine limiter2d_zero(Q)
556  ! mass conserving zero limiter (2D only). to be called just before DSS
557  !
558  ! this routine is called inside a DSS loop, and so Q had already
559  ! been multiplied by the mass matrix. Thus dont include the mass
560  ! matrix when computing the mass = integral of Q over the element
561  !
562  ! ps is only used when advecting Q instead of Qdp
563  ! so ps should be at one timelevel behind Q
564  implicit none
565  real (kind=real_kind), intent(inout) :: q(np,np,nlev)
566 
567  ! local
568  real (kind=real_kind) :: dp(np,np)
569  real (kind=real_kind) :: mass,mass_new,ml
570  integer i,j,k
571 
572  do k = nlev , 1 , -1
573  mass = 0
574  do j = 1 , np
575  do i = 1 , np
576  !ml = Q(i,j,k)*dp(i,j)*spheremp(i,j) ! see above
577  ml = q(i,j,k)
578  mass = mass + ml
579  enddo
580  enddo
581 
582  ! negative mass. so reduce all postive values to zero
583  ! then increase negative values as much as possible
584  if ( mass < 0 ) q(:,:,k) = -q(:,:,k)
585  mass_new = 0
586  do j = 1 , np
587  do i = 1 , np
588  if ( q(i,j,k) < 0 ) then
589  q(i,j,k) = 0
590  else
591  ml = q(i,j,k)
592  mass_new = mass_new + ml
593  endif
594  enddo
595  enddo
596 
597  ! now scale the all positive values to restore mass
598  if ( mass_new > 0 ) q(:,:,k) = q(:,:,k) * abs(mass) / mass_new
599  if ( mass < 0 ) q(:,:,k) = -q(:,:,k)
600  enddo
601  end subroutine limiter2d_zero

606  subroutine advance_hypervis_scalar(elem, hvcoord , hybrid , deriv , nt , nt_qdp , nets , nete , dt2 )
607  ! hyperviscsoity operator for foward-in-time scheme
608  ! take one timestep of:
609  ! Q(:,:,:,np) = Q(:,:,:,np) + dt2*nu*laplacian**order ( Q )
610  !
611  ! For correct scaling, dt2 should be the same 'dt2' used in the leapfrog advace
612  use kinds , only : real_kind
613  use dimensions_mod , only : np, nlev
614  use hybrid_mod , only : hybrid_t
615  use element_mod , only : element_t
616  use derivative_mod , only : derivative_t
617  use bndry_mod , only : bndry_exchangev
618  use perf_mod , only : t_startf, t_stopf ! _EXTERNAL
619  implicit none
620  type (element_t) , intent(inout), target :: elem(:)
621  type (hvcoord_t) , intent(in ) :: hvcoord
622  type (hybrid_t) , intent(in ) :: hybrid
623  type (derivative_t) , intent(in ) :: deriv
624  integer , intent(in ) :: nt
625  integer , intent(in ) :: nt_qdp
626  integer , intent(in ) :: nets
627  integer , intent(in ) :: nete
628  real (kind=real_kind), intent(in ) :: dt2
629 
630  ! local
631  real (kind=real_kind), dimension(np,np,nlev,qsize,nets:nete) :: qtens
632  real (kind=real_kind), dimension(np,np,nlev ) :: dp
633  real (kind=real_kind) :: dt
634  integer :: k , i , j , ie , ic , q
635 
636  if ( nu_q == 0 ) return
637  if ( hypervis_order /= 2 ) return
638 ! call t_barrierf('sync_advance_hypervis_scalar', hybrid%par%comm)
639  call t_startf('advance_hypervis_scalar')
640 
641  !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
642  ! hyper viscosity
643  !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
644  dt = dt2 / hypervis_subcycle_q
645 
646  do ic = 1 , hypervis_subcycle_q
647  do ie = nets , nete
648  ! Qtens = Q/dp (apply hyperviscsoity to dp0 * Q, not Qdp)
649  ! various options:
650  ! 1) biharmonic( Qdp )
651  ! 2) dp0 * biharmonic( Qdp/dp )
652  ! 3) dpave * biharmonic(Q/dp)
653  ! For trace mass / mass consistenciy, we use #2 when nu_p=0
654  ! and #e when nu_p>0, where dpave is the mean mass flux from the nu_p
655  ! contribution from dynamics.
656 
657  if (nu_p>0) then
658 #if (defined COLUMN_OPENMP)
659 !$omp parallel do private(q,k) collapse(2)
660 #endif
661  do q = 1 , qsize
662  do k = 1 , nlev
663  dp(:,:,k) = elem(ie)%derived%dp(:,:,k) - dt2*elem(ie)%derived%divdp_proj(:,:,k)
664  qtens(:,:,k,q,ie) = elem(ie)%derived%dpdiss_ave(:,:,k)*&
665  elem(ie)%state%Qdp(:,:,k,q,nt_qdp) / dp(:,:,k)
666  enddo
667  enddo
668 
669  else
670 #if (defined COLUMN_OPENMP)
671 !$omp parallel do private(q,k) collapse(2)
672 #endif
673  do q = 1 , qsize
674  do k = 1 , nlev
675  dp(:,:,k) = elem(ie)%derived%dp(:,:,k) - dt2*elem(ie)%derived%divdp_proj(:,:,k)
676  qtens(:,:,k,q,ie) = hvcoord%dp0(k)*elem(ie)%state%Qdp(:,:,k,q,nt_qdp) / dp(:,:,k)
677  enddo
678  enddo
679  endif
680  enddo ! ie loop
681 
682  ! compute biharmonic operator. Qtens = input and output
683  call biharmonic_wk_scalar( elem , qtens , deriv , edge_g , hybrid , nets , nete )
684 
685  do ie = nets , nete
686 #if (defined COLUMN_OPENMP)
687 !$omp parallel do private(q,k,j,i)
688 #endif
689  do q = 1 , qsize
690  do k = 1 , nlev
691  do j = 1 , np
692  do i = 1 , np
693  ! advection Qdp. For mass advection consistency:
694  ! DIFF( Qdp) ~ dp0 DIFF (Q) = dp0 DIFF ( Qdp/dp )
695  elem(ie)%state%Qdp(i,j,k,q,nt_qdp) = elem(ie)%state%Qdp(i,j,k,q,nt_qdp) * elem(ie)%spheremp(i,j) &
696  - dt * nu_q * qtens(i,j,k,q,ie)
697  enddo
698  enddo
699  enddo
700 
701  if (limiter_option .ne. 0 ) then
702  ! smooth some of the negativities introduced by diffusion:
703  call limiter2d_zero( elem(ie)%state%Qdp(:,:,:,q,nt_qdp) )
704  endif
705 
706  enddo
707  call edgevpack_nlyr(edge_g , elem(ie)%desc, elem(ie)%state%Qdp(:,:,:,:,nt_qdp) , qsize*nlev , 0 , qsize*nlev )
708  enddo ! ie loop
709 
710  call t_startf('ah_scalar_bexchV')
711  call bndry_exchangev( hybrid , edge_g )
712  call t_stopf('ah_scalar_bexchV')
713 
714  do ie = nets , nete
715  call edgevunpack_nlyr(edge_g , elem(ie)%desc, elem(ie)%state%Qdp(:,:,:,:,nt_qdp) , qsize*nlev , 0, qsize*nlev)
716 #if (defined COLUMN_OPENMP)
717 !$omp parallel do private(q,k) collapse(2)
718 #endif
719  do q = 1 , qsize
720  ! apply inverse mass matrix
721  do k = 1 , nlev
722  elem(ie)%state%Qdp(:,:,k,q,nt_qdp) = elem(ie)%rspheremp(:,:) * elem(ie)%state%Qdp(:,:,k,q,nt_qdp)
723  enddo
724  enddo
725  enddo ! ie loop
726 #ifdef DEBUGOMP
727 #if (defined HORIZ_OPENMP)
728 !$OMP BARRIER
729 #endif
730 #endif
731  enddo
732  call t_stopf('advance_hypervis_scalar')
733  end subroutine advance_hypervis_scalar

740  subroutine advance_physical_vis(elem,hvcoord,hybrid,deriv,nt,nt_qdp,nets,nete,dt,mu)
741  !
742  ! take one timestep of of physical viscosity (single laplace operator) for
743  ! all tracers in both horizontal and vertical
744  !
745  ! as of 2017/5, used only for the supercell test case
746  ! so for now:
747  ! dont bother to optimize
748  ! apply only to perturbation from background state (supercell initial condition)
749  ! uniform spacing in z with delz = 20km/nlev
750  !
751  !
752 
753  use kinds , only : real_kind
754  use dimensions_mod , only : np, nlev
755  use hybrid_mod , only : hybrid_t
756  use element_mod , only : element_t
757  use element_ops , only : state0
758  use derivative_mod , only : derivative_t, laplace_z, laplace_sphere_wk
759  use bndry_mod , only : bndry_exchangev
760  use perf_mod , only : t_startf, t_stopf ! _EXTERNAL
761  implicit none
762  type (element_t) , intent(inout), target :: elem(:)
763  type (hvcoord_t) , intent(in ) :: hvcoord
764  type (hybrid_t) , intent(in ) :: hybrid
765  type (derivative_t) , intent(in ) :: deriv
766  integer , intent(in ) :: nt
767  integer , intent(in ) :: nt_qdp
768  integer , intent(in ) :: nets
769  integer , intent(in ) :: nete
770  real (kind=real_kind), intent(in ) :: dt
771  real (kind=real_kind), intent(in ) :: mu
772 
773  ! local
774  real (kind=real_kind), dimension(np,np,nlev ) :: q_prime,qt
775  real (kind=real_kind) :: dp0, delz
776  integer :: k , i,j,ie,ic,q
777 
778  !if(test_case .ne. 'dcmip2016_test3') call abortmp("dcmip16_mu is currently limited to dcmip16 test 3")
779 
780  !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
781  ! hyper viscosity
782  !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
783  delz=20d3/nlev
784 
785  do ie=nets,nete
786 
787  do q=1,qsize
788  do k=1,nlev
789  qt(:,:,k) =hvcoord%dp0(k)*elem(ie)%state%Qdp(:,:,k,q,nt_qdp)/elem(ie)%state%dp3d(:,:,k,nt)
790  enddo
791  do k=1,nlev
792  ! timestep
793  elem(ie)%state%Qdp(:,:,k,q,nt_qdp)=elem(ie)%state%Qdp(:,:,k,q,nt_qdp)*elem(ie)%spheremp(:,:) &
794  + mu*dt*laplace_sphere_wk(qt(:,:,k),deriv,elem(ie),var_coef=.false.)
795  enddo
796 
797  if (limiter_option .ne. 0 ) then
798  ! smooth some of the negativities introduced by diffusion:
799  call limiter2d_zero( elem(ie)%state%Qdp(:,:,:,q,nt_qdp) )
800  endif
801 
802  enddo
803  call edgevpack_nlyr(edge_g,elem(ie)%desc,elem(ie)%state%Qdp(:,:,:,:,nt_qdp),qsize*nlev,0,qsize*nlev)
804  enddo ! ie loop
805 
806  call bndry_exchangev( hybrid,edge_g )
807 
808  do ie=nets,nete
809  call edgevunpack_nlyr(edge_g,elem(ie)%desc, elem(ie)%state%Qdp(:,:,:,:,nt_qdp),qsize*nlev,0,qsize*nlev)
810  do q=1,qsize
811  ! apply inverse mass matrix
812  do k=1,nlev
813  elem(ie)%state%Qdp(:,:,k,q,nt_qdp)=elem(ie)%rspheremp(:,:) * elem(ie)%state%Qdp(:,:,k,q,nt_qdp)
814  enddo
815 
816  ! apply vertical viscosity
817  do k=1,nlev
818  q_prime(:,:,k) =hvcoord%dp0(k)* ( &
819  elem(ie)%state%Qdp(:,:,k,q,nt_qdp)/elem(ie)%state%dp3d(:,:,k,nt) - &
820  state0(ie)%Qdp(:,:,k,q,1)/state0(ie)%dp3d(:,:,k,1) )
821  enddo
822  call laplace_z(q_prime(:,:,:),qt(:,:,:),1,nlev,delz)
823  ! apply mass matrix and add in horiz laplacian (which has mass matrix built in)
824  do k=1,nlev
825  elem(ie)%state%Qdp(:,:,k,q,nt_qdp)=elem(ie)%state%Qdp(:,:,k,q,nt_qdp) + mu*dt*qt(:,:,k)
826  ! crude sign preserving limiter
827  where(elem(ie)%state%Qdp(:,:,k,q,nt_qdp) < 0 )
828  elem(ie)%state%Qdp(:,:,k,q,nt_qdp)=0
829  end where
830  enddo
831 
832 
833  enddo ! q loop
834  enddo ! ie loop
