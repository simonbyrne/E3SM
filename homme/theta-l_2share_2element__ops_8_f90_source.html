401  subroutine set_thermostate(elem,ps,temperature,hvcoord,qv)
402  !
403  ! Assuming a hydrostatic intital state and given surface pressure,
404  ! and no moisture, compute theta and phi
405  !
406  ! input: ps_v, temperature
407  ! ouput: state variables: vtheta_dp, phi
408  !
409  implicit none
410 
411  type (element_t), intent(inout) :: elem
412  real (kind=real_kind), intent(in) :: temperature(np,np,nlev)
413  type (hvcoord_t), intent(in) :: hvcoord ! hybrid vertical coordinate struct
414  real (kind=real_kind), intent(in) :: ps(np,np)
415  real (kind=real_kind), intent(in), optional :: qv(np,np,nlev) ! water vapor mixing ratio
416 
417  ! local
418  real (kind=real_kind) :: p(np,np,nlev)
419  real (kind=real_kind) :: pi_i(np,np,nlevp)
420  real (kind=real_kind) :: dp(np,np,nlev)
421  real (kind=real_kind) :: rstar(np,np,nlev)
422  integer :: k, nt
423 
424  nt = 1
425  do k=1,nlev
426  p(:,:,k) = hvcoord%hyam(k)*hvcoord%ps0 + hvcoord%hybm(k)*ps(:,:)
427  dp(:,:,k) = ( hvcoord%hyai(k+1) - hvcoord%hyai(k) )*hvcoord%ps0 + &
428  ( hvcoord%hybi(k+1) - hvcoord%hybi(k) )*ps(:,:)
429  enddo
430 
431 ! recompute pressure using same algrebra as EOS:
432 ! improves precision for initial T/theta conversion
433  pi_i(:,:,1)=hvcoord%hyai(1)*hvcoord%ps0
434  do k=1,nlev
435  pi_i(:,:,k+1)=pi_i(:,:,k) + dp(:,:,k)
436  enddo
437  do k=1,nlev
438  p(:,:,k)=pi_i(:,:,k) + dp(:,:,k)/2
439  enddo
440 
441 
442 !set vtheta
443  do k=1,nlev
444  elem%state%vtheta_dp(:,:,k,nt)=dp(:,:,k)*temperature(:,:,k)* &
445  (p(:,:,k)/p0)**(-kappa)
446  elem%state%dp3d(:,:,k,nt)=dp(:,:,k)
447  enddo
448  elem%state%ps_v(:,:,nt)=ps
449 
450  if (present(qv)) then
451  call get_r_star(rstar,qv)
452  elem%state%vtheta_dp(:,:,:,nt)=elem%state%vtheta_dp(:,:,:,nt)*rstar(:,:,:)/rgas
453  endif
454 
455 
456  !set phi, copy from 1st timelevel to all
457  call tests_finalize(elem,hvcoord)
458 
459 
460  ! verify T
461  call get_temperature(elem,p,hvcoord,nt)
462 
463  dp=p-temperature
464  do k=1,nlev
465  if (maxval(abs(dp(:,:,k))) > 1e-9) then
466  write(iulog,*)'WARNING: T/theta initialization inconsistent!'
467  write(iulog,*)k,minval(dp(:,:,k)),maxval(dp(:,:,k))
468  write(iulog,*) 'T,Tnew',temperature(1,1,k),p(1,1,k)
469  endif
470  enddo
471 
472  end subroutine set_thermostate

548  subroutine get_state(u,v,w,T,pnh,dp,ps,rho,zm,zi,g,elem,hvcoord,nt,ntQ)
549  ! get state variables at layer midpoints
550  ! used by idealized tests to compute idealized physics forcing terms
551  ! currently all forcing is done on u,v and T/theta - no forcing
552  ! is computed for interface variables. This routine will have to be updated
553  ! if we add a test case that computes forcing for interface variables
554 
555  real(real_kind), dimension(np,np,nlev), intent(inout) :: u,v,w,t,pnh,dp,zm,rho

627  subroutine set_forcing_rayleigh_friction(elem, zm, zi, ztop, zc, tau, u0,v0, n)
628  !
629  ! test cases which use rayleigh friciton will call this with the relaxation coefficient
630  ! f_d, and the reference state u0,v0. Currently assume w0 = 0
631  !
632  implicit none
633 
634  type(element_t), intent(inout):: elem
635  real(real_kind), intent(in) :: zm(np,np,nlev) ! height at layer midpoints
636  real(real_kind), intent(in) :: zi(np,np,nlevp) ! height at interfaces
637  real(real_kind), intent(in) :: ztop ! top of atm height
638  real(real_kind), intent(in) :: zc ! cutoff height
639  real(real_kind), intent(in) :: tau ! damping timescale
640  real(real_kind), intent(in) :: u0(np,np,nlev) ! reference u
641  real(real_kind), intent(in) :: v0(np,np,nlev) ! reference v
642  integer, intent(in) :: n ! timestep
643 
644  real(real_kind):: f_d(np,np,nlev)
645  integer :: k
646 
647  ! Compute damping as a function of layer-midpoint height
648  f_d=0.0d0
649  where(zm .ge. zc); f_d = sin(dd_pi/2 *(zm - zc)/(ztop - zc))**2; end where
650  where(zm .ge. ztop); f_d = 1.0d0; end where
651  f_d = -f_d/tau
652 
653  elem%derived%FM(:,:,1,:) = f_d * ( elem%state%v(:,:,1,:,n) - u0 )
654  elem%derived%FM(:,:,2,:) = f_d * ( elem%state%v(:,:,2,:,n) - v0)
655 
656  ! compute forcing for 1:nlev. w is determined by boundary condition at nlevp:
657  f_d=0.0d0
658  where(zi(:,:,1:nlev) .ge. zc); f_d = sin(dd_pi/2 *(zm - zc)/(ztop - zc))**2; end where
659  where(zi(:,:,1:nlev) .ge. ztop); f_d = 1.0d0; end where
660  f_d = -f_d/tau
661  elem%derived%FM(:,:,3,:) = f_d * ( elem%state%w_i(:,:,1:nlev,n) )
662  end subroutine
