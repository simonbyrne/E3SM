215  KOKKOS_INLINE_FUNCTION void compute_energy_grad(KernelVariables &kv) const {
216  Kokkos::parallel_for(Kokkos::TeamThreadRange(kv.team, NP * NP),
217  [&](const int idx) {
218  const int igp = idx / NP;
219  const int jgp = idx % NP;
220 
221  Kokkos::parallel_for(Kokkos::ThreadVectorRange(kv.team, NUM_LEV), [&] (const int& ilev) {
222  // pre-fill energy_grad with the pressure(_grad)-temperature part
223  m_buffers.energy_grad(kv.team_idx, 0, igp, jgp, ilev) =
224  PhysicalConstants::Rgas *
225  (m_buffers.temperature_virt(kv.team_idx, igp, jgp, ilev) /
226  m_buffers.pressure(kv.team_idx, igp, jgp, ilev)) *
227  m_buffers.pressure_grad(kv.team_idx, 0, igp, jgp, ilev);
228 
229  m_buffers.energy_grad(kv.team_idx, 1, igp, jgp, ilev) =
230  PhysicalConstants::Rgas *
231  (m_buffers.temperature_virt(kv.team_idx, igp, jgp, ilev) /
232  m_buffers.pressure(kv.team_idx, igp, jgp, ilev)) *
233  m_buffers.pressure_grad(kv.team_idx, 1, igp, jgp, ilev);
234 
235  // Kinetic energy + PHI (geopotential energy) +
236  Scalar k_energy =
237  0.5 * (m_state.m_v(kv.ie, m_data.n0, 0, igp, jgp, ilev) *
238  m_state.m_v(kv.ie, m_data.n0, 0, igp, jgp, ilev) +
239  m_state.m_v(kv.ie, m_data.n0, 1, igp, jgp, ilev) *
240  m_state.m_v(kv.ie, m_data.n0, 1, igp, jgp, ilev));
241  m_buffers.ephi(kv.team_idx, igp, jgp, ilev) += k_energy;
242  });
243  });
244  kv.team_barrier();
245 
246  m_sphere_ops.gradient_sphere_update(kv,
247  Homme::subview(m_buffers.ephi, kv.team_idx),
248  Homme::subview(m_buffers.energy_grad, kv.team_idx));
