263  subroutine coordinates_atomic(elem,gll_points)

271 
272  ! compute the corners in Cartesian coordinates

276  elem%corners3D(i)%y=sy + elem%corners(i)%y * ly
277  elem%corners3D(i)%z=0.0d0
278  enddo
279

282  ! lat=y, lon=x, r=z
283  ! =========================================
284 
285  do i=1,np
286  do j=1,np
287  ! this converts [-1,1] GLL points to [0, dx_ref] or [0, dy_ref], and then adds corner coords
288  elem%cartp(i,j)%x= gll_points(i) * dx_ref/2.0d0 + dx_ref/2.0d0 + elem%corners(1)%x
289  elem%cartp(i,j)%y= gll_points(j) * dy_ref/2.0d0 + dy_ref/2.0d0 + elem%corners(1)%y
290 
291  ! This maps [0,1] to [Sx, Lx + Sx] or [Sy, Ly + Sy]
292  elem%spherep(i,j)%lon = sx + elem%cartp(i,j)%x * lx
293  elem%spherep(i,j)%lat = sy + elem%cartp(i,j)%y * ly
294  elem%spherep(i,j)%r= 0.0d0
295  enddo
296  enddo

302  elem%vec_sphere2cart(:,:,2,1) = 0.0_real_kind
303  elem%vec_sphere2cart(:,:,3,1) = 0.0_real_kind

312  subroutine dmap(D, a,b, corners3D, ref_map, cartp, facenum)
313  use physical_constants, only: dx, dy
314  real (kind=real_kind), intent(out) :: d(2,2)
315  real (kind=real_kind), intent(in) :: a,b
316  type (cartesian3d_t) :: corners3d(4) !x,y,z coords of element corners
317  integer :: ref_map
318  ! only needed for ref_map=0,1
319  type (cartesian2d_t),optional :: cartp(np,np) ! gnomonic coords of element corners
320  integer,optional :: facenum
321 
322 ! factor 1/2 required since HOMME reference element is [-1,1]^2 instead of [0,1]
323 ! this is composition of 2 maps:
324 ! a map from [-1,1]^2 reference ELEMENT to [0,1]^2 reference DOMAIN composed of [ne_x, ne_y] elements
325 ! a map from [0,1]^2 reference DOMAIN to physical [Sx,Lx+Sx] x [Sy,Ly+Sy] DOMAIN
326  d(1,1) = dx/2.0d0
327  d(1,2) = 0.0d0
328  d(2,1) = 0.0d0
329  d(2,2) = dy/2.0d0
330 
331  end subroutine dmap

333  ! =========================================
334  ! metric_atomic:
335  !
336  ! Initialize planar metric terms:
337  ! initialize:
338  ! metdet, rmetdet (analytic) = detD, 1/detD
339  ! met (analytic) D^t D (symmetric)
340  ! metdet (analytic) = detD
341  ! metinv (analytic) Dinv Dinv^t (symmetic)
342  ! D (from subroutine vmap)
343  ! Dinv (computed directly from D)
344  !
345  ! ucontra = Dinv * u = metinv * ucov
346  ! ucov = D^t * u = met * ucontra
347  !
348  ! we also compute DE = D*E, where
349  ! E = eigenvectors of metinv as a basis metinv = E LAMBDA E^t
350  !
351  ! ueig = E^t ucov = E^t D^t u = (DE)^t u
352  !
353  !
354  ! so if we want to tweak the mapping by a factor alpha (so the weights add up to domain size, for example)
355  ! we take:
356  ! NEW OLD
357  ! D = sqrt(alpha) D and then rederive all quantities.
358  ! detD = alpha detD
359  !
360  ! where alpha = domain size/SEMarea, SEMarea = global sum elem(ie)%mv(i,j)*elem(ie)%metdet(i,j)
361  !
362  ! =========================================
363 
364  subroutine metric_atomic(elem,gll_points,alpha)
365  use element_mod, only : element_t
366  use physical_constants, only: lx, ly, dx, dy
367 
368  type (element_t) :: elem
369  real(kind=real_kind) :: alpha
370  real (kind=longdouble_kind) :: gll_points(np)
371  ! Local variables
372  integer ii
373  integer i,j,nn
374  integer iptr
375 
376  real (kind=real_kind) :: r ! distance from origin for point on cube tangent to unit sphere
377 
378  real (kind=real_kind) :: const, norm
379  real (kind=real_kind) :: detd ! determinant of vector field mapping matrix.
380  real (kind=real_kind) :: tmpd(2,2)
381  real (kind=real_kind) :: m(2,2),e(2,2),eig(2),de(2,2),del(2,2),v(2,2), nu1, nu2, lamstar1, lamstar2
382  integer :: imaxm(2)
383  real (kind=real_kind) :: l1, l2, sc,min_svd,max_svd,max_normdinv
384 
385  ! ==============================================
386  ! Initialize differential mapping operator
387  ! to and from vector fields on the physical domain to
388  ! contravariant vector fields on the reference domain
389  ! i.e. dM/dx^i in Sadourney (1972) and it's
390  ! inverse
391  ! ==============================================
392 
393  max_svd = 0.0d0
394  max_normdinv = 0.0d0
395  min_svd = 1d99
396  do j=1,np
397  do i=1,np
398 
399  call dmap(elem%D(i,j,:,:),1.0d0,1.0d0,elem%corners3D,cubed_sphere_map)
400 
401  ! Numerical metric tensor based on analytic D: met = D^T times D
402  ! (D maps between physical plane and reference element)
403  elem%met(i,j,1,1) = elem%D(i,j,1,1)*elem%D(i,j,1,1) + &
404  elem%D(i,j,2,1)*elem%D(i,j,2,1)
405  elem%met(i,j,1,2) = elem%D(i,j,1,1)*elem%D(i,j,1,2) + &
406  elem%D(i,j,2,1)*elem%D(i,j,2,2)
407  elem%met(i,j,2,1) = elem%D(i,j,1,1)*elem%D(i,j,1,2) + &
408  elem%D(i,j,2,1)*elem%D(i,j,2,2)
409  elem%met(i,j,2,2) = elem%D(i,j,1,2)*elem%D(i,j,1,2) + &
410  elem%D(i,j,2,2)*elem%D(i,j,2,2)
411 
412  ! compute D^-1...
413  ! compute determinant of D mapping matrix... if not zero compute inverse
414 
415  detd = elem%D(i,j,1,1)*elem%D(i,j,2,2) - elem%D(i,j,1,2)*elem%D(i,j,2,1)
416 
417  elem%Dinv(i,j,1,1) = elem%D(i,j,2,2)/detd
418  elem%Dinv(i,j,1,2) = -elem%D(i,j,1,2)/detd
419  elem%Dinv(i,j,2,1) = -elem%D(i,j,2,1)/detd
420  elem%Dinv(i,j,2,2) = elem%D(i,j,1,1)/detd
421 
422  ! L2 norm = sqrt max eigenvalue of metinv
423  ! = 1/sqrt(min eigenvalue of met)
424  ! l1 and l2 are eigenvalues of met
425  ! (should both be positive, l1 > l2)
426  l1 = (elem%met(i,j,1,1) + elem%met(i,j,2,2) + sqrt(4.0d0*elem%met(i,j,1,2)*elem%met(i,j,2,1) + &
427  (elem%met(i,j,1,1) - elem%met(i,j,2,2))**2))/2.0d0
428  l2 = (elem%met(i,j,1,1) + elem%met(i,j,2,2) - sqrt(4.0d0*elem%met(i,j,1,2)*elem%met(i,j,2,1) + &
429  (elem%met(i,j,1,1) - elem%met(i,j,2,2))**2))/2.0d0
430  ! Max L2 norm of Dinv is sqrt of max eigenvalue of metinv
431  ! max eigenvalue of metinv is 1/min eigenvalue of met
432  norm = 1.0d0/sqrt(min(abs(l1),abs(l2)))
433  max_svd = max(norm, max_svd)
434  ! Min L2 norm of Dinv is sqrt of min eigenvalue of metinv
435  ! min eigenvalue of metinv is 1/max eigenvalue of met
436  norm = 1.0d0/sqrt(max(abs(l1),abs(l2)))
437  min_svd = min(norm, min_svd)
438 
439  ! some kind of pseudo-norm of Dinv
440  ! C = 1/sqrt(2) sqrt( |g^x|^2 + |g^y|^2 + 2*|g^x dot g^y|)
441  ! = 1/sqrt(2) sqrt( |g_x|^2 + |g_y|^2 + 2*|g_x dot g_y|) / J
442  ! g^x = Dinv(:,1) g_x = D(1,:)
443  ! g^y = Dinv(:,2) g_y = D(2,:)
444  norm = (2*abs(sum(elem%Dinv(i,j,:,1)*elem%Dinv(i,j,:,2))) + sum(elem%Dinv(i,j,:,1)**2) + sum(elem%Dinv(i,j,:,2)**2))
445  norm = sqrt(norm)
446 ! norm = (2*abs(sum(elem%D(1,:,i,j)*elem%D(2,:,i,j))) + sum(elem%D(1,:,i,j)**2) + sum(elem%D(2,:,i,j)**2))
447 ! norm = sqrt(norm)/detD
448  max_normdinv = max(norm,max_normdinv)
449 
450 
451  ! Need inverse of met if not calculated analytically
452  elem%metdet(i,j) = abs(detd)
453  elem%rmetdet(i,j) = 1.0d0/abs(detd)
454 
455  elem%metinv(i,j,1,1) = elem%met(i,j,2,2)/(detd*detd)
456  elem%metinv(i,j,1,2) = -elem%met(i,j,1,2)/(detd*detd)
457  elem%metinv(i,j,2,1) = -elem%met(i,j,2,1)/(detd*detd)
458  elem%metinv(i,j,2,2) = elem%met(i,j,1,1)/(detd*detd)
459 
460  ! matricies for tensor hyper-viscosity
461  ! compute eigenvectors of metinv (probably same as computed above)
462  m = elem%metinv(i,j,:,:)
463 
464  eig(1) = (m(1,1) + m(2,2) + sqrt(4.0d0*m(1,2)*m(2,1) + &
465  (m(1,1) - m(2,2))**2))/2.0d0
466  eig(2) = (m(1,1) + m(2,2) - sqrt(4.0d0*m(1,2)*m(2,1) + &
467  (m(1,1) - m(2,2))**2))/2.0d0
468 
469  ! use DE to store M - Lambda, to compute eigenvectors
470  de=m
471  de(1,1)=de(1,1)-eig(1)
472  de(2,2)=de(2,2)-eig(1)
473 
474  imaxm = maxloc(abs(de))
475  if (maxval(abs(de))==0) then
476  e(1,1)=1; e(2,1)=0;
477  elseif ( imaxm(1)==1 .and. imaxm(2)==1 ) then
478  e(2,1)=1; e(1,1) = -de(2,1)/de(1,1)
479  else if ( imaxm(1)==1 .and. imaxm(2)==2 ) then
480  e(2,1)=1; e(1,1) = -de(2,2)/de(1,2)
481  else if ( imaxm(1)==2 .and. imaxm(2)==1 ) then
482  e(1,1)=1; e(2,1) = -de(1,1)/de(2,1)
483  else if ( imaxm(1)==2 .and. imaxm(2)==2 ) then
484  e(1,1)=1; e(2,1) = -de(1,2)/de(2,2)
485  else
486  call abortmp('Impossible error in planar_mod.F90::metric_atomic()')
487  endif
488 
489  ! the other eigenvector is orthgonal:
490  e(1,2)=-e(2,1)
491  e(2,2)= e(1,1)
492 
493 !normalize columns
494  e(:,1)=e(:,1)/sqrt(sum(e(:,1)*e(:,1)));
495  e(:,2)=e(:,2)/sqrt(sum(e(:,2)*e(:,2)));
496 
497 
498 ! OBTAINING TENSOR FOR HV: follows same approach as in cube_mod, with spherical-specific scaling removed
499 
500 !matrix D*E
501  de(1,1)=sum(elem%D(i,j,1,:)*e(:,1))
502  de(1,2)=sum(elem%D(i,j,1,:)*e(:,2))
503  de(2,1)=sum(elem%D(i,j,2,:)*e(:,1))
504  de(2,2)=sum(elem%D(i,j,2,:)*e(:,2))
505 
506  lamstar1=1/(eig(1)**(hypervis_scaling/4.0d0))
507  lamstar2=1/(eig(2)**(hypervis_scaling/4.0d0))
508 
509 !matrix (DE) * Lam^* * Lam , tensor HV when V is applied at each Laplace calculation
510 ! DEL(1:2,1) = lamStar1*eig(1)*DE(1:2,1)
511 ! DEL(1:2,2) = lamStar2*eig(2)*DE(1:2,2)
512 
513 !matrix (DE) * (Lam^*)^2 * Lam, tensor HV when V is applied only once, at the last Laplace calculation
514 !will only work with hyperviscosity, not viscosity
515  del(1:2,1) = (lamstar1**2) *eig(1)*de(1:2,1)
516  del(1:2,2) = (lamstar2**2) *eig(2)*de(1:2,2)
517

529 ! see Paul Ullrich writeup:
530 ! max_normDinv might be a tighter bound than max_svd for deformed elements
531 ! max_svd >= max_normDinv/sqrt(2), with equality holding if |g^x| = |g^y|
532 ! elem%normDinv=max_normDinv/sqrt(2)
533

535  elem%normDinv=max_svd
536

539  elem%dx_short = 1.0d0/(max_svd*0.5d0*dble(np-1)*1000.0d0)
540  elem%dx_long = 1.0d0/(min_svd*0.5d0*dble(np-1)*1000.0d0)

542  ! Area correction: Bring numerical area from integration weights to
543  ! agreement with geometric area.
544  ! Three different cases:
545  ! (1) alpha == 1, this means that cube_init_atomic wasn't
546  ! called with alpha parameter there will be no correction,
547  ! (2) alpha <> 1 and cubed_sphere_map=0 and correction is so-called
548  ! 'alpha-correction',
549  ! (3) alpha <> 1 and cubed_sphere_map=2 and it is 'epsilon-bubble'
550  ! correction.
551 
552  if( cubed_sphere_map == 0 ) then
553  ! alpha correction for cases (1) and (2).
554  elem%D = elem%D * sqrt(alpha)
555  elem%Dinv = elem%Dinv / sqrt(alpha)
556  elem%metdet = elem%metdet * alpha
557  ! replace "elem%rmetdet = elem%rmetdet / alpha" with the one below,
558  ! to ensure that elem%rmetdet = 1/elem%metdet
559  ! elem%rmetdet = elem%rmetdet / alpha
560  elem%rmetdet = 1.0d0/elem%metdet
561  elem%met = elem%met * alpha
562  elem%metinv = elem%metinv / alpha
563  elseif( cubed_sphere_map == 2 ) then
564  ! eps bubble correction for case (3).
565  do j=2,np-1
566  do i=2,np-1
567  elem%D(i,j,:,:) = elem%D(i,j,:,:) * sqrt(alpha)
568  elem%Dinv(i,j,:,:) = elem%Dinv(i,j,:,:) / sqrt(alpha)
569  elem%metdet(i,j) = elem%metdet(i,j) * alpha
570  elem%rmetdet(i,j) = 1.0d0/elem%metdet(i,j)
571  elem%met(i,j,:,:) = elem%met(i,j,:,:) * alpha
572  elem%metinv(i,j,:,:) = elem%metinv(i,j,:,:) / alpha
573  enddo
574  enddo
575  endif ! end of alpha/eps. bubble correction
