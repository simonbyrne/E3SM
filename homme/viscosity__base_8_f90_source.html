75 subroutine biharmonic_wk_scalar(elem,qtens,deriv,edgeq,hybrid,nets,nete)
76 !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
77 ! compute weak biharmonic operator
78 ! input: qtens = Q
79 ! output: qtens = weak biharmonic of Q
80 !
81 !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
82 type (hybrid_t) , intent(in) :: hybrid
83 type (element_t) , intent(inout), target :: elem(:)
84 integer :: nets,nete
85 real(kind=real_kind), dimension(np,np,nlev,qsize,nets:nete) :: qtens
86 type (EdgeBuffer_t) , intent(inout) :: edgeq
87 type (derivative_t) , intent(in) :: deriv
88 
89 ! local
90 integer :: k,kptr,i,j,ie,ic,q
91 real(kind=real_kind), dimension(np,np) :: lap_p
92 logical var_coef1
93 
94  !if tensor hyperviscosity with tensor V is used, then biharmonic operator is (\grad\cdot V\grad) (\grad \cdot \grad)
95  !so tensor is only used on second call to laplace_sphere_wk
96  var_coef1 = .true.
97  if(hypervis_scaling > 0) var_coef1 = .false.
98 
99 
100 
101  do ie=nets,nete
102 #if (defined COLUMN_OPENMP)
103 !$omp parallel do private(k, q, lap_p)
104 #endif
105  do q=1,qsize
106  do k=1,nlev ! Potential loop inversion (AAM)
107  lap_p(:,:)=qtens(:,:,k,q,ie)
108 ! Original use of qtens on left and right hand sides caused OpenMP errors (AAM)
109  qtens(:,:,k,q,ie)=laplace_sphere_wk(lap_p,deriv,elem(ie),var_coef=var_coef1)
110  enddo
111  call edgevpack_nlyr(edgeq, elem(ie)%desc, qtens(:,:,:,q,ie),nlev,nlev*(q-1),qsize*nlev)
112  enddo
113  enddo
114 
115  call t_startf('biwksc_bexchV')
116  call bndry_exchangev(hybrid,edgeq)
117  call t_stopf('biwksc_bexchV')
118 
119  do ie=nets,nete
120 
121  ! apply inverse mass matrix, then apply laplace again
122 #if (defined COLUMN_OPENMP)
123 !$omp parallel do private(k, q, lap_p)
124 #endif
125  do q=1,qsize
126  call edgevunpack_nlyr(edgeq,elem(ie)%desc,qtens(:,:,:,q,ie),nlev,nlev*(q-1),qsize*nlev)
127  do k=1,nlev ! Potential loop inversion (AAM)
128  lap_p(:,:)=elem(ie)%rspheremp(:,:)*qtens(:,:,k,q,ie)
129  qtens(:,:,k,q,ie)=laplace_sphere_wk(lap_p,deriv,elem(ie),var_coef=.true.)
130  enddo
131  enddo
132  enddo
133 #ifdef DEBUGOMP
134 #if (defined HORIZ_OPENMP)
135 !$OMP BARRIER
136 #endif
137 #endif
138 !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
139 end subroutine

653  enddo
654  enddo

657  enddo
658  if (mod(order_max,2)==0) pstens=-pstens

661  ! ps(t+1) = ps(t) + Minv * DSS * M * RHS
662  ! ps(t+1) = Minv * DSS * M [ ps(t) + RHS ]
663  ! but output of biharminc_wk is of the form M*RHS. rewrite as:
664  ! ps(t+1) = Minv * DSS * M [ ps(t) + M*RHS/M ]
665  ! so we can apply limiter to ps(t) + (M*RHS)/M
666 #if 1

668  mx=pmax(ie)
669  iuse = numcycle+1 ! always apply min/max limiter
670 #endif

672  smooth_phis_nudt*pstens(:,:,ie)/elem(ie)%spheremp(:,:)
673

676  ! but no reason to conserve mean PHI.

679  do j=1,np
680  if (phis(i,j,ie)>mx) phis(i,j,ie)=mx
681  if (phis(i,j,ie)<mn) phis(i,j,ie)=mn
682  enddo
683  enddo
684  endif
685

688  do i=1,np
689  do j=1,np
690  if (phis(i,j,ie)<minf) phis(i,j,ie)=minf
691  enddo
692  enddo
693 
694  phis(:,:,ie)=phis(:,:,ie)*elem(ie)%spheremp(:,:)
695  call edgevpack(edgebuf,phis(:,:,ie),1,0,ie)
696 
697  enddo

700  call bndry_exchangev(hybrid,edgebuf)
701  call t_stopf('smooth_phis_bexchV4')

704  call edgevunpack(edgebuf, phis(:,:,ie), 1, 0, ie)
705  phis(:,:,ie)=phis(:,:,ie)*elem(ie)%rspheremp(:,:)
706  enddo
707  enddo
