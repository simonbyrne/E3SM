117 void BoundaryExchange::set_num_fields (const int num_1d_fields, const int num_2d_fields, const int num_3d_fields, const int num_3d_int_fields)
118 {
119  // We don't allow to call this method twice in a row. If you want to change the number of fields,
120  // you need to call clean_up first, to get a fresh new BoundaryExchange.
121  // Note: if you do call clean_up and then again set_num_field and the new number of fields
122  // are smaller than the previous ones, MpiBuffersManager will NOT shrink the buffers, so
123  // you may be left with buffers that are larger than what you need.
124  assert (m_cleaned_up);
125 
126  // Make sure the connectivity is valid: must at least be a valid pointer, and be initialized, i.e.,
127  // store a valid number of elements, but may be finalized later (before registration_completed call though)
128  assert (m_connectivity && m_connectivity->is_initialized());
129 
130  // We strongly advocate for not using the same BE object for both 'standard' exchange and min/max exchange
131  assert (!(num_1d_fields>0 && (num_2d_fields>0 || num_3d_fields>0 || num_3d_int_fields>0)));
132 
133  // Note: we do not set m_num_1d_fields, m_num_2d_fields and m_num_3d_fields, since we will use them as
134  // progressive indices while adding fields. Then, during registration_completed,
135  // we will check that they match the 2nd dimension of m_1d_fields, m_2d_fields and m_3d_fields.
136 
137  // Create the fields views
138  m_1d_fields = decltype(m_1d_fields)("1d fields", m_num_elems, num_1d_fields);
139  m_2d_fields = decltype(m_2d_fields)("2d fields", m_num_elems, num_2d_fields);
140  if (NUM_LEV==NUM_LEV_P) {
141  // If NUM_LEV=NUM_LEVP, we can use the same 3d buffers for both midpoints and interface quantities
142  m_3d_fields = decltype(m_3d_fields)("3d fields", m_num_elems, num_3d_fields+num_3d_int_fields);
143  m_3d_int_fields = decltype(m_3d_int_fields)("3d interface fields", m_num_elems, 0);
144  } else {
145  m_3d_fields = decltype(m_3d_fields)("3d fields", m_num_elems, num_3d_fields);
146  m_3d_int_fields = decltype(m_3d_int_fields)("3d interface fields", m_num_elems, num_3d_int_fields);
147  }
148 
149  // Now we can start register fields
150  m_registration_started = true;
151  m_registration_completed = false;
152 
153  // We're not all clean
154  m_cleaned_up = false;
155 }

157 void BoundaryExchange::clean_up()
158 {
159  if (m_cleaned_up) {
160  // Perhaps not possible, but just in case
161  return;
162  }
163 
164  // Check that we are not still transmitting
165  assert (!m_send_pending && !m_recv_pending);
166 
167  // Clear stored fields
168  m_1d_fields = decltype(m_1d_fields)("m_1d_fields", 0, 0);
169  m_2d_fields = decltype(m_2d_fields)("m_2d_fields", 0, 0);
170  m_3d_fields = decltype(m_3d_fields)("m_3d_fields", 0, 0);
171  m_3d_int_fields = decltype(m_3d_int_fields)("m_3d_interface_fields", 0, 0);
172 
173  m_num_1d_fields = 0;
174  m_num_2d_fields = 0;
175  m_num_3d_fields = 0;
176  m_num_3d_int_fields = 0;
177 
178  // If we clean up, we need to reset the number of fields
179  m_registration_started = false;
180  m_registration_completed = false;
181 
182  // Clean buffer views and requests
183  clear_buffer_views_and_requests();
184 
185  // Now we're all cleaned
186  m_cleaned_up = true;
187 }

189 void BoundaryExchange::registration_completed()
190 {
191  // If everything is already set up, just return
192  if (m_registration_completed) {
193  // TODO: should we prohibit two consecutive calls of this method? It seems harmless, so I'm allowing it
194  return;
195  }
196 
197  // TODO: should we assert that m_registration_started=true? Or simply return if not? Can calling this
198  // method without a call to registration started be dangerous? Not sure...
199 
200  // At this point, the connectivity MUST be finalized already, and the buffers manager must be set already
201  assert (m_connectivity && m_connectivity->is_finalized());
202  assert (m_buffers_manager);
203 
204  // Create the MPI data types, for corners and edges
205  // Note: this is the size per element, per connection. It is the number of Real's to send/receive to/from the neighbor
206  // Note: for 2d/3d fields, we have 1 Real per GP (per level, in 3d). For 1d fields,
207  // we have 2 Real per level (max and min over element).
208 
209  const int single_ptr_buf_size = m_num_2d_fields + m_num_3d_fields*NUM_LEV*VECTOR_SIZE + m_num_3d_int_fields*NUM_LEV_P*VECTOR_SIZE;
210  m_elem_buf_size[etoi(ConnectionKind::CORNER)] = m_num_1d_fields*2*NUM_LEV*VECTOR_SIZE + single_ptr_buf_size * 1;
211  m_elem_buf_size[etoi(ConnectionKind::EDGE)] = m_num_1d_fields*2*NUM_LEV*VECTOR_SIZE + single_ptr_buf_size * NP;
212 
213  // Determine what kind of BE is this (exchange or exchange_min_max)
214  m_exchange_type = m_num_1d_fields>0 ? MPI_EXCHANGE_MIN_MAX : MPI_EXCHANGE;
215 
216  // Prohibit further registration of fields, and allow exchange
217  m_registration_started = false;
218  m_registration_completed = true;
219 
220  // Optimistically build buffers here. If registration is called with largest
221  // BufferManager user first, then building will occur just once, in the
222  // prim_init2 call.
223  build_buffer_views_and_requests();
224 }

298 void BoundaryExchange::pack_and_send ()
299 {
300  tstart("be pack_and_send");
301  // The registration MUST be completed by now
302  // Note: this also implies connectivity and buffers manager are valid
303  assert (m_registration_completed);
304 
305  // Check that this object is setup to perform exchange and not exchange_min_max
306  assert (m_exchange_type==MPI_EXCHANGE);
307 
308  // I am not sure why and if we could have this scenario, but just in case. I think MPI *may* go bananas in this case
309  if (m_num_2d_fields+m_num_3d_fields+m_num_3d_int_fields==0) {
310  return;
311  }
312 
313  // Check that buffers are not locked by someone else, then lock them
314  assert (!m_buffers_manager->are_buffers_busy());
315  m_buffers_manager->lock_buffers();
316 
317  // If this is the first time we call this method, or if the MpiBuffersManager has performed a reallocation
318  // since the last time this method was called, AND we are calling this method manually, without relying
319  // on the exchange method to call it, then we need to rebuild all our internal buffer views
320  if (!m_buffer_views_and_requests_built) {
321  tstart("be build_buffer_views_and_requests");
322  build_buffer_views_and_requests();
323  tstop("be build_buffer_views_and_requests");
324  }
325 
326  // ---- Pack ---- //
327  // First, pack 2d fields (if any)...
328  auto connections = m_connectivity->get_connections<ExecMemSpace>();
329  if (m_num_2d_fields>0) {
330  auto fields_2d = m_2d_fields;
331  auto send_2d_buffers = m_send_2d_buffers;
332  const ConnectionHelpers helpers;
333  Kokkos::parallel_for(MDRangePolicy<ExecSpace, 3>({0, 0, 0}, {m_num_elems, NUM_CONNECTIONS, m_num_2d_fields}, {1, 1, 1}),
334  KOKKOS_LAMBDA(const int ie, const int iconn, const int ifield) {
335  const ConnectionInfo& info = connections(ie, iconn);
336  const LidGidPos& field_lidpos = info.local;
337  // For the buffer, in case of local connection, use remote info. In fact, while with shared connections the
338  // mpi call will take care of "copying" data to the remote recv buffer in the correct remote element lid,
339  // for local connections we need to manually copy on the remote element lid. We can do it here
340  const LidGidPos& buffer_lidpos = info.sharing==etoi(ConnectionSharing::LOCAL) ? info.remote : info.local;
341 
342  // Note: if it is an edge and the remote edge is in the reverse order, we read the field_lidpos points backwards
343  const auto& pts = helpers.CONNECTION_PTS[info.direction][field_lidpos.pos];
344  for (int k=0; k<helpers.CONNECTION_SIZE[info.kind]; ++k) {
345  send_2d_buffers(buffer_lidpos.lid, ifield, buffer_lidpos.pos)(k) = fields_2d(field_lidpos.lid, ifield)(pts[k].ip, pts[k].jp);
346  }
347  });
348  }
349  // ...then pack 3d fields (if any)...
350  if (m_num_3d_fields>0) {
351  auto fields_3d = m_3d_fields;
352  auto send_3d_buffers = m_send_3d_buffers;
353  const auto num_3d_fields = m_num_3d_fields;
354  if (OnGpu<ExecSpace>::value) {
355  const ConnectionHelpers helpers;
356  Kokkos::parallel_for(
357  Kokkos::RangePolicy<ExecSpace>(0, m_num_elems*m_num_3d_fields*NUM_CONNECTIONS*NUM_LEV),
358  KOKKOS_LAMBDA(const int it) {
359  const int ie = it / (num_3d_fields*NUM_CONNECTIONS*NUM_LEV);
360  const int ifield = (it / (NUM_CONNECTIONS*NUM_LEV)) % num_3d_fields;
361  const int iconn = (it / NUM_LEV) % NUM_CONNECTIONS;
362  const int ilev = it % NUM_LEV;
363  const ConnectionInfo& info = connections(ie, iconn);
364  const LidGidPos& field_lidpos = info.local;
365  // For the buffer, in case of local connection, use remote info. In fact, while with shared connections the
366  // mpi call will take care of "copying" data to the remote recv buffer in the correct remote element lid,
367  // for local connections we need to manually copy on the remote element lid. We can do it here
368  const LidGidPos& buffer_lidpos = info.sharing==etoi(ConnectionSharing::LOCAL) ? info.remote : info.local;
369 
370  // Note: if it is an edge and the remote edge is in the reverse order, we read the field_lidpos points backwards
371  const auto& pts = helpers.CONNECTION_PTS[info.direction][field_lidpos.pos];
372  const auto& sb = send_3d_buffers(buffer_lidpos.lid, ifield, buffer_lidpos.pos);
373  const auto& f3 = fields_3d(field_lidpos.lid, ifield);
374  for (int k=0; k<helpers.CONNECTION_SIZE[info.kind]; ++k) {
375  sb(k, ilev) = f3(pts[k].ip, pts[k].jp, ilev);
376  }
377  });
378  } else {
379  const auto num_parallel_iterations = m_num_elems*m_num_3d_fields;
380  ThreadPreferences tp;
381  tp.max_threads_usable = NUM_CONNECTIONS;
382  tp.max_vectors_usable = NUM_LEV;
383  const auto threads_vectors =
384  DefaultThreadsDistribution<ExecSpace>::team_num_threads_vectors(
385  num_parallel_iterations, tp);
386  const auto policy = Kokkos::TeamPolicy<ExecSpace>(
387  num_parallel_iterations, threads_vectors.first, threads_vectors.second);
388  HOMMEXX_STATIC const ConnectionHelpers helpers;

390  policy,
391  KOKKOS_LAMBDA(const TeamMember& team) {
392  Homme::KernelVariables kv(team, num_3d_fields);
393  const int ie = kv.ie;
394  const int ifield = kv.iq;
395  for (int iconn = 0; iconn < 8; ++iconn) {

397  if (info.kind == etoi(ConnectionSharing::MISSING)) continue;
398  const LidGidPos& field_lidpos = info.local;
399  const LidGidPos& buffer_lidpos = (info.sharing == etoi(ConnectionSharing::LOCAL) ?
400  info.remote :
401  info.local);
402  const auto& pts = helpers.CONNECTION_PTS[info.direction][field_lidpos.pos];
403  const auto& sb = send_3d_buffers(buffer_lidpos.lid, ifield, buffer_lidpos.pos);
404  const auto& f3 = fields_3d(field_lidpos.lid, ifield);
405  Kokkos::parallel_for(
406  Kokkos::TeamThreadRange(kv.team, helpers.CONNECTION_SIZE[info.kind]),
407  [&] (const int& k) {
408  const auto& ip = pts[k].ip;
409  const auto& jp = pts[k].jp;

411  const auto* const f3p = &f3(ip, jp, 0);

414  [&] (const int& ilev) {
415  sbp[ilev] = f3p[ilev];
416  });
417  });
418  }
419  });
420  }
421  }
422  // ...then pack 3d interface fields (if any)
423  if (m_num_3d_int_fields>0) {

430  Kokkos::RangePolicy<ExecSpace>(0, m_num_elems*num_fields*NUM_CONNECTIONS*NUM_LEV_P),
431  KOKKOS_LAMBDA(const int it) {
432  const int ie = it / (num_fields*NUM_CONNECTIONS*NUM_LEV_P);
433  const int ifield = (it / (NUM_CONNECTIONS*NUM_LEV_P)) % num_fields;
434  const int iconn = (it / NUM_LEV_P) % NUM_CONNECTIONS;
435  const int ilev = it % NUM_LEV_P;
436  const ConnectionInfo& info = connections(ie, iconn);
437  const LidGidPos& field_lidpos = info.local;
438  // For the buffer, in case of local connection, use remote info. In fact, while with shared connections the
439  // mpi call will take care of "copying" data to the remote recv buffer in the correct remote element lid,
440  // for local connections we need to manually copy on the remote element lid. We can do it here
441  const LidGidPos& buffer_lidpos = info.sharing==etoi(ConnectionSharing::LOCAL) ? info.remote : info.local;
442 
443  // Note: if it is an edge and the remote edge is in the reverse order, we read the field_lidpos points backwards
444  const auto& pts = helpers.CONNECTION_PTS[info.direction][field_lidpos.pos];
445  const auto& sb = send_buffers(buffer_lidpos.lid, ifield, buffer_lidpos.pos);
446  const auto& f = fields(field_lidpos.lid, ifield);
447  for (int k=0; k<helpers.CONNECTION_SIZE[info.kind]; ++k) {
448  sb(k, ilev) = f(pts[k].ip, pts[k].jp, ilev);
449  }
450  });
451  } else {

453  ThreadPreferences tp;

455  tp.max_vectors_usable = NUM_LEV_P;
456  const auto threads_vectors =
457  DefaultThreadsDistribution<ExecSpace>::team_num_threads_vectors(
458  num_parallel_iterations, tp);

460  num_parallel_iterations, threads_vectors.first, threads_vectors.second);
461  HOMMEXX_STATIC const ConnectionHelpers helpers;

463  policy,

467  const int ifield = kv.iq;

471  const LidGidPos& field_lidpos = info.local;

822  assert (!m_buffers_manager->are_buffers_busy());

830  // on the exchange_min_max method to call it, then we need to rebuild all our internal buffer views
831  if (!m_buffer_views_and_requests_built) {
832  tstart("be build_buffer_views_and_requests");
833  build_buffer_views_and_requests();
834  tstop("be build_buffer_views_and_requests");

854  const LidGidPos& field_lidpos = info.local;

989  Kokkos::parallel_for(

999  const auto& f1 = fields_1d(ie, ifield);

1247 // A slot is the space in a communication buffer for an (element, connection)
1248 // pair. The slot index space numbers slots so that, first, they are contiguous
1249 // by remote PID and, second, within a PID block, each comm partner agrees on
1250 // order of slots.
1251 void BoundaryExchange
1252 ::init_slot_idx_to_elem_conn_pair (
1253  std::vector<int>& slot_idx_to_elem_conn_pair,
1254  std::vector<int>& pids, std::vector<int>& pid_offsets)
1255 {
1256  struct IP {
1257  int i, ord, pid;
1258  bool operator< (const IP& o) const {
1259  if (pid < o.pid) return true;
1260  if (pid > o.pid) return false;
1261  return ord < o.ord;
1262  }
1263  };
1264  std::vector<IP> i2remote(m_num_elems*NUM_CONNECTIONS);
1265 
1266  const auto& connections = m_connectivity->get_connections<HostMemSpace>();
1267  for (int ie = 0; ie < m_num_elems; ++ie)
1268  for (int iconn = 0; iconn < NUM_CONNECTIONS; ++iconn) {
1269  const auto& info = connections(ie, iconn);
1270  const int k = ie*NUM_CONNECTIONS + iconn;
1271  auto& i2r = i2remote[k];
1272  // Original sequence through (element, connection) pairs.
1273  i2r.i = k;
1274  // An ordering of the message buffer upon which both members of the
1275  // communication pair agree.
1276  if (info.local.gid < info.remote.gid)
1277  i2r.ord = info.local.gid*NUM_CONNECTIONS + info.local.pos;
1278  else
1279  i2r.ord = info.remote.gid*NUM_CONNECTIONS + info.remote.pos;
1280  // If local, indicate with -1, which is < the smallest pid of 0.
1281  i2r.pid = -1;
1282  if (info.sharing != etoi(ConnectionSharing::SHARED)) continue;
1283  i2r.pid = info.remote_pid;
1284  }
1285 
1286  // Sort so that, first, all (element, connection) pairs having the same
1287  // remote_pid are contiguous; second, within such a block, ord is
1288  // ascending. The first lets us set up comm buffers so monolithic messages
1289  // slot right in. The second means that the send and recv partners agree on
1290  // how the monolithic message is packed.
1291  std::sort(i2remote.begin(), i2remote.end());
1292 
1293  // Collect the unique remote_pids and get the offsets of the contiguous blocks
1294  // of them.
1295  slot_idx_to_elem_conn_pair.resize(m_num_elems*NUM_CONNECTIONS);
1296  pids.clear();
1297  pid_offsets.clear();
1298  int prev_pid = -2;
1299  for (int k = 0; k < m_num_elems*NUM_CONNECTIONS; ++k) {
1300  const auto& i2r = i2remote[k];
1301  if (i2r.pid > prev_pid && i2r.pid != -1) {
1302  pids.push_back(i2r.pid);
1303  pid_offsets.push_back(k);
1304  prev_pid = i2r.pid;
1305  }
1306  const int ie = i2r.i / NUM_CONNECTIONS;
1307  const int iconn = i2r.i % NUM_CONNECTIONS;
1308  slot_idx_to_elem_conn_pair[k] = ie*NUM_CONNECTIONS + iconn;
1309  }
1310  pid_offsets.push_back(m_num_elems*NUM_CONNECTIONS);
1311 }
