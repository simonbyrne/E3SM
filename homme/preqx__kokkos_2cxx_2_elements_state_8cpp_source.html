40 void ElementsState::randomize(const int seed, const Real max_pressure, const Real ps0, const Real hyai0) {
41  // Check state was inited
42  assert (m_num_elems>0);
43 
44  // Check data makes sense
45  assert (max_pressure>ps0);
46  assert (ps0>0);
47  assert (hyai0>=0);
48 
49  // arbitrary minimum value to generate and minimum determinant allowed
50  constexpr const Real min_value = 0.015625;
51  std::mt19937_64 engine(seed);
52  std::uniform_real_distribution<Real> random_dist(min_value, 1.0 / min_value);
53 
54  genRandArray(m_v, engine, random_dist);
55  genRandArray(m_t, engine, random_dist);
56 
57  // This ensures the pressure in a single column is monotonically increasing
58  // and has fixed upper and lower values
59  const auto make_pressure_partition = [=](
60  ExecViewUnmanaged<Scalar[NUM_LEV]> pt_dp) {
61 
62  auto h_pt_dp = Kokkos::create_mirror_view(pt_dp);
63  Kokkos::deep_copy(h_pt_dp,pt_dp);
64  Real* data = reinterpret_cast<Real*>(h_pt_dp.data());
65  Real* data_end = data + NUM_PHYSICAL_LEV;
66 
67  Real p[NUM_INTERFACE_LEV];
68  Real* p_start = &p[0];
69  Real* p_end = p_start+NUM_INTERFACE_LEV;
70 
71  for (int i=0; i<NUM_PHYSICAL_LEV; ++i) {
72  p[i+1] = data[i];
73  }
74  p[0] = ps0;
75  p[NUM_INTERFACE_LEV-1] = max_pressure;
76 
77  // Put in monotonic order
78  std::sort(p_start, p_end);
79 
80  // Check for no repetitions
81  if (std::unique(p_start,p_end)!=p_end) {
82  return false;
83  }
84 
85  // Compute dp from p (we assume p(last interface)=max_pressure)
86  for (int i=0; i<NUM_PHYSICAL_LEV; ++i) {
87  data[i] = p[i+1]-p[i];
88  }
89 
90  // Check that dp>=dp_min
91  const Real min_dp = std::numeric_limits<Real>::epsilon()*1000;
92  for (auto it=data; it!=data_end; ++it) {
93  if (*it < min_dp) {
94  return false;
95  }
96  }
97 
98  // Fill remainder of last vector pack with quiet nan's
99  Real* alloc_end = data+NUM_LEV*VECTOR_SIZE;
100  for (auto it=data_end; it!=alloc_end; ++it) {
101  *it = std::numeric_limits<Real>::quiet_NaN();
102  }
103 
104  Kokkos::deep_copy(pt_dp,h_pt_dp);
105 
106  return true;
107  };
108 
109  std::uniform_real_distribution<Real> pressure_pdf(ps0, max_pressure);
110 
111  for (int ie = 0; ie < m_num_elems; ++ie) {
112  // Because this constraint is difficult to satisfy for all of the tensors,
113  // incrementally generate the view
114  for (int igp = 0; igp < NP; ++igp) {
115  for (int jgp = 0; jgp < NP; ++jgp) {
116  for (int tl = 0; tl < NUM_TIME_LEVELS; ++tl) {
117  ExecViewUnmanaged<Scalar[NUM_LEV]> pt_dp3d =
118  Homme::subview(m_dp3d, ie, tl, igp, jgp);
119  do {
120  genRandArray(pt_dp3d, engine, pressure_pdf);
121  } while (make_pressure_partition(pt_dp3d)==false);
122  }
123  }
124  }
125  }
126 
127  // Generate ps_v so that it is equal to sum(dp3d).
128  HybridVCoord hvcoord;
129  hvcoord.ps0 = ps0;
130  hvcoord.hybrid_ai0 = hyai0;
131  hvcoord.m_inited = true;
132  auto dp = m_dp3d;
133  auto ps = m_ps_v;
134  Kokkos::parallel_for(Homme::get_default_team_policy<ExecSpace>(m_num_elems*NUM_TIME_LEVELS),
135  KOKKOS_LAMBDA(const TeamMember team){
136  KernelVariables kv(team);
137  const int ie = kv.ie / NUM_TIME_LEVELS;
138  const int tl = kv.ie % NUM_TIME_LEVELS;
139  hvcoord.compute_ps_ref_from_dp(kv,Homme::subview(dp,ie,tl),
140  Homme::subview(ps,ie,tl));
141  });
142 }
