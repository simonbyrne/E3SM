31 void prim_run_subcycle_c (const Real& dt, int& nstep, int& nm1, int& n0, int& np1, const int& next_output_step)
32 {
33  GPTLstart("tl-sc prim_run_subcycle_c");
34 
35  auto& context = Context::singleton();
36 
37  // Get simulation params
38  SimulationParams& params = context.get<SimulationParams>();
39  assert(params.params_set);
40 
41  // Get time info and compute dt for tracers and remap
42  TimeLevel& tl = context.get<TimeLevel>();
43  const Real dt_q = dt*params.qsplit;
44  Real dt_remap = dt_q;
45  int nstep_end = tl.nstep + params.qsplit;
46  if (params.rsplit>0) {
47  dt_remap = dt_q*params.rsplit;
48  nstep_end = tl.nstep + params.qsplit*params.rsplit;
49  }
50 
51  // Check if needed to compute diagnostics or energy
52  bool compute_diagnostics = false;
53  if (nstep_end%params.state_frequency==0 || nstep_end==tl.nstep0 ||
54  nstep_end>=next_output_step) {
55  compute_diagnostics = true;
56  }
57 
58  if (params.disable_diagnostics) {
59  compute_diagnostics = false;
60  }
61 
62  if (compute_diagnostics) {
63  Diagnostics& diags = context.get<Diagnostics>();
64  diags.run_diagnostics(true,2);
65  }
66 
67  tl.update_tracers_levels(params.qsplit);
68 
69 #ifndef CAM
70  apply_test_forcing ();
71 #endif
72 
73  // Apply forcing.
74  // In standalone mode, params.ftype == ForcingAlg::FORCING_DEBUG
75  // Corresponds to ftype == 0 in Fortran
76  if(params.ftype == ForcingAlg::FORCING_DEBUG) {
77  apply_cam_forcing(dt_remap);
78  }
79  // Corresponds to ftype == 2 in Fortran
80  else if(params.ftype == ForcingAlg::FORCING_2) {
81  apply_cam_forcing_dynamics(dt_remap);
82  }
83 
84  if (compute_diagnostics) {
85  Diagnostics& diags = context.get<Diagnostics>();
86  diags.run_diagnostics(true,0);
87  }
88 
89  // Initialize dp3d from ps
90  GPTLstart("tl-sc dp3d-from-ps");
91  Elements& elements = context.get<Elements>();
92  HybridVCoord& hvcoord = context.get<HybridVCoord>();
93  const auto hybrid_ai_delta = hvcoord.hybrid_ai_delta;
94  const auto hybrid_bi_delta = hvcoord.hybrid_bi_delta;
95  const auto ps0 = hvcoord.ps0;
96  const auto ps_v = elements.m_state.m_ps_v;
97  {
98  const auto dp3d = elements.m_state.m_dp3d;
99  const auto tln0 = tl.n0;
100  Kokkos::parallel_for(Kokkos::RangePolicy<ExecSpace> (0,elements.num_elems()*NP*NP*NUM_LEV),
101  KOKKOS_LAMBDA(const int idx) {
102  const int ie = ((idx / NUM_LEV) / NP) / NP;
103  const int igp = ((idx / NUM_LEV) / NP) % NP;
104  const int jgp = (idx / NUM_LEV) % NP;
105  const int ilev = idx % NUM_LEV;
106 
107  dp3d(ie,tln0,igp,jgp,ilev) = hybrid_ai_delta[ilev]*ps0
108  + hybrid_bi_delta[ilev]*ps_v(ie,tln0,igp,jgp);
109  });
110  }
111  ExecSpace::impl_static_fence();
112  GPTLstop("tl-sc dp3d-from-ps");
113 
114  // Loop over rsplit vertically lagrangian timesteps
115  GPTLstart("tl-sc prim_step-loop");
116  prim_step(dt,compute_diagnostics);
117  for (int r=1; r<params.rsplit; ++r) {
118  tl.update_dynamics_levels(UpdateType::LEAPFROG);
119  prim_step(dt,false);
120  }
121  GPTLstop("tl-sc prim_step-loop");
122 
123  tl.update_tracers_levels(params.qsplit);
124 
125  if (compute_diagnostics) {
126  Diagnostics& diags = context.get<Diagnostics>();
127  diags.run_diagnostics(false,3);
128  }
129 
131  // apply vertical remap
132  // always for tracers
133  // if rsplit>0: also remap dynamics and compute reference level ps_v
135  GPTLstart("tl-sc vertical_remap");
136  vertical_remap(dt_remap);
137  GPTLstop("tl-sc vertical_remap");
138 
140  // time step is complete. update some diagnostic variables:
141  // Q (mixing ratio)
143  update_q(tl.np1_qdp,tl.np1);
144 
145  if (compute_diagnostics) {
146  Diagnostics& diags = context.get<Diagnostics>();
147  diags.run_diagnostics(false,1);
148  }
149 
150  // Update dynamics time levels
151  tl.update_dynamics_levels(UpdateType::LEAPFROG);
152 
153  // Update the timelevel info to pass back to fortran
154  nstep = tl.nstep;
155  nm1 = tl.nm1;
156  n0 = tl.n0;
157  np1 = tl.np1;
158 
159  GPTLstop("tl-sc prim_run_subcycle_c");
160 }

178 void update_q (const int np1_qdp, const int np1)
179 {
180  auto& context = Context::singleton();
181 
182  // Get simulation params
183  SimulationParams& params = context.get<SimulationParams>();
184  assert(params.params_set);
185 
186  // Get hybrid vertical coordinate
187  HybridVCoord& hvcoord = context.get<HybridVCoord>();
188  auto hyai_delta = hvcoord.hybrid_ai_delta;
189  auto hybi_delta = hvcoord.hybrid_bi_delta;
190  const Real ps0 = hvcoord.ps0;
191 
192  // Get ps_v from Elements
193  Elements& elements = context.get<Elements>();
194  auto ps_v = elements.m_state.m_ps_v;
195 
196  // Get the tracers concentration and mass from Tracers
197  Tracers& tracers = context.get<Tracers>();
198  auto qdp = tracers.qdp;
199  auto Q = tracers.Q;
200 
201  // Update the device copy of Q, stored in Tracers
202  const int num_elems = elements.num_elems();
203  const int qsize = params.qsize;
204  Kokkos::parallel_for(Kokkos::RangePolicy<ExecSpace>(0,num_elems*qsize*NP*NP*NUM_LEV),
205  KOKKOS_LAMBDA(const int idx) {
206  const int ie = idx / (qsize*NP*NP*NUM_LEV);
207  const int iq = (idx / (NP*NP*NUM_LEV)) % qsize;
208  const int igp = (idx / (NP*NUM_LEV)) % NP;
209  const int jgp = (idx / NUM_LEV) % NP;
210  const int level = idx % NUM_LEV;
211 
212  const Scalar dp = hyai_delta(level)*ps0 + hybi_delta(level)*ps_v(ie,np1,igp,jgp);
213  Q(ie,iq,igp,jgp,level) = qdp(ie,np1_qdp,iq,igp,jgp,level)/dp;
214  });
215 }
