378  subroutine preq_hydrostatic(phi,phis,T_v,p,dp)
379  use kinds, only : real_kind
380  use dimensions_mod, only : np, nlev
381  use physical_constants, only : rgas
382 ! use hybvcoord_mod, only : hvcoord_t
383  implicit none
384 
385 
386  !------------------------------Arguments---------------------------------------------------------------
387  real(kind=real_kind), intent(out) :: phi(np,np,nlev)
388  real(kind=real_kind), intent(in) :: phis(np,np)
389  real(kind=real_kind), intent(in) :: t_v(np,np,nlev)
390  real(kind=real_kind), intent(in) :: p(np,np,nlev)
391  real(kind=real_kind), intent(in) :: dp(np,np,nlev)
392  ! type (hvcoord_t), intent(in) :: hvcoord
393  !------------------------------------------------------------------------------------------------------
394 
395  !---------------------------Local workspace-----------------------------
396  integer i,j,k ! longitude, level indices
397  real(kind=real_kind) hkk,hkl ! diagonal term of energy conversion matrix
398  real(kind=real_kind), dimension(np,np,nlev) :: phii ! Geopotential at interfaces
399  !-----------------------------------------------------------------------
400 
401 #if (defined COLUMN_OPENMP)
402 !$omp parallel do private(k,j,i,hkk,hkl)
403 #endif
404  do j=1,np ! Loop inversion (AAM)
405 
406  do i=1,np
407  hkk = dp(i,j,nlev)*0.5d0/p(i,j,nlev)
408  hkl = 2*hkk
409  phii(i,j,nlev) = rgas*t_v(i,j,nlev)*hkl
410  phi(i,j,nlev) = phis(i,j) + rgas*t_v(i,j,nlev)*hkk
411  end do
412 
413  do k=nlev-1,2,-1
414  do i=1,np
415  ! hkk = dp*ckk
416  hkk = dp(i,j,k)*0.5d0/p(i,j,k)
417  hkl = 2*hkk
418  phii(i,j,k) = phii(i,j,k+1) + rgas*t_v(i,j,k)*hkl
419  phi(i,j,k) = phis(i,j) + phii(i,j,k+1) + rgas*t_v(i,j,k)*hkk
420  end do
421  end do
422 
423  do i=1,np
424  ! hkk = dp*ckk
425  hkk = 0.5d0*dp(i,j,1)/p(i,j,1)
426  phi(i,j,1) = phis(i,j) + phii(i,j,2) + rgas*t_v(i,j,1)*hkk
427  end do
428 
429  end do

539 ! The surface height is zero by definition.
540  do i = 1,ncol
541  zi(i,nlevp) = 0.0_real_kind
542  end do
543 
544 ! Compute zi, zm from bottom up.
545 ! Note, zi(i,k) is the interface above zm(i,k)
546  do k = nlev, 1, -1
547 ! First set hydrostatic elements consistent with dynamics
548  do i = 1,ncol
549  hkl(i) = pdel(i,k) / pmid(i,k)
550  hkk(i) = 0.5_real_kind * hkl(i)
551  end do
552 
553 ! Now compute tv, zm, zi
554  do i = 1,ncol
555  ! tvfac = 1._r8 + zvir * q(i,k)
556  ! tv = t(i,k) * tvfac
557  zm(i,k) = zi(i,k+1) + rog * tv(i,k) * hkk(i)
558  zi(i,k) = zi(i,k+1) + rog * tv(i,k) * hkl(i)
559  end do
560  end do
561 
562  return
563  end subroutine geopotential_t
