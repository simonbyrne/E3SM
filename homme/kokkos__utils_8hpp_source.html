151  template <typename MemberType>
152  KOKKOS_INLINE_FUNCTION
153  int get_workspace_idx(const MemberType& team_member) const
154  {
155 #ifdef HOMMEXX_CUDA_SHARE_BUFFER
156  if (!_need_ws_sharing) {
157  return team_member.league_rank();
158  }
159  else {
160  int ws_idx = 0;
161  Kokkos::single(Kokkos::PerTeam(team_member), [&] () {
162  ws_idx = team_member.league_rank() % _num_ws_slots;
163  if (!Kokkos::atomic_compare_exchange_strong(&_open_ws_slots(ws_idx), (flag_type) 0, (flag_type)1)) {
164  rnd_type rand_gen = _rand_pool.get_state(team_member.league_rank());
165  ws_idx = Kokkos::rand<rnd_type, int>::draw(rand_gen) % _num_ws_slots;
166  while (!Kokkos::atomic_compare_exchange_strong(&_open_ws_slots(ws_idx), (flag_type) 0, (flag_type)1)) {
167  ws_idx = Kokkos::rand<rnd_type, int>::draw(rand_gen) % _num_ws_slots;
168  }
169  }
170  });
171 
172  // broadcast the idx to the team with a simple reduce
173  int ws_idx_max_reduce;
174  Kokkos::parallel_reduce(Kokkos::TeamThreadRange(team_member, 1), [&] (int, int& ws_idx_max) {
175  ws_idx_max = ws_idx;
176  }, Kokkos::Max<int>(ws_idx_max_reduce));
177  team_member.team_barrier();
178  return ws_idx_max_reduce;
179  }
180 #else

182 #endif
183  }
