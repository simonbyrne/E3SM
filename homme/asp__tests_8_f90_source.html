10  !
11  ! Functions for setting up initial conditions for the Jablonowski-Williamson test case.
12  !
13  ! Given longitude (radians), latitude (radians), eta (pressure) and rotation_angle (degrees)
14  ! the functions will return temperature, surface geopotential, zonal and meridional wind
15  ! components, respectively.
16  !
17  ! lperturb=.FALSE. result in initial conditions for the steady-state test case.
18  ! lperturb=.TRUE. result in initial conditions for the baroclinic wave test case.
19  !
20  ! T : FUNCTION temperature (lon,lat,eta,rotation_angle)
21  ! PHIS: FUNCTION surface_geopotential(lon,lat,rotation_angle)
22  ! U : FUNCTION u_wind (lon,lat,eta,lperturb,rotation_angle)
23  ! V : FUNCTION v_wind (lon,lat,eta,lperturb,rotation_angle)
24  !
25  ! The non-rotated (rotation_angle=0) version of the test cases is described in:
26  !
27  ! Jablonowski, C., and D. L. Williamson, 2006: A baroclinic instability
28  ! test case for atmospheric model dynamical cores.
29  ! Quart. J. Roy. Meteor. Soc., 132, 2943-2975.
30  !
31  ! Jablonowski, C., and D. L. Williamson, 2006: A Baroclinic Wave Test Case
32  ! for Dynamical Cores of General Circulation Models: Model Intercomparisons,
33  ! NCAR Technical Note, NCAR/TN-469+STR, 89 pp.
34  !
35  ! The rotated version simply rotates the initial conditions so that the spherical coordinate
36  ! poles do not conicide with the earth's rotation axis. Thereby the Coriolis parameter is
37  ! a function of latitude and longitude:
38  !
39  ! f = 2*Omega*(-cos(lon)*cos(lat)*sin(rotation_angle)+sin(lat)*cos(rotation_angle))
40  !
41  ! where Omega = 7.292 x 10E-5/s and rotation_angle is the angle between the flow direction
42  ! and equator.
43  !
44  ! Author: Peter Hjort Lauritzen (NCAR, pel@ucar.edu)
45  !
46  INTEGER,PARAMETER :: r8 = selected_real_kind(12) ! 8 byte real
47 
48  REAL(r8), PARAMETER :: &
49  eta_strato = 0.2d0 ,&! tropopause level
50  u0 = 35.d0 ,&! 35 m/s
51  t0 = 288.d0 ,&! global mean T at surface
52  p0 = 100000.d0 ,&! global mean surface pressure
53  eta0 = 0.252d0 ,&! center of jets (hybrid)
54  !
55  radius = 10.d0, & ! radius of the perturbation
56  perturbation_amplitude = 1.d0, & ! amplitude of u perturbation 1 m/s
57  perturbation_longitude = 20.d0, & ! longitudinal position, 20E
58  perturbation_latitude = 40.d0, & ! latitudinal position, 40N
59  !
60  !
61  !
62  rd = 287.d0 ,&! gas constant J/(K kg)
63  g = 9.80616d0 ,&! gravitational acceleration (m/s^2)
64  a = 6371229.d0,&! Earth's radius in m
65  omega = 7.29212d-5,&! angular velocity 1/s
66  gamma = 0.005d0 ,&! lapse rate
67  pi = 3.14159265358979323846_r8,& ! pi
68  deg2rad = pi/180.d0, &
69  perturbation_latitude_tracer = 55.d0
70

75  ! Temperature (equation (6) in Jablonowski and Williamson, 2006)
76  !
77  !********************************************************************
78  !
79  REAL(r8) function temperature(lon,lat,eta,rotation_angle)
80  IMPLICIT NONE
81  REAL(r8), INTENT(IN) :: eta, lon, lat, rotation_angle
82  REAL(r8) :: rot_lon, rot_lat
83 
84  IF (abs(rotation_angle)<1.0e-8) THEN
85  rot_lon = lon
86  rot_lat = lat
87  ELSE
88  CALL regrot(lon,lat,rot_lon,rot_lat,0.0d0,-0.5d0*pi+rotation_angle*deg2rad,1)
89  ENDIF
90 
91  temperature = t_mean(eta) + t_deviation(rot_lon,rot_lat,eta)
92  END FUNCTION temperature
93  !
94  ! Horizontally averaged temperature (equation (4) and (5) in Jablonowski and Williamson (2006))
95  !
96  REAL(r8) function t_mean(eta)
97  IMPLICIT NONE
98  REAL(r8), INTENT(IN) :: eta
99  REAL(r8) :: exponent, delta_t
100 
101  exponent = rd*gamma/g
102  delta_t = 480000.d0 ! in K, for T mean calculation
103  IF (eta.gt.(eta_strato)) THEN
104  t_mean = t0*eta**exponent ! mean temperature at each level (troposphere)
105  ELSE
106  t_mean = t0*eta**exponent + delta_t*(eta_strato-eta)**5 ! mean temperature at each level (stratosphere)
107  ENDIF
108  END FUNCTION t_mean
109  !
110  ! Temperature deviation from the horizontal mean
111  ! (equation (6) minus horizontally averaged temperature)
112  !
113  REAL(r8) function t_deviation(lon,lat,eta)
114  IMPLICIT NONE
115  REAL(r8), INTENT(IN) :: eta, lon, lat
116  REAL(r8) :: factor, phi_vertical, rot_lon, rot_lat, a_omega
117 
118  factor = eta*pi*u0/rd
119  phi_vertical = (eta - eta0) * 0.5d0*pi
120  a_omega = a*omega
121 
122  rot_lon = lon
123  rot_lat = lat
124 
125  t_deviation = factor * 1.5d0 * sin(phi_vertical) * (cos(phi_vertical))**0.5d0 * &
126  ((-2.d0*(sin(rot_lat))**6 * ((cos(rot_lat))**2 + 1.d0/3.d0) + 10.d0/63.d0)* &
127  u0 * (cos(phi_vertical))**1.5d0 + &
128  (8.d0/5.d0*(cos(rot_lat))**3 * ((sin(rot_lat))**2 + 2.d0/3.d0) - pi/4.d0)*a_omega*0.5d0 )
129  END FUNCTION t_deviation

138  IMPLICIT NONE
139  REAL(r8), INTENT(IN) :: lon, lat, rotation_angle
140  REAL(r8) :: cos_tmp, eta_sfc, rot_lon, rot_lat, a_omega

152 
153  surface_geopotential = ((-2.d0*(sin(rot_lat))**6 * ((cos(rot_lat))**2 + 1.d0/3.d0) + 10.d0/63.d0)*cos_tmp &
154  + (8.d0/5.d0*(cos(rot_lat))**3 * ((sin(rot_lat))**2 + 2.d0/3.d0) - pi/4.d0)*a_omega)*cos_tmp
155  END FUNCTION surface_geopotential

163  REAL(r8) function u_wind(lon,lat,eta,lperturb,rotation_angle)
164  IMPLICIT NONE
165  REAL(r8), INTENT(IN) :: lon,lat,eta,rotation_angle
166  LOGICAL, INTENT(IN) :: lperturb
167  REAL(r8) :: cos_lat, u_lat, phi_vertical, rot_lon, rot_lat, sin_tmp, cos_tmp, r, u_perturb, v_lat
168  REAL(r8) :: perturb_lon, perturb_lat, v_tmp
169 
170  perturb_lon = perturbation_longitude*deg2rad
171  perturb_lat = perturbation_latitude*deg2rad
172 
173  IF (abs(rotation_angle)<1.0e-8) THEN
174  rot_lon = lon
175  rot_lat = lat
176  ELSE
177  CALL regrot(lon,lat,rot_lon,rot_lat,0.0d0,-0.5d0*pi+rotation_angle*deg2rad,1)
178  ENDIF
179 
180  phi_vertical = (eta - eta0) *0.5d0*pi
181  u_lat = (cos(phi_vertical))**1.5d0 * 4.d0 * u0 * (sin(rot_lat))**2 * (cos(rot_lat))**2
182  u_wind = u_lat
183 
184  IF (lperturb) THEN
185 
186  sin_tmp = sin(perturb_lat)*sin(rot_lat)
187  cos_tmp = cos(perturb_lat)*cos(rot_lat)
188 
189  r = acos( sin_tmp + cos_tmp*cos(rot_lon-perturb_lon) ) ! great circle distance
190  u_perturb = perturbation_amplitude*exp(- (r*radius)**2 )
191  u_lat = u_perturb + u_lat
192  ENDIF
193  IF (abs(rotation_angle)<1.0e-8) THEN
194  u_wind = u_lat
195  ELSE
196  v_lat = 0.0d0
197  !
198  ! rotate wind components
199  !
200  CALL turnwi(u_lat,v_lat, u_wind,v_tmp,lon,lat,rot_lon,rot_lat,0.0d0,-0.5d0*pi+rotation_angle*deg2rad,-1)
201  IF (abs(u_wind)<1.0e-10) u_wind=0.0d0
202  ENDIF
203  END FUNCTION u_wind

331  SUBROUTINE regrot(pxreg,pyreg,pxrot,pyrot,pxcen,pycen,kcall)
332  IMPLICIT NONE
333 !
334 !----------------------------------------------------------------------
335 !
336 !* conversion between regular and rotated spherical coordinates.
337 !*
338 !* pxreg longitudes of the regular coordinates
339 !* pyreg latitudes of the regular coordinates
340 !* pxrot longitudes of the rotated coordinates
341 !* pyrot latitudes of the rotated coordinates
342 !* all coordinates given in degrees n (negative for s)
343 !* and degrees e (negative values for w)
344 !* pxcen regular longitude of the south pole of the rotated grid
345 !* pycen regular latitude of the south pole of the rotated grid
346 !*
347 !* kcall=-1: find regular as functions of rotated coordinates.
348 !* kcall= 1: find rotated as functions of regular coordinates.
349 !
350 !-----------------------------------------------------------------------
351 !
352  integer kxdim,kydim,kx,ky,kcall
353  real(r8) :: pxreg,pyreg,&
354  pxrot,pyrot,&
355  pxcen,pycen
356 !
357 !-----------------------------------------------------------------------
358 !
359  real(r8) zsycen,zcycen,zxmxc,zsxmxc,zcxmxc,zsyreg,zcyreg, &
360  zsyrot,zcyrot,zcxrot,zsxrot,zpi,zpih
361  integer jy,jx
362 
363  zpih = pi*0.5d0
364 !
365  !----------------------------------------------------------------------
366 !
367  zsycen = sin((pycen+zpih))
368  zcycen = cos((pycen+zpih))
369 !
370  IF (kcall.eq.1) then
371 !
372  zxmxc = pxreg - pxcen
373  zsxmxc = sin(zxmxc)
374  zcxmxc = cos(zxmxc)
375  zsyreg = sin(pyreg)
376  zcyreg = cos(pyreg)
377  zsyrot = zcycen*zsyreg - zsycen*zcyreg*zcxmxc
378  zsyrot = max(zsyrot,-1.0_r8)
379  zsyrot = min(zsyrot,+1.0_r8)
380  !
381  pyrot = asin(zsyrot)
382  !
383  zcyrot = cos(pyrot)
384  zcxrot = (zcycen*zcyreg*zcxmxc +zsycen*zsyreg)/zcyrot
385  zcxrot = max(zcxrot,-1.0_r8)
386  zcxrot = min(zcxrot,+1.0_r8)
387  zsxrot = zcyreg*zsxmxc/zcyrot
388  !
389  pxrot = acos(zcxrot)
390  !
391  IF (zsxrot<0.0) pxrot = -pxrot
392  !
393  ELSEIF (kcall.eq.-1) then
394  !
395  zsxrot = sin(pxrot)
396  zcxrot = cos(pxrot)
397  zsyrot = sin(pyrot)
398  zcyrot = cos(pyrot)
399  zsyreg = zcycen*zsyrot + zsycen*zcyrot*zcxrot
400  zsyreg = max(zsyreg,-1.0_r8)
401  zsyreg = min(zsyreg,+1.0_r8)
402  !
403  pyreg = asin(zsyreg)
404  !
405  zcyreg = cos(pyreg)
406  zcxmxc = (zcycen*zcyrot*zcxrot -&
407  zsycen*zsyrot)/zcyreg
408  zcxmxc = max(zcxmxc,-1.0_r8)
409  zcxmxc = min(zcxmxc,+1.0_r8)
410  zsxmxc = zcyrot*zsxrot/zcyreg
411  zxmxc = acos(zcxmxc)
412  IF (zsxmxc<0.0) zxmxc = -zxmxc
413  !
414  pxreg = zxmxc + pxcen
415  !
416  ELSE
417  WRITE(6,'(1x,''invalid kcall in regrot'')')
418  stop
419  ENDIF
420  END SUBROUTINE regrot

422  SUBROUTINE turnwi(puarg,pvarg,pures,pvres, &
423  pxreg,pyreg,pxrot,pyrot, &
424  pxcen,pycen,kcall)
425 !
426  IMPLICIT NONE
427 !
428 !-----------------------------------------------------------------------
429 !
430 !* turn horizontal velocity components between regular and
431 !* rotated spherical coordinates.
432 !
433 !* puarg : input u components
434 !* pvarg : input v components
435 !* pures : output u components
436 !* pvres : output v components
437 !* pa : transformation coefficients
438 !* pb : -"-
439 !* pc : -"-
440 !* pd : -"-
441 !* pxreg : regular longitudes
442 !* pyreg : regular latitudes
443 !* pxrot : rotated longitudes
444 !* pyrot : rotated latitudes
445 !* kxdim : dimension in the x (longitude) direction
446 !* kydim : dimension in the y (latitude) direction
447 !* kx : number of gridpoints in the x direction
448 !* ky : number of gridpoints in the y direction
449 !* pxcen : regular longitude of the south pole of the
450 !* transformed grid
451 !* pycen : regular latitude of the south pole of the
452 !* transformed grid
453 !*
454 !* kcall < 0 : find wind components in regular coordinates
455 !* from wind components in rotated coordinates
456 !* kcall > 0 : find wind components in rotated coordinates
457 !* from wind components in regular coordinates
458 !* note that all coordinates are given in degrees n and degrees e.
459 !* (negative values for s and w)
460 !
461 !-----------------------------------------------------------------------
462 
463  integer kxdim,kydim,kx,ky,kcall
464  real(r8) puarg,pvarg, &
465  pures,pvres, &
466  pa, pb, &
467  pc, pd, &
468  pxreg,pyreg, &
469  pxrot,pyrot
470  real(r8) pxcen,pycen
471 !
472 !-----------------------------------------------------------------------
473 !
474  integer jy,jx
475  real(r8) zpih,zsyc,zcyc,zsxreg,zcxreg,zsyreg,zcyreg,zxmxc,&
476  zsxmxc,zcxmxc,zsxrot,zcxrot,zsyrot,zcyrot
477 !
478 !-----------------------------------------------------------------------
479 !
480  IF (kcall.eq.1) then
481  zpih = pi*0.5d0
482  zsyc = sin(pycen+zpih)
483  zcyc = cos(pycen+zpih)
484  !
485  zsxreg = sin(pxreg)
486  zcxreg = cos(pxreg)
487  zsyreg = sin(pyreg)
488  zcyreg = cos(pyreg)
489  !
490  zxmxc = pxreg - pxcen
491  zsxmxc = sin(zxmxc)
492  zcxmxc = cos(zxmxc)
493  !
494  zsxrot = sin(pxrot)
495  zcxrot = cos(pxrot)
496  zsyrot = sin(pyrot)
497  zcyrot = cos(pyrot)
498  !
499  pa = zcyc*zsxmxc*zsxrot + zcxmxc*zcxrot
500  pb = zcyc*zcxmxc*zsyreg*zsxrot - zsyc*zcyreg*zsxrot - &
501  zsxmxc*zsyreg*zcxrot
502  pc = zsyc*zsxmxc/zcyrot
503  pd = (zsyc*zcxmxc*zsyreg + zcyc*zcyreg)/zcyrot
504  !
505  pures = pa*puarg + pb*pvarg
506  pvres = pc*puarg + pd*pvarg
507  ELSEIF (kcall.eq.-1) then
508  zpih = pi*0.5d0
509  zsyc = sin(pycen+zpih)
510  zcyc = cos(pycen+zpih)
511  !
512  zsxreg = sin(pxreg)
513  zcxreg = cos(pxreg)
514  zsyreg = sin(pyreg)
515  zcyreg = cos(pyreg)
516  !
517  zxmxc = pxreg - pxcen
518  zsxmxc = sin(zxmxc)
519  zcxmxc = cos(zxmxc)
520  !
521  zsxrot = sin(pxrot)
522  zcxrot = cos(pxrot)
523  zsyrot = sin(pyrot)
524  zcyrot = cos(pyrot)
525  !
526  pa = zcxmxc*zcxrot + zcyc*zsxmxc*zsxrot
527  pb = zcyc*zsxmxc*zcxrot*zsyrot + zsyc*zsxmxc*zcyrot -&
528  zcxmxc*zsxrot*zsyrot
529  pc =-zsyc*zsxrot/zcyreg
530  pd = (zcyc*zcyrot - zsyc*zcxrot*zsyrot)/zcyreg
531  !
532  pures = pa*puarg + pb*pvarg
533  pvres = pc*puarg + pd*pvarg
534  ELSE
535  write(6,'(1x,''invalid kcall in turnwi'')')
536  stop
537  ENDIF
538  END SUBROUTINE turnwi

703 ! Rossby_Haurwitz wave, wavenumber 4
704 !==========================================================================================
705  SUBROUTINE rossby_haurwitz (lon, lat, pressure, &
706  u_wind, v_wind, temperature, surface_geopotential, &
707  surface_pressure)
708 !-----------------------------------------------------------------------
709 ! input parameters
710 !-----------------------------------------------------------------------
711  real(r8), intent(in) :: lon, & ! longitude in radians
712  lat, & ! latitude in radians
713  pressure ! pressure at full model level in Pa
714 !-----------------------------------------------------------------------
715 ! input parameters
716 !-----------------------------------------------------------------------
717  real(r8), intent(out) :: u_wind, & ! zonal wind in m/s
718  v_wind, & ! meridional wind in m/s
719  temperature, & ! temperature in K
720  surface_geopotential, & ! surface geopotential in m^2/s^2
721  surface_pressure ! surface pressure in Pa
722 
723 !-----------------------------------------------------------------------
724 ! test case parameters
725 !-----------------------------------------------------------------------
726  real(r8),parameter :: u0 = 50._r8, & ! reference wind
727  t0 = 288._r8, & ! reference temperature
728  n = 4._r8, & ! wavenumber
729  mm = u0/(n*a), & ! parameter M and p_ref=95500.
730  kk = u0/(n*a), & ! parameter K
731  gamma = 0.0065_r8, & ! lapse rate in K/m
732  p_ref = 95500._r8 ! reference pressure
733 
734 !-----------------------------------------------------------------------
735 ! local
736 !-----------------------------------------------------------------------
737  real(r8) :: tmp1, tmp2, tmp3
738  real(r8) :: sin_lat, cos_lat, sin_slat, cos_slat
739  real(r8) :: exponent_1, exponent_2
740  real(r8) :: AA, BB, CC
741  real(r8) :: phis_perturb
742 
743 !-----------------------------------------------------------------------
744 ! initialize the wind components
745 !-----------------------------------------------------------------------
746  cos_lat = cos(lat)
747  sin_lat = sin(lat)
748  tmp1 = a * mm * cos_lat
749  tmp2 = a * kk * cos_lat**(n-1._r8)*(n*sin_lat**2 - cos_lat**2)
750  tmp3 = -a * kk * n * cos_lat**(n-1._r8) * sin_lat
751  u_wind = tmp1 + tmp2 * cos(n*lon)
752  v_wind = tmp3 * sin(n*lon)
753 !-----------------------------------------------------------------------
754 ! initialize surface geopotential
755 !-----------------------------------------------------------------------
756  surface_geopotential = 0._r8
757 !-----------------------------------------------------------------------
758 ! initialize surface pressure and temperature
759 !-----------------------------------------------------------------------
760  tmp1 = gamma/(g*t0)
761  tmp2 = a*a
762  exponent_1 = g/(gamma*rd)
763  exponent_2 = (gamma*rd)/g
764 
765  cos_lat = cos(lat)
766  aa = tmp2 * (0.5_r8 * mm*(2._r8*omega+mm) * cos_lat**2 + 0.25_r8 * kk**2 * cos_lat**(2._r8*n) * &
767  ( (n+1._r8)*cos_lat**2 + (2._r8*n*n - n - 2._r8)) - 0.5_r8*n*n*kk**2 * cos_lat**(2._r8*(n-1)))
768  bb = tmp2 * (2._r8*(omega+mm)*kk/((n+1._r8)*(n+2._r8)) * cos_lat**n * &
769  ( (n*n + 2._r8*n +2._r8) - (n+1._r8)**2 * cos_lat**2 ))
770  cc = tmp2 * (0.25_r8 * kk**2 * cos_lat**(2._r8*n) * ( (n+1._r8)*cos_lat**2 - (n+2._r8)))
771  phis_perturb = aa + bb * cos(n*lon) + cc * cos(2._r8*n*lon)
772  surface_pressure = p_ref * (1._r8 + tmp1*phis_perturb)**exponent_1 ! surface pressure
773  temperature = t0 * (pressure/p_ref)**exponent_2 ! temperature
774 
775  end subroutine rossby_haurwitz
