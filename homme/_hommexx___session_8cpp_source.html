73 void initialize_hommexx_session ()
74 {
75  // If hommexx session is not currently inited, then init it.
76  if (!Session::m_inited) {
77  /* Make certain profiling is only done for code we're working on */
78  profiling_pause();
79 
80  /* Set Environment variables to control how many
81  * threads/processors Kokkos uses */
82  if (Session::m_handle_kokkos) {
83  initialize_kokkos();
84  }
85 
86  // Note: at this point, the Comm *should* already be created.
87  const auto& comm = Context::singleton().get<Comm>();
88  if (comm.root()) {
89  ExecSpace::print_configuration(std::cout, true);
90  print_homme_config_settings ();
91  }
92 
93  Session::m_inited = true;
94  }
95 }
