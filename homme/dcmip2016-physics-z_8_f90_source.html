102  !------------------------------------------------
103  ! Physical Constants - MAY BE MODEL DEPENDENT
104  !------------------------------------------------
105  !REAL(8), PARAMETER :: &
106  ! one = 1.d0, & ! One
107  ! gravit = 9.80616d0, & ! Gravity (m/s^2)
108  ! rair = 287.d0, & ! Gas constant for dry air (J/kg/K)
109  ! cpair = 1004.5d0, & ! Specific heat of dry air (J/kg/K)
110  ! latvap = 2.5d6, & ! Latent heat of vaporization (J/kg)
111  ! rh2o = 461.5d0, & ! Gas constant for water vapor (J/kg/K)
112  ! epsilo = rair/rh2o, & ! Ratio of gas constants for dry air to vapor
113  ! zvir = (rh2o/rair) - 1.d0, & ! Constant for virtual temp. calc. (~0.608)
114  ! a = 6371220.0, & ! Reference Earth's Radius (m)
115  ! omega = 7.29212d-5, & ! Reference rotation rate of the Earth (s^-1)
116  ! pi = 4.d0*atan(1.d0) ! Pi
117 
118  REAL(8), PARAMETER :: &
119  one = 1.d0, & ! One
120  gravit = g, & ! Gravity (m/s^2)
121  rair = rgas, & ! Gas constant for dry air (J/kg/K)
122  cpair = cp, & ! Specific heat of dry air (J/kg/K)
123  latvap = 2.5d6, & ! Latent heat of vaporization (J/kg)
124  rh2o = rwater_vapor, & ! Gas constant for water vapor (J/kg/K)
125  epsilo = rair/rh2o, & ! Ratio of gas constants for dry air to vapor
126  zvir = (rh2o/rair) - 1.d0, & ! Constant for virtual temp. calc. (~0.608)
127  a = rearth0, & ! Reference Earth's Radius (m)
128  omega = omega0, & ! Reference rotation rate of the Earth (s^-1)
129  pi = dd_pi !4.d0*atan(1.d0) ! Pi
130 
131  !------------------------------------------------
132  ! Local Constants for Simple Physics
133  !------------------------------------------------
134  REAL(8), PARAMETER :: &
135  C = 0.0011d0, & ! From Simth and Vogl 2008
136  sst_tc = 302.15d0, & ! Constant Value for SST
137  t0 = 273.16d0, & ! Control temp for calculation of qsat
138  e0 = 610.78d0, & ! Saturation vapor pressure at T0
139  rhow = 1000.0d0, & ! Density of Liquid Water
140  cd0 = 0.0007d0, & ! Constant for Cd calc. Simth and Vogl 2008
141  cd1 = 0.000065d0, & ! Constant for Cd calc. Simth and Vogl 2008
142  cm = 0.002d0, & ! Constant for Cd calc. Simth and Vogl 2008
143  v20 = 20.0d0, & ! Thresh. wind spd. for Cd Smith and Vogl 2008
144  p0 = 100000.0d0, & ! Constant for potential temp calculation
145  pbltop = 85000.d0, & ! Top of boundary layer in p
146  zpbltop = 1000.d0, & ! Top of boundary layer in z
147  pblconst = 10000.d0, & ! Constant for the decay of diffusivity
148  t00 = 288.0d0, & ! Horizontal mean T at surf. for moist baro test
149  u0 = 35.0d0, & ! Zonal wind constant for moist baro test
150  latw = 2.0d0*pi/9.0d0, & ! Halfwidth for for baro test
151  eta0 = 0.252d0, & ! Center of jets (hybrid) for baro test
152  etav = (1.d0-eta0)*0.5d0*pi, & ! Auxiliary variable for baro test
153  q0 = 0.021d0, & ! Maximum specific humidity for baro test
154  kappa = 0.4d0 ! von Karman constant
155 
156  !------------------------------------------------
157  ! Variables used in calculation
158  !------------------------------------------------
159  INTEGER :: k
160 
161  REAL(8) :: &
162  za, & ! Altitude of lowest model level (m)
163  Tsurf, & ! Sea surface temperature (K)
164  ps, & ! Surface pressure (Pa)
165  pres, & ! Pressure on model level (Pa)
166  presi, & ! Pressure on model interface (Pa)
167  rhomi, & ! Moist pressure on model interface (kg/m^3)
168  thetav, & ! Virtual potential temperature (K)
169  dz, & ! Thickness of model level (m)
170  deltaqsv, & ! Change in specific humidity
171  wind, & ! Wind speed in the lowest model level (m/s)
172  Cd, & ! Drag coefficient for momentum
173  qsat, & ! Saturation specific humidity
174  qsats ! Saturation specific humidity at sea surface
175 
176  ! Matrix coefficients for PBL scheme
177  REAL(8) :: &
178  CA, & ! Matrix coefficients for PBL scheme
179  CC, & ! Matrix coefficients for PBL scheme
180  CAm, & ! Matrix coefficients for PBL scheme
181  CAE, & ! Matrix coefficients for PBL scheme
182  CBm, & ! Matrix coefficients for PBL scheme
183  CBE, & ! Matrix coefficients for PBL scheme
184  CCm, & ! Matrix coefficients for PBL scheme
185  CCE ! Matrix coefficients for PBL scheme
186 
187  ! Variables defined on model levels
188  REAL(8), DIMENSION(nz) :: &
189  rhom, & ! Moist density on model levels
190  qsv, & ! Specific humidity (kg/kg)
191  t, & ! Temperature (K)
192 ! theta, & ! Potential temperature on model levels
193  exner, & ! Exner function (p/p0)**(R/cp)
194  cem, & ! Matrix coefficients for PBL scheme
195  cee, & ! Matrix coefficients for PBL scheme
196  cfu, & ! Matrix coefficients for PBL scheme
197  cfv, & ! Matrix coefficients for PBL scheme
198  cft, & ! Matrix coefficients for PBL scheme
199  cfq ! Matrix coefficients for PBL scheme
200 
201  ! Variables defined at interfaces
202  REAL(8), DIMENSION(nz+1) :: &
203  Km, & ! Eddy diffusivity for boundary layer
204  Ke ! Eddy diffusivity for boundary layer
205 
206  if (prec_type < 0) return
207 
208  !------------------------------------------------
209  ! Store altitude of lowest model level
210  !------------------------------------------------
211  za = z(1)
212 
213  !------------------------------------------------
214  ! Calculate sea surface temperature
215  !------------------------------------------------
216 
217  ! Moist baroclinic wave test
218  if (test .eq. 1) then
219  tsurf = &
220  (t00 + pi*u0/rair * 1.5d0 * sin(etav) * (cos(etav))**0.5d0 * &
221  ((-2.d0*(sin(lat))**6 * ((cos(lat))**2 + 1.d0/3.d0) &
222  + 10.d0/63.d0) * &
223  u0 * (cos(etav))**1.5d0 + &
224  (8.d0/5.d0*(cos(lat))**3 * ((sin(lat))**2 + 2.d0/3.d0) &
225  - pi/4.d0)*a*omega*0.5d0 ))/ &
226  (1.d0+zvir*q0*exp(-(lat/latw)**4))
227 
228  ! Tropical cyclone test
229  elseif (test .eq. 2) then
230  tsurf = sst_tc
231 
232  ! Supercell test
233  elseif (test .eq. 3) then
234  tsurf = 0.d0
235 
236  ! Invalid test
237  else
238  write(*,*) 'Invalid test specified in DCMIP2016_PHYSICS', test
239  stop
240  endif
241 
242  !------------------------------------------------
243  ! Initialize precipitation rate to zero
244  !------------------------------------------------
245  precl = 0.d0
246 
247  !--------------------------------------------------------------
248  ! Calculate moist density, specific humidity and temperature
249  !--------------------------------------------------------------
250  do k = 1, nz
251  rhom(k) = rho(k) * (1.0 + qv(k))
252  qsv(k) = qv(k) * rho(k) / rhom(k)
253 
254  !t(k) = p(k) / (rhom(k) * rair * (one + zvir * qv(k)))
255  exner(k) = (p(k) / p0)**(rair/cpair)
256  t(k) = theta(k)*exner(k)
257  enddo
258 
259  !------------------------------------------------
260  ! Large-scale precipitation (Reed-Jablonowski)
261  !------------------------------------------------
262  if (prec_type .eq. 1) then
263  precl = 0.d0
264 
265  do k=1, nz
266  dz = zi(k+1) - zi(k)
267  qsat = epsilo * e0 / p(k) * exp(-(latvap/rh2o) * ((one/t(k))-(one/t0)))
268  if (qsv(k) > qsat) then
269  deltaqsv = (qsv(k) - qsat) &
270  / (one + (latvap/cpair) * epsilo * latvap * qsat / (rair*t(k)**2))
271  t(k) = t(k) + latvap / cpair * deltaqsv
272  qsv(k) = qsv(k) - deltaqsv
273  precl = precl + deltaqsv * rhom(k) * dz / (dt * rhow)
274 
275  qv(k) = qsv(k) / (1.0 - qsv(k))
276  rhom(k) = rho(k) / (1.0 - qsv(k))
277  p(k) = rhom(k) * rair * t(k) * (one + zvir * qv(k))
278  endif
279  enddo
280 
281  !------------------------------------------------
282  ! Large-scale precipitation (Kessler)
283  !------------------------------------------------
284  elseif (prec_type .eq. 0) then
285  !do k=1,nz
286  ! exner(k) = (p(k) / p0)**(rair/cpair)
287  ! theta(k) = t(k) / exner(k)
288  !enddo
289 
290  CALL kessler( &
291  theta, &
292  qv, &
293  qc, &
294  qr, &
295  rho, &
296  exner, &
297  dt, &
298  z, &
299  nz, &
300  precl)
301 
302  ! Convert qv to qsv and theta to pressure and temperature
303  do k = 1,nz
304  qsv(k) = qv(k) / (one + qv(k))
305  rhom(k) = rho(k) / (one - qsv(k))
306  thetav = theta(k) * (one + zvir * qv(k))
307  p(k) = p0 * (rhom(k) * rair * thetav / p0)**(cpair/(cpair-rair))
308  t(k) = p(k) / (rhom(k) * rair * (one + zvir * qv(k)))
309  enddo
310 
311  else
312  write(*,*) 'Invalid prec_type specified in DCMIP2016_PHYSICS', prec_type
313  stop
314  endif
315 
316  !----------------------------------------------------
317  ! Do not apply surface fluxes or PBL for supercell
318  !----------------------------------------------------
319  if (test .eq. 3 .or. pbl_type < 0) then
320  return
321  endif
322 
323  !------------------------------------------------
324  ! Turbulent mixing coefficients
325  !------------------------------------------------
326  wind = sqrt(u(1)**2 + v(1)**2)
327 
328  if (wind .lt. v20) then
329  cd = cd0 + cd1 * wind
330  else
331  cd = cm
332  endif
333 
334  ! Reed-Jablonowski Boundary layer
335  if (pbl_type .eq. 0) then
336  km(1)=0
337  ke(1)=0
338  ke(nz+1)=0
339  km(nz+1)=0
340  do k = 2, nz
341  presi = 0.5d0 * (p(k-1) + p(k))
342  if (presi .ge. pbltop) then
343  km(k) = cd * wind * za
344  ke(k) = c * wind * za
345  else
346  km(k) = cd * wind * za * exp(-(pbltop-presi)**2/pblconst**2)
347  ke(k) = c * wind * za * exp(-(pbltop-presi)**2/pblconst**2)
348  endif
349  enddo
350 
351  ! Bryan Planetary Boundary Layer
352  elseif (pbl_type .eq. 1) then
353 
354  do k = 1, nz+1
355 
356  if (zi(k) .le. zpbltop) then
357  km(k) = kappa * sqrt(cd) * wind * zi(k) &
358  * (one - zi(k)/zpbltop) * (one - zi(k)/zpbltop)
359  ke(k) = kappa * sqrt(c) * wind * zi(k) &
360  * (one - zi(k)/zpbltop) * (one - zi(k)/zpbltop)
361  else
362  km(k) = 0.d0
363  ke(k) = 0.d0
364  endif
365  enddo
366 
367  ! Invalid PBL
368  else
369  write(*,*) 'Invalid pbl_type specified in DCMIP2016_PHYSICS', pbl_type
370  stop
371  endif
372 
373  !------------------------------------------------
374  ! Surface fluxes
375  !------------------------------------------------
376 
377  ! Hydrostatic surface pressure
378  ps = 0.d0
379  do k = 1, nz
380  ps = ps + gravit * rhom(k) * (zi(k+1) - zi(k))
381  enddo
382  qsats = epsilo * e0 / ps * exp(-latvap / rh2o * ((one/tsurf)-(one/t0)))
383 
384  u(1) = u(1) / (one + dt * cd * wind / za)
385  v(1) = v(1) / (one + dt * cd * wind / za)
386  qsv(1) = (qsv(1) + c * wind * qsats * dt / za) / (one + c * wind * dt / za)
387  t(1) = (t(1) + c * wind * tsurf * dt / za) / (one + c * wind * dt / za)
388 
389  qv(1) = qsv(1) / (1.0 - qsv(1))
390  rhom(1) = rho(1) / (1.0 - qsv(1))
391  p(1) = t(1) * rhom(1) * rair * (one + zvir * qv(1))
392 
393  !------------------------------------------------
394  ! Boundary layer
395  !------------------------------------------------
396  do k = 1, nz
397  theta(k) = t(k) * (p0 / p(k))**(rair/cpair)
398  enddo
399 
400  do k = 1, nz
401  ca = 0.d0
402  cc = 0.d0
403 
404  if (k .ne. 1) then
405  rhomi = 0.5d0 * (rhom(k-1) + rhom(k))
406  ca = rhomi / rhom(k) * dt / (z(k) - z(k-1)) / (zi(k+1) - zi(k))
407  endif
408  if (k .ne. nz) then
409  rhomi = 0.5d0 * (rhom(k) + rhom(k+1))
410  cc = rhomi / rhom(k) * dt / (z(k+1) - z(k)) / (zi(k+1) - zi(k))
411  endif
412 
413  cam = ca * km(k)
414  ccm = cc * km(k+1)
415 
416  cae = ca * ke(k)
417  cce = cc * ke(k+1)
418 
419  cbm = one + cam + ccm
420  cbe = one + cae + cce
421 
422  if (k .eq. 1) then
423  cem(1) = ccm / cbm
424  cee(1) = cce / cbe
425 
426  cfu(1) = u(1) / cbm
427  cfv(1) = v(1) / cbm
428  cft(1) = theta(1) / cbe
429  cfq(1) = qsv(1) / cbe
430 
431  else
432  cem(k) = ccm / (cbm - cam * cem(k-1))
433  cee(k) = cce / (cbe - cae * cee(k-1))
434 
435  cfu(k) = (u(k) + cam * cfu(k-1)) / (cbm - cam * cem(k-1))
436  cfv(k) = (v(k) + cam * cfv(k-1)) / (cbm - cam * cem(k-1))
437  cft(k) = (theta(k) + cae * cft(k-1)) / (cbe - cae * cee(k-1))
438  cfq(k) = (qsv(k) + cae * cfq(k-1)) / (cbe - cae * cee(k-1))
439  endif
440  enddo
441 
442  u(nz) = cfu(nz)
443  v(nz) = cfv(nz)
444  theta(nz) = cft(nz)
445  qsv(nz) = cfq(nz)
446 
447  do k=nz-1,1,-1
448  u(k) = cem(k) * u(k+1) + cfu(k)
449  v(k) = cem(k) * v(k+1) + cfv(k)
450  theta(k) = cee(k) * theta(k+1) + cft(k)
451  qsv(k) = cee(k) * qsv(k+1) + cfq(k)
452  enddo
453 
454  ! Convert theta to pressure
455  do k = 1,nz
456  qv(k) = qsv(k) / (one - qsv(k))
457  rhom(k) = rho(k) / (one - qsv(k))
458  thetav = theta(k) * (one + zvir * qv(k))
459  p(k) = p0 * (rhom(k) * rair * thetav / p0)**(cpair/(cpair-rair))
460  enddo
461 
462  return
463
