32 void Connectivity::set_num_elements (const int num_local_elements)
33 {
34  // We don't allow to change the number of elements once set. There may be downstream classes
35  // that already read num_elements from this class, and would not be informed of the change.
36  // Besides, does it really make sense? Does it add an interesting feature? I don't think so.
37  assert (!m_initialized);
38 
39  // Safety check
40  assert (num_local_elements>=0);
41 
42  m_num_local_elements = num_local_elements;
43 
44  m_connections = ExecViewManaged<ConnectionInfo*[NUM_CONNECTIONS]>("Connections", m_num_local_elements);
45  h_connections = Kokkos::create_mirror_view(m_connections);
46 
47  m_num_connections = ExecViewManaged<int[NUM_CONNECTION_SHARINGS+1][NUM_CONNECTION_KINDS+1]>("Connections counts");
48  h_num_connections = Kokkos::create_mirror_view(m_num_connections);
49 
50  // Initialize all connections to MISSING
51  // Note: we still include local element/position, since we need that even for
52  // missing connections! Everything else is set to invalid numbers (for safety)
53  for (int ie=0; ie<m_num_local_elements; ++ie) {
54  for (int iconn=0; iconn<NUM_CONNECTIONS; ++iconn) {
55  ConnectionInfo& info = h_connections(ie,iconn);
56 
57  info.sharing = etoi(ConnectionSharing::MISSING);
58  info.kind = etoi(ConnectionKind::MISSING);
59 
60  info.local.lid = ie;
61  info.local.pos = iconn;
62 
63  info.local.gid = INVALID_ID;
64  info.remote.lid = INVALID_ID;
65  info.remote.gid = INVALID_ID;
66  info.remote.pos = INVALID_ID;
67 
68  }
69  }
70 
71  // Initialize all counters to 0 (even the missing ones)
72  Kokkos::deep_copy (h_num_connections,0);
73 
74  // Connectivity is now initialized
75  m_initialized = true;
76 }

78 void Connectivity::add_connection (const int first_elem_lid, const int first_elem_gid, const int first_elem_pos, const int first_elem_pid,
79  const int second_elem_lid, const int second_elem_gid, const int second_elem_pos, const int second_elem_pid)
80 {
81  // Connectivity must be in initialized state but not in finalized state
82  assert (m_initialized);
83  assert (!m_finalized);
84 
85  // Comm must not be a null comm, otherwise checks on ranks may be misleading
86  assert (m_comm.mpi_comm()!=MPI_COMM_NULL);
87 
88  // I believe edges appear twice in fortran, once per each ordering.
89  // Here, we only need to store them once
90  if (first_elem_pid==m_comm.rank())
91  {
92 #ifndef NDEBUG
93  // There is no edge-to-corner connection. Either the elements share a corner or an edge.
94  assert (m_helpers.CONNECTION_KIND[first_elem_pos]==m_helpers.CONNECTION_KIND[second_elem_pos]);
95 
96  // The elem lid on the local side must be valid!
97  assert (first_elem_lid>=0);
98 #endif
99 
100  ConnectionInfo& info = h_connections(first_elem_lid,first_elem_pos);
101  LidGidPos& local = info.local;
102  LidGidPos& remote = info.remote;
103 
104  // Local and remote elements lid and connection position
105  local.lid = first_elem_lid;
106  local.gid = first_elem_gid;
107  local.pos = first_elem_pos;
108  remote.lid = second_elem_lid;
109  remote.gid = second_elem_gid;
110  remote.pos = second_elem_pos;
111 
112  // Kind
113  info.kind = etoi(m_helpers.CONNECTION_KIND[first_elem_pos]);
114 
115  // Direction
116  info.direction = etoi(m_helpers.CONNECTION_DIRECTION[local.pos][remote.pos]);
117 
118  if (second_elem_pid!=m_comm.rank())
119  {
120  info.sharing = etoi(ConnectionSharing::SHARED);
121  info.remote_pid = second_elem_pid;
122 
123  } else {
124  info.remote_pid = -1;
125  info.sharing = etoi(ConnectionSharing::LOCAL);
126  }
127 
128  ++h_num_connections(info.sharing,info.kind);
129  }
130 }

132 void Connectivity::finalize(const bool sanity_check)
133 {
134  // Sanity check: Homme does not allow less than 2*2 elements on each of the cube's faces, so each element
135  // should have at most ONE missing connection
136  constexpr int corners[NUM_CORNERS] = { etoi(ConnectionName::SWEST), etoi(ConnectionName::SEAST), etoi(ConnectionName::NWEST), etoi(ConnectionName::NEAST)};
137 
138  for (int ie=0; ie<m_num_local_elements; ++ie) {
139  std::array<bool,NUM_CORNERS> missing = {{false, false, false, false}};
140 
141  for (int ic : corners) {
142  if (h_connections(ie,ic).kind == etoi(ConnectionKind::MISSING)) {
143  missing[ic % NUM_CORNERS] = true;
144 
145  // Just for tracking purposes
146  //++h_num_connections(etoi(ConnectionSharing::MISSING),etoi(ConnectionKind::MISSING));
147  h_num_connections(2,2) += 1;//etoi(ConnectionSharing::MISSING),etoi(ConnectionKind::MISSING)) += 1;
148  }
149  }
150 
151  Errors::runtime_check(!sanity_check || std::count(missing.cbegin(),missing.cend(),true)<=1,
152  "Error! On real geometries, each element should have at most ONE missing connection.\n"
153  " If this is a unit test, you may want to call finalize(false) to disable this check.\n",-1);
154  }
155 
156  // Updating counters for groups with same sharing/kind
157  for (int kind=0; kind<NUM_CONNECTION_KINDS; ++kind) {
158  for (int sharing=0; sharing<NUM_CONNECTION_SHARINGS; ++sharing) {
159  h_num_connections(etoi(ConnectionSharing::ANY),kind) += h_num_connections(sharing,kind);
160  h_num_connections(sharing,etoi(ConnectionKind::ANY)) += h_num_connections(sharing,kind);
161  }
162  h_num_connections(etoi(ConnectionKind::ANY),etoi(ConnectionKind::ANY)) += h_num_connections(etoi(ConnectionSharing::ANY),kind);
163  }
164 
165  // Copying to device
166  Kokkos::deep_copy(m_connections, h_connections);
167  Kokkos::deep_copy(m_num_connections, h_num_connections);
168 
169  m_finalized = true;
170 }
