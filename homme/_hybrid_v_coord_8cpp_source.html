71 void HybridVCoord::random_init(int seed) {
72  hybrid_ai_packed = ExecViewManaged<Scalar[NUM_LEV_P]>("Hybrid coordinates; coefficient A_interfaces");
73  hybrid_bi_packed = ExecViewManaged<Scalar[NUM_LEV_P]>("Hybrid coordinates; coefficient B_interfaces");
74 
75  hybrid_ai = ExecViewUnmanaged<Real[NUM_INTERFACE_LEV]>(reinterpret_cast<Real*>(hybrid_ai_packed.data()));
76  hybrid_bi = ExecViewUnmanaged<Real[NUM_INTERFACE_LEV]>(reinterpret_cast<Real*>(hybrid_bi_packed.data()));
77 
78  hybrid_am = ExecViewManaged<Scalar[NUM_LEV]>("Hybrid a_interface coefs");
79  hybrid_bm = ExecViewManaged<Scalar[NUM_LEV]>("Hybrid b_interface coefs");
80 
81  std::mt19937_64 engine(seed);
82  ps0 = 1.0;
83 
84  using urd = std::uniform_real_distribution<Real>;
85 
86  // eta=hybrid_a+hybrid_b increasing ensures p is monotonic (sufficient, not necessary cond).
87  // Also, hybrid_a and hybrid_b must be between 0 and 1.
88  // So easy way: generate uniform eta=a+b in [eta_min,1], with eta_min>0.
89  // That means eta(i) = i*delta_eta, with delta_eta = (1-eta_min)/NUM_PHYSICAL_LEV.
90  // Then pick a = c*b, with c<1 (say c=0.1). Then eta = b+c*b=(c+1)b,
91  // so b(i) = i*delta_eta/(c+1) and a(i) = c*i*delta_eta/(c+1).
92  // To add randomness, perturb sligthly: a(i)=a(i)+da and b(i)=b(i)+db,
93  // without violating eta(i+1)>eta(i). For instance, let da+db<delta_eta/2,
94  // which is true if da<delta_eta/4 and db<delta_eta/4.
95 
96  decltype(hybrid_ai)::HostMirror host_hybrid_ai("Host hybrid ai coefs");
97  decltype(hybrid_bi)::HostMirror host_hybrid_bi("Host hybrid bi coefs");
98  decltype(hybrid_am)::HostMirror host_hybrid_am("Host hybrid am coefs");
99  decltype(hybrid_bm)::HostMirror host_hybrid_bm("Host hybrid bm coefs");
100 
101  // This somewhat odd expression ensures that deta (below) is smaller
102  // than eta_min, so that eta_min + eps*deta is always positive
103  constexpr Real eta_min = (1.0/NUM_PHYSICAL_LEV)+0.001;
104 
105  // The proportionality constant between a and b.
106  Real c;
107  genRandArray(&c,1,engine,urd(0.0001,0.25));
108 
109  // The eta uniform grid spacing
110  Real deta = (1-eta_min)/NUM_PHYSICAL_LEV;
111  Real perturbation = deta/5;
112 
113  for (int i=0; i<NUM_INTERFACE_LEV; ++i) {
114  // Compute eta
115  Real eta_i = eta_min+deta*i;
116 
117  // Create b and a
118  host_hybrid_bi(i) = eta_i/(c+1);
119  host_hybrid_ai(i) = c*host_hybrid_bi(i);
120 
121  // Add some randomness to a and b
122  Real dab[2];
123  Real perturbation_adjusted = std::min(std::min(perturbation,eta_i),std::min(perturbation,1-eta_i));;
124 
125  auto perturbation_dist = urd(-perturbation_adjusted,perturbation_adjusted);
126  genRandArray(dab,2,engine,perturbation_dist);
127  host_hybrid_ai(i) += dab[0];
128  host_hybrid_bi(i) += dab[1];
129 
130  // Enforce 0<a,b<1
131  if (host_hybrid_ai(i)<0) {
132  host_hybrid_ai(i) = 0;
133  }
134  if (host_hybrid_bi(i)<0) {
135  host_hybrid_bi(i) = 0;
136  }
137  if (host_hybrid_ai(i)>1) {
138  host_hybrid_ai(i) = 1;
139  }
140  if (host_hybrid_bi(i)>1) {
141  host_hybrid_bi(i) = 1;
142  }
143  }
144 
145  // Enforce b(0)=0, a(0)=eta_min, and b(end)=1, a(end)=0;
146  host_hybrid_ai(0) = eta_min;
147  host_hybrid_bi(0) = 0;
148  host_hybrid_ai(NUM_PHYSICAL_LEV) = 0;
149  host_hybrid_bi(NUM_PHYSICAL_LEV) = 1;
150 
151  // Safety check: a+b should be monotone here
152  // Also, set midpoints
153  HostViewUnmanaged<Real[NUM_PHYSICAL_LEV]> host_hybrid_am_real(reinterpret_cast<Real*>(host_hybrid_am.data()));
154  HostViewUnmanaged<Real[NUM_PHYSICAL_LEV]> host_hybrid_bm_real(reinterpret_cast<Real*>(host_hybrid_bm.data()));
155  for (int i=1; i<NUM_INTERFACE_LEV; ++i) {
156  Real curr = host_hybrid_ai(i) + host_hybrid_bi(i);
157  Real prev = host_hybrid_ai(i-1) + host_hybrid_bi(i-1);
158 
159  Errors::runtime_check(curr>prev,"Error! hybrid_a+hybrid_b is not increasing.\n", -1);
160 
161  host_hybrid_am_real(i-1) = (host_hybrid_ai(i) + host_hybrid_ai(i))/2.0;
162  host_hybrid_bm_real(i-1) = (host_hybrid_bi(i) + host_hybrid_bi(i))/2.0;
163  }
164 
165  Kokkos::deep_copy(hybrid_ai, host_hybrid_ai);
166  Kokkos::deep_copy(hybrid_bi, host_hybrid_bi);
167  Kokkos::deep_copy(hybrid_am, host_hybrid_am);
168  Kokkos::deep_copy(hybrid_bm, host_hybrid_bm);
169 
170  hybrid_ai0 = host_hybrid_ai(0);
171 
172  compute_deltas();
173  compute_eta();
174 
175  m_inited = true;
176 }

178 void HybridVCoord::compute_deltas ()
179 {
180  // This is obviously not for speed (tiny amount of work, and only at setup),
181  // but rather to delegate the logic to a single place (namely ColumnOps),
182  // so that we avoid making mistakes by replicating the same algorithm
183  hybrid_ai_delta = ExecViewManaged<Scalar[NUM_LEV]>("delta hyai");
184  hybrid_bi_delta = ExecViewManaged<Scalar[NUM_LEV]>("delta hybi");
185  dp0 = ExecViewManaged<Scalar[NUM_LEV]>("dp0");
186 
187  // Create local copies, to avoid issue of lambda on GPU
188  auto hyai = hybrid_ai_packed;
189  auto hybi = hybrid_bi_packed;
190  auto dhyai = hybrid_ai_delta;
191  auto dhybi = hybrid_bi_delta;
192  auto ldp0 = dp0;
193  auto lps0 = ps0;
194  auto policy = Homme::get_default_team_policy<ExecSpace>(1);
195  Kokkos::parallel_for("[HybridVCoord::compute_deltas]",policy,
196  KOKKOS_LAMBDA(const TeamMember& team) {
197  KernelVariables kv(team);
198  Kokkos::parallel_for(Kokkos::TeamThreadRange(team,1),[&](int /*unused*/){
199  ColumnOps::compute_midpoint_delta(kv,hyai,dhyai);
200  ColumnOps::compute_midpoint_delta(kv,hybi,dhybi);
201  Kokkos::parallel_for(Kokkos::ThreadVectorRange(kv.team,NUM_LEV),
202  [&](const int ilev) {
203  ldp0(ilev) = dhyai(ilev) * lps0 + dhybi(ilev) * lps0;
204  });
205  });
206  });
207 }
