263  subroutine set_prescribed_wind(elem,deriv,hybrid,hv,dt,tl,nets,nete,eta_ave_w)
264 
265  type (element_t), intent(inout), target :: elem(:)
266  type (derivative_t), intent(in) :: deriv
267  type (hvcoord_t), intent(inout) :: hv
268  type (hybrid_t), intent(in) :: hybrid
269  real (kind=real_kind), intent(in) :: dt
270  type (timelevel_t) , intent(in) :: tl
271  integer , intent(in) :: nets
272  integer , intent(in) :: nete
273  real (kind=real_kind), intent(in) :: eta_ave_w
274 
275  real (kind=real_kind), parameter :: half = 0.5d0
276 
277  real (kind=real_kind) :: dp(np,np)! pressure thickness, vflux
278  real(kind=real_kind) :: eta_dot_dpdn(np,np,nlevp)
279 
280  integer :: ie,k,n0,np1
281 
282  n0 = tl%n0
283  np1 = tl%np1
284 
285  if (midpoint_eta_dot_dpdn) then
286  ! To get second order in the vertical direction, we need to
287  ! approximate eta_dot_dpdn at the time midpoint.
288  if (dt_remap_factor == 0) then
289  ! Accumulate first part of the midpoint.
290  do ie = nets,nete
291  elem(ie)%derived%eta_dot_dpdn = elem(ie)%derived%eta_dot_dpdn + &
292  half*elem(ie)%derived%eta_dot_dpdn_prescribed*eta_ave_w
293  end do
294  else
295  ! Save previous prescribed value.
296  do ie = nets,nete
297  elem(ie)%derived%eta_dot_dpdn = elem(ie)%derived%eta_dot_dpdn_prescribed
298  end do
299  end if
300  end if
301  call set_test_prescribed_wind(elem,deriv,hybrid,hv,dt,tl,nets,nete)
302  ! accumulate velocities and fluxes over timesteps
303  ! test code only dont bother to openmp thread
304  do ie = nets,nete
305  eta_dot_dpdn(:,:,:)=elem(ie)%derived%eta_dot_dpdn_prescribed(:,:,:)
306  ! accumulate mean fluxes for advection
307  if (midpoint_eta_dot_dpdn) then
308  if (dt_remap_factor == 0) then
309  ! Accumulate second part of the midpoint.
310  elem(ie)%derived%eta_dot_dpdn = elem(ie)%derived%eta_dot_dpdn + &
311  half*elem(ie)%derived%eta_dot_dpdn_prescribed*eta_ave_w
312  else
313  ! Calculate value at time midpoint.
314  eta_dot_dpdn = half*(elem(ie)%derived%eta_dot_dpdn + &
315  elem(ie)%derived%eta_dot_dpdn_prescribed)
316  end if
317  else
318  if (dt_remap_factor == 0) then
319  elem(ie)%derived%eta_dot_dpdn(:,:,:) = &
320  elem(ie)%derived%eta_dot_dpdn(:,:,:) + eta_dot_dpdn(:,:,:)*eta_ave_w
321  end if
322  end if
323  if (dt_remap_factor /= 0) then
324  ! lagrangian case. mean vertical velocity = 0
325  elem(ie)%derived%eta_dot_dpdn(:,:,:) = 0
326  ! update position of floating levels
327  do k=1,nlev
328  elem(ie)%state%dp3d(:,:,k,np1) = elem(ie)%state%dp3d(:,:,k,n0) &
329  + dt*(eta_dot_dpdn(:,:,k+1) - eta_dot_dpdn(:,:,k))
330  enddo
331  end if
332  ! accumulate U*dp
333  do k=1,nlev
334  elem(ie)%derived%vn0(:,:,1,k)=elem(ie)%derived%vn0(:,:,1,k)+&
335  eta_ave_w*elem(ie)%state%v(:,:,1,k,n0)*elem(ie)%state%dp3d(:,:,k,tl%n0)
336  elem(ie)%derived%vn0(:,:,2,k)=elem(ie)%derived%vn0(:,:,2,k)+&
337  eta_ave_w*elem(ie)%state%v(:,:,2,k,n0)*elem(ie)%state%dp3d(:,:,k,tl%n0)
338  enddo
339 
340  enddo
341  end subroutine
