704  subroutine ghost_exchangevfull(par,ithr,buffer)

707 ! element of ghost cell information, including corner

713  use dimensions_mod, only: nelemd

742 
743 
744 #ifdef _MPI

746  pschedule => schedule(1)

748 
749  nsendcycles = pschedule%nSendCycles
750  nrecvcycles = pschedule%nRecvCycles

754  !==================================================
755  do icycle=1,nsendcycles
756  pcycle => pschedule%SendCycle(icycle)

767  endif

769

771  ! Post the Receives
772  !==================================================
773  do icycle=1,nrecvcycles
774  pcycle => pschedule%RecvCycle(icycle)
775  source = pcycle%source - 1
776  length = nlyr * pcycle%lengthP_ghost * buffer%elem_size
777  tag = pcycle%tag

780  call mpi_irecv(buffer%receive(1,1,1,iptr),length,mpireal_t, &

786  endif
787  end do ! icycle

789

792  !==================================================

798  pcycle => pschedule%RecvCycle(icycle)
799  length = pcycle%lengthP_ghost
800  iptr = pcycle%ptrP_ghost
801  do i=0,length-1
802  buffer%buf(:,:,1:nlyr,iptr+i) = buffer%receive(:,:,1:nlyr,iptr+i)
803  enddo
804  end do ! icycle
805 
806 
807 #endif
808  endif ! if (hybrid%ithr == 0)

811 #endif
