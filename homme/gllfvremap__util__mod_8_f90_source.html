98  subroutine set_gll_state(hvcoord, elem, nt1, nt2)
99  ! Set all quantities used in gfr_dyn_to_fv_phys and gfr_fv_phys_to_dyn to
100  ! C^inf functions for convergence testing (as well as property preservation
101  ! testing, but the C^inf part is needed only for convergence testing).
102  ! The C^inf functions are various sinusoidal 3D functions from which the
103  ! values on the 2D sphere are extracted. This is an easy way to get a C^inf
104  ! function on the sphere.
105 
106  use dimensions_mod, only: nlev, qsize
107  use physical_constants, only: g, dd_pi
108  use coordinate_systems_mod, only: cartesian3d_t, change_coordinates
109  use hybvcoord_mod, only: hvcoord_t
110  use element_ops, only: get_field
111 
112  type (hvcoord_t) , intent(in) :: hvcoord
113  type (element_t), intent(inout) :: elem
114  integer, intent(in) :: nt1, nt2
115 
116  type (State_t) :: s1
117  type (cartesian3D_t) :: p
118  real(kind=real_kind) :: wr(np,np,nlev,2)
119  integer :: i, j, k, q, d, tl
120 
121  elem%state%Q(:,:,:,1) = zero ! moisture tracer is 0
122  do j = 1,np
123  do i = 1,np
124  p = change_coordinates(elem%spherep(i,j))
125  do k = 1,nlev
126  do q = 2,qsize
127  elem%state%Q(i,j,k,q) = one + &
128  half*sin((half + modulo(q,2))*p%x)* &
129  sin((half + modulo(q,3))*1.5d0*p%y)* &
130  sin((-2.3d0 + modulo(q,5))*p%z)
131  end do
132  end do
133  s1%ps(i,j) = 1.0d3*(one + 0.05d0*sin(two*p%x+half)*sin(p%y+1.5d0)*sin(3*p%z+2.5d0))
134  s1%phis(i,j) = one + half*sin(p%x-half)*sin(half*p%y+2.5d0)*sin(2*p%z-2.5d0)
135  do k = 1,nlev
136  ! u, v have to be set carefully because they are
137  ! converted to contravariant velocity. Thus, at the
138  ! poles, we need u, v to make sense to measure OOA
139  ! correctly.
140  wr(i,j,k,1) = sin(p%x)*sin(1.5*p%y)*cos(1.7*p%z)
141  wr(i,j,k,2) = sin(half*p%x)*sin(1.5*p%y)*cos(half*dd_pi*p%z)
142  elem%derived%omega_p(i,j,k) = wr(i,j,k,1)
143  end do
144  do k = 1,nlev
145  s1%T(i,j,k) = one + half*sin(p%x+1.5d0)*sin(1.5d0*p%y+half)*sin(two*p%z-half)
146  end do
147  end do
148  end do
149  s1%u = wr(:,:,:,1)
150  s1%v = wr(:,:,:,2)
151  do k = 1,nlev
152  s1%p(:,:,k) = hvcoord%hyam(k)*hvcoord%ps0 + hvcoord%hybm(k)*s1%ps
153  s1%dp(:,:,k) = (hvcoord%hyai(k+1) - hvcoord%hyai(k))*hvcoord%ps0 + &
154  (hvcoord%hybi(k+1) - hvcoord%hybi(k))*s1%ps
155  end do
156  s1%z = zero
157  s1%zi = zero
158  ! a bit of a kludge
159  call set_state(s1, hvcoord, nt1, nt2, nt1, elem)
160  call get_field(elem, 'rho', wr(:,:,:,1), hvcoord, nt1, nt1)
161  s1%w = -elem%derived%omega_p/(wr(:,:,:,1)*g)
162  s1%wi(:,:,:nlev) = s1%w
163  s1%wi(:,:,nlevp) = s1%w(:,:,nlev)
164  call set_state(s1, hvcoord, nt1, nt2, nt1, elem)
165  do q = 1,qsize
166  do tl = nt1,nt2
167  elem%state%Qdp(:,:,:,q,tl) = &
168  elem%state%Q(:,:,:,q)*elem%state%dp3d(:,:,:,nt1)
169  end do
170  end do
171  end subroutine set_gll_state

215  ! Test physgrid API.
216  ! Test that if tendency is 0, then the original field is
217  ! recovered with error eps ~ machine precision.
218  ! OOA for pgN should min(N, 2). The OOA is limited to 2 because
219  ! of the way the tendencies are created. Instead of setting the FV
220  ! value to the average over the subcell, the sampled value at the
221  ! cell midpoint (ref midpoint mapped to sphere) is used. This
222  ! creates a 2nd-order error. Tests 2 and 3 test whether the remaps

241  (hvcoord%hybi(k+1) - hvcoord%hybi(k))* &
242  reshape(pg_data%ps(:,ie), (/nf,nf/))
243  end do
244  end if
245  do j = 1,nf
246  do i = 1,nf
247  col = nf*(j-1) + i
248  call gfr_f_get_cartesian3d(ie, i, j, p)
249  f = 0.25_real_kind*sin(3.2*p%x)*sin(4.2*p%y)*sin(2.3*p%z)
250  pg_data%uv(col,:,:,ie) = f/dt
251  pg_data%T(col,:,ie) = f/dt
252  ! no moisture adjustment => no dp3d adjustment
253  pg_data%q(col,:,1,ie) = zero
254  if (ftype == 0) then
255  do q = 2,qsize
256  pg_data%q(col,:,q,ie) = f*wg(i,j,:)/dt
257  end do
258  else
259  pg_data%q(col,:,2:qsize,ie) = pg_data%q(col,:,2:qsize,ie) + f
260  end if
261  end do
262  end do
263  end do
264  else

275  call gfr_pg1_reconstruct(hybrid, nt2, dt, hvcoord, elem, nets, nete)
276

279  call applycamforcing_tracers(elem(ie), hvcoord, nt2, nt2, dt, logical(ftype /= 0))
280  end do
281  call applycamforcing_dynamics(elem, hvcoord, nt2, dt, nets, nete)
282 
283  ! Test GLL state nt2 vs the original state nt1.
284  if (hybrid%masterthread) write(iulog, '(a,l2)') 'gfrt> tendency', tendency
285  tend = zero
286  mass1 = zero; mass2 = zero
287  qmin1 = one; qmax1 = -one
288  qmin2 = one; qmax2 = -one
289  do q = 2, qsize+4
290  do ie = nets,nete
291  do k = 1,nlev
292  wg(:,:,k) = elem(ie)%spheremp
293  end do

295  do j = 1,np
296  do i = 1,np
297  p = change_coordinates(elem(ie)%spherep(i,j))

308  (elem(ie)%state%v(:,:,qi,:,nt1) + tend))**2)

310  sum(wg*(elem(ie)%state%v(:,:,qi,:,nt1) + tend)**2)
311  elseif (qi == 3) then
312  call get_temperature(elem(ie), wg1, hvcoord, nt1)
313  call get_temperature(elem(ie), wg2, hvcoord, nt2)

315  global_shared_buf(ie,2) = sum(wg*(wg1 + tend)**2)
316  else
317  ! Test omega_p, phis, ps. These were remapped to FV
318  ! but don't get remapped to GLL. Make sure they all

654  function gfr_pgn_to_smoothed_topo(par, elem, output_nphys, intopofn, outtopoprefix) result(stat)
655 #ifndef CAM
656  use common_io_mod, only: varname_len
657  use gllfvremap_mod, only: gfr_init, gfr_finish, gfr_fv_phys_to_dyn_topo, &
658  gfr_dyn_to_fv_phys_topo, gfr_f_get_latlon

660  use physical_constants, only: dd_pi

667 
668  type (parallel_t), intent(in) :: par
669  type (element_t), intent(inout) :: elem(:)

672  integer :: stat
673 
674 #ifndef CAM
675  real(real_kind), allocatable :: gll_fields(:,:,:,:), pg_fields(:,:,:)
676  integer :: intopo_nphys, ie, i, j, k, nvar, nf2
677  character(len=varname_len) :: fieldnames(1)
678  real(real_kind) :: rad2deg
679  logical :: write_latlon

681

683

685  if (write_latlon) nvar = 3

696  call gfr_init(par, elem, intopo_nphys)

698  do ie = 1,nelemd

700  end do

704  end do
705  call gfr_finish()

707  ! Smooth on the GLL grid.
708  hybrid = hybrid_create(par, 0, 1)
709  call smooth_topo_datasets(elem, hybrid, 1, nelemd)
710  do ie = 1,nelemd
711  gll_fields(:,:,ie,1) = elem(ie)%state%phis
712  end do
713 
714  ! Map the GLL data to the target physgrid, e.g., pg2.
715  call gfr_init(par, elem, output_nphys)
716  call gfr_dyn_to_fv_phys_topo(par, elem, pg_fields(:,:,1))
717  if (write_latlon) then
718  rad2deg = 180.0_real_kind/dd_pi
719  do ie = 1,nelemd
720  do j = 1,output_nphys
721  do i = 1,output_nphys
722  k = output_nphys*(j-1) + i
723  call gfr_f_get_latlon(ie, i, j, pg_fields(k,ie,2), pg_fields(k,ie,3))
724  end do
725  end do
726  end do

728  pg_fields(:nf2,:,2:3) = pg_fields(:nf2,:,2:3)*rad2deg
729  do ie = 1,nelemd
730  do j = 1,np
731  do i = 1,np
732  gll_fields(i,j,ie,2) = elem(ie)%spherep(i,j)%lat*rad2deg
733  gll_fields(i,j,ie,3) = elem(ie)%spherep(i,j)%lon*rad2deg
734  end do
735  end do
736  end do
737  end if
738  call gfr_finish()

740  ! Write the netcdf file that will be used as the --smoothed-topography file
741  ! in the second run of cube_to_target. Only ncol, PHIS are needed for
742  ! this. We include PHIS_d for use in the final assembled GLL-physgrid topo
743  ! file. Optionally, we also include physgrid and GLL lat-lon data for easy
744  ! visualization using just this file.
745  call write_physgrid_smoothed_phis_file(outtopoprefix, elem, par, &
746  gll_fields, pg_fields, output_nphys, &
747  'Created from '// trim(intopofn) // ' by HOMME gfr_pgn_to_smoothed_topo', &
748  write_latlon)
749 
750  deallocate(gll_fields, pg_fields)
751 #endif
752  stat = 0
