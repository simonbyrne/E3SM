122 
123 #ifdef _MPI

131 #endif

134  ! Basic MPI initialization
135  ! ================================================
136 
137  call mpi_initialized(running,ierr)
138 
139  if (.not.running) then
140  call mpi_init(ierr)
141  end if
142 
143  par%root = 0
144 
145 #ifdef CAM
146  call mpi_comm_size(mpicom,npes_cam,ierr)
147  if(present(npes_in)) then
148  npes_homme=npes_in
149  else
150  npes_homme=npes_cam
151  end if
152  call mpi_comm_rank(mpicom,iam_cam,ierr)
153  if (present(npes_stride)) npes_cam_stride = npes_stride
154  ! Determine maximum stride and make sure user defined stride is not too big.
155  max_stride = npes_cam/npes_homme
156  if ( npes_cam_stride == 0 .or. npes_cam_stride .gt. max_stride ) npes_cam_stride = max_stride
157  if (mod(iam_cam,npes_cam_stride).eq.0.and.iam_cam.lt.npes_homme*npes_cam_stride) color = 0
158  call mpi_comm_split(mpicom, color, iam_cam, par%comm, ierr)
159  par%dynproc = .false.
160  if (color == 0) par%dynproc = .true.
161 #else
162  par%comm = mpi_comm_world
163  par%dynproc = .true.
164 #endif
165  call mpi_comm_rank(par%comm,par%rank,ierr)
166  call mpi_comm_size(par%comm,par%nprocs,ierr)
167 
168  par%masterproc = .false.
169  if(par%rank .eq. par%root) par%masterproc = .true.
170  if (par%masterproc) write(iulog,*)'number of MPI processes: ',par%nprocs
171  if (par%masterproc) write(iulog,*)'MPI processors stride: ',npes_cam_stride
172 #else
173  par%root = 0
174  par%rank = 0
175  par%nprocs = 1
176  par%comm = -1
177  par%masterproc = .true.
178 #endif

181  subroutine initmp_from_par(par)
182  type (parallel_t),intent(in):: par
183 #ifdef _MPI
184  character(len=MPI_MAX_PROCESSOR_NAME) :: my_name
185  character(len=MPI_MAX_PROCESSOR_NAME), allocatable :: the_names(:)
186  integer(kind=int_kind),allocatable :: tarray(:)
187  integer(kind=int_kind) :: namelen,i
188  integer(kind=int_kind) :: ierr,tmp_min,tmp_max
189  integer :: node_color
190 
191  if (mpi_double_precision==20 .and. mpi_real8==18) then
192  ! LAM MPI defined MPI_REAL8 differently from MPI_DOUBLE_PRECISION
193  ! and LAM MPI's allreduce does not accept on MPI_REAL8
194  mpireal_t = mpi_double_precision
195  else
196  mpireal_t = mpi_real8
197  endif
198 
199  ! this type is only to use mpi_minloc and mpi_maxloc in print_state()
200  ! on a machine where MPIreal_t != MPI_DOUBLE_PRECISION there will be
201  ! truncation in calls with maxloc, minloc and loss of reproducibility
202  mpi2real_t = mpi_2double_precision
203 
204  mpiinteger_t = mpi_integer
205  mpichar_t = mpi_character
206  mpilogical_t = mpi_logical
207 
208  ! ================================================
209  ! Determine where this MPI process is running
210  ! then use this information to determined the
211  ! number of MPI processes per node
212  ! ================================================
213 
214  my_name(:) = ''
215  call mpi_get_processor_name(my_name,namelen,ierr)
216 
217  allocate(the_names(par%nprocs))
218  do i=1,par%nprocs
219  the_names(i)(:) = ''
220  enddo
221  ! ================================================
222  ! Collect all the machine names
223  ! ================================================
224  call mpi_allgather(my_name,mpi_max_processor_name,mpi_character, &
225  the_names,mpi_max_processor_name,mpi_character,par%comm,ierr)
226 
227  ! ======================================================================
228  ! Calculate how many other MPI processes are on my node
229  ! ======================================================================
230  node_color=0
231  nmpi_per_node = 0
232  do i=1,par%nprocs
233  if( trim(adjustl(my_name)) .eq. trim(adjustl(the_names(i))) ) then
234  nmpi_per_node = nmpi_per_node + 1
235  if (node_color==0) node_color=i
236  endif
237  enddo
238  if (node_color==0) call abortmp("initmp: Errror computing procs per node")
239 
240  ! create a communicator of all procs per node
241  ! currently not used, so commenting out
242 ! call mpi_comm_split(par%comm, node_color, par%rank, par%node_comm, ierr)
243 ! call MPI_comm_rank(par%node_comm,par%node_rank,ierr)
244 ! call MPI_comm_size(par%node_comm,par%node_nprocs,ierr)
245 
246  ! =======================================================================
247  ! Verify that everybody agrees on this number otherwise do not do
248  ! the multi-level partitioning
249  ! =======================================================================
250  call mpi_allreduce(nmpi_per_node,tmp_min,1,mpiinteger_t,mpi_min,par%comm,ierr)
251  call mpi_allreduce(nmpi_per_node,tmp_max,1,mpiinteger_t,mpi_max,par%comm,ierr)
252  if (par%masterproc) write(iulog,*)'number of MPI processes per node: min,max=',&
253  tmp_min,tmp_max
254  if(tmp_min .ne. tmp_max) then
255  if (par%masterproc) write(iulog,*)'initmp: disagrement accross nodes for nmpi_per_node'
256  nmpi_per_node = 1
257  partitionfornodes=.false.
258  else
259  partitionfornodes=.true.
260  endif
261 
262 
263  deallocate(the_names)
264 
265 #else
266  nmpi_per_node = 2
267  partitionfornodes = .true.
268 #endif
269  !===================================================
270  ! Kind of lame but set this variable to be 1 based
271  !===================================================
272  iam = par%rank+1
273 
274  end subroutine initmp_from_par
