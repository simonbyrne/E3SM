13 /*
14  * ColumnOps: a series of utility kernels inside an element
15  *
16  * This class is responsible of implementing common kernels used in the
17  * preqx and theta models to compute quantities at level midpoints and
18  * level interfaces. For instance, compute interface quantities from midpoints
19  * ones, or integrate over a column, or compute increments of midpoint
20  * quantities (which will be defined at interfaces).
21  * The kernels are meant to be launched from within a parallel region, with
22  * team policy. More precisely, they are meant to be called from a parallel
23  * region dispatched over the number of thread in a single team. In other words,
24  * you should be inside a TeamThreadRange parallel loop before calling these
25  * kernels, but you should *not* be inside a ThreadVectorRange loop, since these
26  * kernels will attempt to create such loops.
27  *
28  * In the compute_* methods, InputProvider can either be a functor (e.g., a lambda) or a 1d view.
29  * The only requirement is that operator()(int)->Scalar is defined.
30  * For instance, one could use a lambda to compute the midpoint average of the product of two
31  * interface quantities, like this:
32  *
33  * auto prod = [&](const int ilev)->Scalar { return x(ilev)*y(ilev); }
34  * column_ops.compute_midpoint_values(kv,prod,output);
35  *
36  * Note: all methods have a different impl, depending on whether VECTOR_SIZE>1.
37  * The if statement is evaluated at compile-time, so there is no run-time
38  * penalization. The only requirement is that both branches must compile.
39  * Also, the impl for VECTOR_SIZE>1 is not thread safe (no ||for or single),
40  * so it would *not* work on GPU. Hence, the whole class has a static_assert
41  * to make sure we have VECTOR_SIZE=1 or that we are not in a GPU build.
42  */

240  template<CombineMode CM = CombineMode::Replace,
241  typename InputProvider = DefaultMidProvider>
242  KOKKOS_INLINE_FUNCTION
243  static void compute_interface_delta (const KernelVariables& kv,
244  const InputProvider& x_m,
245  const ExecViewUnmanaged<Scalar [NUM_LEV_P]> dx_i,
246  const Real alpha = 1.0, const Real beta = 0.0,
247  const Real bcVal = 0.0)
248  {
249  // Compute increment of midpoint values at interfaces.
250  // Top and bottom interfaces are set to bcVal.
251  if (VECTOR_SIZE==1) {
252  Kokkos::parallel_for(Kokkos::ThreadVectorRange(kv.team,1,NUM_PHYSICAL_LEV),
253  [=](const int& ilev) {
254  combine<CM>(x_m(ilev)-x_m(ilev-1),dx_i(ilev),alpha,beta);
255  });
256 
257  Kokkos::single(Kokkos::PerThread(kv.team),[&](){
258  combine<CM>(bcVal, dx_i(0)[0], alpha, beta);
259  combine<CM>(bcVal, dx_i(NUM_INTERFACE_LEV-1)[0], alpha, beta);
260  });
261  } else {
262  constexpr int LAST_MID_PACK = MIDPOINTS::LastPack;
263  constexpr int LAST_INT_PACK = INTERFACES::LastPack;
264  constexpr int LAST_INT_PACK_END = INTERFACES::LastPackEnd;
265 
266  // Try to use SIMD operations as much as possible
267  // Note: resist the tempation to go all the way to LAST_MID_PACK.
268  // If you do, and LAST_MID_PACK==LAST_INT_PACK, garbage may
269  // be present in LAST_MID_PACK_END+1, which would pollute
270  // values in the last pack of dx_i at LAST_INT_PACK_END
271  for (int ilev=1; ilev<LAST_MID_PACK; ++ilev) {
272  Scalar tmp = x_m(ilev);
273  tmp.shift_right(1);
274  tmp[0] = x_m(ilev-1)[VECTOR_END];
275  combine<CM>(x_m(ilev) - tmp, dx_i(ilev), alpha, beta);
276  }
277 
278  // First and last pack need to be treated separately, due the bc.
279  Scalar tmp = x_m(0);
280  tmp.shift_right(1);
281  tmp = x_m(0)-tmp;
282  tmp[0] = bcVal;
283  combine<CM>(tmp, dx_i(0), alpha, beta);
284 
285  tmp = x_m(LAST_MID_PACK);
286  tmp.shift_right(1);
287  tmp[0] = x_m(LAST_MID_PACK-1)[VECTOR_END];
288  tmp = x_m(LAST_MID_PACK)-tmp;
289  if (LAST_INT_PACK==LAST_MID_PACK) {
290  // Be careful not to pollute the last interface with garbage
291  tmp[LAST_INT_PACK_END] = bcVal;
292  combine<CM>(tmp, dx_i(LAST_INT_PACK), alpha, beta);
293  } else {
294  combine<CM>(tmp, dx_i(LAST_MID_PACK), alpha, beta);
295 
296  // Fix bc as a single Real op.
297  combine<CM>(bcVal,dx_i(LAST_INT_PACK)[LAST_INT_PACK_END], alpha, beta);
298  }
299  }

317  template<int PackLength, bool Forward,bool Inclusive,int LENGTH,typename InputProvider>
318  KOKKOS_INLINE_FUNCTION
319  static typename std::enable_if<(PackLength>1)>::type
320  column_scan_impl (const KernelVariables& /* kv */,
321  const InputProvider& input_provider,
322  const ExecViewUnmanaged<Scalar [ColInfo<LENGTH>::NumPacks]>& sum,
323  const Real s0 = 0.0)
324  {
325  constexpr int OFFSET = Inclusive ? 0 : 1;
326  constexpr int LOOP_RAW_SIZE = LENGTH - OFFSET;
327  constexpr int LOOP_LAST_PACK = ColInfo<LOOP_RAW_SIZE>::LastPack;
328  constexpr int LOOP_LAST_PACK_END = ColInfo<LOOP_RAW_SIZE>::LastPackEnd;
329  constexpr int LOOP_SIZE = ColInfo<LOOP_RAW_SIZE>::NumPacks;
330  constexpr int LAST_PACK = ColInfo<LENGTH>::LastPack;
331  constexpr int LAST_PACK_END = ColInfo<LENGTH>::LastPackEnd;
332 
333  // It is easier to write two loops for Forward true/false. There's no runtime penalty,
334  // since the if is evaluated at compile time, so no big deal.
335  if (Forward) {
336  // Running integral
337  Real integration = s0;
338 
339  for (int ilev = 0; ilev<LOOP_SIZE; ++ilev) {
340  // In all but the last level pack, the loop is over the whole pack
341  const int vec_end = (ilev == LOOP_LAST_PACK ? LOOP_LAST_PACK_END : VECTOR_END);
342 
343  auto input = input_provider(ilev);
344 
345  // Integrate
346  auto& sum_val = sum(ilev);
347  sum_val[0] = integration + (Inclusive ? input[0] : 0.0);
348  for (int iv = 1; iv <= vec_end; ++iv) {
349  sum_val[iv] = sum_val[iv - 1] + (Inclusive ? input[iv] : input[iv-1]);
350  }
351 
352  // Update running integral
353  integration = sum_val[vec_end] + (Inclusive ? 0.0 : input[vec_end]);;
354  }
355 
356  // In an exclusive sum, the procedure above failed to update the last level
357  if (!Inclusive) {
358  sum(LAST_PACK)[LAST_PACK_END] = integration;
359  }
360  } else {
361  // Running integral
362  Real integration = s0;
363 
364  // In an exclusive sum, the procedure below would fail to add
365  // the input's last level to the output's second-to-last level
366  if (!Inclusive) {
367  integration += input_provider(LAST_PACK)[LAST_PACK_END];
368  }
369 
370  for (int ipack=0; ipack<LOOP_SIZE; ++ipack) {
371  const int ilev = LOOP_LAST_PACK-ipack;
372 
373  // In all but the last level pack, the loop is over the whole vector
374  const int vec_start = (ilev == LOOP_LAST_PACK ? LOOP_LAST_PACK_END : VECTOR_END);
375 
376  auto input = input_provider(ilev);
377  // Integrate
378  auto& sum_val = sum(ilev);
379  sum_val[vec_start] = integration + (Inclusive ? input[vec_start] : 0.0);
380  for (int iv = vec_start - 1; iv >= 0; --iv) {
381  sum_val[iv] = sum_val[iv + 1] + (Inclusive ? input[iv] : input[iv+1]);
382  }
383 
384  // Update running integral
385  integration = sum_val[0] + (Inclusive ? 0.0 : input[0]);
386  }
387  }
388  }

390  template<int PackLength,bool Forward,bool Inclusive,int LENGTH,typename InputProvider>
391  KOKKOS_INLINE_FUNCTION
392  static typename std::enable_if<PackLength==1>::type
393  column_scan_impl (const KernelVariables& kv,
394  const InputProvider& input_provider,
395  const ExecViewUnmanaged<Scalar [ColInfo<LENGTH>::NumPacks]>& sum,
396  const Real s0 = 0.0)
397  {
398  if (Forward) {
399  // If exclusive, no need to go to access last input level
400  constexpr int offset = Inclusive ? 0 : 1;
401  constexpr int loop_size = LENGTH - offset;
402 
403  Dispatch<>::parallel_scan(kv.team, loop_size,
404  [&](const int k, Real& accumulator, const bool last) {
405  if (k==0) {
406  // First entry from the bottom: set initial value
407  accumulator = s0;
408  }
409  accumulator += input_provider(k)[0];
410 
411  if (last) {
412  sum(k+offset)[0] = accumulator;
413  }
414  });
415  } else {
416  // If exclusive, no need to go to access first input level
417  constexpr int offset = Inclusive ? 0 : 1;

422  // k_bwd must range in (loop_size,0], while k ranges in [0, loop_size).
423  const int k_bwd = LENGTH - k - 1;

426  // First entry from the top: set initial value
427  accumulator = s0;
428  }
429  accumulator += input_provider(k_bwd)[0];
430 
431  if (last) {

474  template<int LENGTH, typename InputProvider, bool PackedReduction =

476  false

481  KOKKOS_INLINE_FUNCTION
482  static void column_reduction (const KernelVariables& kv,
483  const InputProvider& input,
484  Real& sum)
485  {
486  if (PackedReduction) {
487  // To squeeze some perf out on CPU, do reduction at Scalar level,
488  // then reduce the Scalar at the end. Be CAREFUL: the last pack may
489  // contain some garbage if VECTOR_SIZE does not divide LENGTH!
490 
491  constexpr int NUM_PACKS = ColInfo<LENGTH>::NumPacks;
492  constexpr int LAST_PACK = ColInfo<LENGTH>::LastPack;
493  constexpr int LAST_PACK_LEN = ColInfo<LENGTH>::LastPackLen;
494  constexpr bool HAS_GARBAGE = LAST_PACK_LEN!=VECTOR_SIZE;
495  constexpr int NUM_PACKS_M1 = NUM_PACKS>0 ? NUM_PACKS-1 : 0;
496  constexpr int LOOP_LENGTH = HAS_GARBAGE ? NUM_PACKS_M1 : NUM_PACKS;
497 
498  Scalar packed_sum;
499  Dispatch<>::parallel_reduce(kv.team,Kokkos::ThreadVectorRange(kv.team,LOOP_LENGTH),
500  [&](const int ilev, Scalar& accumulator){
501  accumulator += input(ilev);
502  },packed_sum);
503 
504  // If last pack has garbage, we did not include it in the previous reduction,
505  // so manually add the last pack (only the non-garbage part)
506  if (HAS_GARBAGE) {
