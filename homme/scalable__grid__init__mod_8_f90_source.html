321  subroutine sgi_genspacecurve(ne, Mesh)
322  use parallel_mod, only: abortmp

324

328  integer :: i, i2, j, j2, k, ne2, sfc_index
329 
330  if(isfactorable(ne)) then
331  call genspacecurve(mesh)
332  ! call PrintCurve(Mesh)
333  else
334  ! find the smallest ne2 which is a power of 2 and ne2>ne
335  ne2=2**ceiling( log(real(ne))/log(2d0) )
336  if (ne2<ne) call abortmp('Fatal SFC error')
337 
338  allocate(mesh2(ne2,ne2))
339  allocate(mesh2_map(ne2,ne2,2))
340  allocate(sfcij(0:ne2*ne2,2))
341 
342  call genspacecurve(mesh2) ! SFC partition for ne2
343 
344  ! associate every element on the ne x ne mesh (Mesh)
345  ! with its closest element on the ne2 x ne2 mesh (Mesh2)
346  ! Store this as a map from Mesh2 -> Mesh in Mesh2_map.
347  ! elements in Mesh2 which are not mapped get assigned a value of 0
348  mesh2_map=0
349  do j=1,ne
350  do i=1,ne
351  ! map this element to an (i2,j2) element
352  ! [ (i-.5)/ne , (j-.5)/ne ] = [ (i2-.5)/ne2 , (j2-.5)/ne2 ]
353  i2=nint( ((i-.5)/ne)*ne2 + .5 )
354  j2=nint( ((j-.5)/ne)*ne2 + .5 )
355  if (i2<1) i2=1
356  if (i2>ne2) i2=ne2
357  if (j2<1) j2=1
358  if (j2>ne2) j2=ne2
359  mesh2_map(i2,j2,1)=i
360  mesh2_map(i2,j2,2)=j
361  enddo
362  enddo
363 
364  ! create a reverse index array for Mesh2
365  ! k = Mesh2(i,j)
366  ! (i,j) = (sfcij(k,1),sfci(k,2))
367  do j=1,ne2
368  do i=1,ne2
369  k=mesh2(i,j)
370  sfcij(k,1)=i
371  sfcij(k,2)=j
372  enddo
373  enddo
374 
375  ! generate a SFC for Mesh with the same ordering as the
376  ! elements in Mesh2 which map to Mesh.
377  sfc_index=0
378  do k=0,ne2*ne2-1
379  i2=sfcij(k,1)
380  j2=sfcij(k,2)
381  i=mesh2_map(i2,j2,1)
382  j=mesh2_map(i2,j2,2)
383  if (i/=0) then
384  ! (i2,j2) element maps to (i,j) element
385  mesh(i,j)=sfc_index
386  sfc_index=sfc_index+1
387  endif
388  enddo
389 #if 0
390  print *,'SFC Mapping to non powers of 2,3 used. Mesh:'
391  do j=1,ne
392  write(*,'(99i3)') (mesh(i,j),i=1,ne)
393  enddo
394  call printcurve(mesh2)
395 #endif
396  deallocate(mesh2)
397  deallocate(mesh2_map)
398  deallocate(sfcij)
399  endif
400  end subroutine sgi_genspacecurve

403  ! rank2sfc(rank:rank+1) contains the SFC index inclusive-lower and
404  ! exclusive-upper indices for part 'rank'.

409

685  end if

705  if (i /= 1) call gm_set(gm, igv, i-1, ne, 5, swest)
706  if (i /= ne) call gm_set(gm, igv, i+1, ne, 5, seast)
707  end if
708  if (j == ne .and. k == 5) then ! north edge of 5
709  call gm_set(gm, igv, i, 1, 1, north)
710  if (i /= 1) call gm_set(gm, igv, i-1, 1, 1, nwest)
711  if (i /= ne) call gm_set(gm, igv, i+1, 1, 1, neast)
712  end if

716  if (i /= 1) call gm_set(gm, igv, ne, rev+1, 5, swest)
717  if (i /= ne) call gm_set(gm, igv, ne, rev-1, 5, seast)
718  end if
719  if (i == ne .and. k == 5) then ! east edge of 5
720  rev = ne+1-j
721  call gm_set(gm, igv, rev, 1, 2, east)
722  if (j /= 1) call gm_set(gm, igv, rev+1, 1, 2, seast)
723  if (j /= ne) call gm_set(gm, igv, rev-1, 1, 2, neast)
724  end if
725  if (j == 1 .and. k == 3) then ! south edge of 3
726  rev = ne+1-i
727  call gm_set(gm, igv, rev, 1, 5, south)
728  if (i /= 1) call gm_set(gm, igv, rev+1, 1, 5, swest)
729  if (i /= ne) call gm_set(gm, igv, rev-1, 1, 5, seast)
730  end if
731  if (j == 1 .and. k == 5) then ! south edge of 5
732  rev = ne+1-i
733  call gm_set(gm, igv, rev, 1, 3, south)
734  if (i /= 1) call gm_set(gm, igv, rev+1, 1, 3, swest)
735  if (i /= ne) call gm_set(gm, igv, rev-1, 1, 3, seast)
736  end if
737  if (j == 1 .and. k == 4) then ! south edge of 4
738  call gm_set(gm, igv, 1, i, 5, south)
739  if (i /= 1) call gm_set(gm, igv, 1, i-1, 5, swest)
740  if (i /= ne) call gm_set(gm, igv, 1, i+1, 5, seast)
741  end if
742  if (i == 1 .and. k == 5) then ! west edge of 5
743  call gm_set(gm, igv, j, 1, 4, west)
744  if (j /= 1) call gm_set(gm, igv, j-1, 1, 4, swest)
745  if (j /= ne) call gm_set(gm, igv, j+1, 1, 4, nwest)
746  end if
747  if (j == ne .and. k == 1) then ! north edge of 1
748  call gm_set(gm, igv, i, 1, 6, north)
749  if (i /= 1) call gm_set(gm, igv, i-1, 1, 6, nwest)
750  if (i /= ne) call gm_set(gm, igv, i+1, 1, 6, neast)
751  end if
752  if (j == 1 .and. k == 6) then ! south edge of 6
753  call gm_set(gm, igv, i, ne, 1, south)
754  if (i /= 1) call gm_set(gm, igv, i-1, ne, 1, swest)
755  if (i /= ne) call gm_set(gm, igv, i+1, ne, 1, seast)
756  end if
757  if (j == ne .and. k == 2) then ! north edge of 2
758  call gm_set(gm, igv, ne, i, 6, north)
759  if (i /= 1) call gm_set(gm, igv, ne, i-1, 6, nwest)
760  if (i /= ne) call gm_set(gm, igv, ne, i+1, 6, neast)
761  end if

826  ! Scalable replacement for GridGraph_mod::initgridedge. Only edges belonging
827  ! to owned or remote-used elements are instantiated. Only owned and
828  ! remote-used elements need to be represented in GridVertex.
829  subroutine sgi_initgridedge(gm)
830  use parallel_mod, only : abortmp
831  use dimensions_mod, only : max_corner_elem
832  use kinds, only: iulog
833  use gridgraph_mod, only: num_neighbors
834 
835  type (gridmanager_t), intent(inout) :: gm
836 
837  integer :: i,j,k,iptr,m,n,wgtv,wgtp,gid,igv
838  integer :: nelem,nelem_edge,inbr,igvnbr
839  integer :: mynbr_cnt, cnt, mystart, start
840  logical :: owned_or_used
841  logical, parameter :: verbose = .false.
842 
843  nelem = size(gm%gv)
844 
845  ! Count the number of relevant edges.
846  nelem_edge = 0
847  do j = 1, nelem
848  do i = 1, num_neighbors
849  mynbr_cnt = gm%gv(j)%nbrs_ptr(i+1) - gm%gv(j)%nbrs_ptr(i) ! length of neighbor location
850  mystart = gm%gv(j)%nbrs_ptr(i)
851  do m=0, mynbr_cnt-1
852  if (gm%gv(j)%nbrs_wgt(mystart + m) > 0) then ! want a non-0 weight
853  owned_or_used = gm%gv(j)%processor_number == gm%rank + 1 ! want only owned or used elements
854  if (.not. owned_or_used) then
855  gid = gm%gv(j)%nbrs(mystart + m)
856  igv = gid2igv(gm, gid)
857  owned_or_used = gm%gvid(1,igv) == gid
858  end if
859  if (owned_or_used) then
860  nelem_edge = nelem_edge + 1
861  end if
862  end if
863  end do
864  end do
865  end do
866 
867  ! Allocate enough room for just these relevant edges.
868  allocate(gm%ge(nelem_edge))
869  gm%ge(:)%reverse = .false.
870 
871  ! Fill in the relevant edges.
872  iptr = 1
873  do j = 1, nelem
874  do i = 1, num_neighbors
875  mynbr_cnt = gm%gv(j)%nbrs_ptr(i+1) - gm%gv(j)%nbrs_ptr(i)
876  mystart = gm%gv(j)%nbrs_ptr(i)
877  do m = 0, mynbr_cnt-1
878  if (gm%gv(j)%nbrs_wgt(mystart + m) > 0) then
879  owned_or_used = gm%gv(j)%processor_number == gm%rank + 1
880  if (.not. owned_or_used) then
881  gid = gm%gv(j)%nbrs(mystart + m)
882  igv = gid2igv(gm, gid)
883  owned_or_used = gm%gvid(1,igv) == gid
884  end if
885  if (.not. owned_or_used) cycle
886 
887  gm%ge(iptr)%tail => gm%gv(j)
888  gm%ge(iptr)%tail_face = mystart + m ! needs to be mystart + m (location in array)
889  gm%ge(iptr)%tail_dir = i*max_corner_elem + m ! conversion needed for setcycle
890  inbr = gm%gv(j)%nbrs(mystart+m)
891  igvnbr = gid2igv(gm, inbr)
892  gm%ge(iptr)%head => gm%gv(igvnbr)
893 
894  ! Determine which "face" of the neighbor element the edge links
895  ! (i.e. the "head_face")
896  do k = 1, num_neighbors
897  cnt = gm%gv(igvnbr)%nbrs_ptr(k+1) - gm%gv(igvnbr)%nbrs_ptr(k)
898  start = gm%gv(igvnbr)%nbrs_ptr(k)
899  do n = 0, cnt-1
900  if (gm%gv(igvnbr)%nbrs(start+n) == gm%gv(j)%number) then
901  gm%ge(iptr)%head_face = start+n ! needs to be start + n (location in array)
902  gm%ge(iptr)%head_dir = k*max_corner_elem+n ! conversion (un-done in setcycle)
903  endif
904  enddo
905  enddo
906  iptr=iptr+1
907  end if
908  end do ! m loop
909  end do ! i loop
910  end do ! j loop
911  if (nelem_edge+1 /= iptr) &
912  call abortmp('Error in initgridedge: Number of edges less than expected.')
913 
914  if (verbose) then
915  print *
916  write(iulog,*)"element edge tail,head list: (TEST)"
917  do i=1,nelem_edge
918  write(iulog,*)gm%ge(i)%tail%number,gm%ge(i)%head%number
919  end do
920  print *
921  write(iulog,*)"element edge tail_face, head_face list: (TEST)"
922  do i=1,nelem_edge
923  write(iulog,*)gm%ge(i)%tail_face,gm%ge(i)%head_face
924  end do
925  end if
926  end subroutine sgi_initgridedge
