18 // ================ Subviews of several ranks views ======================= //
19 // Note: we template on ScalarType to allow both Real and Scalar case, and
20 // also to allow const/non-const versions.
21 // Note: we assume to have exactly one runtime dimension.
22 template <typename ScalarType, int DIM1, typename MemSpace,
23  typename... Properties>
24 KOKKOS_INLINE_FUNCTION ViewUnmanaged<ScalarType[DIM1], MemSpace>
25 subview(ViewType<ScalarType * [DIM1], MemSpace, Properties...> v_in, int ie) {
26  assert(v_in.data() != nullptr);
27  assert(ie < v_in.extent_int(0));
28  assert(ie >= 0);
29  return ViewUnmanaged<ScalarType[DIM1], MemSpace>(
30  &v_in.impl_map().reference(ie, 0));
31 }
