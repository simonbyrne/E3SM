5  ! Functions for setting up idealized and perturbed initial conditions
6  ! for the Jablonowski-Williamson baroclinic instability with invariant
7  ! tracers (Jablonowski and Williamson, QJ 2006).
8  !
9  ! Options:
10  ! moist 1 for the moist baroclinic instability
11  ! 0 for the dry baroclinic instability test
12  ! X scale factor of the Earth
13  !
14  ! Given a point specified by:
15  ! lon longitude (radians)
16  ! lat latitude (radians)
17  ! p/z pressure (Pa)/height (m)
18  ! zcoords 1 if z is specified, 0 if p is specified
19  !
20  ! the functions will return:
21  ! p pressure if z is specified and zcoords = 1 (Pa)
22  ! u zonal wind (m s^-1)
23  ! v meridional wind (m s^-1)
24  ! w vertical velocity (m s^-1)
25  ! t temperature (K)
26  ! phis surface geopotential (m^2 s^-2)
27  ! ps surface pressure (Pa)
28  ! rho density (kg m^-3)
29  ! q specific humidity (kg/kg)
30  ! q1 potential temperature tracer (K)
31  ! q2 absolute value of Ertel's potential vorticity (EPV) tracer,
32  ! in PVU units for X=1: PVU = 10^-6 K m^2 kg^-1 s^-1,
33  !
34  ! Authors: Paul Ullrich, Christiane Jablonowski, James Kent

109  SUBROUTINE test4_baroclinic_wave (moist,X,lon,lat,p,z,zcoords,u,v,w,t,phis,ps,rho,q,q1,q2)
110 
111  IMPLICIT NONE
112 
113 !-----------------------------------------------------------------------
114 ! input/output params parameters at given location
115 !-----------------------------------------------------------------------
116  INTEGER, INTENT(IN) :: moist ! Moist (1) or non-moist (0) test case
117 
118  REAL(8), INTENT(IN) :: &
119  lon, & ! Longitude (radians)
120  lat, & ! Latitude (radians)
121  X ! Scale factor, not used in this version since unscaled EPV is selected
122 
123  REAL(8), INTENT(INOUT) :: p,z ! Pressure (Pa), and height(m)
124 
125  INTEGER, INTENT(IN) :: zcoords ! 0 or 1 see below
126 
127  REAL(8), INTENT(OUT) :: &
128  u, & ! Zonal wind (m s^-1)
129  v, & ! Meridional wind (m s^-1)
130  w, & ! Vertical Velocity (m s^-1)
131  t, & ! Temperature (K)
132  phis, & ! Surface Geopotential (m^2 s^-2)
133  ps, & ! Surface Pressure (Pa)
134  rho, & ! density (kg m^-3)
135  q, & ! Specific Humidity (kg/kg)
136  q1, & ! Tracer q1 - Potential temperature (kg/kg)
137  q2 ! Tracer q2 - Ertel's potential vorticity (kg/kg)
138 
139  REAL(8) :: eta
140 
141 !-----------------------------------------------------------------------
142 ! initialize PS (surface pressure)
143 !-----------------------------------------------------------------------
144  ps = p0 ! constant
145 
146 !-----------------------------------------------------------------------
147 ! calculate eta
148 !-----------------------------------------------------------------------
149  if (zcoords .eq. 0) then
150  eta = p / ps
151  z = 0
152  z = geopotential(lon,lat,eta)/g
153  else
154  eta = eta_from_z(lon,lat,z)
155  p = eta * ps
156  end if
157 
158 !-----------------------------------------------------------------------
159 ! initialize wind field with perturbation
160 !-----------------------------------------------------------------------
161  u = u_wind(lon,lat,eta,.true.)
162  v = v_wind(lon,lat,eta,.true.)
163  w = 0.d0
164 
165 !-----------------------------------------------------------------------
166 ! initialize temperature field (virtual temperature for moist conditions)
167 !-----------------------------------------------------------------------
168  t = temperature(lon,lat,eta)
169 
170 !-----------------------------------------------------------------------
171 ! initialize surface geopotential
172 !-----------------------------------------------------------------------
173  phis = surface_geopotential(lon,lat)
174 
175 !-----------------------------------------------------------------------
176 ! initialize density from ideal gas law, t is the virtual temperature in the moist case
177 !-----------------------------------------------------------------------
178  rho = p/(t*rd)
179 
180 !-----------------------------------------------------------------------
181 ! tracer q (specific humidity), moist or dry, and dry temperature
182 !-----------------------------------------------------------------------
183  if (moist == 1) then
184  q = q0 * exp(-(lat/lat_hw)**4) * exp(-((eta-1.d0)*p0/p_hw)**2) ! moist
185  t = t/(1.d0+0.608d0*q) ! convert virtual temperature to temperature
186  else
187  q = 0.d0 ! dry
188  endif
189 
190 !-----------------------------------------------------------------------
191 ! tracer q1, potential temperature
192 !-----------------------------------------------------------------------
193  q1 = theta(lon,lat,eta)
194 
195 !-----------------------------------------------------------------------
196 ! tracer q2, absolute value of EPV
197 !-----------------------------------------------------------------------
198  q2 = abs(epv(lon,lat,eta)) * x ! the value of |EPV| scales with X
199 !
200 ! The absolute value of Ertel's potential vorticity
201 ! is selected to avoid the negative EPV values in the
202 ! Southern Hemisphere. Such negative values might interfere with positive-definite
203 ! constraints in the tracer advection algorithm.
204 
205  end subroutine test4_baroclinic_wave

288 !**************************************************************************
289 ! mean geopotential
290 !**************************************************************************
291  REAL(8) function horiz_mean_geopotential(eta)
292  IMPLICIT NONE
293  REAL(8), INTENT(IN) :: eta
294  REAL(8) :: delta_phi
295 
296  horiz_mean_geopotential = t0 * g / gamma * (1.0d0 - eta**(rd * gamma / g))
297 
298  if (eta < eta_tropo) then
299  delta_phi = rd * delta_t * ( &
300  (log(eta/eta_tropo) + 137.d0/60.d0) * eta_tropo**5 &
301  - 5.d0 * eta_tropo**4 * eta &
302  + 5.d0 * eta_tropo**3 * eta**2 &
303  - 10.d0/3.d0 * eta_tropo**2 * eta**3 &
304  + 5.d0/4.d0 * eta_tropo * eta**4 &
305  - 1.d0/5.d0 * eta**5)
306 
307  horiz_mean_geopotential = horiz_mean_geopotential - delta_phi
308  end if

315  REAL(8) function u_wind(lon,lat,eta,lperturb)
316  IMPLICIT NONE
317  REAL(8), INTENT(IN) :: lon,lat,eta
318  LOGICAL, INTENT(IN) :: lperturb ! if .true. perturbation is added
319  REAL(8) :: phi_vertical, sin_tmp, cos_tmp, r, u_perturb
320  REAL(8) :: perturb_lon, perturb_lat
321 
322 
323 !---------------
324 ! unperturbed u wind
325 !---------------
326  phi_vertical = (eta - eta0) *0.5d0*pi
327  u_wind = (cos(phi_vertical))**1.5d0 * 4.d0 * u0 * (sin(lat))**2 * (cos(lat))**2
328 
329 !---------------
330 ! if needed (lperturb == .TRUE.) add a Gaussian hill perturbation
331 !---------------
332  IF (lperturb) THEN
333 
334  ! center of perturbation, convert from degrees to radians
335  perturb_lon = perturbation_longitude*deg2rad
336  perturb_lat = perturbation_latitude*deg2rad
337 
338  sin_tmp = sin(perturb_lat)*sin(lat)
339  cos_tmp = cos(perturb_lat)*cos(lat)
340 
341  r = acos( sin_tmp + cos_tmp*cos(lon-perturb_lon) ) ! great circle distance without radius 'a'
342  u_perturb = perturbation_amplitude*exp(- (r*radius)**2 ) ! perturbation_amplitude determines strength
343  u_wind = u_perturb + u_wind ! perturbation + unperturbed u wind
344  ENDIF
345 
346  END FUNCTION u_wind

366  REAL(8) function eta_from_z(lon,lat,z)
367  IMPLICIT NONE
368  REAL(8), INTENT(IN) :: lon, lat, z
369  REAL(8) :: eta_new, f, df
370  INTEGER :: iter
371 
372  REAL(8), PARAMETER :: &
373  initial_eta = 1.0d-7, &
374  convergence = 1.0d-13
375 
376  eta_from_z = initial_eta
377 
378  do iter=0, 25
379  f = - g * z + geopotential(lon,lat,eta_from_z)
380  df = - rd / eta_from_z * temperature(lon,lat,eta_from_z)
381 
382  eta_new = eta_from_z - f / df
383 
384  if (abs(eta_from_z - eta_new) < convergence) then
385  exit
386  end if
387 
388  eta_from_z = eta_new
389  enddo
390 
391  END FUNCTION eta_from_z

399  REAL(8) function theta(lon,lat,eta)
400  IMPLICIT NONE
401  REAL(8), INTENT(IN) :: eta, lon, lat
402  REAL(8) :: eta_nu, cos_tmp, y
403 
404  eta_nu = (eta-eta0)*pi*0.5d0
405  cos_tmp = u0 * (cos(eta_nu))**1.5d0
406 
407  y = (-2.d0*(sin(lat))**6 * ((cos(lat))**2 + 1.d0/3.d0) + 10.d0/63.d0)*2.d0*cos_tmp &
408  + (8.d0/5.d0*(cos(lat))**3 * ((sin(lat))**2 + 2.d0/3.d0) - pi/4.d0)*a_omega
409 
410  theta = t_mean(eta) * eta**(-kappa) + 3.d0/4.d0*pi*u0/rd * eta**(1.d0-kappa) &
411  * sin(eta_nu) * sqrt(cos(eta_nu)) * y
412 
413  END FUNCTION theta

415 !-----------------------------------------------------------------------
416 ! Ertel's potential vorticity
417 !-----------------------------------------------------------------------
418  REAL(8) function epv(lon,lat,eta)
419  IMPLICIT NONE
420  REAL(8), INTENT(IN) :: eta, lon, lat
421  REAL(8) :: perturb_lon, perturb_lat
422  REAL(8) :: eta_nu, cos_tmp, y, r, zeta, k, dk, f
423  REAL(8) :: dudeta, dmeanthetadeta, dthetadeta
424  REAL(8) :: dthetadphi1, dthetadphi2, dthetadphi
425  REAL(8) :: epsilon
426 
427  ! center of perturbation, convert from degrees to radians
428  perturb_lon = perturbation_longitude*deg2rad
429  perturb_lat = perturbation_latitude*deg2rad
430 
431  ! initialize epsilon, used to judge the distance of a point to prescribed points
432  epsilon = 1.d-6
433 
434  eta_nu = (eta-eta0)*pi*0.5d0
435  cos_tmp = u0 * (cos(eta_nu))**1.5d0
436 
437  y = (-2.d0*(sin(lat))**6 * ((cos(lat))**2 + 1.d0/3.d0) + 10.d0/63.d0)*2.d0*cos_tmp &
438  + (8.d0/5.d0*(cos(lat))**3 * ((sin(lat))**2 + 2.d0/3.d0) - pi/4.d0)*a_omega
439 
440  ! great circle distance without radius 'a'
441  k = sin(perturb_lat)*sin(lat) + cos(perturb_lat)*cos(lat)*cos(lon-perturb_lon)
442  dk = sin(perturb_lat)*cos(lat) - cos(perturb_lat)*sin(lat)*cos(lon-perturb_lon)
443  r = acos(k)
444 
445  ! relative vorticity
446 
447  ! version v3 (7/20/2012):
448  ! if construct prevents DIVISION BY ZERO in zeta calculation
449 
450  if (abs(lat-perturb_lat).le.epsilon .and. abs(lon-perturb_lon).le.epsilon) then ! grid point is at center position
451  zeta = -4.d0/a*cos_tmp*sin(lat)*cos(lat)*(2.d0-5.d0*(sin(lat))**2) + perturbation_amplitude/a*tan(lat)
452 
453  else if ( (abs(lat+perturb_lat).le.epsilon .and. abs(lon-(perturb_lon+pi)).le.epsilon) & ! antipode
454  .or. abs(lat-pi*0.5d0).le.epsilon & ! north pole
455  .or. abs(lat+pi*0.5d0).le.epsilon) then ! south pole
456  zeta = -4.d0/a*cos_tmp*sin(lat)*cos(lat)*(2.d0-5.d0*(sin(lat))**2)
457 
458  else ! all other positions
459  zeta = -4.d0/a*cos_tmp*sin(lat)*cos(lat)*(2.d0-5.d0*(sin(lat))**2) &
460  + perturbation_amplitude/a*exp(- (r*radius)**2 ) &
461  * (tan(lat) - 2.d0*radius**2*acos(k)*dk/sqrt(1.d0-k**2))
462  endif
463 
464  ! derivative of u with respect to eta
465  dudeta = -u0*(sin(2.d0*lat))**2*3.d0*pi/4.d0*sqrt(cos(eta_nu))*sin(eta_nu)
466 
467  ! derivative of mean theta with respect to eta
468  dmeanthetadeta = t0*(rd*gamma/g - kappa)*eta**(rd*gamma/g-kappa-1.d0)
469 
470  if (eta < eta_tropo) then
471  dmeanthetadeta = dmeanthetadeta &
472  - delta_t * ( &
473  5.d0*(eta_tropo - eta)**4*eta**(-kappa) &
474  + kappa * (eta_tropo - eta)**5 * eta**(-kappa-1.d0))
475  end if
476 
477  ! derivative of theta with respect to eta
478  dthetadeta = dmeanthetadeta &
479  + 3.d0/4.d0*pi*u0/rd*(1.d0-kappa)*eta**(-kappa)*sin(eta_nu)*sqrt(cos(eta_nu))*y &
480  + 3.d0/8.d0*pi*pi/rd*eta**(1.d0-kappa)*cos_tmp*y &
481  - 3.d0/16.d0*pi*pi*u0/rd*eta**(1.d0-kappa)*(sin(eta_nu))**2*(cos(eta_nu))**(-0.5d0)*y &
482  - 9.d0/8.d0*pi*pi*u0*u0/rd*eta**(1.d0-kappa)*(sin(eta_nu))**2*cos(eta_nu) &
483  * (-2.d0*(sin(lat))**6*((cos(lat))**2+1.d0/3.d0) + 10.d0/63.d0)
484 
485  ! derivative of theta with respect to phi
486  dthetadphi1 = 2.d0*cos_tmp &
487  * (- 12.d0*cos(lat)*(sin(lat))**5*((cos(lat))**2+1.d0/3.d0) &
488  + 4.d0*cos(lat)*(sin(lat))**7)
489 
490  dthetadphi2 = a_omega &
491  * (-24.d0/5.d0*sin(lat)*(cos(lat))**2*((sin(lat))**2+2.d0/3.d0) &
492  + 16.d0/5.d0*(cos(lat))**4*sin(lat))
493 
494 ! corrected on June/6/2012
495  dthetadphi = 3.d0/4.d0*pi*u0/rd*eta**(1.d0-kappa)*sin(eta_nu)*sqrt(cos(eta_nu)) &
496  * (dthetadphi1 + dthetadphi2)
497 
498  ! Coriolis parameter
499  f = 2.d0 * omega * sin(lat)
500 
501  ! EPV, unscaled since real-Earth 'a' and 'Omega' were used, EPV will be scaled in calling routine
502  epv = g/p0*(-dudeta*dthetadphi/a - (zeta+f) * dthetadeta)
503 
504  END FUNCTION epv
