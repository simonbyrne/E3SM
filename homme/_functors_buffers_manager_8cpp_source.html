26 void FunctorsBuffersManager:: allocate () {
27  Errors::runtime_check(!m_allocated, "Error! Cannot call 'allocate' more than once.\n");
28 
29  m_buffer = ExecViewManaged<Real*>("",m_size);
30 #ifdef HOMMEXX_BFB_TESTING
31  // Here's the catch: when malloc allocates memory, it *may* get a fresh
32  // new page from the OS. If that happens, the OS will zero-out the provided
33  // memory, for security reasons (e.g., you may get other people passwords).
34  // In functors unit tests, this could accidentally make things work.
35  // E.g., if the functor does not initialize its temps to 0, and sums into
36  // them, a unit test would pass with zero-ed memory.
37  // Therefore, we randomly initialize the buffer, to protect against false
38  // positives.
39  using rngalg = std::mt19937_64;
40  using rpdf = std::uniform_real_distribution<Real>;
41  std::random_device rd;
42  rngalg engine(rd());
43  genRandArray(m_buffer,engine,rpdf(0.1,10.0));
44 #endif
45  m_allocated = true;
46 }
