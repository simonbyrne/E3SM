326  KOKKOS_INLINE_FUNCTION
327  void dpdiss_adjustment (KernelVariables & kv, const TeamMember& team) const {
328 
329  const auto qtens_biharmonic = Homme::subview(m_tracers.qtens_biharmonic, kv.ie, kv.iq);
330  const auto dpdiss_ave = Homme::subview(m_derived_state.m_dpdiss_ave, kv.ie);
331  Kokkos::parallel_for (
332  Kokkos::TeamThreadRange(team, NP*NP),
333  [&] (const int loop_idx) {
334  const int i = loop_idx / NP;
335  const int j = loop_idx % NP;
336  Kokkos::parallel_for(
337  Kokkos::ThreadVectorRange(team, NUM_LEV),
338  [&] (const int& k) {
339  qtens_biharmonic(i,j,k) = qtens_biharmonic(i,j,k) * dpdiss_ave(i,j,k) / m_hvcoord.dp0(k);
340  });
341  });

368  KOKKOS_INLINE_FUNCTION
369  void rhsviss_adjustment (KernelVariables & kv, const TeamMember & team) const {
370  const auto qtens_biharmonic = Homme::subview(m_tracers.qtens_biharmonic, kv.ie, kv.iq);
371  const auto f = -m_data.rhs_viss * m_data.dt * m_data.nu_q;
372  const auto spheremp = Homme::subview(m_geometry.m_spheremp, kv.ie);
373  Kokkos::parallel_for (
374  Kokkos::TeamThreadRange(team, NP*NP),
375  [&] (const int loop_idx) {
376  const int i = loop_idx / NP;
377  const int j = loop_idx % NP;
378  Kokkos::parallel_for(
379  Kokkos::ThreadVectorRange(team, NUM_LEV),
380  [&] (const int& k) {
381  qtens_biharmonic(i,j,k) = (f * m_hvcoord.dp0(k) * qtens_biharmonic(i,j,k) /
382  spheremp(i,j));
383  });
384  });
385  }

582  void euler_step(const int np1_qdp, const int n0_qdp, const Real dt,
583  const Real rhs_multiplier, const DSSOption DSSopt) {
584

587  m_data.dt = dt;
588  m_data.rhs_multiplier = rhs_multiplier;
589  m_data.DSSopt = DSSopt;
590 
591  if (EulerStepFunctor::is_quasi_monotone(m_data.limiter_option)) {
592  // when running lim8, we also need to limit the biharmonic, so that term
593  // needs to be included in each euler step. three possible algorithms
594  // here:
595  // most expensive:
596  // compute biharmonic (which also computes qmin/qmax) during all 3
597  // stages be sure to set rhs_viss=1 cost: 3 biharmonic steps with 3 DSS
598 
599  // cheapest:
600  // compute biharmonic (which also computes qmin/qmax) only on first
601  // stage be sure to set rhs_viss=3 reuse qmin/qmax for all following
602  // stages (but update based on local qmin/qmax) cost: 1 biharmonic
603  // steps with 1 DSS main concern: viscosity
604 
605  // compromise:
606  // compute biharmonic (which also computes qmin/qmax) only on last stage
607  // be sure to set rhs_viss=3
608  // compute qmin/qmax directly on first stage
609  // reuse qmin/qmax for 2nd stage stage (but update based on local
610  // qmin/qmax) cost: 1 biharmonic steps, 2 DSS
611 
612  // NOTE when nu_p=0 (no dissipation applied in dynamics to dp equation),
613  // we should apply dissipation to Q (not Qdp) to preserve Q=1
614  // i.e. laplace(Qdp) ~ dp0 laplace(Q)
615  // for nu_p=nu_q>0, we need to apply dissipation to Q *diffusion_dp
616 
617  // initialize dp, and compute Q from Qdp(and store Q in Qtens_biharmonic)
618 
619  compute_dp();
620  compute_qmin_qmax();
621  if (m_data.rhs_multiplier == 0.0) {
622  neighbor_minmax();
623  } else if (m_data.rhs_multiplier == 2.0) {
624  minmax_and_biharmonic();
625  }
626  }
627  advect_and_limit();
628  exchange_qdp_dss_var();
629  }

761  // Do all the setup and teardown associated with a limiter. Call a limiter
762  // functor to do the actual math given the problem data (mass, minp, maxp, c,
763  // x), where the limiter possibly alters x to place it in the constraint set
764  // {x: (i) minp <= x_k <= maxp and (ii) c'x = mass }.
765  template <typename Limit, typename ArrayGll, typename ArrayGllLvl, typename Array2Lvl>
766  KOKKOS_INLINE_FUNCTION static void
767  with_limiter_shell (const TeamMember& team, const Limit& limit,
768  const ArrayGll& sphweights, const ArrayGllLvl& dpmass,
769  const Array2Lvl& qlim, const ArrayGllLvl& ptens) {
770  const int NP2 = NP * NP;
771 
772  // Size doesn't matter; just need to get a pointer to the start of the
773  // shared memory.
774  Real* const team_data = Memory<ExecSpace>::get_shmem<Real>(team);
775 
776  const auto f = [&] (const int ilev) {
777  const int vpi = ilev / VECTOR_SIZE, vsi = ilev % VECTOR_SIZE;
778 
779  Real* const data = team_data ?
780  team_data + 2 * NP2 * team.team_rank() :
781  nullptr;
782  Memory<ExecSpace>::AutoArray<Real, NP2> x(data), c(data + NP2);
783 
784  Dispatch<>::parallel_for_NP2(team, [&] (const int& k) {
785  const int i = k / NP, j = k % NP;
786  const auto& dpm = dpmass(i,j,vpi)[vsi];
787  c[k] = sphweights(i,j)*dpm;
788  x[k] = ptens(i,j,vpi)[vsi]/dpm;
789  });
790 
791  Kokkos::Real2 sums;
792  Dispatch<>::parallel_reduce_NP2(team, [&] (const int& k, Kokkos::Real2& sums) {
793  sums.v[0] += x[k]*c[k];
794  sums.v[1] += c[k];
795  }, sums);
796  if (sums.v[1] <= 0) return;
797 
798  Real minp = qlim(0,vpi)[vsi], maxp = qlim(1,vpi)[vsi];
799  // This is a slightly different spot than where this comment came from,
800  // but it's logically equivalent to do it here.
803  if (minp < 0)
804  minp = qlim(0,vpi)[vsi] = 0;
805 
809  // This is technically a write race condition, but the same value is
810  // being written, so it doesn't matter.
811  if (sums.v[0] < minp*sums.v[1])
812  minp = qlim(0,vpi)[vsi] = sums.v[0]/sums.v[1];
813  if (sums.v[0] > maxp*sums.v[1])
814  maxp = qlim(1,vpi)[vsi] = sums.v[0]/sums.v[1];
815 
816  const bool modified =
817  limit(team, sums.v[0], minp, maxp, x.data(), c.data());
818 
819  if (modified)
820  Dispatch<>::parallel_for_NP2(team, [&] (const int& k) {
821  const int i = k / NP, j = k % NP;
822  ptens(i,j,vpi)[vsi] = x[k]*dpmass(i,j,vpi)[vsi];
823  });
824  };
825 
826  if (OnGpu<ExecSpace>::value || team.team_size() > 1) {
827  Kokkos::parallel_for (
828  Kokkos::TeamThreadRange(team, NUM_PHYSICAL_LEV),
829  f);
830  } else {
831 VECTOR_SIMD_LOOP
832  for (int ilev = 0; ilev < NUM_PHYSICAL_LEV; ++ilev)
833  f(ilev);
834  }
835  }

965 template <typename ExecSpace>
966 template <int limiter_option, typename ArrayGll, typename ArrayGllLvl, typename Array2Lvl,
967  typename Array2GllLvl>
968 KOKKOS_INLINE_FUNCTION void SerialLimiter<ExecSpace>
969 ::run (const ArrayGll& sphweights, const ArrayGllLvl& idpmass,
970  const Array2Lvl& iqlim, const ArrayGllLvl& iptens,
971  const Array2GllLvl& irwrk) {
972 
973 #define forij for (int i = 0; i < NP; ++i) for (int j = 0; j < NP; ++j)
974 #define forlev for (int lev = 0; lev < NUM_PHYSICAL_LEV; ++lev)
975 
976  ViewUnmanaged<const Real[NP][NP][NUM_LEV*VECTOR_SIZE]>
977  dpmass(&idpmass(0,0,0)[0]);
978  ViewUnmanaged<Real[NP][NP][NUM_LEV*VECTOR_SIZE]>
979  c(&irwrk(0,0,0,0)[0]);
980  ViewUnmanaged<Real[2][NUM_LEV*VECTOR_SIZE]>
981  qlim(&iqlim(0,0)[0]);
982 
983  if (limiter_option == 8) {
984  ViewUnmanaged<Real[NP][NP][NUM_LEV*VECTOR_SIZE]>
985  x(&iptens(0,0,0)[0]);
986  Real mass[NUM_PHYSICAL_LEV] = {0}, sumc[NUM_PHYSICAL_LEV] = {0};
987 
988  forij {
989  const auto& sphij = sphweights(i,j);
990  VECTOR_SIMD_LOOP forlev {
991  const auto& dpm = dpmass(i,j,lev);
992  c(i,j,lev) = sphij*dpm;
993  x(i,j,lev) /= dpm;
994  mass[lev] += c(i,j,lev)*x(i,j,lev);
995  sumc[lev] += c(i,j,lev);
996  }
997  }
998 
999  VECTOR_SIMD_LOOP forlev {
1000  if (qlim(0,lev) < 0)
1001  qlim(0,lev) = 0;
1002  if (mass[lev] < qlim(0,lev)*sumc[lev])
1003  qlim(0,lev) = mass[lev]/sumc[lev];
1004  if (mass[lev] > qlim(1,lev)*sumc[lev])
1005  qlim(1,lev) = mass[lev]/sumc[lev];
1006  }
1007 
1008  static const int maxiter = NP*NP - 1;
1009  static const Real tol_limiter = 5e-14;
1010  int donecnt = 0;
1011  char done[NUM_PHYSICAL_LEV] = {0};
1012  for (int iter = 0; iter < maxiter; ++iter) {
1013  Real addmass[NUM_PHYSICAL_LEV] = {0};
1014 
1015  forij {
1016  VECTOR_SIMD_LOOP forlev {
1017  auto& xij = x(i,j,lev);
1018  Real delta = 0;
1019  if (xij < qlim(0,lev)) {
1020  delta = xij - qlim(0,lev);
1021  xij = qlim(0,lev);
1022  } else if (xij > qlim(1,lev)) {
1023  delta = xij - qlim(1,lev);
1024  xij = qlim(1,lev);
1025  }
1026  addmass[lev] += delta*c(i,j,lev);
1027  }
1028  }
1029 
1030  forlev {
1031  if (std::abs(addmass[lev]) <= tol_limiter*std::abs(mass[lev]) &&
1032  ! done[lev]) {
1033  done[lev] = 1;
1034  ++donecnt;
1035  }
1036  }
1037  if (donecnt == NUM_PHYSICAL_LEV) break;
1038 
1039  Real f[NUM_PHYSICAL_LEV] = {0};
1040  forij {
1041  VECTOR_SIMD_LOOP forlev {
1042  if (done[lev]) continue;
1043  if (addmass[lev] <= 0) {
1044  if (x(i,j,lev) > qlim(0,lev))
1045  f[lev] += c(i,j,lev);
1046  } else {
1047  if (x(i,j,lev) < qlim(1,lev))
1048  f[lev] += c(i,j,lev);
1049  }
1050  }
1051  }
1052 
1053  VECTOR_SIMD_LOOP forlev {
1054  if (f[lev] != 0)
1055  f[lev] = addmass[lev] / f[lev];
1056  }
1057 
1058  forij {
1059  VECTOR_SIMD_LOOP forlev {
1060  if (done[lev]) continue;
1061  if (addmass[lev] <= 0) {
1062  if (x(i,j,lev) > qlim(0,lev))
1063  x(i,j,lev) += f[lev];
1064  } else {
1065  if (x(i,j,lev) < qlim(1,lev))
1066  x(i,j,lev) += f[lev];
1067  }
1068  }
1069  }
1070  }
1071 
1072  forij {
1073  VECTOR_SIMD_LOOP forlev {
1074  x(i,j,lev) *= dpmass(i,j,lev);
1075  }
1076  }
1077  } else if (limiter_option == 9) {
