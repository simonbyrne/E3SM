9 ! SUBROUTINE tropical_cyclone_sample(
10 ! lon,lat,p,z,zcoords,u,v,t,thetav,phis,ps,rho,q)
11 !
12 ! Given a point specified by:
13 ! lon longitude (radians)
14 ! lat latitude (radians)
15 ! p/z pressure (Pa) / height (m)
16 ! zcoords 1 if z is specified, 0 if p is specified
17 !
18 ! the functions will return:
19 ! p pressure if z is specified (Pa)
20 ! z geopotential height if p is specified (m)
21 ! u zonal wind (m s^-1)
22 ! v meridional wind (m s^-1)
23 ! t temperature (K)
24 ! thetav virtual potential temperature (K)
25 ! phis surface geopotential (m^2 s^-2)
26 ! ps surface pressure (Pa)
27 ! rho density (kj m^-3)
28 ! q specific humidity (kg/kg)
29 !
30 ! Initial data are currently identical to:
31 !
32 ! Reed, K. A., and C. Jablonowski, 2011: An analytic
33 ! vortex initialization technique for idealized tropical
34 ! cyclone studies in AGCMs. Mon. Wea. Rev., 139, 689-710.

36 ! Author: Kevin A. Reed

93 ! Evaluate the tropical cyclone initial conditions

95  SUBROUTINE tropical_cyclone_test(lon,lat,p,z,zcoords,u,v,t,thetav,phis,ps,rho,q) &

125  !------------------------------------------------
126  ! Local variables
127  !------------------------------------------------
128  real(8) :: d1, d2, d, vfac, ufac, height, zhere, gr, f, zn
129 
130  integer n
131 
132  !------------------------------------------------
133  ! Define Great circle distance (gr) and
134  ! Coriolis parameter (f)
135  !------------------------------------------------
136  f = 2.d0*omega*sin(cen_lat*deg2rad) ! Coriolis parameter
137  gr = a*acos(sin(cen_lat*deg2rad)*sin(lat) + & ! Great circle radius
138  (cos(cen_lat*deg2rad)*cos(lat)*cos(lon-cen_lon*deg2rad)))
139 
140  !------------------------------------------------
141  ! Initialize PS (surface pressure)
142  !------------------------------------------------
143  ps = p00-dp*exp(-(gr/rp)**exppr)
144 
145  !------------------------------------------------
146  ! Initialize altitude (z) if pressure provided
147  ! or pressure if altitude (z) is provided
148  !------------------------------------------------
149  if (zcoords .eq. 1) then
150 
151  height = z
152 
153  if (height > ztrop) then
154  p = ptrop*exp(-(g*(height-ztrop))/(rd*ttrop))
155  else
156  p = (p00-dp*exp(-(gr/rp)**exppr)*exp(-(height/zp)**exppz)) &
157  * ((t0-gamma*height)/t0)**(1/exponent)
158  end if
159 
160  else
161 
162  height = (t0/gamma)*(1.d0-(p/ps)**exponent)
163 
164  ! If inside a certain distance of the center of the storm
165  ! perform a Fixed-point iteration to calculate the height
166  ! more accurately
167 
168  if (gr < rfpi ) then
169  zhere = height
170  n = 1
171  20 continue
172  n = n+1
173  zn = zhere - fpif(p,gr,zhere)/fpidfdz(gr,zhere)
174  !if (n.gt.20) then
175  if (n.gt.40) then
176 
177  !PRINT *,'FPI did not converge after 20 interations in q & T!!!'
178 print *,'FPI did not converge after 40 interations in q & T!!!'
179  else if ( abs(zn-zhere)/abs(zn) > deltaz) then
180  zhere = zn
181  goto 20
182  end if
183  height = zn
184  end if
185  z=height
186  end if
187 
188  !------------------------------------------------
189  ! Initialize U and V (wind components)
190  !------------------------------------------------
191  d1 = sin(cen_lat*deg2rad)*cos(lat) - &
192  cos(cen_lat*deg2rad)*sin(lat)*cos(lon-cen_lon*deg2rad)
193  d2 = cos(cen_lat*deg2rad)*sin(lon-cen_lon*deg2rad)
194  d = max(epsilon, sqrt(d1**2.d0 + d2**2.d0))
195  ufac = d1/d
196  vfac = d2/d
197 
198  if (height > ztrop) then
199  u = 0.d0
200  v = 0.d0
201  else
202  v = vfac*(-f*gr/2.d0+sqrt((f*gr/2.d0)**(2.d0) &
203  - exppr*(gr/rp)**exppr*rd*(t0-gamma*height) &
204  /(exppz*height*rd*(t0-gamma*height)/(g*zp**exppz) &
205  +(1.d0-p00/dp*exp((gr/rp)**exppr)*exp((height/zp)**exppz)))))
206  u = ufac*(-f*gr/2.d0+sqrt((f*gr/2.d0)**(2.d0) &
207  - exppr*(gr/rp)**exppr*rd*(t0-gamma*height) &
208  /(exppz*height*rd*(t0-gamma*height)/(g*zp**exppz) &
209  +(1.d0-p00/dp*exp((gr/rp)**exppr)*exp((height/zp)**exppz)))))
210  end if
211 
212  !------------------------------------------------
213  ! Initialize water vapor mixing ratio (q)
214  !------------------------------------------------
215  if (height > ztrop) then
216  q = qtrop
217  else
218  q = q0*exp(-height/zq1)*exp(-(height/zq2)**exppz)
219  end if
220 
221  !------------------------------------------------
222  ! Initialize temperature (T)
223  !------------------------------------------------
224  if (height > ztrop) then
225  t = ttrop
226  else
227  t = (t0-gamma*height)/(1.d0+consttv*q)/(1.d0+exppz*rd*(t0-gamma*height)*height &
228  /(g*zp**exppz*(1.d0-p00/dp*exp((gr/rp)**exppr)*exp((height/zp)**exppz))))
229  end if
230 
231  !-----------------------------------------------------
232  ! Initialize virtual potential temperature (thetav)
233  !-----------------------------------------------------
234  thetav = t * (1.d0+consttv*q) * (p0/p)**(rd/cp)
235 
236  !-----------------------------------------------------
237  ! Initialize surface geopotential (PHIS)
238  !-----------------------------------------------------
239  phis = 0.d0 ! constant
240 
241  !-----------------------------------------------------
242  ! Initialize density (rho)
243  !-----------------------------------------------------
244  rho = p/(rd*t*(1.d0+consttv*q))
245 
246  END SUBROUTINE tropical_cyclone_test

251  REAL(8) function fpif(phere, gr, zhere)
252  IMPLICIT NONE

254 
255  fpif = phere-(p00-dp*exp(-(gr/rp)**exppr)*exp(-(zhere/zp)**exppz)) &
256  *((t0-gamma*zhere)/t0)**(g/(rd*gamma))
257 
258  END FUNCTION fpif

268  +g/(rd*t0)*(p00-dp*exp(-(gr/rp)**exppr)*exp(-(zhere/zp)**exppz))*((t0-gamma*zhere)/t0)**(g/(rd*gamma)-1.d0)
