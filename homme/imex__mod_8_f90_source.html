30  ! wh_i = ubar grad phis
31  !

42  ! add wh_i(n0) term to RHS

63  !===================================================================================
64  ! this subroutine solves a stage value equation for a DIRK method which takes the form
65  !
66  ! gi = un0 + dt* sum(1:i-1)(aij n(gj)+a2ij s(gj)) + dt *a2ii s(gi) := y + dt a2ii s(gi)
67  !
68  ! It is assumed that un0 has the value of y and the computed value of gi is stored at
69  ! unp1
70  !
71  ! w_0 = w(np1)
72  ! phi_0 = phi(np1)
73  ! Then solve (ovewriting w(np1),phi(np1):
74  ! w(np1) = w_0 + alphadt_n0 * SW(n0) + alphadt_nm1*SW(nm1) + dt2*SW(np1)
75  ! phi(np1) = phi_0 + alphadt_n0 * SPHI(n0) + alphadt_nm1*SPHI(nm1) + dt2*SPHI(np1)
76  !
77  ! SW(nt) = g*dpnh_dp_i(nt)-1
78  ! SPHI(nt) = g*w(nt) -g*a(k) u(nt) dot grad_phis
79  !
80  ! pnh_and_exner_from_eos() used to compute dpnh_dp_i(nt) in SW(nt)
81  ! compute_gwphis() used to compute g*a(k) u(nt) dot grad_phis in SPHI(nt)
82  !
83  ! We then precompute:
84  ! w_rhs = w_0 + alphadt_n0 * SW(n0) + alphadt_nm1*SW(nm1)
85  ! phi_rhs = w_0 + alphad_n0 * SPHI(n0) + alphadt1_nm1*SPHI(nm1) - dt2*compute_gwphi(np1)
86  ! and solve, via Newton iteration:

414  subroutine get_dirk_jacobian(JacL,JacD,JacU,dt2,dp3d,dphi,pnh,exact,&
415  epsie,hvcoord,dpnh_dp_i,vtheta_dp)
416  !================================================================================
417  ! compute Jacobian of F(phi) = sum(dphi) +const + (dt*g)^2 *(1-dp/dpi) column wise
418  ! with respect to phi
419  !
420  ! This subroutine forms the tridiagonal analytic Jacobian (we actually form the diagonal, sub-, and super-diagonal)
421  ! J for use in a LApack tridiagonal LU factorization and solver to solve J * x = -f either exactly or
422  ! approximately
423  !
424  ! input:
425  ! exact jacobian: dphi, dp3d, pnh
426  ! matrix free jacobian: dphi, dp3d, vtheta_dp, hvcoord, dpnh_dp_i
427  !
428  ! epsie == 1 means exact Jacobian, epsie ~= 1 means finite difference approximate jacobian
429  ! exact,epsie,hvcoord,dpnh_dp,vtheta_dp,pnh,exner,exner_i are only needed as inputs
430  ! if epsie ~=1
431  !
432  ! The rule-of-thumb optimal epsie is epsie = norm(elem)*sqrt(macheps)

627  subroutine solve_strict_diag_dominant_tridiag(dl, d, du, x)
628  ! Solve a strictly diagonally dominant tridiagonal system. On
629  ! input, x is the RHS; on output, the solution. dl contains rows
630  ! 2:nlev of the lower diagonal; d, the diagonal; du, rows 1:nlev-1
631  ! of the upper diagonal. d is overwritten.
632 
633  use dimensions_mod, only: npsq
634 
635  real (kind=real_kind), intent(in) :: dl(npsq,nlev-1), du(npsq,nlev-1)
636  real (kind=real_kind), intent(inout) :: d(npsq,nlev), x(npsq,nlev)
637 
638  real (kind=real_kind) :: dlk(npsq)
639  integer :: k
640 
641  do k = 1,nlev-1
642  dlk = dl(:,k) / d(:,k)
643  d(:,k+1) = d(:,k+1) - dlk * du(:,k)
644  x(:,k+1) = x(:,k+1) - dlk * x(:,k)
645  end do
646  x(:,nlev) = x(:,nlev) / d(:,nlev)
647  do k = nlev-1,1,-1
648  x(:,k) = (x(:,k) - du(:,k) * x(:,k+1)) / d(:,k)
649  end do
650  end subroutine solve_strict_diag_dominant_tridiag
