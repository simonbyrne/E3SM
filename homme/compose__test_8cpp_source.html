66 inline Real great_circle_dist (
67  const Real lat1, const Real lon1, const Real lat2, const Real lon2,
68  const Real R = 1)
69 {
70  Real xA, yA, zA;
71  Real xB, yB, zB;
72  ll2xyz(lat1, lon1, xA, yA, zA);
73  ll2xyz(lat2, lon2, xB, yB, zB);
74  Real cp1, cp2, cp3, cpnorm, dotprod;
75  cp1 = yA*zB - yB*zA;
76  cp2 = xB*zA - xA*zB;
77  cp3 = xA*yB - xB*yA;
78  cpnorm = std::sqrt(cp1*cp1 + cp2*cp2 + cp3*cp3);

80  return R * std::atan2(cpnorm, dotprod);

248 // Convert from (u,v), where u is velocity along latitude and v is velocity
249 // along longitude, to (x,y,z), which is velocity in the global cartesian
250 // coordinate system. Add a w (local vertical) component to push the position
251 // (X,Y,Z) back to the unit sphere.
252 inline void uv2xyz (
253  const Real X, const Real Y, const Real Z, // position
254  const Real u, const Real v, // velocity in tangent coord system
255  Real& x, Real& y, Real& z) // velocity in global coord system
256 {
257  // r should be 1 but will numerically drift, so measure it ...
258  const Real r = std::sqrt(X*X + Y*Y + Z*Z);
259  // ... and then add a local vertical velocity to project back to the sphere.
260  const Real w = (1 - r)/consts::earth_radius_m;
261  Real R[9]; // Row major.
262  // The local vertical is just the position vector.
263  R[2] = X/r; R[5] = Y/r; R[8] = Z/r;
264  // The local along-latitude vector.
265  R[0] = -Y; R[3] = X; R[6] = 0;
266  const Real den = std::sqrt(R[0]*R[0] + R[3]*R[3]);
267  R[0] /= den; R[3] /= den;
268  // Local vertical x along-latitude.
269  R[1] = R[5]*R[6] - R[8]*R[3];
270  R[4] = R[8]*R[0] - R[2]*R[6];
271  R[7] = R[2]*R[3] - R[5]*R[0];
272  // Transform.
273  x = R[0]*u + R[1]*v + R[2]*w;
274  y = R[3]*u + R[4]*v + R[5]*w;
275  z = R[6]*u + R[7]*v + R[8]*w;
276 }
