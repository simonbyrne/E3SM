128 enum class CombineMode {
129  Replace = 0, // out = in
130  Scale, // out = alpha*in
131  Update, // out = beta*out + in
132  ScaleUpdate, // out = beta*out + alpha*in
133  ScaleAdd, // out = out + alpha*in (special case of ScaleUpdate with beta=1)
134  Add, // out = out + in (special case of ScaleAdd/Update with alpha=1, beta=1.0)
135  Multiply, // out = out*in
136  Divide // out = out/in
137 };
138 
139 template<CombineMode CM>

141  switch (CM) {

143  case CombineMode::Scale: return "Scale";

145  case CombineMode::ScaleUpdate: return "ScaleUpdate";
146  case CombineMode::ScaleAdd: return "ScaleAdd";
147  case CombineMode::Add: return "Add";
148  case CombineMode::Multiply: return "Multiply";
149  case CombineMode::Divide: return "Divide";
150  }
151  return "UNKNOWN";
152 }
