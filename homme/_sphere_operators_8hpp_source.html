125  KOKKOS_INLINE_FUNCTION void
126  gradient_sphere_sl (const KernelVariables &kv,
127  const ExecViewUnmanaged<const Real [NP][NP]>& scalar,
128  const ExecViewUnmanaged< Real [2][NP][NP]>& grad_s) const
129  {
130  // Make sure the buffers have been created
131  assert (vector_buf_sl.size()>0);
132 
133  const auto& D_inv = Homme::subview(m_dinv,kv.ie);
134  const auto& temp_v_buf = Homme::subview(vector_buf_sl,kv.team_idx,0);
135  constexpr int np_squared = NP * NP;
136  // TODO: Use scratch space for this
137  Kokkos::parallel_for(Kokkos::TeamThreadRange(kv.team, np_squared),
138  [&](const int loop_idx) {
139  const int j = loop_idx / NP;
140  const int l = loop_idx % NP;
141  Real dsdx(0), dsdy(0);
142  for (int i = 0; i < NP; ++i) {
143  dsdx += dvv(l, i) * scalar(j, i);
144  dsdy += dvv(l, i) * scalar(i, j);
145  }
146  temp_v_buf(0, j, l) = dsdx * PhysicalConstants::rrearth;
147  temp_v_buf(1, l, j) = dsdy * PhysicalConstants::rrearth;
148  });
149  kv.team_barrier();
150 
151  constexpr int grad_iters = 2 * NP * NP;
152  Kokkos::parallel_for(Kokkos::TeamThreadRange(kv.team, grad_iters),
153  [&](const int loop_idx) {
154  const int h = (loop_idx / NP) / NP;
155  const int i = (loop_idx / NP) % NP;
156  const int j = loop_idx % NP;
157  grad_s(h, j, i) = D_inv(h, 0, j, i) * temp_v_buf(0, j, i) +
158  D_inv(h, 1, j, i) * temp_v_buf(1, j, i);
159  });
160  kv.team_barrier();
161  }

201  KOKKOS_INLINE_FUNCTION void
202  divergence_sphere_sl (const KernelVariables &kv,
203  const ExecViewUnmanaged<const Real [2][NP][NP]>& v,
204  const ExecViewUnmanaged< Real [NP][NP]>& div_v) const
205  {
206  // Make sure the buffers have been created
207  assert (vector_buf_sl.size()>0);
208 
209  const auto& metdet = Homme::subview(m_metdet,kv.ie);
210  const auto& D_inv = Homme::subview(m_dinv,kv.ie);
211  const auto& gv_buf = Homme::subview(vector_buf_sl,kv.team_idx,0);
212  constexpr int np_squared = NP * NP;
213  Kokkos::parallel_for(Kokkos::TeamThreadRange(kv.team, np_squared),
214  [&](const int loop_idx) {
215  const int igp = loop_idx / NP;
216  const int jgp = loop_idx % NP;
217  const auto& v0 = v(0,igp,jgp);
218  const auto& v1 = v(1,igp,jgp);
219  gv_buf(0,igp,jgp) = (D_inv(0,0,igp,jgp) * v0 + D_inv(1,0,igp,jgp) * v1) * metdet(igp,jgp);
220  gv_buf(1,igp,jgp) = (D_inv(0,1,igp,jgp) * v0 + D_inv(1,1,igp,jgp) * v1) * metdet(igp,jgp);
221  });
222  kv.team_barrier();
223 
224  constexpr int div_iters = NP * NP;
225  Kokkos::parallel_for(Kokkos::TeamThreadRange(kv.team, div_iters),
226  [&](const int loop_idx) {
227  const int igp = loop_idx / NP;
228  const int jgp = loop_idx % NP;
229  Real dudx = 0.0, dvdy = 0.0;
230  for (int kgp = 0; kgp < NP; ++kgp) {
231  dudx += dvv(jgp, kgp) * gv_buf(0, igp, kgp);
232  dvdy += dvv(igp, kgp) * gv_buf(1, kgp, jgp);
233  }
234  div_v(igp,jgp) = (dudx + dvdy) * ((1.0 / metdet(igp,jgp)) *
235  PhysicalConstants::rrearth);
236  });
237  kv.team_barrier();
238  }

240  KOKKOS_INLINE_FUNCTION void
241  divergence_sphere_wk_sl (const KernelVariables &kv,
242  const ExecViewUnmanaged<const Real [2][NP][NP]>& v,
243  const ExecViewUnmanaged< Real [NP][NP]>& div_v) const
244  {
245  // Make sure the buffers have been created
246  assert (vector_buf_sl.size()>0);
247 
248  const auto& D_inv = Homme::subview(m_dinv,kv.ie);
249  const auto& spheremp = Homme::subview(m_spheremp,kv.ie);
250  const auto& gv_buf = Homme::subview(vector_buf_sl,kv.team_idx,0);
251 
252  // copied from strong divergence as is but without metdet
253  // conversion to contravariant
254  constexpr int np_squared = NP * NP;
255  Kokkos::parallel_for(Kokkos::TeamThreadRange(kv.team, np_squared),
256  [&](const int loop_idx) {
257  const int igp = loop_idx / NP;
258  const int jgp = loop_idx % NP;
259  const auto& v0 = v(0,igp,jgp);
260  const auto& v1 = v(1,igp,jgp);
261  gv_buf(0,igp,jgp) = D_inv(0,0,igp,jgp) * v0 + D_inv(1,0,igp,jgp) * v1;
262  gv_buf(1,igp,jgp) = D_inv(0,1,igp,jgp) * v0 + D_inv(1,1,igp,jgp) * v1;
263  });
264  kv.team_barrier();
265 
266  // in strong div
267  // kgp = i in strong code, jgp=j, igp=l
268  // in weak div, n is like j in strong div,
269  // n(weak)=j(strong)=jgp
270  // m(weak)=l(strong)=igp
271  // j(weak)=i(strong)=kgp
272  constexpr int div_iters = NP * NP;
273  // keeping indices' names as in F
274  Kokkos::parallel_for(Kokkos::TeamThreadRange(kv.team, div_iters),
275  [&](const int loop_idx) {
276  // Note: for this one time, it is better if m strides faster, due to
277  // the way the views are accessed.
278  const int mgp = loop_idx % NP;
279  const int ngp = loop_idx / NP;
280  Real dd = 0.0;
281  for (int jgp = 0; jgp < NP; ++jgp) {
282  dd -= (spheremp(ngp, jgp) * gv_buf(0, ngp, jgp) * dvv(jgp, mgp) +
283  spheremp(jgp, mgp) * gv_buf(1, jgp, mgp) * dvv(jgp, ngp)) *
284  PhysicalConstants::rrearth;
285  }
286  div_v(ngp, mgp) = dd;
287  });
288  kv.team_barrier();
289 
290  } // end of divergence_sphere_wk_sl
