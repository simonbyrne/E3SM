5 ! Simple Physics Package
6 !
7 ! SIMPLE_PHYSICS includes large-scale precipitation, surface fluxes and
8 ! boundary-leyer mixing. The processes are time-split in that order.
9 ! A partially implicit formulation is used to foster numerical
10 ! stability. The routine assumes that the model levels are ordered
11 ! in a top-down approach, e.g. level 1 denotes the uppermost full model
12 ! level.
13 !
14 ! This routine is based on an implementation which was developed for
15 ! the NCAR Community Atmosphere Model (CAM). Adjustments for other
16 ! models may be necessary.
17 !
18 ! The routine provides both updates of the state variables u, v, T, q
19 ! (these are local copies of u,v,T,q within this physics routine) and
20 ! also collects their time tendencies. The latter might be used to
21 ! couple the physics and dynamics in a process-split way. For a
22 ! time-split coupling, the final state should be given to the
23 ! dynamical core for the next time step.
24 !
25 ! Test: 0 = Reed and Jablonowski (2011) tropical cyclone test
26 ! 1 = Moist baroclinic instability test
27 !
28 ! SUBROUTINE SIMPLE_PHYSICS(pcols, pver, dtime, lat, t, q, u, v, pmid, pint, pdel, rpdel, ps, precl, test)
29 !
30 ! Input variables:
31 ! pcols - number of atmospheric columns (#)
32 ! pver - number of model levels (#)
33 ! dtime - time step (s)
34 ! lat - latitude (radians)
35 ! t - temperature at model levels (K)
36 ! q - specific humidity at model levels (gm/gm)
37 ! u - zonal wind at model levels (m/s)
38 ! v - meridional wind at model levels (m/s)
39 ! pmid - pressure at model levels (Pa)
40 ! pint - pressure at interfaces (Pa)
41 ! pdel - layer thickness (Pa)
42 ! rpdel - reciprocal of layer thickness (1/Pa)
43 ! ps - surface pressure (Pa)
44 ! test - test case to use for sea-surface temperatures
45 !
46 ! Output variables:
47 ! Increments are added into t, q, u, v, pmid, pint, pdel, rpdel and ps
48 ! which are returned to the routine from which SIMPLE_PHYSICS was
49 ! called. Precpitation is returned via precl.
50 !
51 ! Change log:
52 ! v2: removal of some NCAR CAM-specific 'use' associations
53 ! v3: corrected precl(i) computation, the precipitation rate is now
54 ! computed via a vertical integral, the previous single-level
55 ! computation in v2 was a bug
56 ! v3: corrected dtdt(i,1) computation, the term '-(i,1)' was missing
57 ! the temperature variable: '-t(i,1)'
58 ! v4: modified and enhanced parameter list to make the routine truly
59 ! standalone, the number of columns and vertical levels have been
60 ! added: pcols, pver
61 ! v4: 'ncol' has been removed, 'pcols' is used instead
62 ! v5: the sea surface temperature (SST) field Tsurf is now an array,
63 ! the SST now depends on the latitude
64 ! v5: addition of the latitude array 'lat' and the flag 'test' in the
65 ! parameter list
66 ! if test = 0: constant SST is used, correct setting for the
67 ! tropical cyclone test
68 ! if test = 1: newly added latitude-dependent SST is used,
69 ! correct setting for the moist baroclinic wave test
70 ! with simple-physics
71 !
72 ! Reference: Reed, K. A. and C. Jablonowski (2012), Idealized tropical cyclone
73 ! simulations of intermediate complexity: A test case for AGCMs,
74 ! J. Adv. Model. Earth Syst., Vol. 4, M04001, doi:10.1029/2011MS000099
75 !-----------------------------------------------------------------------

77 SUBROUTINE simple_physics_v5(pcols, pver, dtime, lat, t, q, u, v, pmid, pint, pdel, rpdel, ps, precl, test)
78 
79  ! use physics_types , only: physics_dme_adjust ! This is for CESM/CAM
80  ! use cam_diagnostics, only: diag_phys_writeout ! This is for CESM/CAM
81 
82  implicit none
83 
84  integer, parameter :: r8 = selected_real_kind(12)
85 
86 !
87 ! Input arguments - MODEL DEPENDENT
88 !
89  integer, intent(in) :: pcols ! Set number of atmospheric columns
90  integer, intent(in) :: pver ! Set number of model levels
91  real(r8), intent(in) :: dtime ! Set model physics timestep
92  real(r8), intent(in) :: lat(pcols) ! Latitude
93  integer, intent(in) :: test ! Test number
94 
95 !
96 ! Input/Output arguments
97 !
98 ! pcols is the maximum number of vertical columns per 'chunk' of atmosphere
99 !
100  real(r8), intent(inout) :: t(pcols,pver) ! Temperature at full-model level (K)
101  real(r8), intent(inout) :: q(pcols,pver) ! Specific Humidity at full-model level (kg/kg)
102  real(r8), intent(inout) :: u(pcols,pver) ! Zonal wind at full-model level (m/s)
103  real(r8), intent(inout) :: v(pcols,pver) ! Meridional wind at full-model level (m/s)
104  real(r8), intent(inout) :: pmid(pcols,pver) ! Pressure is full-model level (Pa)
105  real(r8), intent(inout) :: pint(pcols,pver+1) ! Pressure at model interfaces (Pa)
106  real(r8), intent(inout) :: pdel(pcols,pver) ! Layer thickness (Pa)
107  real(r8), intent(inout) :: rpdel(pcols,pver) ! Reciprocal of layer thickness (1/Pa)
108  real(r8), intent(inout) :: ps(pcols) ! Surface Pressue (Pa)
109 
110 !
111 ! Output arguments
112 !
113  real(r8), intent(out) :: precl(pcols) ! Precipitation rate (m_water / s)
114 
115 !
116 !---------------------------Local workspace-----------------------------
117 !
118 
119 ! Integers for loops
120 
121  integer i,k ! Longitude, level indices
122 
123 ! Physical Constants - Many of these may be model dependent
124 
125  real(r8) gravit ! Gravity
126  real(r8) rair ! Gas constant for dry air
127  real(r8) cpair ! Specific heat of dry air
128  real(r8) latvap ! Latent heat of vaporization
129  real(r8) rh2o ! Gas constant for water vapor
130  real(r8) epsilo ! Ratio of gas constant for dry air to that for vapor
131  real(r8) zvir ! Constant for virtual temp. calc. =(rh2o/rair) - 1
132  real(r8) a ! Reference Earth's Radius (m)
133  real(r8) omega ! Reference rotation rate of the Earth (s^-1)
134  real(r8) pi ! pi
135 
136 ! Simple Physics Specific Constants
137 
138 !++++++++
139  real(r8) Tsurf(pcols) ! Sea Surface Temperature (constant for tropical cyclone)
140 !++++++++ Tsurf needs to be dependent on latitude for the
141  ! moist baroclinic wave test 4-3 with simple-physics, adjust
142 
143  real(r8) SST_tc ! Sea Surface Temperature for tropical cyclone test
144  real(r8) T0 ! Control temp for calculation of qsat
145  real(r8) e0 ! Saturation vapor pressure at T0 for calculation of qsat
146  real(r8) rhow ! Density of Liquid Water
147  real(r8) p0 ! Constant for calculation of potential temperature
148  real(r8) Cd0 ! Constant for calculating Cd from Smith and Vogl 2008
149  real(r8) Cd1 ! Constant for calculating Cd from Smith and Vogl 2008
150  real(r8) Cm ! Constant for calculating Cd from Smith and Vogl 2008
151  real(r8) v20 ! Threshold wind speed for calculating Cd from Smith and Vogl 2008
152  real(r8) C ! Drag coefficient for sensible heat and evaporation
153  real(r8) T00 ! Horizontal mean T at surface for moist baro test
154  real(r8) u0 ! Zonal wind constant for moist baro test
155  real(r8) latw ! halfwidth for for baro test
156  real(r8) eta0 ! Center of jets (hybrid) for baro test
157  real(r8) etav ! Auxiliary variable for baro test
158  real(r8) q0 ! Maximum specific humidity for baro test
159 
160 ! Physics Tendency Arrays
161  real(r8) dtdt(pcols,pver) ! Temperature tendency
162  real(r8) dqdt(pcols,pver) ! Specific humidity tendency
163  real(r8) dudt(pcols,pver) ! Zonal wind tendency
164  real(r8) dvdt(pcols,pver) ! Meridional wind tendency
165 
166 ! Temporary variables for tendency calculations
167 
168  real(r8) tmp ! Temporary
169  real(r8) qsat ! Saturation vapor pressure
170  real(r8) qsats ! Saturation vapor pressure of SST
171 
172 ! Variables for Boundary Layer Calculation
173 
174  real(r8) wind(pcols) ! Magnitude of Wind
175  real(r8) Cd(pcols) ! Drag coefficient for momentum
176  real(r8) Km(pcols,pver+1) ! Eddy diffusivity for boundary layer calculations
177  real(r8) Ke(pcols,pver+1) ! Eddy diffusivity for boundary layer calculations
178  real(r8) rho ! Density at lower/upper interface
179  real(r8) za(pcols) ! Heights at midpoints of first model level
180  real(r8) dlnpint ! Used for calculation of heights
181  real(r8) pbltop ! Top of boundary layer
182  real(r8) pblconst ! Constant for the calculation of the decay of diffusivity
183  real(r8) CA(pcols,pver) ! Matrix Coefficents for PBL Scheme
184  real(r8) CC(pcols,pver) ! Matrix Coefficents for PBL Scheme
185  real(r8) CE(pcols,pver+1) ! Matrix Coefficents for PBL Scheme
186  real(r8) CAm(pcols,pver) ! Matrix Coefficents for PBL Scheme
187  real(r8) CCm(pcols,pver) ! Matrix Coefficents for PBL Scheme
188  real(r8) CEm(pcols,pver+1) ! Matrix Coefficents for PBL Scheme
189  real(r8) CFu(pcols,pver+1) ! Matrix Coefficents for PBL Scheme
190  real(r8) CFv(pcols,pver+1) ! Matrix Coefficents for PBL Scheme
191  real(r8) CFt(pcols,pver+1) ! Matrix Coefficents for PBL Scheme
192  real(r8) CFq(pcols,pver+1) ! Matrix Coefficents for PBL Scheme
193 
194 
195 ! Variable for Dry Mass Adjustment, this dry air adjustment is necessary to
196 ! conserve the mass of the dry air
197 
198  real(r8) qini(pcols,pver) ! Initial specific humidity
199 
200 !===============================================================================
201 !
202 ! Physical Constants - MAY BE MODEL DEPENDENT
203 !
204 !===============================================================================
205  gravit = 9.80616_r8 ! Gravity (9.80616 m/s^2)
206  rair = 287.0_r8 ! Gas constant for dry air: 287 J/(kg K)
207  cpair = 1.0045e3_r8 ! Specific heat of dry air: here we use 1004.5 J/(kg K)
208  latvap = 2.5e6_r8 ! Latent heat of vaporization (J/kg)
209  rh2o = 461.5_r8 ! Gas constant for water vapor: 461.5 J/(kg K)
210  epsilo = rair/rh2o ! Ratio of gas constant for dry air to that for vapor
211  zvir = (rh2o/rair) - 1._r8 ! Constant for virtual temp. calc. =(rh2o/rair) - 1 is approx. 0.608
212  a = 6371220.0_r8 ! Reference Earth's Radius (m)
213  omega = 7.29212d-5 ! Reference rotation rate of the Earth (s^-1)
214  pi = 4._r8*atan(1._r8) ! pi
215 
216 !===============================================================================
217 !
218 ! Local Constants for Simple Physics
219 !
220 !===============================================================================
221  c = 0.0011_r8 ! From Smith and Vogl 2008
222  sst_tc = 302.15_r8 ! Constant Value for SST for tropical cyclone test
223  t0 = 273.16_r8 ! control temp for calculation of qsat
224  e0 = 610.78_r8 ! saturation vapor pressure at T0 for calculation of qsat
225  rhow = 1000.0_r8 ! Density of Liquid Water
226  cd0 = 0.0007_r8 ! Constant for Cd calc. Smith and Vogl 2008
227  cd1 = 0.000065_r8 ! Constant for Cd calc. Smith and Vogl 2008
228  cm = 0.002_r8 ! Constant for Cd calc. Smith and Vogl 2008
229  v20 = 20.0_r8 ! Threshold wind speed for calculating Cd from Smith and Vogl 2008
230  p0 = 100000.0_r8 ! Constant for potential temp calculation
231  pbltop = 85000._r8 ! Top of boundary layer
232  pblconst = 10000._r8 ! Constant for the calculation of the decay of diffusivity
233  t00 = 288.0_r8 ! Horizontal mean T at surface for moist baro test
234  u0 = 35.0_r8 ! Zonal wind constant for moist baro test
235  latw = 2.0_r8*pi/9.0_r8 ! Halfwidth for for baro test
236  eta0 = 0.252_r8 ! Center of jets (hybrid) for baro test
237  etav = (1._r8-eta0)*0.5_r8*pi ! Auxiliary variable for baro test
238  q0 = 0.021 ! Maximum specific humidity for baro test
239 
240 !===============================================================================
241 !
242 ! Definition of local arrays
243 !
244 !===============================================================================
245 !
246 ! Calculate hydrostatic height za of the lowest model level
247 !
248  do i=1,pcols
249  dlnpint = log(ps(i)) - log(pint(i,pver)) ! ps(i) is identical to pint(i,pver+1), note: this is the correct sign (corrects typo in JAMES paper)
250  za(i) = rair/gravit*t(i,pver)*(1._r8+zvir*q(i,pver))*0.5_r8*dlnpint
251  end do
252 !
253 ! Set Initial Specific Humidity - For dry mass adjustment at the end
254 !
255  qini(:pcols,:pver) = q(:pcols,:pver)
256 !--------------------------------------------------------------
257 ! Set Sea Surface Temperature (constant for tropical cyclone)
258 ! Tsurf needs to be dependent on latitude for the
259 ! moist baroclinic wave test 4-3 with simple-physics
260 !--------------------------------------------------------------
261  if (test .eq. 1) then ! moist baroclinic wave with simple-physics
262  do i=1,pcols
263  tsurf(i) = (t00 + pi*u0/rair * 1.5_r8 * sin(etav) * (cos(etav))**0.5_r8 * &
264  ((-2._r8*(sin(lat(i)))**6 * ((cos(lat(i)))**2 + 1._r8/3._r8) + 10._r8/63._r8)* &
265  u0 * (cos(etav))**1.5_r8 + &
266  (8._r8/5._r8*(cos(lat(i)))**3 * ((sin(lat(i)))**2 + 2._r8/3._r8) - pi/4._r8)*a*omega*0.5_r8 ))/ &
267  (1._r8+zvir*q0*exp(-(lat(i)/latw)**4))
268 
269  end do
270  else
271  do i=1,pcols ! constant SST for the tropical cyclone test case
272  tsurf(i) = sst_tc
273  end do
274  end if
275 
276 !===============================================================================
277 !
278 ! Set initial physics time tendencies and precipitation field to zero
279 !
280 !===============================================================================
281  dtdt(:pcols,:pver) = 0._r8 ! initialize temperature tendency with zero
282  dqdt(:pcols,:pver) = 0._r8 ! initialize specific humidity tendency with zero
283  dudt(:pcols,:pver) = 0._r8 ! initialize zonal wind tendency with zero
284  dvdt(:pcols,:pver) = 0._r8 ! initialize meridional wind tendency with zero
285  precl(:pcols) = 0._r8 ! initialize precipitation rate with zero
286 
287 !===============================================================================
288 !
289 ! Large-Scale Condensation and Precipitation Rate
290 !
291 !===============================================================================
292 !
293 ! Calculate Tendencies
294 !
295  do k=1,pver
296  do i=1,pcols
297  qsat = epsilo*e0/pmid(i,k)*exp(-latvap/rh2o*((1._r8/t(i,k))-1._r8/t0)) ! saturation specific humidity
298  if (q(i,k) > qsat) then ! saturated?
299  tmp = 1._r8/dtime*(q(i,k)-qsat)/(1._r8+(latvap/cpair)*(epsilo*latvap*qsat/(rair*t(i,k)**2)))
300  dtdt(i,k) = dtdt(i,k)+latvap/cpair*tmp
301  dqdt(i,k) = dqdt(i,k)-tmp
302  precl(i) = precl(i) + tmp*pdel(i,k)/(gravit*rhow) ! precipitation rate, computed via a vertical integral
303  ! corrected in version 1.3
304  end if
305  end do
306  end do
307 !
308 ! Update moisture and temperature fields from Large-Scale Precipitation Scheme
309 !
310  do k=1,pver
311  do i=1,pcols
312  t(i,k) = t(i,k) + dtdt(i,k)*dtime ! update the state variables T and q
313  q(i,k) = q(i,k) + dqdt(i,k)*dtime
314  end do
315  end do
316 
317 !===============================================================================
318 ! Send variables to history file - THIS PROCESS WILL BE MODEL SPECIFIC
319 !
320 ! note: The variables, as done in many GCMs, are written to the history file
321 ! after the moist physics process. This ensures that the moisture fields
322 ! are somewhat in equilibrium.
323 !===============================================================================
324  ! call diag_phys_writeout(state) ! This is for CESM/CAM
325 
326 !===============================================================================
327 !
328 ! Turbulent mixing coefficients for the PBL mixing of horizontal momentum,
329 ! sensible heat and latent heat
330 !
331 ! We are using Simplified Ekman theory to compute the diffusion coefficients
332 ! Kx for the boundary-layer mixing. The Kx values are calculated at each time step
333 ! and in each column.
334 !
335 !===============================================================================
336 !
337 ! Compute magnitude of the wind and drag coeffcients for turbulence scheme:
338 ! they depend on the conditions at the lowest model level and stay constant
339 ! up to the 850 hPa level. Above this level the coefficients are decreased
340 ! and tapered to zero. At the 700 hPa level the strength of the K coefficients
341 ! is about 10% of the maximum strength.
342 !
343  do i=1,pcols
344  wind(i) = sqrt(u(i,pver)**2+v(i,pver)**2) ! wind magnitude at the lowest level
345  end do
346  do i=1,pcols
347  ke(i,pver+1) = c*wind(i)*za(i)
348  if( wind(i) .lt. v20) then
349  cd(i) = cd0+cd1*wind(i)
350  km(i,pver+1) = cd(i)*wind(i)*za(i)
351  else
352  cd(i) = cm
353  km(i,pver+1) = cm*wind(i)*za(i)
354  endif
355  end do
356 
357  do k=1,pver
358  do i=1,pcols
359  if( pint(i,k) .ge. pbltop) then
360  km(i,k) = km(i,pver+1) ! constant Km below 850 hPa level
361  ke(i,k) = ke(i,pver+1) ! constant Ke below 850 hPa level
362  else
363  km(i,k) = km(i,pver+1)*exp(-(pbltop-pint(i,k))**2/(pblconst)**2) ! Km tapered to 0
364  ke(i,k) = ke(i,pver+1)*exp(-(pbltop-pint(i,k))**2/(pblconst)**2) ! Ke tapered to 0
365  end if
366  end do
367  end do
368 
369 
370 !===============================================================================
371 ! Update the state variables u, v, t, q with the surface fluxes at the
372 ! lowest model level, this is done with an implicit approach
373 ! see Reed and Jablonowski (JAMES, 2012)
374 !
375 ! Sea Surface Temperature Tsurf is constant for tropical cyclone test 5-1
376 ! Tsurf needs to be dependent on latitude for the
377 ! moist baroclinic wave test 4-3 with simple-physics, adjust
378 !===============================================================================
379 
380  do i=1,pcols
381  qsats = epsilo*e0/ps(i)*exp(-latvap/rh2o*((1._r8/tsurf(i))-1._r8/t0)) ! saturation specific humidity at the surface
382  dudt(i,pver) = dudt(i,pver) + (u(i,pver) &
383  /(1._r8+cd(i)*wind(i)*dtime/za(i))-u(i,pver))/dtime
384  dvdt(i,pver) = dvdt(i,pver) + (v(i,pver) &
385  /(1._r8+cd(i)*wind(i)*dtime/za(i))-v(i,pver))/dtime
386  u(i,pver) = u(i,pver)/(1._r8+cd(i)*wind(i)*dtime/za(i))
387  v(i,pver) = v(i,pver)/(1._r8+cd(i)*wind(i)*dtime/za(i))
388  dtdt(i,pver) = dtdt(i,pver) +((t(i,pver)+c*wind(i)*tsurf(i)*dtime/za(i)) &
389  /(1._r8+c*wind(i)*dtime/za(i))-t(i,pver))/dtime
390  t(i,pver) = (t(i,pver)+c*wind(i)*tsurf(i)*dtime/za(i)) &
391  /(1._r8+c*wind(i)*dtime/za(i))
392  dqdt(i,pver) = dqdt(i,pver) +((q(i,pver)+c*wind(i)*qsats*dtime/za(i)) &
393  /(1._r8+c*wind(i)*dtime/za(i))-q(i,pver))/dtime
394  q(i,pver) = (q(i,pver)+c*wind(i)*qsats*dtime/za(i))/(1._r8+c*wind(i)*dtime/za(i))
395  end do
396 !===============================================================================
397 
398 
399 !===============================================================================
400 ! Boundary layer mixing, see Reed and Jablonowski (JAMES, 2012)
401 !===============================================================================
402 ! Calculate Diagonal Variables for Implicit PBL Scheme
403 !
404  do k=1,pver-1
405  do i=1,pcols
406  rho = (pint(i,k+1)/(rair*(t(i,k+1)+t(i,k))/2.0_r8))
407  cam(i,k) = rpdel(i,k)*dtime*gravit*gravit*km(i,k+1)*rho*rho &
408  /(pmid(i,k+1)-pmid(i,k))
409  ccm(i,k+1) = rpdel(i,k+1)*dtime*gravit*gravit*km(i,k+1)*rho*rho &
410  /(pmid(i,k+1)-pmid(i,k))
411  ca(i,k) = rpdel(i,k)*dtime*gravit*gravit*ke(i,k+1)*rho*rho &
412  /(pmid(i,k+1)-pmid(i,k))
413  cc(i,k+1) = rpdel(i,k+1)*dtime*gravit*gravit*ke(i,k+1)*rho*rho &
414  /(pmid(i,k+1)-pmid(i,k))
415  end do
416  end do
417  do i=1,pcols
418  cam(i,pver) = 0._r8
419  ccm(i,1) = 0._r8
420  cem(i,pver+1) = 0._r8
421  ca(i,pver) = 0._r8
422  cc(i,1) = 0._r8
423  ce(i,pver+1) = 0._r8
424  cfu(i,pver+1) = 0._r8
425  cfv(i,pver+1) = 0._r8
426  cft(i,pver+1) = 0._r8
427  cfq(i,pver+1) = 0._r8
428  end do
429  do i=1,pcols
430  do k=pver,1,-1
431  ce(i,k) = cc(i,k)/(1._r8+ca(i,k)+cc(i,k)-ca(i,k)*ce(i,k+1))
432  cem(i,k) = ccm(i,k)/(1._r8+cam(i,k)+ccm(i,k)-cam(i,k)*cem(i,k+1))
433  cfu(i,k) = (u(i,k)+cam(i,k)*cfu(i,k+1)) &
434  /(1._r8+cam(i,k)+ccm(i,k)-cam(i,k)*cem(i,k+1))
435  cfv(i,k) = (v(i,k)+cam(i,k)*cfv(i,k+1)) &
436  /(1._r8+cam(i,k)+ccm(i,k)-cam(i,k)*cem(i,k+1))
437  cft(i,k) = ((p0/pmid(i,k))**(rair/cpair)*t(i,k)+ca(i,k)*cft(i,k+1)) &
438  /(1._r8+ca(i,k)+cc(i,k)-ca(i,k)*ce(i,k+1))
439  cfq(i,k) = (q(i,k)+ca(i,k)*cfq(i,k+1)) &
440  /(1._r8+ca(i,k)+cc(i,k)-ca(i,k)*ce(i,k+1))
441  end do
442  end do
443 
444 !
445 ! Calculate the updated temperature, specific humidity and horizontal wind
446 !
447 ! First we need to calculate the updates at the top model level
448 !
449  do i=1,pcols
450  dudt(i,1) = dudt(i,1)+(cfu(i,1)-u(i,1))/dtime
451  dvdt(i,1) = dvdt(i,1)+(cfv(i,1)-v(i,1))/dtime
452  u(i,1) = cfu(i,1)
453  v(i,1) = cfv(i,1)
454  dtdt(i,1) = dtdt(i,1)+(cft(i,1)*(pmid(i,1)/p0)**(rair/cpair)-t(i,1))/dtime ! corrected in version 1.3
455  t(i,1) = cft(i,1)*(pmid(i,1)/p0)**(rair/cpair)
456  dqdt(i,1) = dqdt(i,1)+(cfq(i,1)-q(i,1))/dtime
457  q(i,1) = cfq(i,1)
458  end do
459 !
460 ! Loop over the remaining level
461 !
462  do i=1,pcols
