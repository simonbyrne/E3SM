385  KOKKOS_INLINE_FUNCTION
386  void operator()(const TagPostExchange&, const int idx) const {
387  // For g
388  using namespace PhysicalConstants;
389 
390  using InfoM = ColInfo<NUM_PHYSICAL_LEV>;
391  using InfoI = ColInfo<NUM_INTERFACE_LEV>;
392  constexpr int LAST_MID_PACK = InfoM::LastPack;
393  constexpr int LAST_MID_PACK_END = InfoM::LastPackEnd;
394  constexpr int LAST_INT_PACK = InfoI::LastPack;
395  constexpr int LAST_INT_PACK_END = InfoI::LastPackEnd;
396 
397  // Note: make sure you run this only in non-hydro mode
398  // KernelVariables kv(team);
399  const int ie = idx / (NP*NP);
400  const int igp = (idx / NP) % NP;
401  const int jgp = idx % NP;
402 
403  auto& u = m_state.m_v(ie,m_data.np1,0,igp,jgp,LAST_MID_PACK)[LAST_MID_PACK_END];
404  auto& v = m_state.m_v(ie,m_data.np1,1,igp,jgp,LAST_MID_PACK)[LAST_MID_PACK_END];
405  auto& w = m_state.m_w_i(ie,m_data.np1,igp,jgp,LAST_INT_PACK)[LAST_INT_PACK_END];
406  const auto& phis_x = m_geometry.m_gradphis(ie,0,igp,jgp);
407  const auto& phis_y = m_geometry.m_gradphis(ie,1,igp,jgp);
408 
409  // Compute dpnh_dp_i on surface
410  auto dpnh_dp_i = 1 + ( ( (u*phis_x + v*phis_y)/g - w) /
411  (g + (phis_x*phis_x+phis_y*phis_y)/(2*g) ) ) / m_data.dt;
412 
413  // Update w_i on bottom interface
414  // Update v on bottom level
415  w += m_data.scale1*m_data.dt*g*(dpnh_dp_i-1.0);
416  u -= m_data.scale1*m_data.dt*(dpnh_dp_i-1.0)*phis_x/2.0;
417  v -= m_data.scale1*m_data.dt*(dpnh_dp_i-1.0)*phis_y/2.0;
418 
419  // TODO: you need to modify the BoundaryExchange class a bit, cause as of today
420  // it exchanges *all* vertical levels. For phi, we don't want/need to
421  // exchange the last level, since phi=phis at surface.
422  // So to make sure we're not messing up, set phi back to phis on last interface
423  // Note: this is *independent* of whether NUM_LEV==NUM_LEV_P or not.
424  auto& phi_surf = m_state.m_phinh_i(ie,m_data.np1,igp,jgp,LAST_INT_PACK)[LAST_INT_PACK_END];
425  phi_surf = m_geometry.m_phis(ie,igp,jgp);
426 
427 #if defined(ENERGY_DIAGNOSTICS) && !defined(NDEBUG)
428  // Check w bc
429  if (fabs( (u*phis_x+v*phis_y)/g - w ) > 1e-10) {
430  printf("[CAAR] WARNING! w b.c. not satisfied at (ie,igp,jgp) = (%d,%d,%d):\n"
431  " w: %3.15f\n"
432  " v*grad(phis)/g: %3.15f\n"
433  " diff: %3.15f\n",
434  ie,igp,jgp,w,(u*phis_x+v*phis_y)/g,fabs( (u*phis_x+v*phis_y)/g - w ));
435  }
436 
437  auto phi = Homme::viewAsReal(Homme::subview(m_state.m_phinh_i,ie,m_data.np1,igp,jgp));
438  for (int k=0; k<NUM_PHYSICAL_LEV; ++k) {
439  if ( (phi(k)-phi(k+1)) < g ) {
440  printf("[CAAR] WARNING! delta z < 1m, at (ie,igp,jgp,k) = (%d,%d,%d,%d):\n"
441  " phi(k): %3.15f\n"
442  " phi(k+1): %3.15f\n",
443  ie,igp,jgp,k,phi(k),phi(k+1));
444  }
445  }
446 #endif

449  KOKKOS_INLINE_FUNCTION
450  void operator()(const TagDp3dLimiter&, const TeamMember &team) const {
451  KernelVariables kv(team, m_tu);
452 
453  // TODO: make this less hard-coded maybe?
454  constexpr Real dp3d_thresh = 0.125;
455  constexpr Real vtheta_thresh = 10; // Kelvin
456 
457  Kokkos::parallel_for(Kokkos::TeamThreadRange(kv.team,NP*NP),
458  [&](const int idx) {
459  const int igp = idx / NP;
460  const int jgp = idx % NP;
461 
462  const auto& spheremp = m_geometry.m_spheremp(kv.ie,igp,jgp);
463 
464  // Check if the minimum dp3d in this column is blow a certain threshold
465  auto dp = Homme::subview(m_state.m_dp3d,kv.ie,m_data.np1,igp,jgp);
466  auto& dp0 = m_hvcoord.dp0;
467  auto diff = Homme::subview(m_buffers.vort,kv.team_idx,igp,jgp);
468  Kokkos::parallel_for(Kokkos::ThreadVectorRange(kv.team,NUM_LEV),
469  [&](const int ilev) {
470  diff(ilev) = (dp(ilev) - dp3d_thresh*dp0(ilev))*spheremp;
471  });
472 
473  Real min_diff = Kokkos::reduction_identity<Real>::min();
474  auto diff_as_real = Homme::viewAsReal(diff);
475  Kokkos::Min<Real,ExecSpace> reducer(min_diff);
476  Kokkos::parallel_reduce(Kokkos::ThreadVectorRange(kv.team,NUM_PHYSICAL_LEV),
477  [&](const int k,Real& result) {
478  result = result<=diff_as_real(k) ? result : diff_as_real(k);
479  }, reducer);
480 
481  auto vtheta_dp = Homme::subview(m_state.m_vtheta_dp,kv.ie,m_data.np1,igp,jgp);
482  if (min_diff<0) {
483  // Compute vtheta = vtheta_dp/dp
484  Kokkos::parallel_for(Kokkos::ThreadVectorRange(kv.team,NUM_LEV),
485  [&](const int ilev) {
486  vtheta_dp(ilev) /= dp(ilev);
487  });
488 
489  // Gotta apply vertical mixing, to prevent levels from getting too thin.
490  Real mass = 0.0;
491  ColumnOps::column_reduction<NUM_PHYSICAL_LEV>(kv.team,diff,mass);
492 
493  if (mass<0) {
494  Kokkos::parallel_for(Kokkos::ThreadVectorRange(kv.team,NUM_LEV),
495  [&](const int ilev) {
496  diff(ilev) *= -1.0;
497  });
498  }
499 
500  // This loop must be done over physical levels, unless we implement
501  // masks, like it has been done in the E3SM/scream project
502  Real mass_new = 0.0;
503  Dispatch<>::parallel_reduce(kv.team,
504  Kokkos::ThreadVectorRange(kv.team,NUM_PHYSICAL_LEV),
505  [&](const int k, Real& accum) {
506  auto& val = diff_as_real(k);
507  val = (val<0 ? 0.0 : val);
508  accum += val;
509  }, mass_new);

568  KOKKOS_INLINE_FUNCTION
569  void compute_scan_quantities (KernelVariables &kv) const {
570  kv.team_barrier();
571  Kokkos::parallel_for(Kokkos::TeamThreadRange(kv.team,NP*NP),
572  [&](const int idx) {
573  const int igp = idx / NP;
574  const int jgp = idx % NP;
575 
576  // At interfaces, pi_i(k+1) = pi_i(k)+dp(k), with pi_i(0)=hyai(0)*ps0;
577  // then, pi(k) = (pi_i(k) + pi_i(k+1))/2. Hence, accumulate midpoints to interfaces,
578  // then do a interface->midpoints average.
579  // For omega, we have omega_i(k+1) = omega_i(k)+div_vdp(k), with omega_i(0)=0;
580  // then, omega(k) = v*grad(pi) - average(omega_i). You can see how we could
581  // avoid computing omega_i, and set the accumulation step to be
582  // omega(k+1) = omega(k)+(div_vdp(k)+div_vdp(k-1))/2, with omega(0)=div_vdp(0)/2.
583  // However, avoiding computing omega_i would lose BFB agreement
584  // with the original F90 implementation. So for now, keep
585  // the two step calculation for omega (interface, then midpoints)
586  // TODO; skip calculation of omega_i
587  // Note: pi_i and omega_i are not needed after computing pi and omega,
588  // so simply grab unused buffers
589  auto dp = Homme::subview(m_state.m_dp3d,kv.ie,m_data.n0,igp,jgp);
590  auto div_vdp = Homme::subview(m_buffers.div_vdp,kv.team_idx,igp,jgp);
591  auto pi = Homme::subview(m_buffers.pi,kv.team_idx,igp,jgp);
592  auto omega_i = Homme::subview(m_buffers.grad_phinh_i,kv.team_idx,0,igp,jgp);
593  auto pi_i = Homme::subview(m_buffers.grad_phinh_i,kv.team_idx,1,igp,jgp);
594 
595  Kokkos::single(Kokkos::PerThread(kv.team),[&]() {
596  pi_i(0)[0] = m_hvcoord.ps0*m_hvcoord.hybrid_ai0;
597  });
598 
599  ColumnOps::column_scan_mid_to_int<true>(kv,dp,pi_i);
600 
601  ColumnOps::compute_midpoint_values(kv,pi_i,pi);
602 
603  Kokkos::single(Kokkos::PerThread(kv.team),[&]() {
604  omega_i(0)[0] = 0.0;
605  });
606 
607  ColumnOps::column_scan_mid_to_int<true>(kv,div_vdp,omega_i);
608  // Average omega_i to midpoints, and change sign, since later
609  // omega=v*grad(pi)-average(omega_i)
610  auto omega = Homme::subview(m_buffers.omega_p,kv.team_idx,igp,jgp);
611  ColumnOps::compute_midpoint_values<CombineMode::Scale>(kv,omega_i,omega,-1.0);
612  });
613  kv.team_barrier();
614 
615  // Compute grad(pi)
616  m_sphere_ops.gradient_sphere(kv,Homme::subview(m_buffers.pi,kv.team_idx),
617  Homme::subview(m_buffers.grad_tmp,kv.team_idx));
618  kv.team_barrier();
619 
620  // Update omega with v*grad(p)
621  Kokkos::parallel_for(Kokkos::TeamThreadRange(kv.team,NP*NP),
622  [&](const int idx) {
623  const int igp = idx / NP;
624  const int jgp = idx % NP;
625 
626  auto omega = Homme::subview(m_buffers.omega_p,kv.team_idx,igp,jgp);
627  auto v = Homme::subview(m_state.m_v,kv.ie,m_data.n0);
628  auto grad_pi = Homme::subview(m_buffers.grad_tmp,kv.team_idx);
629  Kokkos::parallel_for(Kokkos::ThreadVectorRange(kv.team,NUM_LEV),
630  [&](const int ilev) {
631  omega(ilev) += (v(0,igp,jgp,ilev)*grad_pi(0,igp,jgp,ilev) +
632  v(1,igp,jgp,ilev)*grad_pi(1,igp,jgp,ilev));
633  });
634 
635  // Use EOS to compute pnh/exner or exner/phi (depending on whether it's hydro mode).
636  if (m_theta_hydrostatic_mode) {
637  // Recall that, in this case, pnh aliases pi, and pi was already computed
638  m_eos.compute_exner(kv,Homme::subview(m_buffers.pnh,kv.team_idx,igp,jgp),
639  Homme::subview(m_buffers.exner,kv.team_idx,igp,jgp));
640 
641  m_eos.compute_phi_i(kv, m_geometry.m_phis(kv.ie,igp,jgp),
642  Homme::subview(m_state.m_vtheta_dp,kv.ie,m_data.n0,igp,jgp),
643  Homme::subview(m_buffers.exner,kv.team_idx,igp,jgp),
644  Homme::subview(m_buffers.pnh,kv.team_idx,igp,jgp),
645  Homme::subview(m_state.m_phinh_i,kv.ie,m_data.n0,igp,jgp));
646  } else {
647  m_eos.compute_pnh_and_exner(kv,
648  Homme::subview(m_state.m_vtheta_dp,kv.ie,m_data.n0,igp,jgp),
649  Homme::subview(m_state.m_phinh_i,kv.ie,m_data.n0,igp,jgp),
650  Homme::subview(m_buffers.pnh,kv.team_idx,igp,jgp),
651  Homme::subview(m_buffers.exner,kv.team_idx,igp,jgp));
652  }
653 
654  // Compute phi at midpoints
655  ColumnOps::compute_midpoint_values(kv,Homme::subview(m_state.m_phinh_i,kv.ie,m_data.n0,igp,jgp),
656  Homme::subview(m_buffers.phi,kv.team_idx,igp,jgp));
657  });
658  kv.team_barrier();
659  }

773  KOKKOS_INLINE_FUNCTION
774  void compute_v_vadv (KernelVariables& kv, const int& igp, const int& jgp) const {
775  // Note: save v_vadv temp by stuffing directly in vtens
776 
777  auto u = viewAsReal(Homme::subview(m_state.m_v,kv.ie,m_data.n0,0,igp,jgp));
778  auto v = viewAsReal(Homme::subview(m_state.m_v,kv.ie,m_data.n0,1,igp,jgp));
779  auto dp = viewAsReal(Homme::subview(m_state.m_dp3d,kv.ie,m_data.n0,igp,jgp));
780  auto eta_dot_dpdn = viewAsReal(Homme::subview(m_buffers.eta_dot_dpdn,kv.team_idx,igp,jgp));
781  auto u_vadv = viewAsReal(Homme::subview(m_buffers.v_tens,kv.team_idx,0,igp,jgp));
782  auto v_vadv = viewAsReal(Homme::subview(m_buffers.v_tens,kv.team_idx,1,igp,jgp));
783 
784  // TODO: vectorize this code.
785  const auto facp_1 = 0.5*eta_dot_dpdn(1)/dp(0);
786  u_vadv(0) = facp_1*(u(1)-u(0));
787  v_vadv(0) = facp_1*(v(1)-v(0));
788 
789  Kokkos::parallel_for(Kokkos::ThreadVectorRange(kv.team,1,NUM_PHYSICAL_LEV-1),
790  [&](const int k) {
791  const auto facp = 0.5*eta_dot_dpdn(k+1)/dp(k);
792  const auto facm = 0.5*eta_dot_dpdn(k)/dp(k);
793  u_vadv(k) = facp*(u(k+1)-u(k)) + facm*(u(k)-u(k-1));
794  v_vadv(k) = facp*(v(k+1)-v(k)) + facm*(v(k)-v(k-1));
795  });
796 
797  constexpr int last = NUM_PHYSICAL_LEV-1;
798  const auto facm_N = 0.5*eta_dot_dpdn(last)/dp(last);
799  u_vadv(last) = facm_N*(u(last)-u(last-1));
800  v_vadv(last) = facm_N*(v(last)-v(last-1));
801  }

857  KOKKOS_INLINE_FUNCTION
858  void compute_accumulated_quantities(KernelVariables &kv) const {
859  // Compute omega = v*grad(p) + average(omega_i)
860  // Accmuulate: dereived.omega_p += eta_ave_w*omega
861  // Accmuulate: dereived.eta_dot_dpdn += eta_ave_w*eta_dot_dpdn
862  // Accumulate: vn0 += eta_ave_w*v*dp
863  Kokkos::parallel_for(Kokkos::TeamThreadRange(kv.team,NP*NP),
864  [&](const int idx) {
865  const int igp = idx / NP;
866  const int jgp = idx % NP;

868  // If rsplit>0, m_buffers.eta_dot_dpdn is identically 0, so skip this step
869  if (m_rsplit==0) {
870  // Accumulate eta_dot_dpdn
871  // Note: m_buffers.eta_dot_dpdn is 0 at top/bottom, so we can just accumulate NUM_LEV packs

875  m_data.eta_ave_w*m_buffers.eta_dot_dpdn(kv.team_idx,igp,jgp,ilev);
876  });
877  }

880  // Accumulate v*dp in vn0
881  // Note: so far omega already contains average(omega_i), since we didn't even
882  // bother storing omega_i, and computed directly the average

889  m_derived.m_vn0(kv.ie,1,igp,jgp,ilev) += m_data.eta_ave_w*m_buffers.vdp(kv.team_idx,1,igp,jgp,ilev);
890  });
891  });
892  }

899  // Compute w_tens = scale1*(-w_vadv_i - v*grad(w_i)) - scale2*g*(1-dpnh_dp_i)

907  grad_w_i);
908  kv.team_barrier();
909 
910  auto v_i = Homme::subview(m_buffers.v_i,kv.team_idx);
911  Kokkos::parallel_for(Kokkos::TeamThreadRange(kv.team,NP*NP),
912  [&](const int idx) {
913  const int igp = idx / NP;
914  const int jgp = idx % NP;
915 
916  auto w_tens = Homme::subview(m_buffers.w_tens,kv.team_idx,igp,jgp);
917  auto phi_tens = Homme::subview(m_buffers.phi_tens,kv.team_idx,igp,jgp);

1021  // Compute theta_tens=scale1*(-theta_vadv-div(v*vtheta_dp)

1031  return vtheta_dp(igp,jgp,ilev) / dp(igp,jgp,ilev);
1032  };
1033 
1034  auto v_vtheta_dp = [&](const int icomp, const int igp, const int jgp, const int ilev)->Scalar {
1035  return v(icomp,igp,jgp,ilev) * vtheta_dp(igp,jgp,ilev);
1036  };
1037 
1038  if (m_theta_advection_form==AdvectionForm::Conservative) {
1039  if (m_rsplit==0) {
1040  using CM = CombineMode;
1041  // If you want a CombineMode different than Replace, unfortunately you have to specify
1042  // all the template args, since the CombineMode is the last one...
1043  m_sphere_ops.divergence_sphere_cm<CM::Add>(kv,v_vtheta_dp,
1044  Homme::subview(m_buffers.theta_tens,kv.team_idx));
1045  } else {
1046  m_sphere_ops.divergence_sphere(kv,v_vtheta_dp,
1047  Homme::subview(m_buffers.theta_tens,kv.team_idx));
1048  }
1049  } else {
1050  m_sphere_ops.gradient_sphere(kv,vtheta,
1051  Homme::subview(m_buffers.grad_tmp,kv.team_idx));
1052  }
1053  kv.team_barrier();
1054 
1055  Kokkos::parallel_for(Kokkos::TeamThreadRange(kv.team,NP*NP),
1056  [&](const int idx) {
1057  const int igp = idx / NP;
1058  const int jgp = idx % NP;
1059 
1060  auto dp_tens = Homme::subview(m_buffers.dp_tens,kv.team_idx,igp,jgp);
1061  auto theta_tens = Homme::subview(m_buffers.theta_tens,kv.team_idx,igp,jgp);
1062 
1063  auto div_vdp = Homme::subview(m_buffers.div_vdp,kv.team_idx,igp,jgp);
1064 
1065  if (m_rsplit==0) {
1066  auto eta_dot_dpdn = Homme::subview(m_buffers.eta_dot_dpdn,kv.team_idx,igp,jgp);
1067  ColumnOps::compute_midpoint_delta(kv,eta_dot_dpdn,dp_tens);
1068  }

1071  [&](const int ilev) {
1072  // Compute dp_tens
1073  if (m_rsplit>0) {
1074  dp_tens(ilev) = div_vdp(ilev);
1075  } else {
1076  dp_tens(ilev) += div_vdp(ilev);
1077  }
1078 
1079  // Compute theta_tens
1080  // NOTE: if the condition is false, then theta_tens already contains div(v*theta*dp) already
1081  if (m_theta_advection_form==AdvectionForm::NonConservative) {
1082  // We need a temp, since, if rsplit=0, theta_tens is already storing theta_vadv
1083

1085  temp += m_buffers.grad_tmp(kv.team_idx,0,igp,jgp,ilev)*m_buffers.vdp(kv.team_idx,0,igp,jgp,ilev);

1087  if (m_rsplit>0) {

1089  } else {
1090  theta_tens(ilev) += temp;
1091  }
1092  }
1093  });
1094  });

1131  KOKKOS_INLINE_FUNCTION
1132  void compute_v_tens (KernelVariables &kv) const {
1133  // Not necessarily in this order,
1134  // - Compute vort=vorticity(v)
1135  // - Compute wvor = grad[average(w_i^2/2)] - average[w_i*grad(w_i)]
1136  // - Compute gradKE = grad(v*v/2)
1137  // - Compute gradExner = grad(exner)
1138  // - Compute mgrad = average[dpnh_dp_i*gradphinh_i]
1139  // - Compute v_tens =
1140  // scale1*(-v_vadv + v2*(fcor+vort)-gradKE -mgrad -cp*vtheta*gradExner - wvor)
1141  // scale1*(-v_vadv - v1*(fcor+vort)-gradKE -mgrad -cp*vtheta*gradExner - wvor)
1142 
1143  auto v_tens = Homme::subview(m_buffers.v_tens,kv.team_idx);
1144  auto vort = Homme::subview(m_buffers.vort,kv.team_idx);
1145  auto wvor = Homme::subview(m_buffers.vdp,kv.team_idx);
1146 
1147  // Compute vorticity(v)
1148  m_sphere_ops.vorticity_sphere(kv, Homme::subview(m_state.m_v,kv.ie,m_data.n0),
1149  vort);
1150 
1151  if (m_theta_hydrostatic_mode) {
1152  // In nh mode, gradphinh has already been computed, but in hydro mode
1153  // we skip the whole compute_w_and_phi_tens call
1154  m_sphere_ops.gradient_sphere(kv,Homme::subview(m_state.m_phinh_i,kv.ie,m_data.n0),
1155  Homme::subview(m_buffers.grad_phinh_i,kv.team_idx));
1156  kv.team_barrier();
1157  } else {
1158  // Compute average(w^2/2)
1159  Kokkos::parallel_for(Kokkos::TeamThreadRange(kv.team,NP*NP),
1160  [&](const int idx) {
1161  const int igp = idx / NP;
1162  const int jgp = idx % NP;
1163  auto w_i = Homme::subview(m_state.m_w_i,kv.ie,m_data.n0,igp,jgp);
1164 
1165  // Use lambda as a provider for w_i*w_i
1166  const auto w_sq = [&w_i] (const int ilev) -> Scalar{
1167  return w_i(ilev)*w_i(ilev);
1168  };
1169 
1170  // Rescale by 2 (we are averaging kinetic energy w_i*w_i/2, but the provider has no '/2')
1171  ColumnOps::compute_midpoint_values<CombineMode::Scale>(kv, w_sq, Homme::subview(m_buffers.temp,kv.team_idx,igp,jgp),0.5);
1172  });
1173  kv.team_barrier();
1174 
1175  // Compute grad(average(w^2/2))
1176  m_sphere_ops.gradient_sphere(kv, Homme::subview(m_buffers.temp,kv.team_idx),
1177  wvor);
1178 
1179  // Compute grad(w)
1180  m_sphere_ops.gradient_sphere(kv, Homme::subview(m_state.m_w_i,kv.ie,m_data.n0),
1181  Homme::subview(m_buffers.v_i,kv.team_idx));
1182  kv.team_barrier();
1183  }
1184 
1185  // Scalar w_vor,mgrad,w_gradw,gradw2;
1186  Kokkos::parallel_for(Kokkos::TeamThreadRange(kv.team,NP*NP),
1187  [&](const int idx) {
1188  const int igp = idx / NP;
1189  const int jgp = idx % NP;
1190 
1191  auto wvor_x = Homme::subview(wvor,0,igp,jgp);
1192  auto wvor_y = Homme::subview(wvor,1,igp,jgp);
1193  if (!m_theta_hydrostatic_mode) {
1194  // Compute wvor = grad(average(w^2/2)) - average(w*grad(w))
1195  // Note: vtens is already storing grad(avg(w^2/2))
1196  auto gradw_x = Homme::subview(m_buffers.v_i,kv.team_idx,0,igp,jgp);
1197  auto gradw_y = Homme::subview(m_buffers.v_i,kv.team_idx,1,igp,jgp);
1198  auto w_i = Homme::subview(m_state.m_w_i,kv.ie,m_data.n0,igp,jgp);
1199 
1200  const auto w_gradw_x = [&gradw_x,&w_i] (const int ilev) -> Scalar {
1201  return gradw_x(ilev)*w_i(ilev);
1202  };
1203  const auto w_gradw_y = [&gradw_y,&w_i] (const int ilev) -> Scalar {
1204  return gradw_y(ilev)*w_i(ilev);
1205  };
1206 
1207  ColumnOps::compute_midpoint_values<CombineMode::ScaleAdd>(kv,
1208  w_gradw_x, wvor_x, -1.0);
1209  ColumnOps::compute_midpoint_values<CombineMode::ScaleAdd>(kv,
1210  w_gradw_y, wvor_y, -1.0);
1211  }
1212 
1213  // Compute average(dpnh_dp_i*grad(phinh_i)), and add to wvor
1214  const auto phinh_i_x = Homme::subview(m_buffers.grad_phinh_i,kv.team_idx,0,igp,jgp);
1215  const auto phinh_i_y = Homme::subview(m_buffers.grad_phinh_i,kv.team_idx,1,igp,jgp);

1225  return phinh_i_y(ilev)*dpnh_dp_i(ilev);
1226  };

1229  ColumnOps::compute_midpoint_values<CombineMode::Add>(kv,prod_y,wvor_y);
1230  }

1233  auto u = Homme::subview(m_state.m_v,kv.ie,m_data.n0,0,igp,jgp);
1234  auto v = Homme::subview(m_state.m_v,kv.ie,m_data.n0,1,igp,jgp);

1240  KE(ilev) += v(ilev)*v(ilev);
1241  KE(ilev) /= 2.0;
1242 
1243  vort(igp,jgp,ilev) += fcor;
1244  });
1245  });

1251  m_sphere_ops.gradient_sphere_update(kv, Homme::subview(m_buffers.temp,kv.team_idx),
1252  Homme::subview(m_buffers.v_tens,kv.team_idx));

1256  Homme::subview(m_buffers.v_tens,kv.team_idx));

1262  m_sphere_ops.gradient_sphere(kv, Homme::subview(m_buffers.exner,kv.team_idx),
1263  grad_exner);
1264  kv.team_barrier();
1265 
1266  Kokkos::parallel_for(Kokkos::TeamThreadRange(kv.team,NP*NP),
1267  [&](const int idx) {
1268  const int igp = idx / NP;
1269  const int jgp = idx % NP;
1270

1272  // Note: right now, vtens already contains gradKE+v_vadv
1273  // (or just gradKE if rsplit>0)
1274  auto u_tens = Homme::subview(m_buffers.v_tens,kv.team_idx,0,igp,jgp);
