33 {
34  GPTLstart("tl-ae prim_advance_exp");
35 
36 #ifdef ARKODE
37  Errors::runtime_abort("'ARKODE' support not yet available in C++ build.\n",
38  Errors::err_not_implemented);

40 
41  auto& context = Context::singleton();
42 
43  // Get simulation params
44  SimulationParams& params = context.get<SimulationParams>();
45 
46  // Determine the tracers time level
47  tl.update_tracers_levels(params.qsplit);
48 
49  // Set eta_ave_w
50  Real eta_ave_w = 1.0/params.qsplit;
51 
52  // From f90 code: "this should not be needed, but in case physics update u without updating w b.c."
53  if (!params.theta_hydrostatic_mode) {
54  auto e = context.get<Elements>();
55  auto w_i = e.m_state.m_w_i;
56  auto v = e.m_state.m_v;
57  auto gradphis = e.m_geometry.m_gradphis;
58  auto n0 = tl.n0;
59  constexpr auto LAST_LEV_P = ColInfo<NUM_INTERFACE_LEV>::LastPack;
60  constexpr auto LAST_LEV = ColInfo<NUM_PHYSICAL_LEV>::LastPack;
61  constexpr auto LAST_INTERFACE_VEC_IDX = ColInfo<NUM_INTERFACE_LEV>::LastPackEnd;
62  constexpr auto LAST_MIDPOINT_VEC_IDX = ColInfo<NUM_PHYSICAL_LEV>::LastPackEnd;
63  Kokkos::parallel_for(Kokkos::RangePolicy<ExecSpace>(0,NP*NP*e.m_geometry.num_elems()),
64  KOKKOS_LAMBDA(const int idx) {
65  const int ie = idx / (NP*NP);
66  const int igp = (idx / NP) % NP;
67  const int jgp = idx % NP;
68  w_i(ie,n0,igp,jgp,LAST_LEV_P)[LAST_INTERFACE_VEC_IDX] =
69  (v(ie,n0,0,igp,jgp,LAST_LEV)[LAST_MIDPOINT_VEC_IDX]*gradphis(ie,0,igp,jgp) +
70  v(ie,n0,1,igp,jgp,LAST_LEV)[LAST_MIDPOINT_VEC_IDX]*gradphis(ie,1,igp,jgp))/PhysicalConstants::g;
71  });
72  }
73 
74 #ifndef CAM
75  // if "prescribed wind" set dynamics explicitly and skip time-integration
76  if (params.prescribed_wind) {
77  Errors::runtime_abort("'prescribed wind' functionality not yet available in C++ build.\n",
78  Errors::err_not_implemented);
79  }
80 #endif
81 
82  switch (params.time_step_type) {
83  case TimeStepType::RK2:
84  RK2_timestep (tl, dt, eta_ave_w);
85  break;
86  case TimeStepType::IMEX_KG254_EX:
87  imex_KG254_explicit_timestep (tl, dt, eta_ave_w);
88  break;
89  case TimeStepType::ULLRICH_RK35:
90  u3_5stage_timestep (tl, dt, eta_ave_w);
91  break;
92  case TimeStepType::IMEX_KG243:
93  imex_KG243_timestep (tl, dt, eta_ave_w);
94  break;
95  case TimeStepType::IMEX_KG254:
96  imex_KG254_timestep (tl, dt, eta_ave_w);
97  break;
98  case TimeStepType::IMEX_KG255:
99  imex_KG255_timestep (tl, dt, eta_ave_w);
100  break;
101  default:
102  {
103  std::string msg = "[prim_advance_exp]:";
104  msg += "missing some code for time step method ";
105  msg += etoi(params.time_step_type);
106  msg += ".\n";
107  Errors::runtime_abort(msg,Errors::err_not_implemented);
108  }
109  }
110 
111  if (compute_diagnostics) {
112  auto& diags = context.get<Diagnostics>();
113  diags.run_diagnostics(false,4);
114  }
115 
116  if (params.hypervis_order==2 && params.nu>0) {
117  HyperviscosityFunctor& functor = context.get<HyperviscosityFunctor>();
118  GPTLstart("tl-ae advance_hypervis_dp");
119  functor.run(tl.np1,dt,eta_ave_w);
120  GPTLstop("tl-ae advance_hypervis_dp");
121  }
122 
123  if (params.dcmip16_mu>0) {
124  Errors::runtime_abort("'dcmip16_mu>0' functionality not yet available in C++ build.\n",
125  Errors::err_not_implemented);
126  }
127 
128  if (compute_diagnostics) {
129  auto& diags = context.get<Diagnostics>();
130  diags.run_diagnostics(false,5);
131  }
132 
133  GPTLstop("tl-ae prim_advance_exp");
134 }

154 void u3_5stage_timestep(const TimeLevel& tl, const Real dt, const Real eta_ave_w)
155 {
156  GPTLstart("tl-ae U3-5stage_timestep");
157  // Get elements structure
158  Elements& elements = Context::singleton().get<Elements>();
159  SimulationParams& params = Context::singleton().get<SimulationParams>();
160 
161  // Create the functor
162  CaarFunctor& functor = Context::singleton().get<CaarFunctor>();
163 
164  const int nm1 = tl.nm1;
165  const int n0 = tl.n0;
166  const int np1 = tl.np1;
167  const int qn0 = tl.n0_qdp;
168 
169  // ===================== RK STAGES ===================== //
170 
171  // Stage 1: u1 = u0 + dt/5 RHS(u0), t_rhs = t
172  functor.run(RKStageData(n0, n0, nm1, qn0, dt/5.0, eta_ave_w/4.0));
173 
174  // Stage 2: u2 = u0 + dt/5 RHS(u1), t_rhs = t + dt/5
175  functor.run(RKStageData(n0, nm1, np1, qn0, dt/5.0, 0.0));
176 
177  // Stage 3: u3 = u0 + dt/3 RHS(u2), t_rhs = t + dt/5 + dt/5
178  functor.run(RKStageData(n0, np1, np1, qn0, dt/3.0, 0.0));
179 
180  // Stage 4: u4 = u0 + 2dt/3 RHS(u3), t_rhs = t + dt/5 + dt/5 + dt/3
181  functor.run(RKStageData(n0, np1, np1, qn0, 2.0*dt/3.0, 0.0));
182 
183  // Compute (5u1-u0)/4 and store it in timelevel nm1
184  {
185  const auto v = elements.m_state.m_v;
186  const auto w = elements.m_state.m_w_i;
187  const auto vtheta_dp = elements.m_state.m_vtheta_dp;
188  const auto phinh = elements.m_state.m_phinh_i;
189  const auto dp3d = elements.m_state.m_dp3d;
190  const auto hydrostatic_mode = params.theta_hydrostatic_mode;
191 
192  Kokkos::parallel_for(
193  Kokkos::RangePolicy<ExecSpace>(0, elements.num_elems()*NP*NP*NUM_LEV),
194  KOKKOS_LAMBDA(const int it) {
195  const int ie = it / (NP*NP*NUM_LEV);
196  const int igp = (it / (NP*NUM_LEV)) % NP;
197  const int jgp = (it / NUM_LEV) % NP;
198  const int ilev = it % NUM_LEV;
199  v(ie,nm1,0,igp,jgp,ilev) = (5.0*v(ie,nm1,0,igp,jgp,ilev)-v(ie,n0,0,igp,jgp,ilev))/4.0;
200  v(ie,nm1,1,igp,jgp,ilev) = (5.0*v(ie,nm1,1,igp,jgp,ilev)-v(ie,n0,1,igp,jgp,ilev))/4.0;
201  vtheta_dp(ie,nm1,igp,jgp,ilev) = (5.0*vtheta_dp(ie,nm1,igp,jgp,ilev)-vtheta_dp(ie,n0,igp,jgp,ilev))/4.0;
202  dp3d(ie,nm1,igp,jgp,ilev) = (5.0*dp3d(ie,nm1,igp,jgp,ilev)-dp3d(ie,n0,igp,jgp,ilev))/4.0;
203  if (!hydrostatic_mode) {
204  w(ie,nm1,igp,jgp,ilev) = (5.0*w(ie,nm1,igp,jgp,ilev)-w(ie,n0,igp,jgp,ilev))/4.0;
205  phinh(ie,nm1,igp,jgp,ilev) = (5.0*phinh(ie,nm1,igp,jgp,ilev)-phinh(ie,n0,igp,jgp,ilev))/4.0;
206  }
207  });
208  // If NUM_LEV==NUM_LEV_P, the code above will take care also of the last interface
209  if (NUM_LEV_P>NUM_LEV && !hydrostatic_mode) {
210  const int LAST_INT = NUM_LEV_P-1;
211  Kokkos::parallel_for(
212  Kokkos::RangePolicy<ExecSpace>(0, elements.num_elems()*NP*NP),
213  KOKKOS_LAMBDA(const int it) {
214  const int ie = it / (NP*NP);
215  const int igp = (it / NP) % NP;
216  const int jgp = it % NP;
217  w(ie,nm1,igp,jgp,LAST_INT) = (5.0*w(ie,nm1,igp,jgp,LAST_INT)-w(ie,n0,igp,jgp,LAST_INT))/4.0;
218  });
219  }
220  }
221  ExecSpace::impl_static_fence();
222 
223  // Stage 5: u5 = (5u1-u0)/4 + 3dt/4 RHS(u4), t_rhs = t + dt/5 + dt/5 + dt/3 + 2dt/3
224  functor.run(RKStageData(nm1, np1, np1, qn0, 3.0*dt/4.0, 3.0*eta_ave_w/4.0));
225  GPTLstop("tl-ae U3-5stage_timestep");
226 }

245 void imex_KG255_timestep(const TimeLevel& tl,
246  const Real dt_dyn,
247  const Real eta_ave_w)
248 {
249  GPTLstart("IMEX_KG255");
250 
251  // The context
252  const auto& c = Context::singleton();
253 
254  // Get elements, hvcoord, and functors
255  auto& elements = c.get<Elements>();
256  auto& hvcoord = c.get<HybridVCoord>();
257  auto& dirk = c.get<DirkFunctor>();
258  auto& caar = c.get<CaarFunctor>();
259 
260  const int nm1 = tl.nm1;
261  const int n0 = tl.n0;
262  const int np1 = tl.np1;
263  const int qn0 = tl.n0_qdp;
264 
265  // ===================== IMEX STAGES ===================== //
266 
268 // Time level indices
269 // caar: nm1, n0, np1
270 // dirk: nm1, n0, np1
272 
273  // Stage 1
274  Real dt = dt_dyn/4.0;
275 
276  caar.run(RKStageData(n0, n0, nm1, qn0, dt, 0.0, 1.0, 0.0, 1.0));
277  dirk.run(nm1, 0.0, n0, 0.0, nm1, dt, elements, hvcoord);
278 
279  // Stage 2
280  dt = dt_dyn/6.0;
281 
282  caar.run(RKStageData(n0, nm1, np1, qn0, dt, 0.0, 1.0, 0.0, 1.0));
283  dirk.run(nm1, 0.0, n0, 0.0, np1, dt, elements, hvcoord);
284 
285  // Stage 3
286  dt = 3.0*dt_dyn/8.0;
287 
288  caar.run(RKStageData(n0, np1, np1, qn0, dt, 0.0, 1.0, 0.0, 1.0));
289  dirk.run(nm1, 0.0, n0, 0.0, np1, dt, elements, hvcoord);
290 
291  // Stage 4
292  dt = dt_dyn/2.0;
293 
294  caar.run(RKStageData(n0, np1, np1, qn0, dt, 0.0, 1.0, 0.0, 1.0));
295  dirk.run(nm1, 0.0, n0, 0.0, np1, dt, elements, hvcoord);
296 
297  // Stage 5
298  Real a1 = 0.24362;
299  Real a2 = 0.34184;
300  Real a3 = 1-(a1+a2);
301  dt = dt_dyn;
302 
303  caar.run(RKStageData(n0, np1, np1, qn0, dt, eta_ave_w, 1.0, 0.0, 1.0));
304  dirk.run(nm1, a2*dt, n0, a1*dt, np1, a3*dt, elements, hvcoord);
305 
306  GPTLstop("IMEX_KG255");
307 }
