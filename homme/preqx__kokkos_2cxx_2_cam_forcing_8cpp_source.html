20 void state_forcing(
21  const ExecViewUnmanaged<const Scalar * [NP][NP][NUM_LEV]> &f_t,
22  const ExecViewUnmanaged<const Scalar * [2][NP][NP][NUM_LEV]> &f_m,
23  const int &np1, const Real &dt,
24  const ExecViewUnmanaged<Scalar * [NUM_TIME_LEVELS][NP][NP][NUM_LEV]> &t,
25  const ExecViewUnmanaged<Scalar * [NUM_TIME_LEVELS][2][NP][NP][NUM_LEV]> &
26  v) {
27  const int num_e = f_t.extent_int(0);
28  Kokkos::parallel_for(
29  "state temperature forcing",
30  Kokkos::RangePolicy<ExecSpace>(0, num_e * NP * NP * NUM_LEV),
31  KOKKOS_LAMBDA(const int & idx) {
32  const int ie = ((idx / NUM_LEV) / NP) / NP;
33  const int igp = ((idx / NUM_LEV) / NP) % NP;
34  const int jgp = (idx / NUM_LEV) % NP;

36  t(ie, np1, igp, jgp, k) += dt * f_t(ie, igp, jgp, k);
37  });
38  Kokkos::parallel_for(
39  "state velocity forcing",
40  Kokkos::RangePolicy<ExecSpace>(0, num_e * 2 * NP * NP * NUM_LEV),
41  KOKKOS_LAMBDA(const int & idx) {
42  const int ie = (((idx / NUM_LEV) / NP) / NP) / 2;
43  const int dim = (((idx / NUM_LEV) / NP) / NP) % 2;
44  const int igp = ((idx / NUM_LEV) / NP) % NP;
45  const int jgp = (idx / NUM_LEV) % NP;
46  const int k = idx % NUM_LEV;
47  v(ie, np1, dim, igp, jgp, k) += dt * f_m(ie, dim, igp, jgp, k);
48  });

51 void tracer_forcing(
52  const ExecViewUnmanaged<const Scalar * [QSIZE_D][NP][NP][NUM_LEV]> &f_q,
53  const HybridVCoord &hvcoord, const TimeLevel &tl, const int &num_q,
54  const MoistDry &moisture, const double &dt,
55  const ExecViewManaged<Real * [NUM_TIME_LEVELS][NP][NP]> &ps_v,
56  const ExecViewManaged<
57  Scalar * [Q_NUM_TIME_LEVELS][QSIZE_D][NP][NP][NUM_LEV]> &qdp,
58  const ExecViewManaged<Scalar * [QSIZE_D][NP][NP][NUM_LEV]> &Q) {
59 
60  const int num_e = ps_v.extent_int(0);
61  const int np1 = tl.n0;
62  const int np1_qdp = tl.n0_qdp;
63 
64  if (moisture == MoistDry::MOIST) {
65  // Remove the m_fq_ps_v buffer since it's not actually needed.
66  // Instead apply the forcing to m_ps_v directly
67  // Bonus - one less parallel reduce in dry cases!
68 
69  // This block must go force so that qdp_s, which is read but not written,
70  // has its original value.
71 
72  // This conserves the dry mass in the process of forcing tracer 0
73  const auto policy = Homme::get_default_team_policy<ExecSpace>(num_e);
74  Kokkos::parallel_for("tracer FQ_PS forcing", policy,
75  KOKKOS_LAMBDA(const TeamMember & team) {
76  KernelVariables kv(team);
77  const int &ie = kv.ie;
78 
79  Kokkos::parallel_for(Kokkos::TeamThreadRange(kv.team, NP * NP),
80  [&](const int pt_idx) {
81  const int igp = pt_idx / NP;
82  const int jgp = pt_idx % NP;
83 
84  Real ps_v_forcing = 0.0;
85 
86  Dispatch<ExecSpace>::parallel_reduce(
87  kv.team, Kokkos::ThreadVectorRange(kv.team, NUM_PHYSICAL_LEV),
88  [&](const int &k, Real &accumulator) {
89  const int ilev = k / VECTOR_SIZE;
90  const int vlev = k % VECTOR_SIZE;
91  Real v1 = dt * f_q(ie, 0, igp, jgp, ilev)[vlev];
92  const Real &qdp_s =
93  qdp(ie, np1_qdp, 0, igp, jgp, ilev)[vlev];
94  if (qdp_s + v1 < 0.0 && v1 < 0.0) {
95  if (qdp_s < 0.0) {
96  v1 = 0.0;
97  } else {
98  v1 = -qdp_s;
99  }
100  }
101  accumulator += v1/dt;
102  },
103  ps_v_forcing);
104  ps_v(ie, np1, igp, jgp) += dt*ps_v_forcing;
105  });
106  });
107  }
108 
109  Kokkos::parallel_for(
110  "tracer qdp forcing",
111  Kokkos::RangePolicy<ExecSpace>(0, num_e * num_q * NP * NP * NUM_LEV),
112  KOKKOS_LAMBDA(const int & idx) {
113  const int ie = (((idx / NUM_LEV) / NP) / NP) / num_q;
114  const int iq = (((idx / NUM_LEV) / NP) / NP) % num_q;
115  const int igp = ((idx / NUM_LEV) / NP) % NP;
116  const int jgp = (idx / NUM_LEV) % NP;
117  const int k = idx % NUM_LEV;
118  Scalar v1 = dt * f_q(ie, iq, igp, jgp, k);
119  Scalar &qdp_s = qdp(ie, np1_qdp, iq, igp, jgp, k);
120  VECTOR_SIMD_LOOP
121  for (int vlev = 0; vlev < VECTOR_SIZE; ++vlev) {
122  if (qdp_s[vlev] + v1[vlev] < 0.0 && v1[vlev] < 0.0) {
123  if (qdp_s[vlev] < 0.0) {
124  v1[vlev] = 0.0;
125  } else {
126  v1[vlev] = -qdp_s[vlev];
127  }
128  }
129  qdp_s[vlev] += v1[vlev];
130  }
131  });
132 
133  Kokkos::parallel_for("tracer forcing ps_v",
134  Kokkos::RangePolicy<ExecSpace>(0, num_e * num_q * NP *
135  NP * NUM_LEV),
136  KOKKOS_LAMBDA(const int & idx) {
137  const int ie = (((idx / NUM_LEV) / NP) / NP) / num_q;
138  const int iq = (((idx / NUM_LEV) / NP) / NP) % num_q;
139  const int igp = ((idx / NUM_LEV) / NP) % NP;
140  const int jgp = (idx / NUM_LEV) % NP;
141  const int k = idx % NUM_LEV;
142 
143  const Scalar dp = hvcoord.hybrid_ai_delta(k) * hvcoord.ps0 +
144  hvcoord.hybrid_bi_delta(k) * ps_v(ie, np1, igp, jgp);
145  Q(ie, iq, igp, jgp, k) = qdp(ie, np1_qdp, iq, igp, jgp, k) / dp;
146  });
147 }

150  GPTLstart("ApplyCAMForcing");

152  const TimeLevel &tl = Context::singleton().get<TimeLevel>();
153 
154  state_forcing(elems.m_forcing.m_ft, elems.m_forcing.m_fm, tl.n0, dt, elems.m_state.m_t, elems.m_state.m_v);
155 
156  const SimulationParams &sim_params =
157  Context::singleton().get<SimulationParams>();
158  const HybridVCoord &hvcoord = Context::singleton().get<HybridVCoord>();
159  Tracers &tracers = Context::singleton().get<Tracers>();
160  if(tracers.fq.data() == nullptr) {
161  tracers.fq = decltype(tracers.fq)("fq", elems.num_elems());
162  }
163  tracer_forcing(tracers.fq, hvcoord, tl, tracers.num_tracers(),
164  sim_params.moisture, dt, elems.m_state.m_ps_v, tracers.qdp, tracers.Q);
165  GPTLstop("ApplyCAMForcing");
166 }
