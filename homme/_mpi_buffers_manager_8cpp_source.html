70 void MpiBuffersManager::allocate_buffers ()
71 {
72  // If views are marked as valid, they are already allocated, and no other
73  // customer has requested a larger size
74  if (m_views_are_valid) {
75  return;
76  }
77 
78  // The buffers used for packing/unpacking
79  m_send_buffer = ExecViewManaged<Real*>("send buffer", m_mpi_buffer_size);
80  m_recv_buffer = ExecViewManaged<Real*>("recv buffer", m_mpi_buffer_size);
81  m_local_buffer = ExecViewManaged<Real*>("local buffer", m_local_buffer_size);
82 
83  // The buffers used in MPI calls
84  m_mpi_send_buffer = Kokkos::create_mirror_view(decltype(m_mpi_send_buffer)::execution_space(),m_send_buffer);
85  m_mpi_recv_buffer = Kokkos::create_mirror_view(decltype(m_mpi_recv_buffer)::execution_space(),m_recv_buffer);
86 
87  m_views_are_valid = true;
88 
89  // Tell to all our customers that they need to redo the setup of the internal buffer views
90  for (auto& be_ptr : m_customers) {
91  // Invalidate buffer views and requests in the customer (if none built yet, it's a no-op)
92  be_ptr.first->clear_buffer_views_and_requests ();
93  }
94 }

152 void MpiBuffersManager::update_requested_sizes (typename std::map<BoundaryExchange*,CustomerNeeds>::value_type& customer)
153 {
154  // Make sure connectivity is valid
155  assert (m_connectivity && m_connectivity->is_finalized());
156 
157  // Make sure this is a customer
158  assert (m_customers.find(customer.first)!=m_customers.end());
159 
160  // Get the number of fields that this customer has
161  const int num_1d_fields = customer.first->get_num_1d_fields();
162  const int num_2d_fields = customer.first->get_num_2d_fields();
163  const int num_3d_fields = customer.first->get_num_3d_fields();
164  const int num_3d_int_fields = customer.first->get_num_3d_int_fields();
165 
166  // Compute the requested buffers sizes and compare with stored ones
167  required_buffer_sizes (num_1d_fields, num_2d_fields, num_3d_fields, num_3d_int_fields, customer.second.mpi_buffer_size, customer.second.local_buffer_size);
168  if (customer.second.mpi_buffer_size>m_mpi_buffer_size) {
169  // Update the total
170  m_mpi_buffer_size = customer.second.mpi_buffer_size;
171 
172  // Mark the views as invalid
173  m_views_are_valid = false;
174  }
175 
176  if(customer.second.local_buffer_size>m_local_buffer_size) {
177  // Update the total
178  m_local_buffer_size = customer.second.local_buffer_size;
179 
180  // Mark the views as invalid
181  m_views_are_valid = false;
182  }
183 }

185 void MpiBuffersManager::required_buffer_sizes (const int num_1d_fields, const int num_2d_fields,
186  const int num_3d_fields, const int num_3d_interface_fields,
187  size_t& mpi_buffer_size, size_t& local_buffer_size) const
188 {
189  mpi_buffer_size = local_buffer_size = 0;
190 
191  // The buffer size for each connection kind
192  // Note: for 2d/3d fields, we have 1 Real per GP (per level, in 3d). For 1d fields,
193  // we have 2 Real per level (max and min over element).
194  int elem_buf_size[2];
195  const int pt_buf_size = num_2d_fields + num_3d_fields*NUM_LEV*VECTOR_SIZE + num_3d_interface_fields*NUM_LEV_P*VECTOR_SIZE;
196  elem_buf_size[etoi(ConnectionKind::CORNER)] = num_1d_fields*2*NUM_LEV*VECTOR_SIZE + pt_buf_size * 1;
197  elem_buf_size[etoi(ConnectionKind::EDGE)] = num_1d_fields*2*NUM_LEV*VECTOR_SIZE + pt_buf_size * NP;
198 
199  // Compute the requested buffers sizes and compare with stored ones
200  mpi_buffer_size += elem_buf_size[etoi(ConnectionKind::CORNER)] * m_connectivity->get_num_connections<HostMemSpace>(ConnectionSharing::SHARED,ConnectionKind::CORNER);
201  mpi_buffer_size += elem_buf_size[etoi(ConnectionKind::EDGE)] * m_connectivity->get_num_connections<HostMemSpace>(ConnectionSharing::SHARED,ConnectionKind::EDGE);
202 
203  local_buffer_size += elem_buf_size[etoi(ConnectionKind::CORNER)] * m_connectivity->get_num_connections<HostMemSpace>(ConnectionSharing::LOCAL,ConnectionKind::CORNER);
204  local_buffer_size += elem_buf_size[etoi(ConnectionKind::EDGE)] * m_connectivity->get_num_connections<HostMemSpace>(ConnectionSharing::LOCAL,ConnectionKind::EDGE);
205 }
