127  KOKKOS_INLINE_FUNCTION
128  void preprocess_state (const KernelVariables& kv,
129  const int istate,
130  const int np1,
131  ExecViewUnmanaged<const Scalar[NP][NP][NUM_LEV]> dp) const {
132  assert (m_process_nh_vars);
133  // Note: w/phi are the state 3/4, but when it comes to pre/post processing they are 0/1
134  if (istate==0) {
135  // Compute delta_w
136  Kokkos::parallel_for(Kokkos::TeamThreadRange(kv.team,NP*NP),
137  [&](const int idx) {
138  const int igp = idx / NP;
139  const int jgp = idx % NP;
140  auto w_i = Homme::subview(m_state.m_w_i,kv.ie,np1,igp,jgp);
141  auto delta_w = Homme::subview(m_delta_w,kv.ie,igp,jgp);
142 
143  ColumnOps::compute_midpoint_delta(kv,w_i,delta_w);
144  });
145  } else if (istate==1) {
146  // Compute delta_phinh
147  Kokkos::parallel_for(Kokkos::TeamThreadRange(kv.team,NP*NP),
148  [&](const int idx) {
149  const int igp = idx / NP;
150  const int jgp = idx % NP;
151 
152  // Subviews (note: recycle phi_ref for the temporary p_i)
153  auto phinh_i = Homme::subview(m_state.m_phinh_i,kv.ie,np1,igp,jgp);
154  auto delta_phinh = Homme::subview(m_delta_phinh,kv.ie,igp,jgp);
155  auto phi_ref = Homme::subview(m_phi_ref,kv.team_idx,igp,jgp);
156  auto dp_pt = Homme::subview(dp,igp,jgp);
157  auto p_i = Homme::subview(m_phi_ref,kv.team_idx,igp,jgp);
158  auto p = Homme::subview(m_temp,kv.team_idx,igp,jgp);
159 
160  // Compute hydrostatic phi
161  m_elem_ops.compute_hydrostatic_p(kv,dp_pt,p_i,p);
162  m_eos.compute_phi_i(kv,m_geometry.m_phis(kv.ie,igp,jgp),
163  Homme::subview(m_state.m_vtheta_dp,kv.ie,np1,igp,jgp),
164  p,
165  phi_ref);
166 
167  Kokkos::parallel_for(Kokkos::ThreadVectorRange(kv.team,NUM_LEV_P),
168  [&](const int ilev) {
169  phinh_i(ilev) -= phi_ref(ilev);
170  });
171 
172  // Now compute phinh_i increments
173  ColumnOps::compute_midpoint_delta(kv,phinh_i,delta_phinh);
174  });
175  }
176  }

178  KOKKOS_INLINE_FUNCTION
179  void postprocess_state (const KernelVariables& kv,
180  const int istate,
181  const int np1,
182  ExecViewUnmanaged<const Scalar[NP][NP][NUM_LEV]> dp) const {
183  assert (m_process_nh_vars);
184  using InfoI = ColInfo<NUM_INTERFACE_LEV>;
185  using InfoM = ColInfo<NUM_PHYSICAL_LEV>;
186  // Note: w/phi are the state 3/4, but when it comes to pre/post processing they are 0/1
187  if (istate==0) {
188  // Update w_i
189  Kokkos::parallel_for(Kokkos::TeamThreadRange(kv.team,NP*NP),
190  [&](const int idx) {
191  const int igp = idx / NP;
192  const int jgp = idx % NP;
193  auto w_i = Homme::subview(m_state.m_w_i,kv.ie,np1,igp,jgp);
194  auto delta_w = Homme::subview(m_delta_w,kv.ie,igp,jgp);
195 
196  auto minus_delta = [&](const int ilev)->Scalar { return -delta_w(ilev); };
197  ColumnOps::column_scan_mid_to_int<false>(kv,minus_delta,w_i);
198 
199  // Since u changed, update w_i b.c. at the surface
200  Kokkos::single(Kokkos::PerThread(kv.team),[&](){
201  constexpr int LAST_MID_PACK = InfoM::LastPack;
202  constexpr int LAST_MID_PACK_END = InfoM::LastPackEnd;
203  constexpr int LAST_INT_PACK = InfoI::LastPack;
204  constexpr int LAST_INT_PACK_END = InfoI::LastPackEnd;
205  constexpr auto g = PhysicalConstants::g;
206 
207  const auto gradphis = Homme::subview(m_geometry.m_gradphis,kv.ie);
208  const auto v = Homme::subview(m_state.m_v,kv.ie,np1);
209 
210  w_i(LAST_INT_PACK)[LAST_INT_PACK_END] =
211  (v(0,igp,jgp,LAST_MID_PACK)[LAST_MID_PACK_END]*gradphis(0,igp,jgp) +
212  v(1,igp,jgp,LAST_MID_PACK)[LAST_MID_PACK_END]*gradphis(1,igp,jgp)) / g;
213  });
214  });
215  } else if (istate==1) {
216  // Update phinh_i
217  Kokkos::parallel_for(Kokkos::TeamThreadRange(kv.team,NP*NP),
218  [&](const int idx) {
219  const int igp = idx / NP;
220  const int jgp = idx % NP;
221 
222  // Subviews (note: recycle phi_ref for the temporary p_i)
223  auto phinh_i = Homme::subview(m_state.m_phinh_i,kv.ie,np1,igp,jgp);
224  auto delta_phinh = Homme::subview(m_delta_phinh,kv.ie,igp,jgp);
225  auto phi_ref = Homme::subview(m_phi_ref,kv.team_idx,igp,jgp);
226  auto dp_pt = Homme::subview(dp,igp,jgp);
227  auto p_i = Homme::subview(m_phi_ref,kv.team_idx,igp,jgp);
228  auto p = Homme::subview(m_temp,kv.team_idx,igp,jgp);
229 
230  // Recompute hydrostatic phi with new theta
231  m_elem_ops.compute_hydrostatic_p(kv,dp_pt,p_i,p);
232  m_eos.compute_phi_i(kv,m_geometry.m_phis(kv.ie,igp,jgp),
233  Homme::subview(m_state.m_vtheta_dp,kv.ie,np1,igp,jgp),
234  p,
235  phi_ref);
236 
237  // // phinh_i(k) = phinh_i(k+1) - delta_phinh(k), so do a backward scan sum of -delta_phinh
238  auto minus_delta = [&](const int ilev)->Scalar { return -delta_phinh(ilev); };
239  ColumnOps::column_scan_mid_to_int<false>(kv,minus_delta,phinh_i);
240 
241  // Add hydrostatic phi
242  Kokkos::parallel_for(Kokkos::ThreadVectorRange(kv.team,NUM_LEV_P),
243  [&](const int ilev) {
244  phinh_i(ilev) += phi_ref(ilev);
245  });
246  });
247  }
248  }

250  KOKKOS_INLINE_FUNCTION
251  ExecViewUnmanaged<Scalar[NP][NP][NUM_LEV]>
252  get_state(const KernelVariables &kv, int np1, int var) const {
253  assert(var>=0 && var<=4);
254  switch (var) {
255  case 0:
256  return Homme::subview(m_state.m_v, kv.ie, np1, 0);
257  case 1:
258  return Homme::subview(m_state.m_v, kv.ie, np1, 1);
259  case 2:
260  return Homme::subview(m_state.m_vtheta_dp, kv.ie, np1);
261  case 3:
262  return Homme::subview(m_delta_w, kv.ie);
263  case 4:
264  return Homme::subview(m_delta_phinh, kv.ie);
265  default:
266  Kokkos::abort("RemapStateProvider: invalid variable index.\n");
267  return ExecViewUnmanaged<Scalar[NP][NP][NUM_LEV]>();
268  }
