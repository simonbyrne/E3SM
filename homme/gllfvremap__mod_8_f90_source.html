5 module gllfvremap_mod
6  ! High-order, mass-conserving, optionally shape-preserving
7  ! FV physics <-> GLL dynamics
8  ! remap.
9  !
10  ! This module implements algorithms to remap state variables and
11  ! tendencies between the spectral element (SE) dynamics grid and the
12  ! finite volume (FV) physics grid. The two grids are the same at the
13  ! element level. The dynamics grid has np x np GLL grid points
14  ! inside each element. The physics grid has nphys x nphys FV
15  ! subcells inside each element. We frequently shorten nphys to nf or
16  ! just N in the pgN physics grid specification. Thus, the dynamics
17  ! grid specification may be written ne30np4 and the physics grid one
18  ! as ne30pg2.
19  ! The remap algorithms have the following properties:
20  ! * Conserve mass.
21  ! * Have order of accuracy (OOA) N for N in pgN without a limiter.
22  ! * Optionally prevent new mixing ratio (Q) extrema by a
23  ! nonlinear limiter. OOA is then min(N,2). The extrema are
24  ! limited in the full state, not just the tendency, so Q state
25  ! or tendency returned from the physics is assured to be
26  ! physically valid.
27  ! * pg1's OOA is boosted to ~1.6.
28  ! This module works with cubed_sphere_map 0 and 2, but only 2 will
29  ! work in the fully coupled E3SM.
30  ! This module supports all ftype values 0 to 4.
31  ! We find in practice that pg1 is too coarse. To see this, run the
32  ! Homme-standalone test dcmip2016_test1_pg1 and compare results with
33  ! dcmip2016_test1_pg2 and dcmip2016_test1 (np4). pg2 and np4 fields are
34  ! nearly identical out to day 30, whereas pg1 fields differ visibly.
35  !
36  ! AMB 2019/07-2020/06 Initial

840  subroutine eval_lagrange_bases(gll, np, x, y)

855  f = one

865  subroutine gfr_init_m_gf(np, nphys, M_gf, scale)
866  ! Compute the mixed mass matrix with range the FV subcells and
867  ! domain the GLL nodes.
868 
869  use quadrature_mod, only : gausslobatto, quadrature_t
870 
871  integer, intent(in) :: np, nphys
872  real(kind=real_kind), intent(out) :: m_gf(:,:,:,:)
873  logical, intent(in) :: scale
874 
875  type (quadrature_t) :: gll
876  integer :: gi, gj, fi, fj, qi, qj
877  real(kind=real_kind) :: xs, xe, ys, ye, ref, bi(np), bj(np)
878 
879  gll = gausslobatto(np)
880 
881  m_gf = zero
882 
883  do fj = 1,nphys
884  ! The subcell is [xs,xe]x[ys,ye].
885  xs = two*real(fj-1, real_kind)/real(nphys, real_kind) - one
886  xe = two*real(fj, real_kind)/real(nphys, real_kind) - one
887  do fi = 1,nphys
888  ys = two*real(fi-1, real_kind)/real(nphys, real_kind) - one
889  ye = two*real(fi, real_kind)/real(nphys, real_kind) - one
890  ! Use GLL quadrature within this subcell.
891  do qj = 1,np
892  ! (xref,yref) are w.r.t. the [-1,1]^2 reference domain mapped to
893  ! the subcell.
894  ref = xs + half*(xe - xs)*(one + gll%points(qj))
895  call eval_lagrange_bases(gll, np, ref, bj)
896  do qi = 1,np
897  ref = ys + half*(ye - ys)*(one + gll%points(qi))
898  call eval_lagrange_bases(gll, np, ref, bi)
899  do gj = 1,np
900  do gi = 1,np
901  ! Accumulate each GLL basis's contribution to this
902  ! subcell.
903  m_gf(gi,gj,fi,fj) = m_gf(gi,gj,fi,fj) + &
904  gll%weights(qi)*gll%weights(qj)*bi(gi)*bj(gj)
905  end do
906  end do
907  end do
908  end do
909  end do
910  end do
911 
912  m_gf = m_gf/real(nphys*nphys, real_kind)
913 
914  if (scale) then
915  ! Scale so the sum over FV subcells gives the GLL weights to machine
916  ! precision.
917  do gj = 1,np
918  do gi = 1,np
919  m_gf(gi,gj,:nphys,:nphys) = m_gf(gi,gj,:nphys,:nphys)* &
920  ((gll%weights(gi)*gll%weights(gj))/ &
921  sum(m_gf(gi,gj,:nphys,:nphys)))
922  end do
923  end do
924  end if
925 
926  call gll_cleanup(gll)
927  end subroutine gfr_init_m_gf

929  subroutine gfr_init_r(np, nphys, w_gg, M_gf, R, tau)
930  ! We want to solve
931  ! min_g 1/2 g'M_gg g - g' M_gf f
932  ! st M_gf' g = M_ff f,
933  ! which gives
934  ! [M_gg -M_gf] [g] = [M_gf f]
935  ! [M_gf' 0 ] [y] [M_ff f].
936  ! Recall M_gg, M_ff are diag. Let
937  ! S = M_gf' inv(M_gg) M_gf.
938  ! Then
939  ! g = inv(M_gg) M_gf inv(S) M_ff f.
940  ! Compute the QR factorization sqrt(inv(M_gg)) M_gf = Q R so that S =
941  ! R'R. In this module, we can take M_gg = diag(w_gg) and M_ff = diag(w_ff)
942  ! with no loss of accuracy.
943  ! If nphys = np, then the problem reduces to
944  ! M_gf' g = M_ff f.
945  ! M_gf is symmetric. We could use the same computations as above
946  ! or, to gain a bit more accuracy, compute the simpler
947  ! M_gf = Q R
948  ! and later solve
949  ! R'Q' g = M_ff f.
950  ! In either case, all of this is one-time initialization; during
951  ! time stepping, just a matvec is computed.
952  !
953  !assume nphys <= np
954 
955  integer, intent(in) :: np, nphys
956  real(kind=real_kind), intent(in) :: w_gg(:,:), m_gf(:,:,:,:)
957  real(kind=real_kind), intent(out) :: r(:,:), tau(:)
958 
959  real(kind=real_kind) :: wrk(np*np*nphys*nphys), v
960  integer :: gi, gj, fi, fj, npsq, info
961 
962  do fj = 1,nphys
963  do fi = 1,nphys
964  do gi = 1,np
965  do gj = 1,np
966  v = m_gf(gi,gj,fi,fj)
967  if (nphys < np) v = v/sqrt(w_gg(gi,gj))
968  r(np*(gi-1) + gj, nphys*(fi-1) + fj) = v
969  end do
970  end do
971  end do
972  end do
973  call dgeqrf(np*np, nphys*nphys, r, size(r,1), tau, wrk, np*np*nphys*nphys, info)
974  end subroutine gfr_init_r

977  ! Compute the matrix that interpolates from the npi-GLL nodes to
978  ! the np-GLL nodes.
979 
980  use quadrature_mod, only : gausslobatto, quadrature_t
981

1033  subroutine gfr_f2g_remapd_op(gfr, R, tau, f, g)

1035  ! problem described in gfr_init_R.
1036 
1037  !assume nphys <= np
1038

1052  ! g = inv(M_sgsg) M_sgf inv(S) M_ff f
1053  wrk = reshape(gfr%w_ff(:nf2), (/nf,nf/))*f(:nf,:nf)
1054  if (nf == npi) then
1055  call dtrsm('l', 'u', 't', 'n', nf2, 1, one, r, size(r,1), wrk, nf2)
1056  call dormqr('l', 'n', nf2, 1, nf2, r, size(r,1), tau, wrk, nf2, wr, np2, info)
1057  g(:npi,:npi) = wrk
1058  else

1060  call dtrtrs('u', 'n', 'n', nf2, 1, r, size(r,1), wrk, nf2, info)

1067  end do
1068  end do
1069  end do
1070  end do
1071  end if

1613  subroutine gfr_pg1_solve(gfr, s, g)
1614  ! Solve
1615  ! min_g* M44 g* = M42 g(corners) ! project a bilinear function onto 4-GLL basis
1616  ! st spheremp'g* = spheremp g ! subject to conserving mass
1617  ! where g(corners) extracts the corner values from the 4-GLL
1618  ! coefficient vector g. (In this explantation, for concreteness I
1619  ! assume np=4, but it doesn't have to be.) This problem
1620  ! reconstructs the field in the element using just the corner
1621  ! values and while conserving mass. This boosts the OOA of pg1.
1622 
1623  type (GllFvRemap_t), intent(in) :: gfr
1624  type (Pg1SolverData_t), intent(in) :: s
1625  real(kind=real_kind), intent(inout) :: g(:,:)
1626 
1627  real(kind=real_kind) :: x(np*np), mass, wr(4)
1628  integer :: np2, i, n, info
1629 
1630  np2 = np*np
1631  n = np2
1632 
1633  ! Form RHS M42 g(corners).
1634  wr(1) = g(1,1); wr(2) = g(np,1); wr(3) = g(1,np); wr(4) = g(np,np)
1635  do i = 1,n
1636  x(i) = sum(s%B(:4,i)*wr(:4))
1637  end do
1638  mass = sum(gfr%w_gg*g)
1639 
1640  ! Solve R'z = b.
1641  call dtrtrs('u', 't', 'n', n, 1, s%Achol, size(s%Achol,1), x, np*np, info)
1642  ! Assemble z + (d - s'z)/(s's) s.
1643  x(:n) = x(:n) + ((mass - sum(s%s(:n)*x(:n)))/s%sts)*s%s(:n)
1644  ! Solve R x = z + (d - s'z)/(s's) s.
1645  call dtrtrs('u', 'n', 'n', n, 1, s%Achol, size(s%Achol,1), x, np*np, info)
1646 
1647  ! Extract g(I).
1648  g = reshape(x(:n), (/np,np/))
1649  end subroutine gfr_pg1_solve

2096  subroutine limiter1_clip_and_sum(n, spheremp, qmin, qmax, dp, q)

2098  ! mixing ratio. Solve
2099  ! min_q* norm(dp q - dp q*, 1)
2100  ! st spheremp'(dp q*) = spheremp'(dp q)
2101  ! qmin < q* < qmax
2102

2104  real (kind=real_kind), intent(in) :: spheremp(:), dp(:)
2105  real (kind=real_kind), intent(inout) :: qmin, qmax, q(:)
2106

2110  real(kind=real_kind) :: x(n*n), c(n*n), v(n*n)
2111

2113  x = q(:n2)

2115 
2116  sumc = sum(c)
2117  mass = sum(c*x)
2118  ! In the case of an infeasible problem, prefer to conserve mass
2119  ! and violate a bound.
2120  if (mass < qmin*sumc) qmin = mass / sumc
2121  if (mass > qmax*sumc) qmax = mass / sumc

2132  elseif (x(k1) < qmin) then
2133  modified = .true.
2134  addmass = addmass + (x(k1) - qmin)*c(k1)
2135  x(k1) = qmin
2136  end if

2138  if (.not. modified) return
2139

2141  ! Determine weights.
2142  if (addmass > zero) then
2143  v = qmax - x
2144  else
2145  v = x - qmin
2146  end if
2147  den = sum(v*c)
2148  if (den > zero) x = x + addmass*(v/den)
2149  end if
2150
