22 // Since we're initializing from inside a Fortran code and don't have access to
23 // char** args to pass to Kokkos::initialize, we need to do some work on our
24 // own. As a side benefit, we'll end up running on GPU platforms optimally
25 // without having to specify --kokkos-ndevices on the command line.
26 void initialize_kokkos () {
27  // This is in fact const char*, but Kokkos::initialize requires char*.
28  std::vector<char*> args;
29 
30  // This is the only way to get the round-robin rank assignment Kokkos
31  // provides, as that algorithm is hardcoded in Kokkos::initialize(int& narg,
32  // char* arg[]). Once the behavior is exposed in the InitArguments version of
33  // initialize, we can remove this string code.
34  // If for some reason we're running on a GPU platform, have Cuda enabled,
35  // but are using a different execution space, this initialization is still
36  // OK. The rank gets a GPU assigned and simply will ignore it.
37 #ifdef KOKKOS_ENABLE_CUDA
38  int nd;
39  const auto ret = cudaGetDeviceCount(&nd);
40  if (ret != cudaSuccess) {
41  // It isn't a big deal if we can't get the device count.
42  nd = 1;
43  }
44  std::stringstream ss;
45  ss << "--kokkos-num-devices=" << nd;
46  const auto key = ss.str();
47  std::vector<char> str(key.size()+1);
48  std::copy(key.begin(), key.end(), str.begin());
49  str.back() = 0;
50  args.push_back(const_cast<char*>(str.data()));
51 #endif
52 
53  const char* silence = "--kokkos-disable-warnings";
54  args.push_back(const_cast<char*>(silence));
55 
56  int narg = args.size();
57  Kokkos::initialize(narg, args.data());
58 }

69 std::pair<int, int>
70 team_num_threads_vectors_from_pool (
71  const int pool_size, const int num_parallel_iterations,
72  const ThreadPreferences tp)
73 {
74  assert(pool_size >= 1);
75  assert(num_parallel_iterations >= 0);
76  assert(tp.max_threads_usable >= 1 && tp.max_vectors_usable >= 1);
77 
78  const int num_threads_per_element = ( (num_parallel_iterations > 0 &&
79  pool_size > num_parallel_iterations) ?
80  pool_size / num_parallel_iterations :
81  1 );
82 
83  const int num_threads = ( (tp.max_threads_usable > num_threads_per_element) ?
84  num_threads_per_element :
85  tp.max_threads_usable );
86 
87  const int p1 = num_threads, p2 = num_threads_per_element / num_threads;
88  return tp.prefer_threads ? std::make_pair(p1, p2) : std::make_pair(p2, p1);
89 }

91 std::pair<int, int>
92 team_num_threads_vectors_for_gpu (
93  const int num_warps_total, const int num_threads_per_warp,
94  const int min_num_warps, const int max_num_warps,
95  const int num_parallel_iterations, const ThreadPreferences tp)
96 {
97  using Homme::nextpow2;
98  using Homme::prevpow2;
99 
100  assert(num_warps_total > 0 && num_threads_per_warp > 0 && min_num_warps > 0);
101  assert(num_parallel_iterations >= 0);
102  assert(min_num_warps <= max_num_warps);
103  assert(num_warps_total >= max_num_warps);
104  assert(tp.max_threads_usable >= 1 && tp.max_vectors_usable >= 1);
105 
106  int num_warps;
107  if (tp.prefer_larger_team) {
108  const int num_warps_usable =
109  (tp.max_threads_usable *
110  // Vector size is limited to threads/warp.
111  std::min(num_threads_per_warp, tp.max_vectors_usable) +
112  num_threads_per_warp - 1) /
113  num_threads_per_warp;
114  num_warps =
115  std::max<int>( min_num_warps,
116  nextpow2(
117  std::min( max_num_warps,
118  std::max( 1, num_warps_usable ))));
119  } else {
120  num_warps =
121  // Min and max keep num_warps in bounds.
122  std::max<int>( min_num_warps,
123  // We want num_warps to divide 32, the number of cores per SM, so
124  // apply nextpow2.
125  nextpow2(
126  std::min( max_num_warps,
127  ( (num_parallel_iterations > 0 &&
128  num_warps_total > num_parallel_iterations) ?
129  (num_warps_total / num_parallel_iterations) :
130  1 ))));
131  }
132 
133  const int num_device_threads = num_warps * num_threads_per_warp;
134 
135  if (tp.prefer_threads) {
136  const int num_threads = ( (tp.max_threads_usable > num_device_threads) ?
137  num_device_threads :
138  tp.max_threads_usable );
139  return std::make_pair( num_threads,
140  prevpow2(num_device_threads / num_threads) );
141  } else {
142  const int num_vectors = prevpow2( (tp.max_vectors_usable > num_device_threads) ?
143  num_device_threads :
144  tp.max_vectors_usable );
145  return std::make_pair( num_device_threads / num_vectors,
146  num_vectors );
147  }
148 }

156  // It appears we can't use Kokkos to tell us this. On current devices, using
157  // fewer than 4 warps/thread block limits the thread occupancy to that
158  // number/4. That seems to be in Cuda specs, but I don't know of a function
159  // that provides this number. Use a configuration option that defaults to 4.
160  const int min_num_warps = HOMMEXX_CUDA_MIN_WARP_PER_TEAM;
161 #ifdef KOKKOS_ENABLE_CUDA
162  const int num_warps_device = Kokkos::Impl::cuda_internal_maximum_concurrent_block_count();
163  const int num_threads_warp = Kokkos::Impl::CudaTraits::WarpSize;
164  // The number on P100 is 16, but for some reason
165  // Kokkos::Impl::cuda_internal_maximum_grid_count() returns 8. I may be
166  // misusing the function. I have an open issue with the Kokkos team to resolve
167  // this. For now:
168 # ifdef KOKKOS_ENABLE_DEBUG
169  // Work around spurious "terminate called after throwing an instance
170  // of 'std::runtime_error'" message.
171  const int max_num_warps = 8;
172 # else
173  const int max_num_warps = HOMMEXX_CUDA_MAX_WARP_PER_TEAM; //Kokkos::Impl::cuda_internal_maximum_grid_count();
174 # endif
175 #else
176  // I want thread-distribution rules to be unit-testable even when Cuda is
177  // off. Thus, make up a P100-like machine:
178  const int num_warps_device = 1792;
179  const int num_threads_warp = 32;
180  const int max_num_warps = 16;
181 #endif
182 
183  return Parallel::team_num_threads_vectors_for_gpu(
184  num_warps_device, num_threads_warp,
185  min_num_warps, max_num_warps,
186  num_parallel_iterations, tp);
187 }
