106  ! Taylor formula [ (1+x)^a = sum (a choose k) x^k ] holds for -1<x<1, so 0<(1+x)<2
107  ! For better convergence, we ask that -0.5<x<0.5, so 0.5<(1+x)<1.5, compute
108  ! the factor (2^e)^k separately, and then combine the two
109  ! Note: all of this is tailored (or, if you wish, taylored...eheh) for 0<e<2
110 
111  if (val .eq. 0) then
112  if (a .eq. 0) then
113  print *, "Cannot do 0^0, sorry."
114  endif
115  y = 0.0
116  else
117  factor = 1.0
118  ! Taylor formula [ (1+x)^a = sum (a choose k) x^k ] holds for -1<x<1, so 0<(1+x)<2
119  ! To converge faster, we enforce 0.5<1+x<1.5. We set x = 2^k * tmp, with tmp in
120  ! the [0.5,1.5] bound. Then raise the two separately to the power e.
121  tmp = val
122  e = a
123 
124  if (e < 0) then
125  e = -e
126  tmp = 1.0d0 / tmp
127  endif
128 
129  if (tmp .ge. 1.5) then
130  k = 0
131  do while (tmp .ge. 16.0)
132  k = k+4
133  tmp = tmp/16.0
134  enddo
135  do while (tmp .ge. 1.5)
136  k = k+1
137  tmp = tmp/2.0
138  enddo
139 
140  factor = int_pow(power_of_two(e),k)
141  else if (tmp .le. 0.5) then
142  k = 0
143  do while(tmp .le. 0.0625)
144  k = k+4
145  tmp = tmp*16.0
146  enddo
147  do while (tmp .le. 0.5)
148  k = k+1
149  tmp = tmp*2.0
150  enddo
151 
152  factor = 1.0/int_pow(power_of_two(e),k)
153  endif
154 
155  ! Note: if order = 0, y=1, which is the initial value.
156  x = tmp - 1
157  y = 1.0d0
158  do n=order,1,-1
159  y = y * (( (e-(n-1))/n ) * x)
160  y = y + 1.0d0
161  enddo
