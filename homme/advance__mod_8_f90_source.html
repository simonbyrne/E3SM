591  ! uses specified global minimum
592  !

594  use dimensions_mod, only : np, nlev
595  use control_mod, only : test_case
596 
597  implicit none
598  real (kind=real_kind), intent(inout) :: q(np,np,nlev)

601

603  integer k

605 
606  do k=nlev,1,-1
607  qmin=0
608  if (test_case=='swirl') then
609  if (k.eq.1) qmin=0.0d0 ! lifted cosine bell
610  if (k.eq.3) qmin=0.1d0 ! lifted slotted cylinder
611  endif
612  if (kmass==-1) then
613  call limiter2d_min_onelevel(q(:,:,k),spheremp(:,:),qmin)
614  else
615  if(k.ne.kmass)then
616  q(:,:,k)=q(:,:,k)/q(:,:,kmass)
617  call limiter2d_min_onelevel(q(:,:,k),spheremp(:,:)*q(:,:,kmass),qmin)

625 !-------------------------------------------------------------------------------
626  subroutine limiter2d_min_onelevel(Q,spheremp,qmin)
627  use kinds, only : real_kind
628  use dimensions_mod, only : np
629 
630  implicit none
631  real (kind=real_kind), intent(inout) :: q(np,np)
632  real (kind=real_kind), intent(in) :: spheremp(np,np)
633  real (kind=real_kind), intent(in) :: qmin
634 
635  ! local
636  integer i,j
637  real (kind=real_kind) :: mass,mass_added,area,mass2
638 
639  if ( minval(q(:,:)) < qmin ) then
640  mass=sum( q(:,:)*spheremp(:,:) )
641  area=sum( spheremp(:,:) )
642  mass2 = mass - area*qmin
643  q(:,:)=q(:,:)-qmin
644 
645  ! negative mass. so reduce all postive values to zero
646  ! then increase negative values as much as possible
647  if (mass2 < 0) q(:,:)=-q(:,:)
648  mass_added=0
649  do j=1,np
650  do i=1,np
651  if (q(i,j)<0) then
652  q(i,j)=0
653  else
654  mass_added = mass_added + q(i,j)*spheremp(i,j)
655  endif
656  enddo
657  enddo
658  ! now scale the all positive values to restore mass
659  if (mass_added>0) q(:,:) = q(:,:)*abs(mass2)/mass_added
660  if (mass2 < 0) q(:,:)=-q(:,:)
661  q(:,:)=q(:,:)+qmin
662  endif
663 
664  end subroutine limiter2d_min_onelevel

714  subroutine advance_hypervis(edge3,elem,hybrid,deriv,vtens,ptens,nt,nets,nete,dt2)
715  !
716  ! take one timestep of:
717  ! u(:,:,:,np) = u(:,:,:,np) + dt2*nu*laplacian**order ( u )
718  ! h(:,:,:,np) = h(:,:,:,np) + dt2*nu_s*laplacian**order ( h )
719  !
720  ! For height equation, we want to dissipate P.E. = gH^2, where H = surface height = h + h_surface
721  ! since dH/dt = dh/dt, we can solve:
722  ! h(:,:,:,np) = h(:,:,:,np) + dt2*nu_s*laplacian**order ( H )
723  !
724  ! (to understand this, think of a flow at rest with topography. H=constant,
725  ! laplace(H)=0. But h= -h_surface and laplacian(h) <> 0
726  !
727  ! For correct scaling, dt2 should be the same 'dt2' used in the leapfrog advace
728  !
729  !
730  use kinds, only : real_kind
731  use dimensions_mod, only : np, nlev
732  use control_mod, only : nu, nu_s, hypervis_order, hypervis_subcycle, limiter_option,&
733  test_case, kmass
734  use hybrid_mod, only : hybrid_t
735  use element_mod, only : element_t
736  use derivative_mod, only : derivative_t, laplace_sphere_wk, vlaplace_sphere_wk
737  use edge_mod, only : edgevpack_nlyr, edgevunpack_nlyr
738  use edgetype_mod, only : edgebuffer_t
739  use bndry_mod, only : bndry_exchangev
740  use viscosity_mod, only : biharmonic_wk, neighbor_minmax
741  implicit none
742 
743  type (hybrid_t) , intent(in) :: hybrid
744  type (element_t) , intent(inout), target :: elem(:)
745  integer :: nt,nets,nete
746  real (kind=real_kind), dimension(np,np,2,nlev,nets:nete) :: vtens
747  real (kind=real_kind), dimension(np,np,nlev,nets:nete) :: ptens
748  type (EdgeBuffer_t) , intent(inout) :: edge3
749  type (derivative_t) , intent(in) :: deriv
750  real (kind=real_kind) :: dt2
751 
752 
753  ! local
754  integer :: k,kptr,i,j,ie,ic
755  real (kind=real_kind), dimension(:,:), pointer :: spheremp,rspheremp
756  real (kind=real_kind), dimension(np,np) :: ptot
757  real (kind=real_kind), dimension(np,np) :: lap_p
758  real (kind=real_kind), dimension(np,np,2) :: lap_v
759  real (kind=real_kind) :: v1,v2,dt
760 
761  if (hypervis_order == 0 ) return;
762  if (nu_s == 0 .and. nu == 0 ) return;
763 
764 
765  do ie=nets,nete
766  do k=1,nlev
767 
768  ! contra -> latlon
769  do j=1,np
770  do i=1,np
771  v1 = elem(ie)%state%v(i,j,1,k,nt) ! contra
772  v2 = elem(ie)%state%v(i,j,2,k,nt) ! contra
773  elem(ie)%state%v(i,j,1,k,nt)=elem(ie)%D(i,j,1,1)*v1 + elem(ie)%D(i,j,1,2)*v2 ! contra->latlon
774  elem(ie)%state%v(i,j,2,k,nt)=elem(ie)%D(i,j,2,1)*v1 + elem(ie)%D(i,j,2,2)*v2 ! contra->latlon
775  enddo
776  enddo
777  enddo
778  enddo
779 
780 
781  dt=dt2/hypervis_subcycle
782 
783 !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
784  ! regular viscosity
785 !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
786  if (hypervis_order == 1) then
787 
788  do ic=1,hypervis_subcycle
789  do ie=nets,nete
790 
791  spheremp => elem(ie)%spheremp
792 
793  do k=1,nlev
794  ! filter surface height, not thickness
795  do j=1,np
796  do i=1,np
797  ptot(i,j)=elem(ie)%state%p(i,j,k,nt) + elem(ie)%state%ps(i,j)
798  enddo
799  enddo
800 
801  lap_p=laplace_sphere_wk(ptot,deriv,elem(ie),var_coef=.false.)
802  lap_v=vlaplace_sphere_wk(elem(ie)%state%v(:,:,:,k,nt),deriv,elem(ie),var_coef=.false.)
803 
804  ! advace in time. (note: DSS commutes with time stepping, so we
805  ! can time advance and then DSS. this has the advantage of
806  ! not letting any discontinuties accumulate in p,v via tol
807  do j=1,np
808  do i=1,np
809  elem(ie)%state%p(i,j,k,nt)=elem(ie)%state%p(i,j,k,nt)*spheremp(i,j) + dt*nu_s*lap_p(i,j)
810  elem(ie)%state%v(i,j,1,k,nt)=elem(ie)%state%v(i,j,1,k,nt)*spheremp(i,j) + dt*nu*lap_v(i,j,1)
811  elem(ie)%state%v(i,j,2,k,nt)=elem(ie)%state%v(i,j,2,k,nt)*spheremp(i,j) + dt*nu*lap_v(i,j,2)
812  enddo
813  enddo
814  enddo
815 
816  kptr=0
817  call edgevpack_nlyr(edge3,elem(ie)%desc,elem(ie)%state%p(:,:,:,nt),nlev,kptr,3*nlev)
818  kptr=nlev
819  call edgevpack_nlyr(edge3,elem(ie)%desc,elem(ie)%state%v(:,:,:,:,nt),2*nlev,kptr,3*nlev)
820  enddo
821 
822  call bndry_exchangev(hybrid,edge3)
823 
824  do ie=nets,nete
825  rspheremp => elem(ie)%rspheremp
826 
827  kptr=0
828  call edgevunpack_nlyr(edge3,elem(ie)%desc,elem(ie)%state%p(:,:,:,nt), nlev, kptr,3*nlev)
829  kptr=nlev
830  call edgevunpack_nlyr(edge3,elem(ie)%desc,elem(ie)%state%v(:,:,:,:,nt), 2*nlev, kptr,3*nlev)
831 
832  ! apply inverse mass matrix
833  do k=1,nlev
834  do j=1,np
835  do i=1,np
836  elem(ie)%state%p(i,j,k,nt)=rspheremp(i,j)*elem(ie)%state%p(i,j,k,nt)
837  elem(ie)%state%v(i,j,1,k,nt)=rspheremp(i,j)*elem(ie)%state%v(i,j,1,k,nt)
838  elem(ie)%state%v(i,j,2,k,nt)=rspheremp(i,j)*elem(ie)%state%v(i,j,2,k,nt)
839  enddo
840  enddo
841  enddo
842  enddo
843  enddo ! subcycle
844  endif
845 
846 
847 !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
848  ! hyper viscosity
849 !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
850  if (hypervis_order == 2) then
851  do ic=1,hypervis_subcycle
852  call biharmonic_wk(elem,ptens,vtens,deriv,edge3,hybrid,nt,nets,nete)
853 
854  do ie=nets,nete
855  spheremp => elem(ie)%spheremp
856  do k=1,nlev
857  ! advace in time.
858  ! note: DSS commutes with time stepping, so we can time advance and then DSS.
859  ! note: weak operators alreayd have mass matrix "included"
860 
861  do j=1,np
862  do i=1,np
863  elem(ie)%state%p(i,j,k,nt) = elem(ie)%state%p(i,j,k,nt)*spheremp(i,j) -dt*nu_s*ptens(i,j,k,ie)
864  elem(ie)%state%v(i,j,1,k,nt)=elem(ie)%state%v(i,j,1,k,nt)*spheremp(i,j) - dt*nu*vtens(i,j,1,k,ie)
865  elem(ie)%state%v(i,j,2,k,nt)=elem(ie)%state%v(i,j,2,k,nt)*spheremp(i,j) - dt*nu*vtens(i,j,2,k,ie)
866  enddo
867  enddo
868  enddo
869 
870  kptr=0
871  call edgevpack_nlyr(edge3,elem(ie)%desc,elem(ie)%state%p(:,:,:,nt),nlev,kptr,3*nlev)
872  kptr=nlev
873  call edgevpack_nlyr(edge3,elem(ie)%desc,elem(ie)%state%v(:,:,:,:,nt),2*nlev,kptr,3*nlev)
874  enddo
875 
876  call bndry_exchangev(hybrid,edge3)
877 
878  do ie=nets,nete
879  rspheremp => elem(ie)%rspheremp
880 
881  kptr=0
882  call edgevunpack_nlyr(edge3,elem(ie)%desc,elem(ie)%state%p(:,:,:,nt), nlev, kptr,3*nlev)
883  kptr=nlev
884  call edgevunpack_nlyr(edge3,elem(ie)%desc,elem(ie)%state%v(:,:,:,:,nt), 2*nlev, kptr,3*nlev)
885 
886  ! apply inverse mass matrix
887  do k=1,nlev
888  do j=1,np
889  do i=1,np
890  elem(ie)%state%p(i,j,k,nt)=rspheremp(i,j)*elem(ie)%state%p(i,j,k,nt)
891  elem(ie)%state%v(i,j,1,k,nt)=rspheremp(i,j)*elem(ie)%state%v(i,j,1,k,nt)
892  elem(ie)%state%v(i,j,2,k,nt)=rspheremp(i,j)*elem(ie)%state%v(i,j,2,k,nt)
893  enddo
894  enddo
895  enddo
896  enddo
897  enddo
898  endif
899 
900 
901  ! convert lat-lon -> contra variant
902  do ie=nets,nete
903  do k=1,nlev
904  do j=1,np
905  do i=1,np
906  v1=elem(ie)%state%v(i,j,1,k,nt)
907  v2=elem(ie)%state%v(i,j,2,k,nt)
908  elem(ie)%state%v(i,j,1,k,nt) = elem(ie)%Dinv(i,j,1,1)*v1 + elem(ie)%Dinv(i,j,1,2)*v2
909  elem(ie)%state%v(i,j,2,k,nt) = elem(ie)%Dinv(i,j,2,1)*v1 + elem(ie)%Dinv(i,j,2,2)*v2
910  enddo
911  enddo
912  enddo
913  enddo
914 
915 
916  ! lat-lon conversion 1740 20%
917  ! opss+pack+mass 5937 68%
918  ! unpack+mass_inv 1015 12%
919  ! total 8692
920  end subroutine advance_hypervis

958  subroutine compute_and_apply_rhs(np1,nm1,n0,dt2,real_time,edge3,elem,pmean,hybrid,deriv,vtens,ptens,nets,nete)

967  ! take a forward euler step, overwriting the input with the output.

970  ! and u(np1) is not changed.
971  !
972  ! Combining the RHS and DSS pack operation in one routine
973  ! allows us to fuse these two loops for more cache reuse
974  !
975  ! Combining the dt advance and DSS unpack operation in one routine
976  ! allows us to fuse these two loops for more cache reuse
977  !
978  ! note: for prescribed velocity case, velocity will be computed at
979  ! "real_time", which should be the time of timelevel n0.
980  ! ===================================
981  use kinds, only : real_kind
982  use dimensions_mod, only : np, nlev
983  use hybrid_mod, only : hybrid_t
984  use element_mod, only : element_t
985  use derivative_mod, only : derivative_t, divergence_sphere, gradient_sphere, vorticity_sphere
986  use edge_mod, only : edgevpack_nlyr, edgevunpack_nlyr

996  real (kind=real_kind), dimension(np,np,nlev,nets:nete) :: ptens
997  type (EdgeBuffer_t) , intent(inout) :: edge3

1000 
1001  ! local
1002  ! pointer ...

1004  real (kind=real_kind), dimension(np,np,2) :: grade ! strong kinetic energy gradient
1005  real (kind=real_kind), dimension(np,np,2) :: pv ! p*v lat-lon
1006  real (kind=real_kind), dimension(np,np) :: e ! kinetic energy term
1007  real (kind=real_kind), dimension(np,np) :: zeta ! relative vorticity
1008  real (kind=real_kind), dimension(np,np) :: div
1009  real (kind=real_kind), dimension(np,np,2) :: ulatlon
1010 
1011  integer i,j,k,kptr,ie
1012  real (kind=real_kind) :: v1,v2
1013  real (kind=real_kind) :: vtens1,vtens2
1014 
1015  call t_startf('compute_and_apply_rhs')
1016 
1017 
1018  ! ===================================
1019  ! construct v tendencies and v.grad(p)
1020  ! on the velocity grid...
1021  ! ===================================
1022  do ie=nets,nete
1023  fcor => elem(ie)%fcor
1024  spheremp => elem(ie)%spheremp
1025 
1026  call set_prescribed_velocity(elem(ie),n0,real_time)
1027 
1028  do k=1,nlev
1029  ! ==============================================
1030  ! Compute kinetic energy term
1031  ! ==============================================
1032  do j=1,np
1033  do i=1,np
1034 
1035  v1 = elem(ie)%state%v(i,j,1,k,n0) ! contra
1036  v2 = elem(ie)%state%v(i,j,2,k,n0) ! contra
1037  ulatlon(i,j,1)=elem(ie)%D(i,j,1,1)*v1 + elem(ie)%D(i,j,1,2)*v2 ! contra->latlon
1038  ulatlon(i,j,2)=elem(ie)%D(i,j,2,1)*v1 + elem(ie)%D(i,j,2,2)*v2 ! contra->latlon
1039 
1040  e(i,j) = 0.5d0*(ulatlon(i,j,1)**2 + ulatlon(i,j,2)**2) +&
1041  elem(ie)%state%p(i,j,k,n0) + elem(ie)%state%ps(i,j)
1042

1044  pv(i,j,2) = ulatlon(i,j,2)*(pmean+elem(ie)%state%p(i,j,k,n0))
1045  end do
1046  end do
1047  grade = gradient_sphere(e,deriv,elem(ie)%Dinv) ! scalar -> latlon vector
