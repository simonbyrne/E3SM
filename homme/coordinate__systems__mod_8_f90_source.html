308  function unit_face_based_cube_to_unit_sphere(cart, face_no) result(sphere)
309 
310 ! Note: Output spherical longitude is [-pi,pi]
311 
312 ! Project from a UNIT cube to a UNIT sphere. ie, the lenght of the cube edge is 2.
313 ! Face 1 of the cube touches the sphere at longitude, latitude (0,0). The negative
314 ! x axis is negative longitude (ie. going west is negative), the positive x axis
315 ! is increasing longitude. Face 1 maps the Face 1 to the lat,lon on the sphere:
316 ! [-1,1] x [-1,1] => [-\pi/4,\pi/4] x [-\pi/4, \pi/4]
317 
318 ! Face 2 continues with increasing longitude (ie to the east of Face 1).
319 ! The left edge of Face 2 (negative x) is the right edge of Face 1 (positive x)
320 ! The latitude is the same as Face 1, but the longitude increases:
321 ! [-1,1] x [-1,1] => [\pi/4, 3\pi/4] x [-\pi/4, \pi/4]
322 
323 ! Face 3 continues with increasing longitude (ie to the east of Face 2).
324 ! Face 3 is like Face 1, but the x coordinates are reversed, ie. decreasing x
325 ! is increasing longitude:
326 ! [-1,1] x [-1,1] = [-1,0] x [-1,1] U [0,1] x [-1,1] =>
327 ! [3\pi/4,\pi] x [-\pi, -3\pi/4]
328 
329 ! Face 4 finally connects Face 3 to Face 1. Like Face 2, but wtih opposite x
330 ! [-1,1] x [-1,1] => [-3\pi/4, -\pi/4] x [-\pi/4, \pi/4]
331 
332 ! Face 5 is along the bottom edges of Faces 1,2,3,and 4 so the latitude goes from
333 ! -\pi/4 to -\pi/2. The tricky part is lining up the longitude. The zero longitude
334 ! must line up with the center of Face 1. ATAN2(x,1) = 0 => x = 0.
335 ! So the (0,1) point on Face 5 is the zero longitude on the sphere. The top edge of
336 ! Face 5 is the bottom edge of Face 1.
337 ! ATAN(x,0) = \pi/2 => x = 1, so the right edge of Face 5 is the bottom of Face 2.
338 ! Continueing, the bottom edge of 5 is the bottom of 3. Left of 5 is bottom of 4.
339 
340 ! Face 6 is along the top edges of Faces 1,2,3 and 4 so the latitude goes from
341 ! \pi/4 to \pi/2. The zero longitude must line up with the center of Face 1.
342 ! This is just like Face 5, but the y axis is reversed. So the bottom edge of Face 6
343 ! is the top edge of Face 1. The right edge of Face 6 is the top of Face 2. The
344 ! top of 6 the top of 3 and the left of 6 the top of 4.
345 
346  use physical_constants, only : dd_pi
347  use parallel_mod, only : abortmp
348  implicit none
349  type (cartesian2d_t), intent(in) :: cart ! On face_no of a unit cube
350  integer, intent(in) :: face_no
351 
352  type (spherical_polar_t) :: sphere
353 
354  integer i,j
355  real(kind=real_kind) :: r!, l_inf
356 
357 ! MNL: removing check that points are on the unit cube because we allow
358 ! spherical grids to map beyond the extent of the cube (though we probably
359 ! should still have an upper bound for how far past the edge the element lies)
360 ! l_inf = MAX(ABS(cart%x), ABS(cart%y))
361 ! if (1.01 < l_inf) then
362 ! call abortmp('unit_face_based_cube_to_unit_sphere: Input not on unit cube.')
363 ! end if
364 
365  sphere%r=one
366  r = sqrt( one + (cart%x)**2 + (cart%y)**2)
367  select case (face_no)
368  case (1)
369  sphere%lat=asin((cart%y)/r)
370  sphere%lon=atan2(cart%x,one)
371  case (2)
372  sphere%lat=asin((cart%y)/r)
373  sphere%lon=atan2(one,-cart%x)
374  case (3)
375  sphere%lat=asin((cart%y)/r)
376  sphere%lon=atan2(-cart%x,-one)
377  case (4)
378  sphere%lat=asin((cart%y)/r)
379  sphere%lon=atan2(-one,cart%x)
380  case (5)
381  if (abs(cart%y) > dist_threshold .or. abs(cart%x) > dist_threshold ) then
382  sphere%lon=atan2(cart%x, cart%y )
383  else
384  sphere%lon= 0.0d0 ! longitude is meaningless at south pole set to 0.0
385  end if
386  sphere%lat=asin(-one/r)
387  case (6)
388  if (abs(cart%y) > dist_threshold .or. abs(cart%x) > dist_threshold ) then
389  sphere%lon = atan2(cart%x, -cart%y)
390  else
391  sphere%lon= 0.0d0 ! longitude is meaningless at north pole set to 0.0
392  end if
393  sphere%lat=asin(one/r)
394  case default
395  call abortmp('unit_face_based_cube_to_unit_sphere: Face number not 1 to 6.')
396  end select
397 
398  if (sphere%lon < 0.0d0) then
399  sphere%lon=sphere%lon + two*dd_pi
400  end if
401 
402  end function unit_face_based_cube_to_unit_sphere

553  real(kind=real_kind) :: lat,lon
554  real(kind=real_kind) :: pi,twopi, pi2, pi3, pi4
555 
556  lat = sphere%lat
557  lon = sphere%lon
558 
559  pi = dd_pi
560  twopi = 2.0d0 * pi
561  pi2 = pi * 0.5d0
562  pi3 = pi * 1.5d0
563  pi4 = pi * 0.25d0
564

568  if (pi < lon) xp = lon - twopi !if lon in [0,2\pi]

570  case (2)
571  xp = lon - pi2
572  yp = atan(tan(lat)/cos(xp))
573  case (3)
574  xp = lon - pi
575  if (lon < 0) xp = lon + pi !if lon in [0,2\pi]
576  yp = atan(tan(lat)/cos(xp))
577  case (4)
578  xp = lon - pi3
579  if (lon < 0) xp = lon + pi2 !if lon in [0,2\pi]
580  yp = atan(tan(lat)/cos(xp))
581  case (5)
582  xp = atan(-sin(lon)/tan(lat))
583  yp = atan(-cos(lon)/tan(lat))
584  case (6)
585  xp = atan( sin(lon)/tan(lat))
586  yp = atan(-cos(lon)/tan(lat))

651 ! This function divides three dimentional space up into
652 ! six sectors. These sectors are then considered as the
653 ! faces of the cube. It should work for any (x,y,z) coordinate
654 ! if on a sphere or on a cube.
655  pure function cube_face_number_from_cart(cart) result(face_no)
656 
657  implicit none
658  type(cartesian3d_t),intent(in) :: cart
659  integer :: face_no
660 
661  real(real_kind) :: x,y,z
662  x=cart%x
663  y=cart%y
664  z=cart%z
665 
666 ! Divide the X-Y plane into for quadrants of
667 ! [-\pi/2,\pi/2], [\pi/2,3\pi/2], .....
668 ! based on the lines X=Y and X=-Y. This divides
669 ! 3D space up into four sections. Doing the same
670 ! for the XZ and YZ planes divides space into six
671 ! sections. Can also be thought of as conic sections
672 ! in the L_infinity norm.
673 
674  if (y<x .and. y>-x) then ! x>0, Face 1,5 or 6
675  if (z>x) then
676  face_no=6 ! north pole
677  else if (z<-x) then
678  face_no=5 ! south pole
679  else
680  face_no = 1
681  endif
682  else if (y>x .and. y<-x) then ! x<0
683  if (z>-x) then
684  face_no=6 ! north pole
685  else if (z<x) then
686  face_no=5 ! south pole
687  else
688  face_no=3
689  endif
690  else if (y>x .and. y>-x) then ! y>0
691  if (z>y) then
692  face_no=6 ! north pole
693  else if (z<-y) then
694  face_no = 5 ! south pole
695  else
696  face_no=2
697  endif
698  else if (y<x .and. y<-x) then ! y<0
699  if (z>-y) then
700  face_no=6 ! north pole
701  else if (z<y) then
702  face_no=5 ! south pole
703  else
704  face_no=4
705  endif
706  else
707  ! abs(y) = abs(x). point is on cube edge, or on face 5 or 6:
708  if (abs(x)<z) then
709  face_no=6 ! north pole
710  else if (z<-abs(x)) then
711  face_no=5 ! south pole
712  else if (0 < x .and. 0 < y) then
713  face_no = 1
714  else if (x < 0 .and. 0 < y) then
715  face_no = 2
716  else if (x < 0 .and. y < 0) then
717  face_no = 3
718  else
719  face_no = 4
720  endif
721  endif
722 
723  end function cube_face_number_from_cart

772  subroutine sphere_tri_area( v1, v2, v3, area )
773  ! input: v1(3),v2(3),v3(3) cartesian coordinates of triangle
774  ! output: area
775  ! based on formulas in STRI_QUAD:
776  ! http://people.sc.fsu.edu/~burkardt/f_src/stri_quad/stri_quad.html
777  !
778  ! MT: note that the usual Girard formula used below is ill-conditioned
779  ! for small nearly flat triangles (which is probably our case).
780  ! I put in a crude fix below.
781  ! We should instead be using l'Huiller's formula,
782  ! see: http://williams.best.vwh.net/avform.htm
783  use physical_constants, only : dd_pi
784  implicit none
785  real(kind=real_kind) area
786  real(kind=real_kind) a,b,c,al,bl,cl,sina,sinb,sinc,sins,a1,b1,c1
787  type (cartesian3d_t) v1,v2,v3
788 
789  ! compute great circle lengths
790  al = acos( v2%x * v3%x + v2%y * v3%y + v2%z * v3%z )
791  bl = acos( v3%x * v1%x + v3%y * v1%y + v3%z * v1%z )
792  cl = acos( v1%x * v2%x + v1%y * v2%y + v1%z * v2%z )
793 
794  ! compute angles
795  sina = sin( (bl+cl-al)/2 ) ! sin(sl-al)
796  sinb = sin( (al+cl-bl)/2 ) ! sin(sl-bl)
797  sinc = sin( (al+bl-cl)/2 )
798  sins = sin( (al+bl+cl)/2 )
799 
800 
801 #if 0
802  a = 2*atan2(sqrt(sinb*sinc), sqrt(sins*sina) )
803  b = 2*atan2(sqrt(sina*sinc), sqrt(sins*sinb) )
804  c = 2*atan2(sqrt(sina*sinb), sqrt(sins*sinc) )
805  ! apply Girard's theorem
806  area = a+b+c - dd_pi
807 #endif
808 
809  ! for small areas, formula above looses precision.
810  ! 2atan(x) + 2atan(1/x) = pi
811  ! 2atan(x) - pi = -2atan(1/x)
812  a = sqrt( (sinb*sinc) / (sins*sina) )
813  b = sqrt( (sina*sinc) / (sins*sinb) )
814  c = sqrt( (sina*sinb) / (sins*sinc) )
815 
816 
817  a1 = 2*atan(a)
818  b1 = 2*atan(b)
819  c1 = 2*atan(c)
820 
821  if (a.gt.b.and.a.gt.c) then
822  a1 = -2*atan(1/a)
823  else if (b.gt.c) then
824  b1 = -2*atan(1/b)
825  else
826  c1 = -2*atan(1/c)
827  endif
828  ! apply Girard's theorem
829  area = a1+b1+c1
830 
831 
832  end subroutine sphere_tri_area
